119 /* Change the value field of the node to the content of buffer value
120  of size LEN. With VALUE of NULL or LEN of 0 the value field is
121  deleted */
122 void
123 _ksba_asn_set_value (AsnNode node,
124  enum asn_value_type vtype, const void *value, size_t len)
125 {
126  return_if_fail (node);
127 
128  if (node->valuetype)
129  {
130  if (node->valuetype == VALTYPE_CSTR)
131  xfree (node->value.v_cstr);
132  else if (node->valuetype == VALTYPE_MEM)
133  xfree (node->value.v_mem.buf);
134  node->valuetype = 0;
135  }
136 
137  switch (vtype)
138  {
139  case VALTYPE_NULL:
140  break;
141  case VALTYPE_BOOL:
142  return_if_fail (len);
143  node->value.v_bool = !!(const unsigned *)value;
144  break;
145  case VALTYPE_CSTR:
146  node->value.v_cstr = xstrdup (value);
147  break;
148  case VALTYPE_MEM:
149  node->value.v_mem.len = len;
150  if (len)
151  {
152  node->value.v_mem.buf = xmalloc (len);
153  memcpy (node->value.v_mem.buf, value, len);
154  }
155  else
156  node->value.v_mem.buf = NULL;
157  break;
158  case VALTYPE_LONG:
159  return_if_fail (sizeof (long) == len);
160  node->value.v_long = *(long *)value;
161  break;
162 
163  case VALTYPE_ULONG:
164  return_if_fail (sizeof (unsigned long) == len);
165  node->value.v_ulong = *(unsigned long *)value;
166  break;
167 
168  default:
169  return_if_fail (0);
170  }
171  node->valuetype = vtype;
172 }

174 static void
175 copy_value (AsnNode d, const AsnNode s)
176 {
177  char helpbuf[1];
178  const void *buf = NULL;
179  size_t len = 0;
180 
181  return_if_fail (d != s);
182 
183  switch (s->valuetype)
184  {
185  case VALTYPE_NULL:
186  break;
187  case VALTYPE_BOOL:
188  len = 1;
189  helpbuf[0] = s->value.v_bool;
190  buf = helpbuf;
191  break;
192  case VALTYPE_CSTR:
193  buf = s->value.v_cstr;
194  break;
195  case VALTYPE_MEM:
196  len = s->value.v_mem.len;
197  buf = len? s->value.v_mem.buf : NULL;
198  break;
199  case VALTYPE_LONG:
200  len = sizeof (long);
201  buf = &s->value.v_long;
202  break;
203  case VALTYPE_ULONG:
204  len = sizeof (unsigned long);
205  buf = &s->value.v_ulong;
206  break;
207 
208  default:
209  return_if_fail (0);
210  }
211  _ksba_asn_set_value (d, s->valuetype, buf, len);
212  d->off = s->off;
213  d->nhdr = s->nhdr;
214  d->len = s->len;
215 }

291 /* find the node with the given name. A name part of "?LAST" matches
292  the last element of a set of */
293 static AsnNode
294 find_node (AsnNode root, const char *name, int resolve)
295 {
296  AsnNode p;
297  const char *s;
298  char buf[129];
299  int i;
300 
301  if (!name || !name[0])
302  return NULL;
303 
304  /* find the first part */
305  s = name;
306  for (i=0; *s && *s != '.' && i < DIM(buf)-1; s++)
307  buf[i++] = *s;
308  buf[i] = 0;
309  return_null_if_fail (i < DIM(buf)-1);
310 
311  for (p = root; p && (!p->name || strcmp (p->name, buf)); p = p->right)
312  ;
313 
314  /* find other parts */
315  while (p && *s)
316  {
317  assert (*s == '.');
318  s++; /* skip the dot */
319 
320  if (!p->down)
321  return NULL; /* not found */
322  p = p->down;
323 
324  for (i=0; *s && *s != '.' && i < DIM(buf)-1; s++)
325  buf[i++] = *s;
326  buf[i] = 0;
327  return_null_if_fail (i < DIM(buf)-1);
328 
329  if (!*buf)
330  {
331  /* a double dot can be used to get over an unnamed sequence
332  in a set - Actually a hack to workaround a bug. We should
333  rethink the entire node naming issue */
334  }
335  else if (!strcmp (buf, "?LAST"))
336  {
337  if (!p)
338  return NULL;
339  while (p->right)
340  p = p->right;
341  }
342  else
343  {
344  for (; p ; p = p->right)
345  {
346  if (p->name && !strcmp (p->name, buf))
347  break;
348  if (resolve && p->name && p->type == TYPE_IDENTIFIER)
349  {
350  AsnNode p2;
351 
352  p2 = resolve_identifier (root, p, 0);
353  if (p2 && p2->name && !strcmp (p2->name, buf))
354  break;
355  }
356  }
357 
358  if (resolve && p && p->type == TYPE_IDENTIFIER)
359  p = resolve_identifier (root, p, 0);
360  }
361  }
362 
363  return p;
364 }

400 static void
401 print_value (AsnNode node, FILE *fp)
402 {
403  if (!node->valuetype)
404  return;
405  fprintf (fp, " vt=%d val=", node->valuetype);
406  switch (node->valuetype)
407  {
408  case VALTYPE_BOOL:
409  fputs (node->value.v_bool? "True":"False", fp);
410  break;
411  case VALTYPE_CSTR:
412  fputs (node->value.v_cstr, fp);
413  break;
414  case VALTYPE_MEM:
415  {
416  size_t n;
417  unsigned char *p;
418  for (p=node->value.v_mem.buf, n=node->value.v_mem.len; n; n--, p++)
419  fprintf (fp, "%02X", *p);
420  }
421  break;
422  case VALTYPE_LONG:
423  fprintf (fp, "%ld", node->value.v_long);
424  break;
425  case VALTYPE_ULONG:
426  fprintf (fp, "%lu", node->value.v_ulong);
427  break;
428  default:
429  return_if_fail (0);
430  }

654 int
655 _ksba_asn_delete_structure (AsnNode root)
656 {
657  AsnNode p, p2, p3;
658 
659  if (root == NULL)
660  return gpg_error (GPG_ERR_ELEMENT_NOT_FOUND);
661 
662  p = root;
663  while (p)
664  {
665  if (p->down)
666  {
667  p = p->down;
668  }
669  else
670  { /* no down */
671  p2 = p->right;
672  if (p != root)
673  {
674  p3 = find_up (p);
675  set_down (p3, p2);
676  _ksba_asn_remove_node (p);
677  p = p3;
678  }
679  else
680  { /* p==root */
681  p3 = _asn1_find_left (p);
682  if (!p3)
683  {
684  p3 = find_up (p);
685  if (p3)
686  set_down (p3, p2);
687  else
688  {
689  if (p->right)
690  p->right->left = NULL;
691  }
692  }
693  else
694  set_right (p3, p2);
695  _ksba_asn_remove_node (p);
696  p = NULL;
697  }
698  }
699  }
700  return 0;
701 }

704 /* check that all identifiers referenced in the tree are available */
705 int
706 _ksba_asn_check_identifier (AsnNode node)
707 {
708  AsnNode p, p2;
709  char name2[129];
710 
711  if (!node)
712  return gpg_error (GPG_ERR_ELEMENT_NOT_FOUND);
713 
714  for (p = node; p; p = _ksba_asn_walk_tree (node, p))
715  {
716  if (p->type == TYPE_IDENTIFIER && p->valuetype == VALTYPE_CSTR)
717  {
718  if (strlen (node->name)+strlen(p->value.v_cstr)+2 > DIM(name2))
719  return gpg_error (GPG_ERR_BUG); /* well identifier too long */
720  strcpy (name2, node->name);
721  strcat (name2, ".");
722  strcat (name2, p->value.v_cstr);
723  p2 = _ksba_asn_find_node (node, name2);
724  if (!p2)
725  {
726  fprintf (stderr,"reference to `%s' not found\n", name2);
727  return gpg_error (GPG_ERR_IDENTIFIER_NOT_FOUND);
728  }
729 /* fprintf (stdout,"found reference for `%s' (", name2); */
730 /* print_node (p2, stdout); */
731 /* fputs (")\n", stdout); */
732  }
733  else if (p->type == TYPE_OBJECT_ID && p->flags.assignment)
734  { /* an object ID in an assignment */
735  p2 = p->down;
736  if (p2 && (p2->type == TYPE_CONSTANT))
737  {
738  if (p2->valuetype == VALTYPE_CSTR && !isdigit (p2->value.v_cstr[0]))
739  { /* the first constand below is a reference */
740  if (strlen (node->name)
741  +strlen(p->value.v_cstr)+2 > DIM(name2))
742  return gpg_error (GPG_ERR_BUG); /* well identifier too long */
743  strcpy (name2, node->name);
744  strcat (name2, ".");
745  strcat (name2, p2->value.v_cstr);
746  p2 = _ksba_asn_find_node (node, name2);
747  if (!p2)
748  {
749  fprintf (stderr,"object id reference `%s' not found\n",
750  name2);
751  return gpg_error (GPG_ERR_IDENTIFIER_NOT_FOUND);
752  }
753  else if ( p2->type != TYPE_OBJECT_ID
754  || !p2->flags.assignment )
755  {
756  fprintf (stderr,"`%s' is not an object id\n", name2);
757  return gpg_error (GPG_ERR_IDENTIFIER_NOT_FOUND);
758  }
759 /* fprintf (stdout,"found objid reference for `%s' (", name2); */
760 /* print_node (p2, stdout); */
761 /* fputs (")\n", stdout); */
762  }
763  }
764  }
765  }
766 
767  return 0;
768 }
