81 /* Initialize the ath subsystem. This is called as part of the
82  Libgcrypt initialization. It's purpose is to initialize the
83  locking system. It returns 0 on sucess or an ERRNO value on error.
84  In the latter case it is not defined whether ERRNO was changed.
85 
86  Note: This should be called as early as possible because it is not
87  always possible to detect the thread model to use while already
88  running multi threaded. */
89 int
90 ath_init (void)
91 {
92  int err = 0;
93 
94  if (thread_model)
95  return 0; /* Already initialized - no error. */
96 
97  if (0)
98  ;
99 #if USE_POSIX_THREADS_WEAK
100  else if (pthread_cancel)
101  {
102  thread_model = ath_model_pthreads_weak;
103  }
104 #endif
105  else
106  {
107 #if HAVE_W32_SYSTEM
108  thread_model = ath_model_w32;
109 #elif USE_POSIX_THREADS && !USE_POSIX_THREADS_WEAK
110  thread_model = ath_model_pthreads;
111 #else
112  /* Assume a single threaded application. */
113  thread_model = ath_model_none;
114 #endif
115  }
116 
117  return err;
118 }

140 /* This function was used in old Libgcrypt versions (via
141  GCRYCTL_SET_THREAD_CBS) to register the thread callback functions.
142  It is not anymore required. However to allow existing code to
143  continue to work, we keep this function and check that no user
144  defined callbacks are used and that the requested thread system
145  matches the one Libgcrypt is using. */
146 gpg_err_code_t
147 ath_install (struct ath_ops *ath_ops)
148 {
149  gpg_err_code_t rc;
150  unsigned int thread_option;
151 
152  /* Fist call ath_init so that we know our thread model. */
153  rc = ath_init ();
154  if (rc)
155  return rc;
156 
157  /* Check if the requested thread option is compatible to the
158  thread option we are already committed to. */
159  thread_option = ath_ops? GET_OPTION (ath_ops->option) : 0;
160 
161  /* Return an error if the requested thread model does not match the
162  configured one. */
163  if (0)
164  ;
165 #if USE_POSIX_THREADS
166  else if (thread_model == ath_model_pthreads
167  || thread_model == ath_model_pthreads_weak)
168  {
169  if (thread_option == ATH_THREAD_OPTION_PTHREAD)
170  return 0; /* Okay - compatible. */
171  if (thread_option == ATH_THREAD_OPTION_PTH)
172  return 0; /* Okay - compatible. */
173  }
174 #endif /*USE_POSIX_THREADS*/
175  else if (thread_option == ATH_THREAD_OPTION_PTH)
176  {
177  if (thread_model == ath_model_none)
178  return 0; /* Okay - compatible. */
179  }
180  else if (thread_option == ATH_THREAD_OPTION_DEFAULT)
181  return 0; /* No thread support requested. */
182 #if HAVE_W32_SYSTEM
183  else
184  return 0; /* It is always enabled. */
185 #endif /*HAVE_W32_SYSTEM*/
186 
187  return GPG_ERR_NOT_SUPPORTED;
188 }

191 /* Initialize a new mutex. This function returns 0 on success or an
192  system error code (i.e. an ERRNO value). ERRNO may or may not be
193  changed on error. */
194 int
195 ath_mutex_init (ath_mutex_t *lock)
196 {
197  int err;
198 
199  switch (thread_model)
200  {
201  case ath_model_none:
202  *lock = MUTEX_UNLOCKED;
203  err = 0;
204  break;
205 
206 #if USE_POSIX_THREADS
207  case ath_model_pthreads:
208  case ath_model_pthreads_weak:
209  {
210  pthread_mutex_t *plck;
211 
212  plck = malloc (sizeof *plck);
213  if (!plck)
214  err = errno? errno : ENOMEM;
215  else
216  {
217  err = pthread_mutex_init (plck, NULL);
218  if (err)
219  free (plck);
220  else
221  *lock = (void*)plck;
222  }
223  }
224  break;
225 #endif /*USE_POSIX_THREADS*/
226 
227 #if HAVE_W32_SYSTEM
228  case ath_model_w32:
229  {
230  CRITICAL_SECTION *csec;
231 
232  csec = malloc (sizeof *csec);
233  if (!csec)
234  err = errno? errno : ENOMEM;
235  else
236  {
237  InitializeCriticalSection (csec);
238  *lock = (void*)csec;
239  err = 0;
240  }
241  }
242  break;
243 #endif /*HAVE_W32_SYSTEM*/
244 
245  default:
246  err = EINVAL;
247  break;
248  }
249 
250  return err;
251 }

254 /* Destroy a mutex. This function is a NOP if LOCK is NULL. If the
255  mutex is still locked it can't be destroyed and the function
256  returns EBUSY. ERRNO may or may not be changed on error. */
257 int
258 ath_mutex_destroy (ath_mutex_t *lock)
259 {
260  int err;
261 
262  if (!*lock)
263  return 0;
264 
265  switch (thread_model)
266  {
267  case ath_model_none:
268  if (*lock != MUTEX_UNLOCKED)
269  err = EBUSY;
270  else
271  {
272  *lock = MUTEX_DESTROYED;
273  err = 0;
274  }
275  break;
276 
277 #if USE_POSIX_THREADS
278  case ath_model_pthreads:
279  case ath_model_pthreads_weak:
280  {
281  pthread_mutex_t *plck = (pthread_mutex_t*) (*lock);
282 
283  err = pthread_mutex_destroy (plck);
284  if (!err)
285  {
286  free (plck);
287  lock = NULL;
288  }
289  }
290  break;
291 #endif /*USE_POSIX_THREADS*/
292 
293 #if HAVE_W32_SYSTEM
294  case ath_model_w32:
295  {
296  CRITICAL_SECTION *csec = (CRITICAL_SECTION *)(*lock);
297 
298  DeleteCriticalSection (csec);
299  free (csec);
300  err = 0;
301  }
302  break;
303 #endif /*HAVE_W32_SYSTEM*/
304 
305  default:
306  err = EINVAL;
307  break;
308  }
309 
310  return err;
311 }

314 /* Lock the mutex LOCK. On success the function returns 0; on error
315  an error code. ERRNO may or may not be changed on error. */
316 int
317 ath_mutex_lock (ath_mutex_t *lock)
318 {
319  int err;
320 
321  switch (thread_model)
322  {
323  case ath_model_none:
324  if (*lock == MUTEX_NOTINIT)
325  err = EINVAL;
