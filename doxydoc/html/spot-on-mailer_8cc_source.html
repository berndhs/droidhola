336 void spoton_mailer::slotRetrieveMail
337 (const QByteArray &data,
338  const QByteArray &publicKeyHash,
339  const QByteArray &timestamp,
340  const QByteArray &signature,
341  const QPairByteArrayByteArray &adaptiveEchoPair)
342 {
343  /*
344  ** We must locate the public key that's associated with the provided
345  ** public key hash. Remember, publicKeyHash is the hash of the signature
346  ** public key.
347  */
348 
349  QByteArray publicKey
350  (spoton_misc::publicKeyFromHash(publicKeyHash,
351  spoton_kernel::s_crypts.value("email",
352  0)));
353 
354  if(publicKey.isEmpty())
355  {
356  spoton_misc::logError("spoton_mailer::slotRetrieveMail(): "
357  "empty public key from hash.");
358  return;
359  }
360 
361  if(!spoton_crypt::isValidSignature(data,
362  publicKey,
363  signature))
364  {
365  spoton_misc::logError("spoton_mailer::slotRetrieveMail(): "
366  "invalid signature.");
367  return;
368  }
369 
370  publicKey = spoton_misc::publicKeyFromSignaturePublicKeyHash
371  (publicKeyHash, spoton_kernel::s_crypts.value("email", 0));
372 
373  if(publicKey.isEmpty())
374  {
375  spoton_misc::logError("spoton_mailer::slotRetrieveMail(): "
376  "empty public key from signature hash.");
377  return;
378  }
379 
380  QByteArray hash;
381  bool ok = true;
382 
383  hash = spoton_crypt::sha512Hash(publicKey, &ok);
384 
385  if(!ok)
386  {
387  spoton_misc::logError
388  ("spoton_mailer::slotRetrieveMail(): "
389  "spoton_crypt::sha512Hash() failure.");
390  return;
391  }
392 
393  QDateTime dateTime
394  (QDateTime::fromString(timestamp.constData(), "MMddyyyyhhmmss"));
395 
396  if(!dateTime.isValid())
397  {
398  spoton_misc::logError
399  ("spoton_mailer::slotRetrieveMail(): "
400  "invalid dateTime object.");
401  return;
402  }
403 
404  QDateTime now(QDateTime::currentDateTimeUtc());
405 
406  dateTime.setTimeSpec(Qt::UTC);
407  now.setTimeSpec(Qt::UTC);
408 
409  qint64 secsTo = qAbs(now.secsTo(dateTime));
410 
411  if(!(secsTo <= static_cast<qint64> (spoton_common::
412  MAIL_TIME_DELTA_MAXIMUM)))
413  {
414  spoton_misc::logError
415  (QString("spoton_mailer::slotRetrieveMail(): "
416  "large time delta (%1).").arg(secsTo));
417  return;
418  }
419  else if(spoton_kernel::duplicateEmailRequests(data))
420  {
421  spoton_misc::logError
422  ("spoton_mailer::slotRetrieveMail(): duplicate requests.");
423  return;
424  }
425 
426  spoton_kernel::emailRequestCacheAdd(data);
427 
428  QList<QByteArray> list;
429 
430  list << hash << adaptiveEchoPair.first << adaptiveEchoPair.second;
431 
432  if(!m_publicKeyHashesAdaptiveEchoPairs.contains(list))
433  m_publicKeyHashesAdaptiveEchoPairs.append(list);
434 
435  if(!m_retrieveMailTimer.isActive())
436  m_retrieveMailTimer.start();
437 }

439 void spoton_mailer::slotRetrieveMailTimeout(void)
440 {
441  if(m_publicKeyHashesAdaptiveEchoPairs.isEmpty())
442  {
443  m_retrieveMailTimer.stop();
444  return;
445  }
446 
447  /*
448  ** We're assuming that only authenticated participants
449  ** can request their e-mail. Let's hope our implementation
450  ** of digital signatures is correct.
451  */
452 
453  QString connectionName("");
454 
455  {
456  QSqlDatabase db = spoton_misc::database(connectionName);
457 
458  db.setDatabaseName
459  (spoton_misc::homePath() + QDir::separator() + "email.db");
460 
461  if(db.open())
462  {
463  QByteArray publicKeyHash(m_publicKeyHashesAdaptiveEchoPairs.
464  first().value(0));
465  QSqlQuery query(db);
466 
467  query.setForwardOnly(true);
468  query.prepare("SELECT message_bundle, OID FROM post_office "
469  "WHERE recipient_hash = ?");
470  query.bindValue(0, publicKeyHash.toBase64());
471 
472  if(query.exec())
473  {
474  if(query.next())
475  {
476  spoton_crypt *s_crypt =
477  spoton_kernel::s_crypts.value("email", 0);
478 
479  if(s_crypt)
480  {
481  /*
482  ** Is this a letter?
483  */
484 
485  QByteArray message;
486  bool ok = true;
487 
488  message = s_crypt->
489  decryptedAfterAuthenticated
490  (QByteArray::fromBase64(query.value(0).toByteArray()),
491  &ok);
492 
493  if(ok)
494  {
495  QPair<QByteArray, QByteArray> pair;
496 
497  pair.first = m_publicKeyHashesAdaptiveEchoPairs.
498  first().value(1);
499  pair.second = m_publicKeyHashesAdaptiveEchoPairs.
500  first().value(2);
501  emit sendMailFromPostOffice(message, pair);
502 
503  QSqlQuery deleteQuery(db);
504 
505  deleteQuery.exec("PRAGMA secure_delete = ON");
506  deleteQuery.prepare("DELETE FROM post_office "
507  "WHERE recipient_hash = ? AND "
508  "OID = ?");
509  deleteQuery.bindValue(0, publicKeyHash.toBase64());
510  deleteQuery.bindValue(1, query.value(1));
511  deleteQuery.exec();
512  }
513  }
514  }
515  else
516  m_publicKeyHashesAdaptiveEchoPairs.removeFirst();
517  }
518  }
519 
520  db.close();
521  }
522 
523  QSqlDatabase::removeDatabase(connectionName);
524 
525  if(m_publicKeyHashesAdaptiveEchoPairs.isEmpty())
526  m_retrieveMailTimer.stop();
527 }
