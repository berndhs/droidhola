47 static wchar_t *
48 utf8_to_wchar (const char *string)
49 {
50  int n;
51  size_t nbytes;
52  wchar_t *result;
53 
54  if (!string)
55  return NULL;
56 
57  n = MultiByteToWideChar (CP_UTF8, 0, string, -1, NULL, 0);
58  if (n < 0)
59  {
60  gpg_err_set_errno (EINVAL);
61  return NULL;
62  }
63 
64  nbytes = (size_t)(n+1) * sizeof(*result);
65  if (nbytes / sizeof(*result) != (n+1))
66  {
67  gpg_err_set_errno (ENOMEM);
68  return NULL;
69  }
70  result = malloc (nbytes);
71  if (!result)
72  return NULL;
73 
74  n = MultiByteToWideChar (CP_UTF8, 0, string, -1, result, n);
75  if (n < 0)
76  {
77  free (result);
78  gpg_err_set_errno (EINVAL);
79  result = NULL;
80  }
81  return result;
82 }

113 /* Prepare a pipe. Returns a handle which is, depending on WRITE_END,
114  will either act the read or as the write end of the pipe. The
115  other value returned is a rendezvous id used to complete the pipe
116  creation with _assuan_w32ce_finish_pipe. The rendezvous id may be
117  passed to another process and that process may finish the pipe
118  creation. This creates the interprocess pipe. The rendezvous id
119  is not a handle but a plain number; there is no release function
120  and care should be taken not to pass it to a function expecting a
121  handle. */
122 HANDLE
123 _assuan_w32ce_prepare_pipe (int *r_rvid, int write_end)
124 {
125  HANDLE hd;
126  LONG rvid;
127 
128  ActivateDevice (L"Drivers\\GnuPG_Device", 0);
129 
130  /* Note: Using "\\$device\\GPG1" should be identical to "GPG1:".
131  However this returns an invalid parameter error without having
132  called GPG_Init in the driver. The docs mention something about
133  RegisterAFXEx but that API is not documented. */
134  hd = CreateFile (L"GPG1:", write_end? GENERIC_WRITE : GENERIC_READ,
135  FILE_SHARE_READ | FILE_SHARE_WRITE,
136  NULL, OPEN_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
137  if (hd != INVALID_HANDLE_VALUE)
138  {
139  if (!DeviceIoControl (hd, GPGCEDEV_IOCTL_GET_RVID,
140  NULL, 0, &rvid, sizeof rvid, NULL, NULL))
141  {
142  DWORD lastrc = GetLastError ();
143  CloseHandle (hd);
144  hd = INVALID_HANDLE_VALUE;
145  SetLastError (lastrc);
146  }
147  else
148  *r_rvid = rvid;
149  }
150 
151  return hd;
152 }

188 /* WindowsCE does not provide a pipe feature. However we need
189  something like a pipe to convey data between processes and in some
190  cases within a process. This replacement is not only used by
191  libassuan but exported and thus usable by gnupg and gpgme as well. */
192 DWORD
193 _assuan_w32ce_create_pipe (HANDLE *read_hd, HANDLE *write_hd,
194  LPSECURITY_ATTRIBUTES sec_attr, DWORD size)
195 {
196  HANDLE hd[2] = {INVALID_HANDLE_VALUE, INVALID_HANDLE_VALUE};
197  int rvid;
198  int rc = 0;
199 
200  hd[0] = _assuan_w32ce_prepare_pipe (&rvid, 0);
201  if (hd[0] != INVALID_HANDLE_VALUE)
202  {
203  hd[1] = _assuan_w32ce_finish_pipe (rvid, 1);
204  if (hd[1] != INVALID_HANDLE_VALUE)
205  rc = 1;
206  else
207  {
208  DWORD lastrc = GetLastError ();
209  CloseHandle (hd[0]);
210  hd[0] = INVALID_HANDLE_VALUE;
211  SetLastError (lastrc);
212  }
213  }
214 
215  *read_hd = hd[0];
216  *write_hd = hd[1];
217  return rc;
218 }

221 /* Create a pipe with one inheritable end. Default implementation.
222  If INHERIT_IDX is 0, the read end of the pipe is made inheritable;
223  with INHERIT_IDX is 1 the write end will be inheritable. The
224  question now is how we create an inheritable pipe end under windows
225  CE were handles are process local objects? The trick we employ is
226  to defer the actual creation to the other end: We create an
227  incomplete pipe and pass a rendezvous id to the other end
228  (process). The other end now uses the rendezvous id to lookup the
229  pipe in our device driver, creates a new handle and uses that one
230  to finally establish the pipe. */
231 int
232 __assuan_pipe (assuan_context_t ctx, assuan_fd_t fd[2], int inherit_idx)
233 {
234  HANDLE hd;
235  int rvid;
236 
237  hd = _assuan_w32ce_prepare_pipe (&rvid, !inherit_idx);
238  if (hd == INVALID_HANDLE_VALUE)
239  {
240  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_pipe", ctx,
241  "CreatePipe failed: %s", _assuan_w32_strerror (ctx, -1));
242  gpg_err_set_errno (EIO);
243  return -1;
244  }
245 
246  if (inherit_idx)
247  {
248  fd[0] = hd;
249  fd[1] = (void*)rvid;
250  }
251  else
252  {
253  fd[0] = (void*)rvid;
254  fd[1] = hd;
255  }
256  return 0;
257 }

285 ssize_t
286 __assuan_read (assuan_context_t ctx, assuan_fd_t fd, void *buffer, size_t size)
287 {
288  /* Due to the peculiarities of the W32 API we can't use read for a
289  network socket and thus we try to use recv first and fallback to
290  read if recv detects that it is not a network socket. */
291  int res;
292 
293  TRACE_BEG3 (ctx, ASSUAN_LOG_SYSIO, "__assuan_read", ctx,
294  "fd=0x%x, buffer=%p, size=%i", fd, buffer, size);
295 
296 #ifdef HAVE_W32CE_SYSTEM
297  /* This is a bit of a hack to support stdin over ssh. Note that
298  fread buffers fully while getchar is line buffered. Weird, but
299  that's the way it is. ASSUAN_STDIN and ASSUAN_STDOUT are
300  special handle values that shouldn't occur in the wild. */
301  if (fd == ASSUAN_STDIN)
302  {
303  int i = 0;
304  int chr;
305  while (i < size)
306  {
307  chr = getchar();
308  if (chr == EOF)
309  break;
310  ((char*)buffer)[i++] = (char) chr;
311  if (chr == '\n')
312  break;
313  }
314  return TRACE_SYSRES (i);
315  }
316 #endif
317 
318  res = recv (HANDLE2SOCKET (fd), buffer, size, 0);
319  if (res == -1)
320  {
321  TRACE_LOG1 ("recv failed: rc=%d", (int)WSAGetLastError ());
322  switch (WSAGetLastError ())
323  {
324  case WSAENOTSOCK:
325  {
326  DWORD nread = 0;
327 
328  res = ReadFile (fd, buffer, size, &nread, NULL);
329  if (! res)
330  {
331  TRACE_LOG1 ("ReadFile failed: rc=%d", (int)GetLastError ());
332  switch (GetLastError ())
333  {
334  case ERROR_BROKEN_PIPE:
335  gpg_err_set_errno (EPIPE);
336  break;
337 
338  case ERROR_PIPE_NOT_CONNECTED:
339  case ERROR_BUSY:
340  gpg_err_set_errno (EAGAIN);
341  break;
342 
343  default:
344  gpg_err_set_errno (EIO);
345  }
346  res = -1;
347  }
348  else
349  res = (int) nread;
350  }
351  break;
352 
353  case WSAEWOULDBLOCK:
354  gpg_err_set_errno (EAGAIN);
355  break;
356 
357  case ERROR_BROKEN_PIPE:
358  gpg_err_set_errno (EPIPE);
359  break;
360 
361  default:
362  gpg_err_set_errno (EIO);
363  break;
364  }
365  }
366  return TRACE_SYSRES (res);
367 }

387  special handle values that shouldn't occur in the wild. */

455 /* Build a command line for use with W32's CreateProcess. On success
456  CMDLINE gets the address of a newly allocated string. */
457 static int
458 build_w32_commandline (assuan_context_t ctx, const char * const *argv,
459  assuan_fd_t fd0, assuan_fd_t fd1, assuan_fd_t fd2,
460  int fd2_isnull,
461  char **cmdline)
462 {
463  int i, n;
464  const char *s;
465  char *buf, *p;
466  char fdbuf[3*30];
467 
468  p = fdbuf;
469  *p = 0;
470  if (fd0 != ASSUAN_INVALID_FD)
471  {
472  snprintf (p, 25, "-&S0=%d ", (int)fd0);
473  p += strlen (p);
474  }
475  if (fd1 != ASSUAN_INVALID_FD)
476  {
477  snprintf (p, 25, "-&S1=%d ", (int)fd1);
478  p += strlen (p);
479  }
480  if (fd2 != ASSUAN_INVALID_FD)
481  {
482  if (fd2_isnull)
483  strcpy (p, "-&S2=null ");
484  else
485  snprintf (p, 25, "-&S2=%d ", (int)fd2);
486  p += strlen (p);
487  }
488 
489  *cmdline = NULL;
490  n = strlen (fdbuf);
491  for (i=0; (s = argv[i]); i++)
492  {
493  if (!i)
494  continue; /* Ignore argv[0]. */
495  n += strlen (s) + 1 + 2; /* (1 space, 2 quoting) */
496  for (; *s; s++)
497  if (*s == '\"')
498  n++; /* Need to double inner quotes. */
499  }
500  n++;
501 
502  buf = p = _assuan_malloc (ctx, n);
503  if (! buf)
504  return -1;
505 
506  p = stpcpy (p, fdbuf);
507  for (i = 0; argv[i]; i++)
508  {
509  if (!i)
510  continue; /* Ignore argv[0]. */
511  if (i > 1)
512  p = stpcpy (p, " ");
513 
514  if (! *argv[i]) /* Empty string. */
515  p = stpcpy (p, "\"\"");
516  else if (strpbrk (argv[i], " \t\n\v\f\""))
517  {
518  p = stpcpy (p, "\"");
519  for (s = argv[i]; *s; s++)
520  {
521  *p++ = *s;
522  if (*s == '\"')
523  *p++ = *s;
524  }
525  *p++ = '\"';
526  *p = 0;
527  }
528  else
529  p = stpcpy (p, argv[i]);
530  }
531 
532  *cmdline = buf;
533  return 0;
534 }

537 int
538 __assuan_spawn (assuan_context_t ctx, pid_t *r_pid, const char *name,
539  const char **argv,
540  assuan_fd_t fd_in, assuan_fd_t fd_out,
541  assuan_fd_t *fd_child_list,
542  void (*atfork) (void *opaque, int reserved),
543  void *atforkvalue, unsigned int flags)
544 {
545  PROCESS_INFORMATION pi =
546  {
547  NULL, /* Returns process handle. */
548  0, /* Returns primary thread handle. */
549  0, /* Returns pid. */
550  0 /* Returns tid. */
551  };
552  assuan_fd_t fd;
553  assuan_fd_t *fdp;
554  assuan_fd_t fd_err;
555  int fd_err_isnull = 0;
556  char *cmdline;
557 
558  /* Dup stderr to /dev/null unless it is in the list of FDs to be
559  passed to the child. Well we don't actually open nul because
560  that is not available on Windows, but use our hack for it.
561  Because an RVID of 0 is an invalid value and HANDLES will never
562  have this value either, we test for this as well. */
563 
564  /* FIXME: As long as we can't decide whether a handle is a real
565  handler or an rendezvous id we can't do anything with the
566  FD_CHILD_LIST. We can't do much with stderr either, thus we
567  better don't pass stderr to the child at all. If we would do so
568  and it is not a rendezvous id the client would run into
569  problems. */
570  fd = assuan_fd_from_posix_fd (fileno (stderr));
571  fdp = fd_child_list;
572  if (fdp)
573  {
574  for (; *fdp != ASSUAN_INVALID_FD && *fdp != 0 && *fdp != fd; fdp++)
575  ;
576  }
577  if (!fdp || *fdp == ASSUAN_INVALID_FD)
578  fd_err_isnull = 1;
579  fd_err = ASSUAN_INVALID_FD;
580 
581  if (build_w32_commandline (ctx, argv, fd_in, fd_out, fd_err, fd_err_isnull,
582  &cmdline))
583  {
584  return -1;
585  }
586 
587  TRACE2 (ctx, ASSUAN_LOG_SYSIO, "__assuan_spawn", ctx,
588  "path=`%s' cmdline=`%s'", name, cmdline);
589 
590  {
591  wchar_t *wcmdline, *wname;
592 
593  wcmdline = utf8_to_wchar (cmdline);
594  _assuan_free (ctx, cmdline);
595  if (!wcmdline)
596  return -1;
597 
598  wname = utf8_to_wchar (name);
599  if (!wname)
600  {
601  free_wchar (wcmdline);
602  return -1;
603  }
604 
605  if (!CreateProcess (wname, /* Program to start. */
606  wcmdline, /* Command line arguments. */
607  NULL, /* (not supported) */
608  NULL, /* (not supported) */
609  FALSE, /* (not supported) */
610  (CREATE_SUSPENDED), /* Creation flags. */
611  NULL, /* (not supported) */
612  NULL, /* (not supported) */
613  NULL, /* (not supported) */
614  &pi /* Returns process information.*/
615  ))
616  {
617  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_spawn", ctx,
618  "CreateProcess failed: %s", _assuan_w32_strerror (ctx, -1));
619  free_wchar (wname);
620  free_wchar (wcmdline);
621  gpg_err_set_errno (EIO);
622  return -1;
623  }
624  free_wchar (wname);
625  free_wchar (wcmdline);
626  }
627 
628  ResumeThread (pi.hThread);
629 
630  TRACE4 (ctx, ASSUAN_LOG_SYSIO, "__assuan_spawn", ctx,
631  "CreateProcess ready: hProcess=%p hThread=%p"
632  " dwProcessID=%d dwThreadId=%d\n",
633  pi.hProcess, pi.hThread, (int) pi.dwProcessId, (int) pi.dwThreadId);
634 
635  CloseHandle (pi.hThread);
636 
637  *r_pid = (pid_t) pi.hProcess;
638  return 0;
639 }
