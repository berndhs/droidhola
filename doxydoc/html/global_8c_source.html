75 /* This is our handmade constructor. It gets called by any function
76  likely to be called at startup. The suggested way for an
77  application to make sure that this has been called is by using
78  gcry_check_version. */
79 static void
80 global_init (void)
81 {
82  gcry_error_t err = 0;
83 
84  if (any_init_done)
85  return;
86  any_init_done = 1;
87 
88  /* Tell the random module that we have seen an init call. */
89  _gcry_set_preferred_rng_type (0);
90 
91  /* Initialize our portable thread/mutex wrapper. */
92  err = ath_init ();
93  if (err)
94  {
95  err = gpg_error_from_errno (err);
96  goto fail;
97  }
98 
99  /* See whether the system is in FIPS mode. This needs to come as
100  early as possible but after ATH has been initialized. */
101  _gcry_initialize_fips_mode (force_fips_mode);
102 
103  /* Before we do any other initialization we need to test available
104  hardware features. */
105  _gcry_detect_hw_features ();
106 
107  /* Initialize the modules - this is mainly allocating some memory and
108  creating mutexes. */
109  err = _gcry_cipher_init ();
110  if (err)
111  goto fail;
112  err = _gcry_md_init ();
113  if (err)
114  goto fail;
115  err = _gcry_pk_init ();
116  if (err)
117  goto fail;
118  err = _gcry_primegen_init ();
119  if (err)
120  goto fail;
121  err = _gcry_secmem_module_init ();
122  if (err)
123  goto fail;
124  err = _gcry_mpi_init ();
125  if (err)
126  goto fail;
127 
128  return;
129 
130  fail:
131  BUG ();
132 }

135 /* This function is called by the macro fips_is_operational and makes
136  sure that the minimal initialization has been done. This is far
137  from a perfect solution and hides problems with an improper
138  initialization but at least in single-threaded mode it should work
139  reliable.
140 
141  The reason we need this is that a lot of applications don't use
142  Libgcrypt properly by not running any initialization code at all.
143  They just call a Libgcrypt function and that is all what they want.
144  Now with the FIPS mode, that has the side effect of entering FIPS
145  mode (for security reasons, FIPS mode is the default if no
146  initialization has been done) and bailing out immediately because
147  the FSM is in the wrong state. If we always run the init code,
148  Libgcrypt can test for FIPS mode and at least if not in FIPS mode,
149  it will behave as before. Note that this on-the-fly initialization
150  is only done for the cryptographic functions subject to FIPS mode
151  and thus not all API calls will do such an initialization. */
152 int
153 _gcry_global_is_operational (void)
154 {
155  if (!any_init_done)
156  {
157 #ifdef HAVE_SYSLOG
158  syslog (LOG_USER|LOG_WARNING, "Libgcrypt warning: "
159  "missing initialization - please fix the application");
160 #endif /*HAVE_SYSLOG*/
161  global_init ();
162  }
163  return _gcry_fips_is_operational ();
164 }

171 /* This function parses the first portion of the version number S and
172  stores it in *NUMBER. On success, this function returns a pointer
173  into S starting with the first character, which is not part of the
174  initial number portion; on failure, NULL is returned. */
175 static const char*
176 parse_version_number( const char *s, int *number )
177 {
178  int val = 0;
179 
180  if( *s == '0' && isdigit(s[1]) )
181  return NULL; /* leading zeros are not allowed */
182  for ( ; isdigit(*s); s++ ) {
183  val *= 10;
184  val += *s - '0';
185  }
186  *number = val;
187  return val < 0? NULL : s;
188 }
189 
190 /* This function breaks up the complete string-representation of the
191  version number S, which is of the following struture: <major
192  number>.<minor number>.<micro number><patch level>. The major,
193  minor and micro number components will be stored in *MAJOR, *MINOR
194  and *MICRO.
195 
196  On success, the last component, the patch level, will be returned;
197  in failure, NULL will be returned. */
198 
199 static const char *
200 parse_version_string( const char *s, int *major, int *minor, int *micro )
201 {
202  s = parse_version_number( s, major );
203  if( !s || *s != '.' )
204  return NULL;
205  s++;
206  s = parse_version_number( s, minor );
207  if( !s || *s != '.' )
208  return NULL;
209  s++;
210  s = parse_version_number( s, micro );
211  if( !s )
212  return NULL;
213  return s; /* patchlevel */
214 }

216 /* If REQ_VERSION is non-NULL, check that the version of the library
217  is at minimum the requested one. Returns the string representation
218  of the library version if the condition is satisfied; return NULL
219  if the requested version is newer than that of the library.
220 
221  If a NULL is passed to this function, no check is done, but the
222  string representation of the library is simply returned. */
223 const char *
224 _gcry_check_version (const char *req_version)
225 {
226  const char *ver = VERSION;
227  int my_major, my_minor, my_micro;
228  int rq_major, rq_minor, rq_micro;
229  const char *my_plvl;
230 
231  if (req_version && req_version[0] == 1 && req_version[1] == 1)
232  return _gcry_compat_identification ();
233 
234  /* Initialize library. */
235  global_init ();
236 
237  if ( !req_version )
238  /* Caller wants our version number. */
239  return ver;
240 
241  /* Parse own version number. */
242  my_plvl = parse_version_string( ver, &my_major, &my_minor, &my_micro );
243  if ( !my_plvl )
244  /* very strange our own version is bogus. Shouldn't we use
245  assert() here and bail out in case this happens? -mo. */
246  return NULL;
247 
248  /* Parse requested version number. */
249  if (!parse_version_string (req_version, &rq_major, &rq_minor, &rq_micro))
250  return NULL; /* req version string is invalid, this can happen. */
251 
252  /* Compare version numbers. */
253  if ( my_major > rq_major
254  || (my_major == rq_major && my_minor > rq_minor)
255  || (my_major == rq_major && my_minor == rq_minor && my_micro > rq_micro)
256  || (my_major == rq_major && my_minor == rq_minor
257  && my_micro == rq_micro))
258  {
259  return ver;
260  }
261 
262  return NULL;
263 }
