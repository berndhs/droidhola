439 void spoton_starbeam_writer::slotReadKeys(void)
440 {
441  spoton_crypt *s_crypt = spoton_kernel::s_crypts.value("chat", 0);
442 
443  if(!s_crypt)
444  {
445  spoton_misc::logError("spoton_starbeam_writer::slotReadKeys(): "
446  "s_crypt is zero.");
447  return;
448  }
449 
450  QString connectionName("");
451 
452  {
453  QSqlDatabase db = spoton_misc::database(connectionName);
454 
455  db.setDatabaseName
456  (spoton_misc::homePath() + QDir::separator() + "starbeam.db");
457 
458  if(db.open())
459  {
460  QWriteLocker locker(&m_keyMutex);
461 
462  m_magnets.clear();
463  m_novas.clear();
464  locker.unlock();
465 
466  QSqlQuery query(db);
467 
468  query.setForwardOnly(true);
469  query.prepare("SELECT magnet FROM magnets");
470 
471  if(query.exec())
472  while(query.next())
473  {
474  QByteArray data
475  (QByteArray::fromBase64(query.value(0).toByteArray()));
476  bool ok = true;
477 
478  data = s_crypt->decryptedAfterAuthenticated(data, &ok);
479 
480  if(!ok)
481  continue;
482 
483  QHash<QString, QByteArray> elements;
484  QList<QByteArray> list
485  (data.remove(0, static_cast<int> (qstrlen("magnet:?"))).
486  split('&'));
487 
488  while(!list.isEmpty())
489  {
490  QByteArray bytes(list.takeFirst());
491 
492  if(bytes.startsWith("ct=")) // Cipher Type
493  {
494  bytes.remove(0, 3);
495  elements.insert("ct", bytes);
496  }
497  else if(bytes.startsWith("ek=")) // Encryption Key
498  {
499  bytes.remove(0, 3);
500  elements.insert("ek", bytes);
501  }
502  else if(bytes.startsWith("ht=")) // Hash Type
503  {
504  bytes.remove(0, 3);
505  elements.insert("ht", bytes);
506  }
507  else if(bytes.startsWith("mk=")) // MAC Key
508  {
509  bytes.remove(0, 3);
510  elements.insert("mk", bytes);
511  }
512  else if(bytes.startsWith("xt="))
513  {
514  bytes.remove(0, 3);
515 
516  if(bytes == "urn:starbeam")
517  elements.insert("xt", bytes);
518  }
519  }
520 
521  if(elements.contains("xt"))
522  {
523  QWriteLocker locker(&m_keyMutex);
524 
525  m_magnets.append(elements);
526  }
527  }
528 
529  query.prepare("SELECT nova FROM received_novas");
530 
531  if(query.exec())
532  while(query.next())
533  {
534  QByteArray data
535  (QByteArray::fromBase64(query.value(0).toByteArray()));
536  bool ok = true;
537 
538  data = s_crypt->decryptedAfterAuthenticated(data, &ok);
539 
540  if(!ok)
541  continue;
542 
543  QWriteLocker locker(&m_keyMutex);
544 
545  m_novas.append(data);
546  }
547  }
548 
549  db.close();
550  }
551 
552  QSqlDatabase::removeDatabase(connectionName);
553 }
