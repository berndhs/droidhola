82  void initializeMulticast(const QHostAddress &address, const quint16 port)
83  {
84  if(address.protocol() == QAbstractSocket::IPv4Protocol)
85  {
86  quint32 a = address.toIPv4Address();
87 
88  if(!((a & 0xf0000000) == 0xe0000000))
89  return;
90  }
91  else if(address.protocol() == QAbstractSocket::IPv6Protocol)
92  {
93 #ifdef Q_OS_OS2
94  return;
95 #endif
96 
97  Q_IPV6ADDR a6 = address.toIPv6Address();
98 
99  if(a6.c[0] != 0xff)
100  return;
101  }
102  else
103  return;
104 
105  if(m_multicastSocket)
106  m_multicastSocket->deleteLater();
107 
108  m_multicastSocket = new (std::nothrow) QUdpSocket(this);
109 
110  if(m_multicastSocket)
111  {
112  if(!m_multicastSocket->bind(address, port,
113  QUdpSocket::ReuseAddressHint |
114  QUdpSocket::ShareAddress))
115  {
116  m_multicastSocket->deleteLater();
117  spoton_misc::logError
118  (QString("spoton_neighbor_udp_socket::initializeMulticast(): "
119  "bind() failure for %1:%2.").
120  arg(address.toString()).arg(port));
121  return;
122  }
123 
124 #if QT_VERSION >= 0x040800
125  if(!m_multicastSocket->joinMulticastGroup(address))
126  {
127  m_multicastSocket->deleteLater();
128  spoton_misc::logError
129  (QString("spoton_neighbor_udp_socket::initializeMulticast(): "
130  "joinMulticastGroup() failure for %1:%2.").
131  arg(address.toString()).arg(port));
132  }
133  else
134  m_multicastSocket->setSocketOption
135  (QAbstractSocket::MulticastLoopbackOption, 1);
136 #else
137  if(!spoton_misc::joinMulticastGroup(address,
138  1, // Enable loopback.
139  m_multicastSocket->
140  socketDescriptor(),
141  port))
142  m_multicastSocket->deleteLater();
143 #endif
144  }
145  }
