204 static void
205 check (void)
206 {
207  unsigned char buf[20];
208  unsigned char *ret;
209  gcry_error_t err;
210  gcry_sexp_t sexp;
211  unsigned int i;
212  int repn;
213 
214  for (i = 0; i < (sizeof (key_grips) / sizeof (*key_grips)); i++)
215  {
216  if (gcry_pk_test_algo (key_grips[i].algo))
217  {
218  if (verbose)
219  fprintf (stderr, "algo %d not available; test skipped\n",
220  key_grips[i].algo);
221  continue;
222  }
223  err = gcry_sexp_sscan (&sexp, NULL, key_grips[i].key,
224  strlen (key_grips[i].key));
225  if (err)
226  die ("scanning data %d failed: %s\n", i, gpg_strerror (err));
227 
228  for (repn=0; repn < repetitions; repn++)
229  {
230  ret = gcry_pk_get_keygrip (sexp, buf);
231  if (!ret)
232  die ("gcry_pk_get_keygrip failed for %d\n", i);
233 
234  if ( memcmp (key_grips[i].grip, buf, sizeof (buf)) )
235  {
236  print_hex ("keygrip: ", buf, sizeof buf);
237  die ("keygrip for %d does not match\n", i);
238  }
239  }
240 
241  gcry_sexp_release (sexp);
242  }
243 }
