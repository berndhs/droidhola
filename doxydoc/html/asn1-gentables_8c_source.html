76 static size_t
77 insert_string (const char *name)
78 {
79  struct name_list_s *item;
80  size_t off, n;
81 
82  if (!string_table_tail)
83  {
84  string_table_tail = &string_table;
85  insert_string ("");
86  }
87 
88  if (string_table_offset && !*name)
89  return 0;
90 
91  for (item = string_table,off = 0; item; item = item->next)
92  {
93  for (n=0; item->name[n]; n++)
94  if (!strcmp (item->name+n, name))
95  return off + n;
96  off += strlen (item->name) + 1;
97  }
98 
99  item = xmalloc ( sizeof *item + strlen (name));
100  strcpy (item->name, name);
101  item->next = NULL;
102  *string_table_tail = item;
103  string_table_tail = &item->next;
104  off = string_table_offset;
105  string_table_offset += strlen (name) + 1;
106  return off;
107 }

118 static void
119 sort_string_table (void)
120 {
121  struct name_list_s *item;
122  struct name_list_s **array;
123  size_t i, arraylen;
124 
125  if (!string_table || !string_table->next)
126  return; /* Nothing to sort. */
127 
128  for (item = string_table,arraylen = 0; item; item = item->next)
129  arraylen++;
130  array = xcalloc (arraylen, sizeof *array);
131  for (item = string_table,arraylen = 0; item; item = item->next)
132  array[arraylen++] = item;
133  qsort (array, arraylen, sizeof *array, cmp_string);
134  /* Replace table by sorted one. */
135  string_table_tail = NULL;
136  string_table = NULL;
137  string_table_offset = 0;
138  for (i=0; i < arraylen; i++)
139  insert_string (array[i]->name);
140  xfree (array);
141  /* for (item = string_table,arraylen = 0; item; item = item->next) */
142  /* fprintf (stderr, " `%s'\n", item->name); */
143 }

175 create_static_structure (AsnNode pointer, const char *file_name, FILE *fp)
176 {
177  AsnNode p;
178  struct name_list_s *structure_name;
179  const char *char_p, *slash_p, *dot_p;
180  char numbuf[50];
181 
182  char_p = file_name;
183  slash_p = file_name;
184  while ((char_p = strchr (char_p, '/')))
185  {
186  char_p++;
187  slash_p = char_p;
188  }
189 
190  char_p = slash_p;
191  dot_p = file_name + strlen (file_name);
192 
193  while ((char_p = strchr (char_p, '.')))
194  {
195  dot_p = char_p;
196  char_p++;
197  }
198 
199  structure_name = xmalloc (sizeof *structure_name + dot_p - slash_p + 100);
200  structure_name->next = NULL;
201  memcpy (structure_name->name, slash_p, dot_p - slash_p);
202  structure_name->name[dot_p - slash_p] = 0;
203 
204  fprintf (fp, "static const static_asn %s_asn1_tab[] = {\n",
205  structure_name->name);
206 
207  for (p = pointer; p; p = _ksba_asn_walk_tree (pointer, p))
208  {
209  /* set the help flags */
210  p->flags.help_down = !!p->down;
211  p->flags.help_right = !!p->right;
212 
213  /* write a structure line */
214  fputs (" {", fp);
215  if (p->name)
216  fprintf (fp, "%u", (unsigned int)insert_string (p->name));
217  else
218  fprintf (fp, "0");
219  fprintf (fp, ",%u", p->type);
220 
221  fputs (", {", fp);
222  fprintf (fp, "%u", p->flags.class);
223  fputs (p->flags.explicit ? ",1":",0", fp);

225  fputs (p->flags.has_imports ? ",1":",0", fp);
226  fputs (p->flags.assignment ? ",1":",0", fp);
227  fputs (p->flags.one_param ? ",1":",0", fp);
228  fputs (p->flags.has_tag ? ",1":",0", fp);
229  fputs (p->flags.has_size ? ",1":",0", fp);
230  fputs (p->flags.has_list ? ",1":",0", fp);
231  fputs (p->flags.has_min_max ? ",1":",0", fp);
232  fputs (p->flags.has_defined_by ? ",1":",0", fp);
233  fputs (p->flags.is_false ? ",1":",0", fp);

236  fputs (p->flags.is_optional ? ",1":",0", fp);

239  fputs (p->flags.in_choice ? ",1":",0", fp);
240  fputs (p->flags.in_array ? ",1":",0", fp);
241  fputs (p->flags.is_any ? ",1":",0", fp);
242  fputs (p->flags.not_used ? ",1":",0", fp);
243  fputs (p->flags.help_down ? ",1":",0", fp);
244  fputs (p->flags.help_right ? ",1":",0", fp);
245  fputs ("}", fp);
246 
247  if (p->valuetype == VALTYPE_CSTR)
248  fprintf (fp, ",%u",
249  (unsigned int)insert_string (p->value.v_cstr));
250  else if (p->valuetype == VALTYPE_LONG
251  && p->type == TYPE_INTEGER && p->flags.assignment)
252  {
253  snprintf (numbuf, sizeof numbuf, "%ld", p->value.v_long);
254  fprintf (fp, ",%u", (unsigned int)insert_string (numbuf));
255  }
256  else if (p->valuetype == VALTYPE_ULONG)
257  {
258  snprintf (numbuf, sizeof numbuf, "%lu", p->value.v_ulong);
259  fprintf (fp, ",%u", (unsigned int)insert_string (numbuf));
260  }
261  else
262  {
263  if (p->valuetype)
264  print_error ("can't store a value of type %d\n", p->valuetype);
265  fprintf (fp, ",0");
266  }
267  fputs ("},\n", fp);
268  }
269 
270  fprintf (fp, " {0,0}\n};\n");
271 
272  return structure_name;
273 }

277 static struct name_list_s *
278 one_file (const char *fname, int *count, FILE *fp)
279 {
280  ksba_asn_tree_t tree;

282 
283  rc = ksba_asn_parse_file (fname, &tree, check_only);
284  if (rc)
285  print_error ("error parsing `%s': %s\n", fname, gpg_strerror (rc) );
286  else if (!check_only)
287  {
288  if (dump_only)
289  ksba_asn_tree_dump (tree, dump_only==2? "<":NULL, fp);
290  else
291  {
292  if (!*count)
293  fprintf (fp,"\n"
294  "#include <config.h>\n"
295  "#include <stdio.h>\n"
296  "#include <string.h>\n"
297  "#include \"ksba.h\"\n"
298  "#include \"asn1-func.h\"\n"
299  "\n");
300  ++*count;
301  return create_static_structure (tree->parse_tree, fname, fp);
302  }
303  }
