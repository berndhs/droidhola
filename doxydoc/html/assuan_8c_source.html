108 /* Create a new Assuan context. The initial parameters are all needed
109  in the creation of the context. */
110 gpg_error_t
111 assuan_new_ext (assuan_context_t *r_ctx, gpg_err_source_t err_source,
112  assuan_malloc_hooks_t malloc_hooks, assuan_log_cb_t log_cb,
113  void *log_cb_data)
114 {
115  struct assuan_context_s wctx;
116  assuan_context_t ctx;
117 
118  /* Set up a working context so we can use standard functions. */
119  memset (&wctx, 0, sizeof (wctx));
120  wctx.err_source = err_source;
121  wctx.malloc_hooks = *malloc_hooks;
122  wctx.log_cb = log_cb;
123  wctx.log_cb_data = log_cb_data;
124 
125  /* Need a new block for the trace macros to work. */
126  {
127  TRACE_BEG8 (&wctx, ASSUAN_LOG_CTX, "assuan_new_ext", r_ctx,
128  "err_source = %i (%s), malloc_hooks = %p (%p, %p, %p), "
129  "log_cb = %p, log_cb_data = %p", err_source,
130  gpg_strsource (err_source), malloc_hooks, malloc_hooks->malloc,
131  malloc_hooks->realloc, malloc_hooks->free, log_cb, log_cb_data);
132 
133  *r_ctx = NULL;
134  ctx = _assuan_malloc (&wctx, sizeof (*ctx));
135  if (!ctx)
136  return TRACE_ERR (gpg_err_code_from_syserror ());
137 
138  memcpy (ctx, &wctx, sizeof (*ctx));
139  ctx->system = _assuan_system_hooks;
140 
141  /* FIXME: Delegate to subsystems/engines, as the FDs are not our
142  responsibility (we don't deallocate them, for example). */
143  ctx->input_fd = ASSUAN_INVALID_FD;
144  ctx->output_fd = ASSUAN_INVALID_FD;
145  ctx->inbound.fd = ASSUAN_INVALID_FD;
146  ctx->outbound.fd = ASSUAN_INVALID_FD;
147  ctx->listen_fd = ASSUAN_INVALID_FD;
148 
149  *r_ctx = ctx;
150 
151  return TRACE_SUC1 ("ctx=%p", ctx);
152  }
153 }

181 /* Release all resources associated with the given context. */
182 void
183 assuan_release (assuan_context_t ctx)
184 {
185  if (! ctx)
186  return;
187 
188  TRACE (ctx, ASSUAN_LOG_CTX, "assuan_release", ctx);
189 
190  _assuan_reset (ctx);
191  /* None of the members that are our responsibility requires
192  deallocation. To avoid sensitive data in the line buffers we
193  wipe them out, though. Note that we can't wipe the entire
194  context because it also has a pointer to the actual free(). */
195  wipememory (&ctx->inbound, sizeof ctx->inbound);
196  wipememory (&ctx->outbound, sizeof ctx->outbound);
197  _assuan_free (ctx, ctx);
198 }

206 static const char*
207 parse_version_number (const char *s, int *number)
208 {
209  int val = 0;
210 
211  if (*s == '0' && digitp (s[1]))
212  return NULL; /* Leading zeros are not allowed. */
213  for (; digitp (*s); s++)
214  {
215  val *= 10;
216  val += *s - '0';
217  }
218  *number = val;
219  return val < 0 ? NULL : s;
220 }

223 static const char *
224 parse_version_string (const char *s, int *major, int *minor, int *micro)
225 {
226  s = parse_version_number (s, major);
227  if (!s || *s != '.')
228  return NULL;
229  s++;
230  s = parse_version_number (s, minor);
231  if (!s || *s != '.')
232  return NULL;
233  s++;
234  s = parse_version_number (s, micro);
235  if (!s)
236  return NULL;
237  return s; /* Patchlevel. */
238 }

241 static const char *
242 compare_versions (const char *my_version, const char *req_version)
243 {
244  int my_major, my_minor, my_micro;
245  int rq_major, rq_minor, rq_micro;
246  const char *my_plvl, *rq_plvl;
247 
248  if (!req_version)
249  return my_version;
250  if (!my_version)
251  return NULL;
252 
253  my_plvl = parse_version_string (my_version, &my_major, &my_minor, &my_micro);
254  if (!my_plvl)
255  return NULL; /* Very strange: our own version is bogus. */
256  rq_plvl = parse_version_string(req_version,
257  &rq_major, &rq_minor, &rq_micro);
258  if (!rq_plvl)
259  return NULL; /* Requested version string is invalid. */
260 
261  if (my_major > rq_major
262  || (my_major == rq_major && my_minor > rq_minor)
263  || (my_major == rq_major && my_minor == rq_minor
264  && my_micro > rq_micro)
265  || (my_major == rq_major && my_minor == rq_minor
266  && my_micro == rq_micro))
267  {
268  return my_version;
269  }
270  return NULL;
271 }
