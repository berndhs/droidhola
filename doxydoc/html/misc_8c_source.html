113 /****************
114  * This is our log function which prints all log messages to stderr or
115  * using the function defined with gcry_set_log_handler().
116  */
117 void
118 _gcry_logv( int level, const char *fmt, va_list arg_ptr )
119 {
120  if (log_handler)
121  log_handler (log_handler_value, level, fmt, arg_ptr);
122  else
123  {
124  switch (level)
125  {
126  case GCRY_LOG_CONT: break;
127  case GCRY_LOG_INFO: break;
128  case GCRY_LOG_WARN: break;
129  case GCRY_LOG_ERROR: break;
130  case GCRY_LOG_FATAL: fputs("Fatal: ",stderr ); break;
131  case GCRY_LOG_BUG: fputs("Ohhhh jeeee: ", stderr); break;
132  case GCRY_LOG_DEBUG: fputs("DBG: ", stderr ); break;
133  default: fprintf(stderr,"[Unknown log level %d]: ", level ); break;
134  }
135  vfprintf(stderr,fmt,arg_ptr) ;
136  }
137 
138  if ( level == GCRY_LOG_FATAL || level == GCRY_LOG_BUG )
139  {
140  fips_signal_fatal_error ("internal error (fatal or bug)");
141  _gcry_secmem_term ();
142  abort ();
143  }
144 }

271 /* Helper for _gcry_log_printhex and _gcry_log_printmpi. */
272 static void
273 do_printhex (const char *text, const char *text2,
274  const void *buffer, size_t length)
275 {
276  int wrap = 0;
277  int cnt = 0;
278 
279  if (text && *text)
280  {
281  wrap = 1;
282  log_debug ("%s:%s", text, text2);
283  if (text2[1] == '[' && length && buffer)
284  {
285  /* Start with a new line so that we get nice output for
286  opaque MPIS:
287  "value: [31 bit]"
288  " 01020300" */
289  log_printf ("\n");
290  text2 = " ";
291  log_debug ("%*s ", (int)strlen(text), "");
292  }
293  }
294  if (length)
295  {
296  const unsigned char *p = buffer;
297  for (; length--; p++)
298  {
299  log_printf ("%02x", *p);
300  if (wrap && ++cnt == 32 && length)
301  {
302  cnt = 0;
303  log_printf (" \\\n");
304  log_debug ("%*s %*s",
305  (int)strlen(text), "", (int)strlen(text2), "");
306  }
307  }
308  }
309  if (text)
310  log_printf ("\n");
311 }

324 /* Print MPI in hex notation. To make clear that the output is an MPI
325  a sign is always printed. With TEXT of NULL print just the raw dump
326  without any wrapping, with TEXT an empty string, print a trailing
327  linefeed, otherwise print an entire debug line. */
328 void
329 _gcry_log_printmpi (const char *text, gcry_mpi_t mpi)
330 {
331  unsigned char *rawmpi;
332  unsigned int rawmpilen;
333  int sign;
334 
335  if (!mpi)
336  do_printhex (text? text:" ", " (null)", NULL, 0);
337  else if (mpi_is_opaque (mpi))
338  {
339  unsigned int nbits;
340  const unsigned char *p;
341  char prefix[30];
342 
343  p = mpi_get_opaque (mpi, &nbits);
344  snprintf (prefix, sizeof prefix, " [%u bit]", nbits);
345  do_printhex (text? text:" ", prefix, p, (nbits+7)/8);
346  }
347  else
348  {
349  rawmpi = _gcry_mpi_get_buffer (mpi, 0, &rawmpilen, &sign);
350  if (!rawmpi)
351  do_printhex (text? text:" ", " [out of core]", NULL, 0);
352  else
353  {
354  if (!rawmpilen)
355  do_printhex (text, sign? "-":"+", "", 1);
356  else
357  do_printhex (text, sign? "-":"+", rawmpi, rawmpilen);
358  xfree (rawmpi);
359  }
360  }
361 }

379 /* Print SEXP in human readabale format. With TEXT of NULL print just the raw
380  dump without any wrappping, with TEXT an empty string, print a
381  trailing linefeed, otherwise print the full debug output. */
382 void
383 _gcry_log_printsxp (const char *text, gcry_sexp_t sexp)
384 {
385  int with_lf = 0;
386 
387  if (text && *text)
388  {
389  if ((with_lf = !!strchr (text, '\n')))
390  log_debug ("%s", text);
391  else
392  log_debug ("%s: ", text);
393  }
394  if (sexp)
395  {
396  int any = 0;
397  int n_closing;
398  char *buf, *pend;
399  const char *p;
400  size_t size;
401 
402  size = sexp_sprint (sexp, GCRYSEXP_FMT_ADVANCED, NULL, 0);
403  p = buf = xmalloc (size);
404  sexp_sprint (sexp, GCRYSEXP_FMT_ADVANCED, buf, size);
405 
406  do
407  {
408  if (any && !with_lf)
409  log_debug ("%*s ", (int)strlen(text), "");
410  else
411  any = 1;
412  pend = strchr (p, '\n');
413  size = pend? (pend - p) : strlen (p);
414  if (with_lf)
415  log_debug ("%.*s", (int)size, p);
416  else
417  log_printf ("%.*s", (int)size, p);
418  if (pend)
419  p = pend + 1;
420  else
421  p += size;
422  n_closing = count_closing_parens (p);
423  if (n_closing)
424  {
425  while (n_closing--)
426  log_printf (")");
427  p = "";
428  }
429  log_printf ("\n");
430  }
431  while (*p);
432  xfree (buf);
433  }
434  else if (text)
435  log_printf ("\n");
436 }
