71 gpg_error_t
72 assuan_client_read_response (assuan_context_t ctx,
73  char **line_r, int *linelen_r)
74 {
75  gpg_error_t rc;
76  char *line = NULL;
77  int linelen = 0;
78 
79  *line_r = NULL;
80  *linelen_r = 0;
81 
82  do
83  {
84  do
85  {
86  rc = _assuan_read_line (ctx);
87  }
88  while (_assuan_error_is_eagain (ctx, rc));
89  if (rc)
90  return rc;
91  line = ctx->inbound.line;
92  linelen = ctx->inbound.linelen;
93  }
94  while (!linelen);
95 
96  /* For data lines, we deescape immediately. The user will never
97  have to worry about it. */
98  if (linelen >= 1 && line[0] == 'D' && line[1] == ' ')
99  {
100  char *s, *d;
101  for (s=d=line; linelen; linelen--)
102  {
103  if (*s == '%' && linelen > 2)
104  { /* handle escaping */
105  s++;
106  *d++ = xtoi_2 (s);
107  s += 2;
108  linelen -= 2;
109  }
110  else
111  *d++ = *s++;
112  }
113  *d = 0; /* add a hidden string terminator */
114 
115  linelen = d - line;
116  ctx->inbound.linelen = linelen;
117  }
118 
119  *line_r = line;
120  *linelen_r = linelen;
121 
122  return 0;
123 }

126 gpg_error_t
127 assuan_client_parse_response (assuan_context_t ctx, char *line, int linelen,
128  assuan_response_t *response, int *off)
129 {
130  *response = ASSUAN_RESPONSE_ERROR;
131  *off = 0;
132 
133  if (linelen >= 1
134  && line[0] == 'D' && line[1] == ' ')
135  {
136  *response = ASSUAN_RESPONSE_DATA; /* data line */
137  *off = 2;
138  }
139  else if (linelen >= 1
140  && line[0] == 'S'
141  && (line[1] == '\0' || line[1] == ' '))
142  {
143  *response = ASSUAN_RESPONSE_STATUS;
144  *off = 1;
145  while (line[*off] == ' ')
146  ++*off;
147  }
148  else if (linelen >= 2
149  && line[0] == 'O' && line[1] == 'K'
150  && (line[2] == '\0' || line[2] == ' '))
151  {
152  *response = ASSUAN_RESPONSE_OK;
153  *off = 2;
154  while (line[*off] == ' ')
155  ++*off;
156  }
157  else if (linelen >= 3
158  && line[0] == 'E' && line[1] == 'R' && line[2] == 'R'
159  && (line[3] == '\0' || line[3] == ' '))
160  {
161  *response = ASSUAN_RESPONSE_ERROR;
162  *off = 3;
163  while (line[*off] == ' ')
164  ++*off;
165  }
166  else if (linelen >= 7
167  && line[0] == 'I' && line[1] == 'N' && line[2] == 'Q'
168  && line[3] == 'U' && line[4] == 'I' && line[5] == 'R'
169  && line[6] == 'E'
170  && (line[7] == '\0' || line[7] == ' '))
171  {
172  *response = ASSUAN_RESPONSE_INQUIRE;
173  *off = 7;
174  while (line[*off] == ' ')
175  ++*off;
176  }
177  else if (linelen >= 3
178  && line[0] == 'E' && line[1] == 'N' && line[2] == 'D'
179  && (line[3] == '\0' || line[3] == ' '))
180  {
181  *response = ASSUAN_RESPONSE_END;
182  *off = 3;
183  }
184  else if (linelen >= 1 && line[0] == '#')
185  {
186  *response = ASSUAN_RESPONSE_COMMENT;
187  *off = 1;
188  }
189  else
190  return _assuan_error (ctx, GPG_ERR_ASS_INV_RESPONSE);
191 
192  return 0;
193 }

198  int *off, int convey_comments)

236 gpg_error_t
237 assuan_transact (assuan_context_t ctx,
238  const char *command,
239  gpg_error_t (*data_cb)(void *, const void *, size_t),
240  void *data_cb_arg,
241  gpg_error_t (*inquire_cb)(void*, const char *),
242  void *inquire_cb_arg,
243  gpg_error_t (*status_cb)(void*, const char *),
244  void *status_cb_arg)
245 {
246  gpg_error_t rc;
247  assuan_response_t response;
248  int off;
249  char *line;
250  int linelen;
251 
252  rc = assuan_write_line (ctx, command);
253  if (rc)
254  return rc;
255 
256  if (*command == '#' || !*command)
257  return 0; /* Don't expect a response for a comment line. */
258 
259  again:
260  rc = _assuan_read_from_server (ctx, &response, &off,
261  ctx->flags.convey_comments);
262  if (rc)
263  return rc; /* error reading from server */
264 
265  line = ctx->inbound.line + off;
266  linelen = ctx->inbound.linelen - off;
267 
268  if (response == ASSUAN_RESPONSE_ERROR)
269  rc = atoi (line);
270  else if (response == ASSUAN_RESPONSE_DATA)
271  {
272  if (!data_cb)
273  rc = _assuan_error (ctx, GPG_ERR_ASS_NO_DATA_CB);
274  else
275  {
276  rc = data_cb (data_cb_arg, line, linelen);
277  if (!rc)
278  goto again;
279  }
280  }
281  else if (response == ASSUAN_RESPONSE_INQUIRE)
282  {
283  if (!inquire_cb)
284  {
285  assuan_write_line (ctx, "END"); /* get out of inquire mode */
286  _assuan_read_from_server (ctx, &response, &off, 0); /* dummy read */
287  rc = _assuan_error (ctx, GPG_ERR_ASS_NO_INQUIRE_CB);
288  }
289  else
290  {
291  rc = inquire_cb (inquire_cb_arg, line);
292  if (!rc)
293  rc = assuan_send_data (ctx, NULL, 0); /* flush and send END */
294  else
295  { /* Flush and send CAN. */
296  /* Note that in this error case we don't want to return
297  an error code from sending the cancel. The dummy
298  read is to remove the response from the server which
299  we are not interested in. */
300  assuan_send_data (ctx, NULL, 1);
301  _assuan_read_from_server (ctx, &response, &off, 0);
302  }
303  if (!rc)
304  goto again;
305  }
306  }
307  else if (response == ASSUAN_RESPONSE_STATUS)
308  {
309  if (status_cb)
310  rc = status_cb (status_cb_arg, line);
311  if (!rc)
312  goto again;
313  }
314  else if (response == ASSUAN_RESPONSE_COMMENT && ctx->flags.convey_comments)
315  {
316  line -= off; /* Send line with the comment marker. */
317  if (status_cb)
318  rc = status_cb (status_cb_arg, line);
319  if (!rc)
320  goto again;
321  }
322  else if (response == ASSUAN_RESPONSE_END)
323  {
324  if (!data_cb)
325  rc = _assuan_error (ctx, GPG_ERR_ASS_NO_DATA_CB);
326  else
327  {
328  rc = data_cb (data_cb_arg, NULL, 0);
329  if (!rc)
330  goto again;
331  }
332  }
333 
334  return rc;
335 }
