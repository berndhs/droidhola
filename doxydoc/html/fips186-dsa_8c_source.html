109 /* Convert STRING consisting of hex characters into its binary
110  representation and return it as an allocated buffer. The valid
111  length of the buffer is returned at R_LENGTH. The string is
112  delimited by end of string. The function returns NULL on
113  error. */
114 static void *
115 data_from_hex (const char *string, size_t *r_length)
116 {
117  const char *s;
118  unsigned char *buffer;
119  size_t length;
120 
121  buffer = gcry_xmalloc (strlen(string)/2+1);
122  length = 0;
123  for (s=string; *s; s +=2 )
124  {
125  if (!hexdigitp (s) || !hexdigitp (s+1))
126  die ("error parsing hex string `%s'\n", string);
127  ((unsigned char*)buffer)[length++] = xtoi_2 (s);
128  }
129  *r_length = length;
130  return buffer;
131 }

134 static void
135 extract_cmp_mpi (gcry_sexp_t sexp, const char *name, const char *expected)
136 {
137  gcry_sexp_t l1;
138  gcry_mpi_t a, b;
139 
140  l1 = gcry_sexp_find_token (sexp, name, 0);
141  a = gcry_sexp_nth_mpi (l1, 1, GCRYMPI_FMT_USG);
142  b = mpi_from_string (expected);
143  if (!a)
144  fail ("parameter \"%s\" missing in key\n", name);
145  else if ( gcry_mpi_cmp (a, b) )
146  fail ("parameter \"%s\" does not match expected value\n", name);
147  gcry_mpi_release (b);
148  gcry_mpi_release (a);
149  gcry_sexp_release (l1);
150 }

153 static void
154 extract_cmp_data (gcry_sexp_t sexp, const char *name, const char *expected)
155 {
156  gcry_sexp_t l1;
157  const void *a;
158  size_t alen;
159  void *b;
160  size_t blen;
161 
162  l1 = gcry_sexp_find_token (sexp, name, 0);
163  a = gcry_sexp_nth_data (l1, 1, &alen);
164  b = data_from_hex (expected, &blen);
165  if (!a)
166  fail ("parameter \"%s\" missing in key\n", name);
167  else if ( alen != blen || memcmp (a, b, alen) )
168  fail ("parameter \"%s\" does not match expected value\n", name);
169  gcry_free (b);
170  gcry_sexp_release (l1);
171 }
