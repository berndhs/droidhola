62 char *
63 ksba_oid_to_str (const char *buffer, size_t length)
64 {
65  const unsigned char *buf = buffer;
66  char *string, *p;
67  int n = 0;
68  unsigned long val, valmask;
69 
70  valmask = (unsigned long)0xfe << (8 * (sizeof (valmask) - 1));
71 
72  /* To calculate the length of the string we can safely assume an
73  upper limit of 3 decimal characters per byte. Two extra bytes
74  account for the special first octect */
75  string = p = xtrymalloc (length*(1+3)+2+1);
76  if (!string)
77  return NULL;
78  if (!length)
79  {
80  *p = 0;
81  return string;
82  }
83 
84  if (buf[0] < 40)
85  p += sprintf (p, "0.%d", buf[n]);
86  else if (buf[0] < 80)
87  p += sprintf (p, "1.%d", buf[n]-40);
88  else {
89  val = buf[n] & 0x7f;
90  while ( (buf[n]&0x80) && ++n < length )
91  {
92  if ( (val & valmask) )
93  goto badoid; /* Overflow. */
94  val <<= 7;
95  val |= buf[n] & 0x7f;
96  }
97  if (val < 80)
98  goto badoid;
99  val -= 80;
100  sprintf (p, "2.%lu", val);
101  p += strlen (p);
102  }
103  for (n++; n < length; n++)
104  {
105  val = buf[n] & 0x7f;
106  while ( (buf[n]&0x80) && ++n < length )
107  {
108  if ( (val & valmask) )
109  goto badoid; /* Overflow. */
110  val <<= 7;
111  val |= buf[n] & 0x7f;
112  }
113  sprintf (p, ".%lu", val);
114  p += strlen (p);
115  }
116 
117  *p = 0;
118  return string;
119 
120  badoid:
121  /* Return a special OID (gnu.gnupg.badoid) to indicate the error
122  case. The OID is broken and thus we return one which can't do
123  any harm. Formally this does not need to be a bad OID but an OID
124  with an arc that can't be represented in a 32 bit word is more
125  than likely corrupt. */
126  xfree (string);
127  return xtrystrdup ("1.3.6.1.4.1.11591.2.12242973");
128 }

142 static size_t
143 make_flagged_int (unsigned long value, char *buf, size_t buflen)
144 {
145  int more = 0;
146  int shift;
147 
148  /* fixme: figure out the number of bits in an ulong and start with
149  that value as shift (after making it a multiple of 7) a more
150  straigtforward implementation is to do it in reverse order using
151  a temporary buffer - saves a lot of compares */
152  for (more=0, shift=28; shift > 0; shift -= 7)
153  {
154  if (more || value >= (1<<shift))
155  {
156  buf[buflen++] = 0x80 | (value >> shift);
157  value -= (value >> shift) << shift;
158  more = 1;
159  }
160  }
161  buf[buflen++] = value;
162  return buflen;
163 }

182 gpg_error_t
183 ksba_oid_from_str (const char *string, unsigned char **rbuf, size_t *rlength)
184 {
185  unsigned char *buf;
186  size_t buflen;
187  unsigned long val1, val;
188  const char *endp;
189  int arcno;
190 
191  if (!string || !rbuf || !rlength)
192  return gpg_error (GPG_ERR_INV_VALUE);
193  *rbuf = NULL;
194  *rlength = 0;
195 
196  /* we allow the OID to be prefixed with either "oid." or "OID." */
197  if ( !strncmp (string, "oid.", 4) || !strncmp (string, "OID.", 4))
198  string += 4;
199 
200  if (!*string)
201  return gpg_error (GPG_ERR_INV_VALUE);
202 
203  /* we can safely assume that the encoded OID is shorter than the string */
204  buf = xtrymalloc ( strlen(string) + 2);
205  if (!buf)
206  return gpg_error (GPG_ERR_ENOMEM);
207  buflen = 0;
208 
209  val1 = 0; /* avoid compiler warnings */
210  arcno = 0;
211  do {
212  arcno++;
213  val = strtoul (string, (char**)&endp, 10);
214  if (!digitp (string) || !(*endp == '.' || !*endp))
215  {
216  xfree (buf);
217  return gpg_error (GPG_ERR_INV_OID_STRING);
218  }
219  if (*endp == '.')
220  string = endp+1;
221 
222  if (arcno == 1)
223  {
224  if (val > 2)
225  break; /* not allowed, error catched below */
226  val1 = val;
227  }
228  else if (arcno == 2)
229  { /* need to combine the first to arcs in one octet */
230  if (val1 < 2)
231  {
232  if (val > 39)
233  {
234  xfree (buf);
235  return gpg_error (GPG_ERR_INV_OID_STRING);
236  }
237  buf[buflen++] = val1*40 + val;
238  }
239  else
240  {
241  val += 80;
242  buflen = make_flagged_int (val, buf, buflen);
243  }
244  }
245  else
246  {
247  buflen = make_flagged_int (val, buf, buflen);
248  }
249  } while (*endp == '.');
250 
251  if (arcno == 1)
252  { /* it is not possible to encode only the first arc */
253  xfree (buf);
254  return gpg_error (GPG_ERR_INV_OID_STRING);
255  }
256 
257  *rbuf = buf;
258  *rlength = buflen;
259  return 0;
260 }
