344 QHash<QString, QByteArray> spoton_starbeam_reader::elementsFromMagnet
345 (const QByteArray &magnet, spoton_crypt *s_crypt)
346 {
347  QByteArray data;
348  QHash<QString, QByteArray> elements;
349  QList<QByteArray> list;
350  bool ok = true;
351 
352  if(!s_crypt)
353  goto done_label;
354 
355  data = s_crypt->decryptedAfterAuthenticated(magnet, &ok);
356 
357  if(!ok)
358  goto done_label;
359 
360  list = data.remove
361  (0, static_cast<int> (qstrlen("magnet:?"))).split('&');
362 
363  while(!list.isEmpty())
364  {
365  QByteArray bytes(list.takeFirst());
366 
367  if(bytes.startsWith("ct=")) // Cipher Type
368  {
369  bytes.remove(0, 3);
370  elements.insert("ct", bytes);
371  }
372  else if(bytes.startsWith("ek=")) // Encryption Key
373  {
374  bytes.remove(0, 3);
375  elements.insert("ek", bytes);
376  }
377  else if(bytes.startsWith("ht=")) // Hash Type
378  {
379  bytes.remove(0, 3);
380  elements.insert("ht", bytes);
381  }
382  else if(bytes.startsWith("mk=")) // MAC Key
383  {
384  bytes.remove(0, 3);
385  elements.insert("mk", bytes);
386  }
387  else if(bytes.startsWith("xt="))
388  {
389  bytes.remove(0, 3);
390 
391  if(bytes == "urn:starbeam")
392  elements.insert("xt", bytes);
393  }
394  }
395 
396  if(!elements.contains("xt"))
397  {
398  elements.clear();
399  goto done_label;
400  }
401 
402  done_label:
403  return elements;
404 }

406 void spoton_starbeam_reader::pulsate(const QByteArray &buffer,
407  const QString &fileName,
408  const QString &pulseSize,
409  const QString &fileSize,
410  const QByteArray &magnet,
411  const QByteArray &nova,
412  const QByteArray &hash,
413  const QSqlDatabase &db,
414  const qint64 rc,
415  spoton_crypt *s_crypt)
416 {
417  if(m_position < 0)
418  {
419  spoton_misc::logError("spoton_starbeam_reader::pulsate(): "
420  "m_position is negative.");
421  return;
422  }
423 
424  QHash<QString, QByteArray> elements(elementsFromMagnet(magnet, s_crypt));
425 
426  if(elements.isEmpty())
427  {
428  spoton_misc::logError("spoton_starbeam_reader::pulsate(): "
429  "elements is empty.");
430  return;
431  }
432 
433  QByteArray bytes;
434  QByteArray data(buffer.mid(0, static_cast<int> (rc)));
435  QByteArray messageCode;
436  QDataStream stream(&bytes, QIODevice::WriteOnly);
437  QString status("completed");
438  bool ok = true;
439  int size = 0;
440  spoton_crypt crypt(elements.value("ct").constData(),
441  elements.value("ht").constData(),
442  QByteArray(),
443  elements.value("ek"),
444  elements.value("mk"),
445  0,
446  0,
447  "");
448 
449  data = qCompress(data, 9);
450  size = data.length();
451  stream << QByteArray("0060")
452  << QFileInfo(fileName).fileName().toUtf8()
453  << QByteArray::number(m_position)
454  << QByteArray::number(size)
455  << fileSize.toLatin1()
456  << data
457  << pulseSize.toLatin1()
458  << hash;
459 
460  if(stream.status() != QDataStream::Ok)
461  ok = false;
462  else
463  ok = true;
464 
465  if(nova.isEmpty())
466  {
467  if(ok)
468  data = crypt.encrypted(bytes, &ok);
469  }
470  else
471  {
472  QPair<QByteArray, QByteArray> pair;
473 
474  pair.first = nova.mid
475  (0, static_cast<int> (spoton_crypt::
476  cipherKeyLength("aes256")));
477  pair.second = nova.mid(pair.first.length());
478 
479  {
480  spoton_crypt crypt("aes256",
481  "sha512",
482  QByteArray(),
483  pair.first,
484  pair.second,
485  0,
486  0,
487  "");
488 
489  if(ok)
490  data = crypt.encrypted(bytes, &ok);
491 
492  if(ok)
493  data = data + crypt.keyedHash(data, &ok);
494  }
495 
496  if(ok)
497  data = crypt.encrypted(data, &ok);
498  }
499 
500  if(ok)
501  messageCode = crypt.keyedHash(data, &ok);
502 
503  if(ok)
504  data = data.toBase64() + "\n" + messageCode.toBase64();
505 
506  if(ok)
507  {
508  if(spoton_kernel::instance())
509  spoton_kernel::instance()->writeMessage0060
510  (data, m_fragmented ? &m_neighborIndex : 0, &ok);
511  else
512  ok = false;
513  }
514 
515  if(ok)
516  {
517  if(m_missingLinksIterator)
518  {
519  if(!m_missingLinksIterator->hasNext())
520  m_position = QFileInfo(fileName).size();
521  }
522  else
523  m_position = qAbs(m_position + rc); // +=
524  }
525 
526  if(m_position < QFileInfo(fileName).size())
527  status = "transmitting";
528 
529  if(ok)
530  savePositionAndStatus(status, db);
531 }
