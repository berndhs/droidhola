80 static libspoton_error_t initialize_libgcrypt
81 (const int secure_memory_pool_size)
82 {
83  /*
84  ** Initialize the gcrypt library if it has not yet been
85  ** initialized.
86  */
87 
88  libspoton_error_t rerr = LIBSPOTON_ERROR_NONE;
89 
90  if(!gcryctl_set_thread_cbs_set)
91  {
92  gcry_error_t err = 0;
93 
94 #if !defined(GCRYPT_VERSION_NUMBER) || GCRYPT_VERSION_NUMBER < 0x010600
95  err = gcry_control(GCRYCTL_SET_THREAD_CBS, &gcry_threads_pthread, 0);
96 #endif
97 
98  if(err == 0)
99  gcryctl_set_thread_cbs_set = true;
100  else
101  fprintf(stderr, "libspoton::initialize_libgcrypt(): "
102  "error initializing threads. Proceeding.\n");
103  }
104 
105  if(!gcry_control(GCRYCTL_INITIALIZATION_FINISHED_P))
106  {
107  gcry_control(GCRYCTL_ENABLE_M_GUARD);
108 
109  if(!gcry_check_version(GCRYPT_VERSION))
110  rerr = LIBSPOTON_ERROR_GCRY_CHECK_VERSION;
111  else if(secure_memory_pool_size == 0)
112  {
113  fprintf(stderr, "libspoton::initialize_libgcrypt(): "
114  "disabling secure memory.\n");
115  gcry_control(GCRYCTL_DISABLE_SECMEM, 0);
116  gcry_control(GCRYCTL_INITIALIZATION_FINISHED, 0);
117  }
118  else
119  {
120  gcry_control(GCRYCTL_SUSPEND_SECMEM_WARN);
121 #ifdef LIBSPOTON_IGNORE_GCRY_CONTROL_GCRYCTL_INIT_SECMEM_RETURN_VALUE
122  gcry_control(GCRYCTL_INIT_SECMEM, abs(secure_memory_pool_size), 0);
123 #else
124  if(gcry_control(GCRYCTL_INIT_SECMEM,
125  abs(secure_memory_pool_size), 0) != 0)
126  rerr = LIBSPOTON_ERROR_GCRY_CONTROL;
127 #endif
128  gcry_control(GCRYCTL_RESUME_SECMEM_WARN);
129  gcry_control(GCRYCTL_INITIALIZATION_FINISHED, 0);
130  }
131  }
132 
133  return rerr;
134 }

294 libspoton_error_t libspoton_init_a(const char *databasePath,
295  const char *cipherType,
296  const char *key,
297  const size_t keyLength,
298  libspoton_handle_t *libspotonHandle,
299  const int secure_memory_pool_size)
300 {
301  int rv = 0;
302  libspoton_error_t rerr = LIBSPOTON_ERROR_NONE;
303 
304  if(!libspotonHandle)
305  {
306  rerr = LIBSPOTON_ERROR_NULL_LIBSPOTON_HANDLE;
307  goto error_label;
308  }
309 
310  libspotonHandle->m_cipherAlgorithm = 0;
311  libspotonHandle->m_key = 0;
312  libspotonHandle->m_keyLength = 0;
313  libspotonHandle->m_sqliteHandle = 0;
314 
315  /*
316  ** Initialize libgcrypt.
317  */
318 
319  if((rerr = initialize_libgcrypt(secure_memory_pool_size)) !=
320  LIBSPOTON_ERROR_NONE)
321  goto error_label;
322 
323  if(key && keyLength > 0)
324  {
325  if((libspotonHandle->m_cipherAlgorithm =
326  gcry_cipher_map_name(cipherType)) == 0)
327  {
328  rerr = LIBSPOTON_ERROR_GCRY_CIPHER_MAP_NAME;
329  goto error_label;
330  }
331 
332  libspotonHandle->m_keyLength = keyLength;
333 
334  if(!(libspotonHandle->m_key =
335  gcry_calloc_secure(libspotonHandle->m_keyLength,
336  sizeof(char))))
337  {
338  rerr = LIBSPOTON_ERROR_GCRY_CALLOC_SECURE;
339  goto error_label;
340  }
341  else
342  memcpy(libspotonHandle->m_key, key, keyLength);
343  }
344 
345  /*
346  ** Create the shared.db database.
347  */
348 
349  pthread_mutex_lock(&sqlite_mutex);
350  rv = sqlite3_open_v2(databasePath,
351  &libspotonHandle->m_sqliteHandle,
352  SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE,
353  0);
354  pthread_mutex_unlock(&sqlite_mutex);
355 
356  if(rv != SQLITE_OK)
357  {
358  rerr = LIBSPOTON_ERROR_SQLITE_OPEN_V2;
359  goto error_label;
360  }
361 
362  error_label:
363 
364  if(rerr != LIBSPOTON_ERROR_NONE)
365  libspoton_close(libspotonHandle);
366 
367  return rerr;
368 }

370 libspoton_error_t libspoton_init_b(const char *databasePath,
371  const char *cipherType,
372  const char *hashType,
373  const char *passphrase,
374  const char *salt,
375  const size_t passphraseLength,
376  const size_t saltLength,
377  const unsigned long iterationCount,
378  libspoton_handle_t *libspotonHandle,
379  const int secure_memory_pool_size)
380 {
381  int hashAlgorithm = 0;
382  int rv = 0;
383  libspoton_error_t rerr = LIBSPOTON_ERROR_NONE;
384 
385  if(!libspotonHandle)
386  {
387  rerr = LIBSPOTON_ERROR_NULL_LIBSPOTON_HANDLE;
388  goto error_label;
389  }
390 
391  libspotonHandle->m_cipherAlgorithm = 0;
392  libspotonHandle->m_key = 0;
393  libspotonHandle->m_keyLength = 0;
394  libspotonHandle->m_sqliteHandle = 0;
395 
396  /*
397  ** Initialize libgcrypt.
398  */
399 
400  if((rerr =
401  initialize_libgcrypt(secure_memory_pool_size)) != LIBSPOTON_ERROR_NONE)
402  goto error_label;
403 
404  if(cipherType && hashType && iterationCount > 0 &&
405  passphrase && salt && passphraseLength > 0 && saltLength > 0)
406  {
407  if((libspotonHandle->m_cipherAlgorithm =
408  gcry_cipher_map_name(cipherType)) == 0)
409  {
410  rerr = LIBSPOTON_ERROR_GCRY_CIPHER_MAP_NAME;
411  goto error_label;
412  }
413 
414  if((hashAlgorithm = gcry_md_map_name(hashType)) == 0)
415  {
416  rerr = LIBSPOTON_ERROR_GCRY_MD_MAP_NAME;
417  goto error_label;
418  }
419 
420  if((libspotonHandle->m_keyLength =
421  gcry_cipher_get_algo_keylen(libspotonHandle->
422  m_cipherAlgorithm)) == 0)
423  {
424  rerr = LIBSPOTON_ERROR_GCRY_CIPHER_GET_ALGO_KEYLEN;
425  goto error_label;
426  }
427 
428  if(!(libspotonHandle->m_key =
429  gcry_calloc_secure(libspotonHandle->m_keyLength,
430  sizeof(char))))
431  {
432  rerr = LIBSPOTON_ERROR_GCRY_CALLOC_SECURE;
433  goto error_label;
434  }
435 
436  gcry_fast_random_poll();
437 
438  if(gcry_kdf_derive((const void *) passphrase,
439  passphraseLength,
440  GCRY_KDF_PBKDF2,
441  hashAlgorithm,
442  (const void *) salt,
443  saltLength,
444  iterationCount,
445  libspotonHandle->m_keyLength,
446  (void *) libspotonHandle->m_key) != 0)
447  {
448  rerr = LIBSPOTON_ERROR_GCRY_KDF_DERIVE;
449  goto error_label;
450  }
451  }
452 
453  /*
454  ** Create the shared.db database.
455  */
456 
457  pthread_mutex_lock(&sqlite_mutex);
458  rv = sqlite3_open_v2(databasePath,
459  &libspotonHandle->m_sqliteHandle,
460  SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE,
461  0);
462  pthread_mutex_unlock(&sqlite_mutex);
463 
464  if(rv != SQLITE_OK)
465  {
466  rerr = LIBSPOTON_ERROR_SQLITE_OPEN_V2;
467  goto error_label;
468  }
469 
470  error_label:
471 
472  if(rerr != LIBSPOTON_ERROR_NONE)
473  libspoton_close(libspotonHandle);
474 
475  return rerr;
476 }

478 libspoton_error_t libspoton_register_kernel
479 (const pid_t pid,
480  const bool forceRegistration,
481  libspoton_handle_t *libspotonHandle)
482 {
483  const char *sql = "INSERT OR REPLACE INTO kernel_registration (pid) "
484  "VALUES (?)";
485  int rv = 0;
486  libspoton_error_t rerr = LIBSPOTON_ERROR_NONE;
487  sqlite3_stmt *stmt = 0;
488 
489  if(!libspotonHandle)
490  {
491  rerr = LIBSPOTON_ERROR_NULL_LIBSPOTON_HANDLE;
492  goto error_label;
493  }
494  else if(!libspotonHandle->m_sqliteHandle)
495  {
496  rerr = LIBSPOTON_ERROR_NOT_CONNECTED_TO_SQLITE_DATABASE;
497  goto error_label;
498  }
499 
500  pthread_mutex_lock(&sqlite_mutex);
501  rv = sqlite3_exec(libspotonHandle->m_sqliteHandle,
502  "CREATE TABLE IF NOT EXISTS kernel_registration ("
503  "pid INTEGER PRIMARY KEY NOT NULL)",
504  0,
505  0,
506  0);
507  pthread_mutex_unlock(&sqlite_mutex);
508 
509  if(rv != SQLITE_OK)
510  {
511  rerr = LIBSPOTON_ERROR_SQLITE_CREATE_KERNEL_REGISTRATION_TABLE;
512  goto error_label;
513  }
514 
515  /*
516  ** The kernel_registration table must contain only one entry.
517  */
518 
519  pthread_mutex_lock(&sqlite_mutex);
520  rv = sqlite3_exec(libspotonHandle->m_sqliteHandle,
521  "CREATE TRIGGER IF NOT EXISTS kernel_registration_trigger "
522  "BEFORE INSERT ON kernel_registration "
523  "BEGIN "
524  "DELETE FROM kernel_registration; "
525  "END",
526  0,
527  0,
528  0);
529  pthread_mutex_unlock(&sqlite_mutex);
530

534  goto error_label;

544  if(rerr != LIBSPOTON_ERROR_NONE)
545  goto error_label;
546 
547  pthread_mutex_lock(&sqlite_mutex);

552  0);

554 
555  if(rv != SQLITE_OK)
556  {
557  rerr = LIBSPOTON_ERROR_SQLITE_PREPARE_V2;
558  goto error_label;
559  }
560 
561  if(sqlite3_bind_int64(stmt, 1, pid) != SQLITE_OK)
562  {
563  rerr = LIBSPOTON_ERROR_SQLITE_BIND_INT64;
564  goto error_label;
565  }
566 
567  rv = sqlite3_step(stmt);
568 
569  if(!(rv == SQLITE_OK || rv == SQLITE_DONE))
570  {
571  rerr = LIBSPOTON_ERROR_SQLITE_STEP;
572  goto error_label;
573  }
574 
575  error_label:
576  sqlite3_finalize(stmt);
577  return rerr;
578 }
