203 static inline u64
204 ROTR (u64 x, u64 n)
205 {
206  return ((x >> n) | (x << (64 - n)));
207 }

209 static inline u64
210 Ch (u64 x, u64 y, u64 z)
211 {
212  return ((x & y) ^ ( ~x & z));
213 }
214 
215 static inline u64
216 Maj (u64 x, u64 y, u64 z)
217 {
218  return ((x & y) ^ (x & z) ^ (y & z));
219 }

606 /* The routine final terminates the computation and
607  * returns the digest.
608  * The handle is prepared for a new cycle, but adding bytes to the
609  * handle will the destroy the returned buffer.
610  * Returns: 64 bytes representing the digest. When used for sha384,
611  * we take the leftmost 48 of those bytes.
612  */
613 
614 static void
615 sha512_final (void *context)
616 {
617  SHA512_CONTEXT *hd = context;
618  unsigned int stack_burn_depth;
619  u64 t, th, msb, lsb;
620  byte *p;
621 
622  _gcry_md_block_write (context, NULL, 0); /* flush */ ;
623 
624  t = hd->bctx.nblocks;
625  /* if (sizeof t == sizeof hd->bctx.nblocks) */
626  th = hd->bctx.nblocks_high;
627  /* else */
628  /* th = hd->bctx.nblocks >> 64; In case we ever use u128 */
629 
630  /* multiply by 128 to make a byte count */
631  lsb = t << 7;
632  msb = (th << 7) | (t >> 57);
633  /* add the count */
634  t = lsb;
635  if ((lsb += hd->bctx.count) < t)
636  msb++;
637  /* multiply by 8 to make a bit count */
638  t = lsb;
639  lsb <<= 3;
640  msb <<= 3;
641  msb |= t >> 61;
642 
643  if (hd->bctx.count < 112)
644  { /* enough room */
645  hd->bctx.buf[hd->bctx.count++] = 0x80; /* pad */
646  while (hd->bctx.count < 112)
647  hd->bctx.buf[hd->bctx.count++] = 0; /* pad */
648  }
649  else
650  { /* need one extra block */
651  hd->bctx.buf[hd->bctx.count++] = 0x80; /* pad character */
652  while (hd->bctx.count < 128)
653  hd->bctx.buf[hd->bctx.count++] = 0;
654  _gcry_md_block_write (context, NULL, 0); /* flush */ ;
655  memset (hd->bctx.buf, 0, 112); /* fill next block with zeroes */
656  }
657  /* append the 128 bit count */
658  buf_put_be64(hd->bctx.buf + 112, msb);
659  buf_put_be64(hd->bctx.buf + 120, lsb);
660  stack_burn_depth = transform (hd, hd->bctx.buf);
661  _gcry_burn_stack (stack_burn_depth);
662 
663  p = hd->bctx.buf;
664 #define X(a) do { *(u64*)p = be_bswap64(hd->state.h##a) ; p += 8; } while (0)
665  X (0);
666  X (1);
667  X (2);
668  X (3);
669  X (4);
670  X (5);
671  /* Note that these last two chunks are included even for SHA384.
672  We just ignore them. */
673  X (6);
674  X (7);
675 #undef X
676 }
