121 /* Note: Keeps the next two macros in sync
122  with their counterparts in gpg-error.h. */
123 #define _gpgrt_getc_unlocked(stream) \
124  (((!(stream)->flags.writing) \
125  && ((stream)->data_offset < (stream)->data_len) \
126  && (! (stream)->unread_data_len)) \
127  ? ((int) (stream)->buffer[((stream)->data_offset)++]) \
128  : _gpgrt__getc_underflow ((stream)))
129 
130 #define _gpgrt_putc_unlocked(c, stream) \
131  (((stream)->flags.writing \
132  && ((stream)->data_offset < (stream)->buffer_size) \
133  && (c != '\n')) \
134  ? ((int) ((stream)->buffer[((stream)->data_offset)++] = (c))) \
135  : _gpgrt__putc_overflow ((c), (stream)))
