334  res->sign = rsign;
335

341  if (ep_marker)
342  _gcry_mpi_free_limb_space( ep_marker, ep_nlimbs );

344  _gcry_mpi_free_limb_space( xp_marker, xp_nlimbs );
345  if (tspace)
