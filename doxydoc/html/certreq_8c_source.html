130 /* Store the serial number. If this function is used, a real X.509
131  certificate will be built instead of a pkcs#10 certificate signing
132  request. SN must be a simple canonical encoded s-expression with
133  the serial number as its only item. Note that this function allows
134  to set a negative serial number, which is not forbidden but
135  probably not a good idea. */
136 gpg_error_t
137 ksba_certreq_set_serial (ksba_certreq_t cr, ksba_const_sexp_t sn)
138 {
139  const char *p = (const char *)sn;
140  unsigned long n;
141  char *endp;
142 
143  if (!cr || !sn || !p || *p != '(')
144  return gpg_error (GPG_ERR_INV_VALUE);
145 
146  p++;
147  n = strtoul (p, &endp, 10);
148  p = endp;
149  if (*p++ != ':' || !n)
150  return gpg_error (GPG_ERR_INV_VALUE);
151 
152  /* Remove invalid leading zero bytes. */
153  for (; n > 1 && !*p && !(p[1] & 0x80); n--, p++)
154  ;
155 
156  if (cr->x509.serial.der)
157  return gpg_error (GPG_ERR_CONFLICT); /* Already set */
158  cr->x509.serial.der = xtrymalloc (n);
159  if (!cr->x509.serial.der)
160  return gpg_error_from_syserror ();
161  memcpy (cr->x509.serial.der, p, n);
162  cr->x509.serial.derlen = n;
163 
164  return 0;
165 }

224 /* Store the subject's name. Does perform some syntactic checks on
225  the name. The first added subject is the real one, all subsequent
226  calls add subjectAltNames.
227 
228  NAME must be a valid RFC-2253 encoded DN name for the first one or an
229  email address enclosed in angle brackets for all further calls.
230  */
231 gpg_error_t
232 ksba_certreq_add_subject (ksba_certreq_t cr, const char *name)
233 {
234  unsigned long namelen;
235  size_t n, n1;
236  struct general_names_s *gn;
237  unsigned char *der;
238  int tag;
239  const char *endp;
240 
241  if (!cr || !name)
242  return gpg_error (GPG_ERR_INV_VALUE);
243  if (!cr->subject.der)
244  return _ksba_dn_from_str (name, &cr->subject.der, &cr->subject.derlen);
245  /* This is assumed to be an subjectAltName. */
246 
247  /* Note that the way we pass the name should match what
248  ksba_cert_get_subject() returns. In particular we expect that it
249  is a real string and thus a canonical S-expression is
250  additionally terminated by a 0. */
251  namelen = strlen (name);
252  if (*name == '<' && name[namelen-1] == '>'
253  && namelen >= 4 && strchr (name, '@'))
254  {
255  name++;
256  namelen -= 2;
257  tag = 1; /* rfc822Name */
258  }
259  else if (!strncmp (name, "(8:dns-name", 11))
260  {
261  tag = 2; /* dNSName */
262  namelen = strtoul (name+11, (char**)&endp, 10);
263  name = endp;
264  if (!namelen || *name != ':')
265  return gpg_error (GPG_ERR_INV_SEXP);
266  name++;
267  }
268  else if (!strncmp (name, "(3:uri", 6))
269  {
270  tag = 6; /* uRI */
271  namelen = strtoul (name+6, (char**)&endp, 10);
272  name = endp;
273  if (!namelen || *name != ':')
274  return gpg_error (GPG_ERR_INV_SEXP);
275  name++;
276  }
277  else
278  return gpg_error (GPG_ERR_INV_VALUE);
279 
280  n1 = _ksba_ber_count_tl (tag, CLASS_CONTEXT, 0, namelen);
281  n1 += namelen;
282 
283  gn = xtrymalloc (sizeof *gn + n1 - 1);
284  if (!gn)
285  return gpg_error_from_errno (errno);
286  gn->tag = tag;
287  gn->datalen = n1;
288  der = (unsigned char *)gn->data;
289  n = _ksba_ber_encode_tl (der, tag, CLASS_CONTEXT, 0, namelen);
290  if (!n)
291  return gpg_error (GPG_ERR_BUG);
292  der += n;
293  memcpy (der, name, namelen);
294  assert (der + namelen - (unsigned char*)gn->data == n1);
295 
296  gn->next = cr->subject_alt_names;
297  cr->subject_alt_names = gn;
298 
299  return 0;
300 }

303 /* Add the GeneralNames object GNAMES to the list of extensions in CR.
304  Use OID as object identifier for the extensions. */
305 static gpg_error_t
306 add_general_names_to_extn (ksba_certreq_t cr, struct general_names_s *gnames,
307  const char *oid)
308 {
309  struct general_names_s *g;
310  size_t n, n1, n2;
311  struct extn_list_s *e;
312  unsigned char *der;
313 
314  /* Calculate the required size. */
315  n1 = 0;
316  for (g=gnames; g; g = g->next)
317  n1 += g->datalen;
318 
319  n2 = _ksba_ber_count_tl (TYPE_SEQUENCE, CLASS_UNIVERSAL, 1, n1);
320  n2 += n1;
321 
322  /* Allocate memory and encode all. */
323  e = xtrymalloc (sizeof *e + n2 - 1);
324  if (!e)
325  return gpg_error_from_errno (errno);
326  e->oid = oid;
327  e->critical = 0;
328  e->derlen = n2;
329  der = e->der;
330  n = _ksba_ber_encode_tl (der, TYPE_SEQUENCE, CLASS_UNIVERSAL, 1, n1);
331  if (!n)
332  return gpg_error (GPG_ERR_BUG); /* (no need to cleanup after a bug) */
333  der += n;
334 
335  for (g=gnames; g; g = g->next)
336  {
337  memcpy (der, g->data, g->datalen);
338  der += g->datalen;
339  }
340  assert (der - e->der == n2);
341 
342  e->next = cr->extn_list;
343  cr->extn_list = e;
344  return 0;
345 }

395  * r_sig = (sig-val
396  * (<algo>
397  * (<param_name1> <mpi>)
398  * ...
399  * (<param_namen> <mpi>)
400  * ))
401  * The sexp must be in canonical form.
402  * Fixme: The code is mostly duplicated from cms.c
403  * Note, that <algo> must be given as a stringified OID or the special

405 */
406 gpg_error_t
407 ksba_certreq_set_sig_val (ksba_certreq_t cr, ksba_const_sexp_t sigval)
408 {
409  const char *s, *endp;
410  unsigned long n;
411 
412  if (!cr)
413  return gpg_error (GPG_ERR_INV_VALUE);
414 
415  s = sigval;
416  if (*s != '(')
417  return gpg_error (GPG_ERR_INV_SEXP);
418  s++;
419 
420  n = strtoul (s, (char**)&endp, 10);
421  s = endp;
422  if (!n || *s!=':')
423  return gpg_error (GPG_ERR_INV_SEXP); /* we don't allow empty lengths */
424  s++;
425  if (n != 7 || memcmp (s, "sig-val", 7))
426  return gpg_error (GPG_ERR_UNKNOWN_SEXP);
427  s += 7;
428  if (*s != '(')
429  return gpg_error (digitp (s)? GPG_ERR_UNKNOWN_SEXP : GPG_ERR_INV_SEXP);
430  s++;
431 
432  /* break out the algorithm ID */
433  n = strtoul (s, (char**)&endp, 10);
434  s = endp;
435  if (!n || *s != ':')
436  return gpg_error (GPG_ERR_INV_SEXP); /* we don't allow empty lengths */
437  s++;
438  xfree (cr->sig_val.algo);
439  if (n==3 && s[0] == 'r' && s[1] == 's' && s[2] == 'a')
440  { /* kludge to allow "rsa" to be passed as algorithm name */
441  cr->sig_val.algo = xtrystrdup ("1.2.840.113549.1.1.5");
442  if (!cr->sig_val.algo)
443  return gpg_error (GPG_ERR_ENOMEM);
444  }
445  else
446  {
447  cr->sig_val.algo = xtrymalloc (n+1);
448  if (!cr->sig_val.algo)
449  return gpg_error (GPG_ERR_ENOMEM);
450  memcpy (cr->sig_val.algo, s, n);
451  cr->sig_val.algo[n] = 0;
452  }
453  s += n;
454 
455  /* And now the values - FIXME: For now we only support one */
456  /* fixme: start loop */
457  if (*s != '(')
458  return gpg_error (digitp (s)? GPG_ERR_UNKNOWN_SEXP : GPG_ERR_INV_SEXP);
459  s++;
460  n = strtoul (s, (char**)&endp, 10);
461  s = endp;
462  if (!n || *s != ':')
463  return gpg_error (GPG_ERR_INV_SEXP);
464  s++;
465  s += n; /* ignore the name of the parameter */
466 
467  if (!digitp(s))
468  return gpg_error (GPG_ERR_UNKNOWN_SEXP); /* but may also be an invalid one */
469  n = strtoul (s, (char**)&endp, 10);
470  s = endp;
471  if (!n || *s != ':')
472  return gpg_error (GPG_ERR_INV_SEXP);
473  s++;
474  if (n > 1 && !*s)
475  { /* We might have a leading zero due to the way we encode
476  MPIs - this zero should not go into the BIT STRING. */
477  s++;
478  n--;
479  }
480  xfree (cr->sig_val.value);
481  cr->sig_val.value = xtrymalloc (n);
482  if (!cr->sig_val.value)
483  return gpg_error (GPG_ERR_ENOMEM);
484  memcpy (cr->sig_val.value, s, n);
485  cr->sig_val.valuelen = n;
486  s += n;
487  if ( *s != ')')
488  return gpg_error (GPG_ERR_UNKNOWN_SEXP); /* but may also be an invalid one */
489  s++;
490  /* fixme: end loop over parameters */
491 
492  /* we need 2 closing parenthesis */
493  if ( *s != ')' || s[1] != ')')
494  return gpg_error (GPG_ERR_INV_SEXP);
495 
496  return 0;
497 }

501 /* Build the extension block and return it in R_DER and R_DERLEN. IF
502  CERTMODE is true build X.509 certificate extension instead. */
503 static gpg_error_t
504 build_extensions (ksba_certreq_t cr, int certmode,
505  void **r_der, size_t *r_derlen)
506 {
507  gpg_error_t err;
508  ksba_writer_t writer, w=NULL;
509  struct extn_list_s *e;
510  unsigned char *value = NULL;
511  size_t valuelen;
512  unsigned char *p;
513  size_t n;
514 
515  *r_der = NULL;
516  *r_derlen = 0;
517  err = ksba_writer_new (&writer);
518  if (err)
519  goto leave;
520  err = ksba_writer_set_mem (writer, 2048);
521  if (err)
522  goto leave;
523  err = ksba_writer_new (&w);
524  if (err)
525  goto leave;
526 
527  for (e=cr->extn_list; e; e = e->next)
528  {
529  err = ksba_writer_set_mem (w, e->derlen + 100);
530  if (err)
531  goto leave;
532 
533  err = ksba_oid_from_str (e->oid, &p, &n);
534  if(err)
535  goto leave;
536  err = _ksba_ber_write_tl (w, TYPE_OBJECT_ID, CLASS_UNIVERSAL, 0, n);
537  if (!err)
538  err = ksba_writer_write (w, p, n);
539  xfree (p);
540 
541  if (e->critical)
542  {
543  err = _ksba_ber_write_tl (w, TYPE_BOOLEAN, CLASS_UNIVERSAL, 0, 1);
544  if (!err)
545  err = ksba_writer_write (w, "\xff", 1);
546  if(err)
547  goto leave;
548  }
549 
550  err = _ksba_ber_write_tl (w, TYPE_OCTET_STRING, CLASS_UNIVERSAL,
551  0, e->derlen);
552  if (!err)
553  err = ksba_writer_write (w, e->der, e->derlen);
554  if(err)
555  goto leave;
556 
557  p = ksba_writer_snatch_mem (w, &n);
558  if (!p)
559  {
560  err = gpg_error (GPG_ERR_ENOMEM);
561  goto leave;
562  }
563  err = _ksba_ber_write_tl (writer, TYPE_SEQUENCE, CLASS_UNIVERSAL,
564  1, n);
565  if (!err)
566  err = ksba_writer_write (writer, p, n);
567  xfree (p); p = NULL;
568  if (err)
569  goto leave;
570  }
571 
572  /* Embed all the sequences into another sequence */
573  value = ksba_writer_snatch_mem (writer, &valuelen);
574  if (!value)
575  {
576  err = gpg_error (GPG_ERR_ENOMEM);
577  goto leave;
578  }
579  err = ksba_writer_set_mem (writer, valuelen+10);
580  if (err)
581  goto leave;
582  err = _ksba_ber_write_tl (writer, TYPE_SEQUENCE, CLASS_UNIVERSAL,
583  1, valuelen);
584  if (!err)
585  err = ksba_writer_write (writer, value, valuelen);
586  if (err)
587  goto leave;
588 
589  xfree (value);
590  value = ksba_writer_snatch_mem (writer, &valuelen);
591  if (!value)
592  {
593  err = gpg_error (GPG_ERR_ENOMEM);
594  goto leave;
595  }
596 
597  if (!certmode)
598  {
599  /* Now create the extension request sequence content */
600  err = ksba_writer_set_mem (writer, valuelen+100);
601  if (err)
602  goto leave;
603  err = ksba_oid_from_str (oidstr_extensionReq, &p, &n);
604  if(err)
605  goto leave;
606  err = _ksba_ber_write_tl (writer, TYPE_OBJECT_ID, CLASS_UNIVERSAL, 0, n);
607  if (!err)
608  err = ksba_writer_write (writer, p, n);
609  xfree (p); p = NULL;
610  if (err)
611  return err;
612  err = _ksba_ber_write_tl (writer, TYPE_SET, CLASS_UNIVERSAL, 1, valuelen);
613  if (!err)
614  err = ksba_writer_write (writer, value, valuelen);
615 
616  /* Put this all into a SEQUENCE */
617  xfree (value);
618  value = ksba_writer_snatch_mem (writer, &valuelen);
619  if (!value)
620  {
621  err = gpg_error (GPG_ERR_ENOMEM);
622  goto leave;
623  }
624  err = ksba_writer_set_mem (writer, valuelen+10);
625  if (err)
626  goto leave;
627  err = _ksba_ber_write_tl (writer, TYPE_SEQUENCE, CLASS_UNIVERSAL,
628  1, valuelen);
629  if (!err)
630  err = ksba_writer_write (writer, value, valuelen);
631  if (err)
632  goto leave;
633 
634  xfree (value);
635  value = ksba_writer_snatch_mem (writer, &valuelen);
636  if (!value)
637  {
638  err = gpg_error (GPG_ERR_ENOMEM);
639  goto leave;
640  }
641  }
642 
643  *r_der = value;
644  *r_derlen = valuelen;
645  value = NULL;
646 
647 
648  leave:
649  ksba_writer_release (writer);
650  ksba_writer_release (w);
651  xfree (value);
652  return err;
653 }

657 static gpg_error_t
658 build_cri (ksba_certreq_t cr)
659 {

662  void *value = NULL;

916 {

920  size_t valuelen;

972  if (err)

982  else

984 
985  leave:

993 /* The main function to build a certificate request. It used used in
994  a loop so allow for interaction between the function and the caller */
995 gpg_error_t
996 ksba_certreq_build (ksba_certreq_t cr, ksba_stop_reason_t *r_stopreason)
997 {
998  enum {
999  sSTART,
1000  sHASHING,
1001  sGOTSIG,
1002  sERROR
1003  } state = sERROR;
1004  gpg_error_t err = 0;
1005  ksba_stop_reason_t stop_reason;
1006 
1007  if (!cr || !r_stopreason)
1008  return gpg_error (GPG_ERR_INV_VALUE);
1009 
1010  if (!cr->any_build_done)
1011  { /* first time initialization of the stop reason */
1012  *r_stopreason = 0;
1013  cr->any_build_done = 1;
1014  }
1015 
1016  /* Calculate state from last reason */
1017  stop_reason = *r_stopreason;
1018  *r_stopreason = KSBA_SR_RUNNING;
1019  switch (stop_reason)
1020  {
1021  case 0:
1022  state = sSTART;
1023  break;
1024  case KSBA_SR_NEED_HASH:
1025  state = sHASHING;
1026  break;
1027  case KSBA_SR_NEED_SIG:
1028  if (!cr->sig_val.algo)
1029  err = gpg_error (GPG_ERR_MISSING_ACTION);
1030  else
1031  state = sGOTSIG;
1032  break;
1033  case KSBA_SR_RUNNING:
1034  err = gpg_error (GPG_ERR_INV_STATE);
1035  break;
1036  default:
1037  err = gpg_error (GPG_ERR_BUG);
1038  break;
1039  }
1040  if (err)
1041  return err;
1042 
1043  /* Do the action */
1044  switch (state)
1045  {
1046  case sSTART:
1047  err = build_cri (cr);
1048  break;
1049  case sHASHING:
1050  err = hash_cri (cr);
1051  break;
1052  case sGOTSIG:
1053  err = sign_and_write (cr);
1054  break;
1055  default:
1056  err = gpg_error (GPG_ERR_INV_STATE);
1057  break;
1058  }
1059  if (err)
1060  return err;
1061 
1062  /* Calculate new stop reason */
1063  switch (state)
1064  {
1065  case sSTART:
1066  stop_reason = KSBA_SR_NEED_HASH; /* caller should set the hash function*/
1067  break;
1068  case sHASHING:
1069  stop_reason = KSBA_SR_NEED_SIG;
1070  break;
1071  case sGOTSIG:
1072  stop_reason = KSBA_SR_READY;
1073  break;
1074  default:
1075  break;
1076  }
1077 
1078  *r_stopreason = stop_reason;
1079  return 0;
1080 }
