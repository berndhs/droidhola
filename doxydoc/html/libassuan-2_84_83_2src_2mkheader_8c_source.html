28 /* Include the file NAME form the source directory. The included file
29  is not further expanded. It may have comments indicated by a
30  double hash mark at the begin of a line. */
31 static void
32 include_file (const char *fname, int lnr, const char *name)
33 {
34  FILE *fp;
35  char *incfname;
36  char line[LINESIZE];
37 
38  incfname = malloc (strlen (srcdir) + strlen (name) + 1);
39  if (!incfname)
40  {
41  fputs (PGM ": out of core\n", stderr);
42  exit (1);
43  }
44  strcpy (incfname, srcdir);
45  strcat (incfname, name);
46 
47  fp = fopen (incfname, "r");
48  if (!fp)
49  {
50  fprintf (stderr, "%s:%d: error including `%s': %s\n",
51  fname, lnr, incfname, strerror (errno));
52  exit (1);
53  }
54 
55  while (fgets (line, LINESIZE, fp))
56  {
57  if (line[0] == '#' && line[1] == '#')
58  {
59  if (!strncmp (line+2, "EOF##", 5))

61  }

63  fputs (line, stdout);
64  }
65  if (ferror (fp))
66  {
67  fprintf (stderr, "%s:%d: error reading `%s': %s\n",
68  fname, lnr, incfname, strerror (errno));
