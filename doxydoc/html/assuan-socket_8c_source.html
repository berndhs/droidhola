160 /* Insert SOCKFD into the table of Cygwin sockets. Return 0 on
161  success or -1 on error. */
162 static int
163 insert_cygwin_fd (assuan_fd_t sockfd)
164 {
165  int ret = 0;
166  int mark = -1;
167  int i;
168 
169  EnterCriticalSection (&cygwin_fdtable_cs);
170 
171  for (i=0; i < DIM(cygwin_fdtable); i++)
172  {
173  if (cygwin_fdtable[i] == sockfd)
174  goto leave; /* Already in table. */
175  else if (cygwin_fdtable[i] == ASSUAN_INVALID_FD)
176  mark = i;
177  }
178  if (mark == -1)
179  {
180  gpg_err_set_errno (EMFILE);
181  ret = -1;
182  }
183  else
184  cygwin_fdtable[mark] = sockfd;
185 
186  leave:
187  LeaveCriticalSection (&cygwin_fdtable_cs);
188  return ret;
189 }

335 /* W32: The buffer for NONCE needs to be at least 16 bytes. Returns 0
336  on success and sets errno on failure. If FNAME has a Cygwin socket
337  descriptor True is stored at CYGWIN. */
338 static int
339 read_port_and_nonce (const char *fname, unsigned short *port, char *nonce,
340  int *cygwin)
341 {
342  FILE *fp;
343  char buffer[50], *p;
344  size_t nread;
345  int aval;
346 
347  *cygwin = 0;
348  fp = fopen (fname, "rb");
349  if (!fp)
350  return -1;
351  nread = fread (buffer, 1, sizeof buffer - 1, fp);
352  fclose (fp);
353  if (!nread)
354  {
355  gpg_err_set_errno (ENOENT);
356  return -1;
357  }
358  buffer[nread] = 0;
359  if (!strncmp (buffer, "!<socket >", 10))
360  {
361  /* This is the Cygwin compatible socket emulation. The format
362  * of the file is:
363  *
364  * "!<socket >%u %c %08x-%08x-%08x-%08x\x00"
365  *
366  * %d for port number, %c for kind of socket (s for STREAM), and
367  * we have 16-byte random bytes for nonce. We only support
368  * stream mode.
369  */
370  unsigned int u0;
371  int narr[4];
372 
373  if (sscanf (buffer+10, "%u s %08x-%08x-%08x-%08x",
374  &u0, narr+0, narr+1, narr+2, narr+3) != 5
375  || u0 < 1 || u0 > 65535)
376  {
377  gpg_err_set_errno (EINVAL);
378  return -1;
379  }
380  *port = u0;
381  memcpy (nonce, narr, 16);
382  *cygwin = 1;
383  }
384  else
385  {
386  /* This is our own socket emulation. */
387  aval = atoi (buffer);
388  if (aval < 1 || aval > 65535)
389  {
390  gpg_err_set_errno (EINVAL);
391  return -1;
392  }
393  *port = (unsigned int)aval;
394  for (p=buffer; nread && *p != '\n'; p++, nread--)
395  ;
396  if (*p != '\n' || nread != 17)
397  {
398  gpg_err_set_errno (EINVAL);
399  return -1;
400  }
401  p++; nread--;
402  memcpy (nonce, p, 16);
403  }
404 
405  return 0;
406 }

411 /* Find a redirected socket name for fname and return a malloced setup
412  filled sockaddr. If this does not work out NULL is returned and
413  ERRNO is set. If the file seems to be a redirect True is stored at
414  R_REDIRECT. Note that this function uses the standard malloc and
415  not the assuan wrapped one. The format of the file is:
416 
417  %Assuan%
418  socket=NAME
419 
420  where NAME is the actual socket to use. No white spaces are
421  allowed, both lines must be terminated by a single LF, extra lines
422  are not allowed. Environment variables are interpreted in NAME if
423  given in "${VAR} notation; no escape characters are defined, if
424  "${" shall be used verbatim, you need to use an environment
425  variable with that content.
426 
427  The use of an absolute NAME is strongly suggested. The length of
428  the file is limited to 511 bytes which is more than sufficient for
429  that common value of 107 for sun_path. */
430 static struct sockaddr_un *
431 eval_redirection (const char *fname, int *r_redirect)
432 {
433  FILE *fp;
434  char buffer[512], *name;
435  size_t n;
436  struct sockaddr_un *addr;
437  char *p, *pend;
438  const char *s;
439 
440  *r_redirect = 0;
441 
442  fp = fopen (fname, "rb");
443  if (!fp)
444  return NULL;
445  n = fread (buffer, 1, sizeof buffer - 1, fp);
446  fclose (fp);
447  if (!n)
448  {
449  gpg_err_set_errno (ENOENT);
450  return NULL;
451  }
452  buffer[n] = 0;
453 
454  /* Check that it is a redirection file. We also check that the
455  first byte of the name is not a LF because that would lead to an
456  zero length name. */
457  if (n < 17 || buffer[n-1] != '\n'
458  || memcmp (buffer, "%Assuan%\nsocket=", 16)
459  || buffer[16] == '\n')
460  {
461  gpg_err_set_errno (EINVAL);
462  return NULL;
463  }
464  buffer[n-1] = 0;
465  name = buffer + 16;
466 
467  *r_redirect = 1;
468 
469  addr = calloc (1, sizeof *addr);
470  if (!addr)
471  return NULL;
472  addr->sun_family = AF_LOCAL;
473 
474  n = 0;
475  for (p=name; *p; p++)
476  {
477  if (*p == '$' && p[1] == '{')
478  {
479  p += 2;
480  pend = strchr (p, '}');
481  if (!pend)
482  {
483  free (addr);
484  gpg_err_set_errno (EINVAL);
485  return NULL;
486  }
487  *pend = 0;
488  if (*p && (s = getenv (p)))
489  {
490  for (; *s; s++)
491  {
492  if (n < sizeof addr->sun_path - 1)
493  addr->sun_path[n++] = *s;
494  else
495  {
496  free (addr);
497  gpg_err_set_errno (ENAMETOOLONG);
498  return NULL;
499  }
500  }
501  }
502  p = pend;
503  }
504  else if (*p == '\n')
505  break; /* Be nice and stop at the first LF. */
506  else if (n < sizeof addr->sun_path - 1)
507  addr->sun_path[n++] = *p;
508  else
509  {
510  free (addr);
511  gpg_err_set_errno (ENAMETOOLONG);
512  return NULL;
513  }
514  }
515 
516  return addr;
517 }
518 #endif

633 /* Read NBYTES from SOCKFD into BUFFER. Return 0 on success. Handle
634  EAGAIN and EINTR. */
635 static int
636 do_readn (assuan_context_t ctx, assuan_fd_t sockfd,
637  void *buffer, size_t nbytes)
638 {
639  char *p = buffer;
640  ssize_t n;
641 
642  while (nbytes)
643  {
644  n = _assuan_read (ctx, sockfd, p, nbytes);
645  if (n < 0 && errno == EINTR)
646  ;
647  else if (n < 0 && errno == EAGAIN)
648  _assuan_usleep (ctx, 100000); /* 100ms */
649  else if (n < 0)
650  return -1;
651  else if (!n)
652  {
653  gpg_err_set_errno (EIO);
654  return -1;
655  }
656  else
657  {
658  p += n;
659  nbytes -= n;
660  }
661  }
662  return 0;
663 }

687 /* Connect using the SOCKS5 protocol. */
688 static int
689 socks5_connect (assuan_context_t ctx, assuan_fd_t sock,
690  unsigned short socksport,
691  const char *credentials,
692  const char *hostname, unsigned short hostport,
693  struct sockaddr *addr, socklen_t length)
694 {
695  int ret;
696  /* struct sockaddr_in6 proxyaddr_in6; */
697  struct sockaddr_in proxyaddr_in;
698  struct sockaddr *proxyaddr;
699  size_t proxyaddrlen;
700  struct sockaddr_in6 *addr_in6;
701  struct sockaddr_in *addr_in;
702  unsigned char buffer[22+512]; /* The extra 512 gives enough space
703  for username/password or the
704  hostname. */
705  size_t buflen, hostnamelen;
706  int method;
707 
708  /* memset (&proxyaddr_in6, 0, sizeof proxyaddr_in6); */
709  memset (&proxyaddr_in, 0, sizeof proxyaddr_in);
710 
711  /* Either HOSTNAME or ADDR may be given. */
712  if (hostname && addr)
713  {
714  gpg_err_set_errno (EINVAL);
715  return -1;
716  }
717 
718  /* If a hostname is given it must fit into our buffer and it must be
719  less than 256 so that its length can be encoded in one byte. */
720  hostnamelen = hostname? strlen (hostname) : 0;
721  if (hostnamelen > 255)
722  {
723  gpg_err_set_errno (ENAMETOOLONG);
724  return -1;
725  }
726 
727  /* Connect to local host. */
728  /* Fixme: First try to use IPv6 but note that
729  _assuan_sock_connect_byname created the socket with AF_INET. */
730  proxyaddr_in.sin_family = AF_INET;
731  proxyaddr_in.sin_port = htons (socksport);
732  proxyaddr_in.sin_addr.s_addr = htonl (INADDR_LOOPBACK);
733  proxyaddr = (struct sockaddr *)&proxyaddr_in;
734  proxyaddrlen = sizeof proxyaddr_in;
735  ret = _assuan_connect (ctx, HANDLE2SOCKET (sock), proxyaddr, proxyaddrlen);
736  if (ret && socksport == TOR_PORT && errno == ECONNREFUSED)
737  {
738  /* Standard Tor port failed - try the Tor browser port. */
739  proxyaddr_in.sin_port = htons (TOR_PORT2);
740  ret = _assuan_connect (ctx, HANDLE2SOCKET (sock),
741  proxyaddr, proxyaddrlen);
742  }
743  if (ret)
744  return ret;
745  buffer[0] = 5; /* RFC-1928 VER field. */
746  buffer[1] = 1; /* NMETHODS */
747  if (credentials)
748  method = 2; /* Method: username/password authentication. */
749  else
750  method = 0; /* Method: No authentication required. */
751  buffer[2] = method;
752 
753  /* Negotiate method. */
754  ret = do_writen (ctx, sock, buffer, 3);
755  if (ret)
756  return ret;
757  ret = do_readn (ctx, sock, buffer, 2);
758  if (ret)
759  return ret;
760  if (buffer[0] != 5 || buffer[1] != method )
761  {
762  /* Socks server returned wrong version or does not support our
763  requested method. */
764  gpg_err_set_errno (ENOTSUP); /* Fixme: Is there a better errno? */
765  return -1;
766  }
767 
768  if (credentials)
769  {
770  const char *password;
771  int ulen, plen;
772 
773  password = strchr (credentials, ':');
774  if (!password)
775  {
776  gpg_err_set_errno (EINVAL); /* No password given. */
777  return -1;
778  }
779  ulen = password - credentials;
780  password++;
781  plen = strlen (password);
782  if (!ulen || ulen > 255 || !plen || plen > 255)
783  {
784  gpg_err_set_errno (EINVAL);
785  return -1;
786  }
787 
788  buffer[0] = 1; /* VER of the sub-negotiation. */
789  buffer[1] = ulen;
790  buflen = 2;
791  memcpy (buffer+buflen, credentials, ulen);
792  buflen += ulen;
793  buffer[buflen++] = plen;
794  memcpy (buffer+buflen, password, plen);
795  buflen += plen;
796  ret = do_writen (ctx, sock, buffer, buflen);
797  wipememory (buffer, buflen);
798  if (ret)
799  return ret;
800  ret = do_readn (ctx, sock, buffer, 2);
801  if (ret)
802  return ret;
803  if (buffer[0] != 1)
804  {
805  /* SOCKS server returned wrong version. */
806  gpg_err_set_errno (EPROTO);
807  return -1;
808  }
809  if (buffer[1])
810  {
811  /* SOCKS server denied access. */
812  gpg_err_set_errno (EACCES);
813  return -1;
814  }
815  }
816 
817  if (hostname && !*hostname && !hostport)
818  {
819  /* Empty hostname given. Stop right here to allow the caller to
820  do the actual proxy request. */
821  return 0;
822  }
823 
824  /* Send request details (rfc-1928, 4). */
825  buffer[0] = 5; /* VER */
826  buffer[1] = 1; /* CMD = CONNECT */
827  buffer[2] = 0; /* RSV */
828  if (hostname)
829  {
830  buffer[3] = 3; /* ATYP = DOMAINNAME */
831  buflen = 4;
832  buffer[buflen++] = hostnamelen;
833  memcpy (buffer+buflen, hostname, hostnamelen);
834  buflen += hostnamelen;
835  buffer[buflen++] = (hostport >> 8); /* DST.PORT */
836  buffer[buflen++] = hostport;
837  }
838  else if (addr->sa_family == AF_INET6)
839  {
840  addr_in6 = (struct sockaddr_in6 *)addr;
841 
842  buffer[3] = 4; /* ATYP = IPv6 */
843  memcpy (buffer+ 4, &addr_in6->sin6_addr.s6_addr, 16); /* DST.ADDR */
844  memcpy (buffer+20, &addr_in6->sin6_port, 2); /* DST.PORT */
845  buflen = 22;
846  }
847  else
848  {
849  addr_in = (struct sockaddr_in *)addr;
850 
851  buffer[3] = 1; /* ATYP = IPv4 */
852  memcpy (buffer+4, &addr_in->sin_addr.s_addr, 4); /* DST.ADDR */
853  memcpy (buffer+8, &addr_in->sin_port, 2); /* DST.PORT */
854  buflen = 10;
855  }
856  ret = do_writen (ctx, sock, buffer, buflen);
857  if (ret)
858  return ret;
859  ret = do_readn (ctx, sock, buffer, 10 /* Length for IPv4 */);
860  if (ret)
861  return ret;
862  if (buffer[0] != 5 || buffer[2] != 0 )
863  {
864  /* Socks server returned wrong version or the reserved field is
865  not zero. */
866  gpg_err_set_errno (EPROTO);
867  return -1;
868  }
869  if (buffer[1])
870  {
871  switch (buffer[1])
872  {
873  case 0x01: /* General SOCKS server failure. */
874  gpg_err_set_errno (ENETDOWN);
875  break;
876  case 0x02: /* Connection not allowed by ruleset. */
877  gpg_err_set_errno (EACCES);
878  break;
879  case 0x03: /* Network unreachable */
880  gpg_err_set_errno (ENETUNREACH);
881  break;
882  case 0x04: /* Host unreachable */
883  gpg_err_set_errno (EHOSTUNREACH);
884  break;
885  case 0x05: /* Connection refused */
886  gpg_err_set_errno (ECONNREFUSED);
887  break;
888  case 0x06: /* TTL expired */
889  gpg_err_set_errno (ETIMEDOUT);
890  break;
891  case 0x08: /* Address type not supported */
892  gpg_err_set_errno (EPROTONOSUPPORT);
893  break;
894  case 0x07: /* Command not supported */
895  default:
896  gpg_err_set_errno (ENOTSUP); /* Fixme: Is there a better errno? */
897  }
898  return -1;
899  }
900  if (buffer[3] == 4)
901  {
902  /* ATYP indicates a v6 address. We need to read the remaining
903  12 bytes. */
904  ret = do_readn (ctx, sock, buffer+10, 12);
905  if (ret)
906  return ret;
907  }
908 
909  /* FIXME: We have not way to store the actual address used by the
910  server. */
911 
912 
913  return 0;
914 }

917 /* Return true if SOCKS shall be used. This is the case if tor_mode
918  is enabled and the desired address is not the loopback
919  address. */
920 static int
921 use_socks (struct sockaddr *addr)
922 {
923  if (!tor_mode)
924  return 0;
925  else if (addr->sa_family == AF_INET6)
926  {
927  struct sockaddr_in6 *addr_in6 = (struct sockaddr_in6 *)addr;
928  const unsigned char *s;
929  int i;
930 
931  s = (unsigned char *)&addr_in6->sin6_addr.s6_addr;
932  if (s[15] != 1)
933  return 1; /* Last octet is not 1 - not the loopback address. */
934  for (i=0; i < 15; i++, s++)
935  if (*s)
936  return 1; /* Non-zero octet found - not the loopback address. */
937 
938  return 0; /* This is the loopback address. */
939  }
940  else if (addr->sa_family == AF_INET)
941  {
942  struct sockaddr_in *addr_in = (struct sockaddr_in *)addr;
943 
944  if (*(unsigned char*)&addr_in->sin_addr.s_addr == 127)
945  return 0; /* Loopback (127.0.0.0/8) */
946 
947  return 1;
948  }
949  else
950  return 0;
951 }

955 _assuan_sock_connect (assuan_context_t ctx, assuan_fd_t sockfd,
956  struct sockaddr *addr, int addrlen)
957 {
958 #ifdef HAVE_W32_SYSTEM
959  if (addr->sa_family == AF_LOCAL || addr->sa_family == AF_UNIX)
960  {
961  struct sockaddr_in myaddr;
962  struct sockaddr_un *unaddr;
963  unsigned short port;
964  char nonce[16];
965  int cygwin;
966  int ret;
967

969  if (read_port_and_nonce (unaddr->sun_path, &port, nonce, &cygwin))
970  return -1;
971 
972  myaddr.sin_family = AF_INET;
973  myaddr.sin_port = htons (port);
974  myaddr.sin_addr.s_addr = htonl (INADDR_LOOPBACK);
975 
976  /* Set return values. */
977  unaddr->sun_family = myaddr.sin_family;
978  unaddr->sun_port = myaddr.sin_port;
979  unaddr->sun_addr.s_addr = myaddr.sin_addr.s_addr;
980

982  (struct sockaddr *)&myaddr, sizeof myaddr);
983  if (!ret)
984  {
985  /* Send the nonce. */
986  ret = do_writen (ctx, sockfd, nonce, 16);
987  if (!ret && cygwin)
988  {
989  char buffer[16];
990 
991  /* The client sends the nonce back - not useful. We do
992  a dummy read. */
993  ret = do_readn (ctx, sockfd, buffer, 16);
994  if (!ret)
995  {
996  /* Send our credentials. */
997  int n = getpid ();
998  memcpy (buffer, &n, 4);
999  memset (buffer+4, 0, 4); /* uid = gid = 0 */
1000  ret = do_writen (ctx, sockfd, buffer, 8);
1001  if (!ret)
1002  {
1003  /* Receive credentials. We don't need them. */
1004  ret = do_readn (ctx, sockfd, buffer, 8);
1005  }
1006  }
1007  }
1008  }
1009  return ret;
1010  }
1011  else if (use_socks (addr))
1012  {
1013  return socks5_connect (ctx, sockfd, tor_mode,
1014  NULL, NULL, 0, addr, addrlen);
1015  }
1016  else
1017  {
1018  return _assuan_connect (ctx, HANDLE2SOCKET (sockfd), addr, addrlen);
1019  }
1020 #else
1021 # if HAVE_STAT
1022  if (addr->sa_family == AF_LOCAL || addr->sa_family == AF_UNIX)
1023  {
1024  struct sockaddr_un *unaddr;
1025  struct stat statbuf;
1026  int redirect, res;
1027

1030  && !S_ISSOCK (statbuf.st_mode)
1031  && S_ISREG (statbuf.st_mode))
1032  {

1034  We use the content of that file to redirect to another
1035  socket file. This can be used to use sockets on file
1036  systems which do not support sockets or if for example a
1037  home directory is shared by several machines. */
1038  unaddr = eval_redirection (unaddr->sun_path, &redirect);
1039  if (unaddr)
1040  {
1041  res = _assuan_connect (ctx, sockfd, (struct sockaddr *)unaddr,
1042  SUN_LEN (unaddr));
1043  free (unaddr);
1044  return res;
1045  }
1046  if (redirect)
1047  return -1;
1048  /* Continue using the standard connect. */
1049  }
1050 
1051  }
1052 # endif /*HAVE_STAT*/
1053 
1054  if (use_socks (addr))
1055  {
1056  return socks5_connect (ctx, sockfd, tor_mode,
1057  NULL, NULL, 0, addr, addrlen);
1058  }
1059  else
1060  {
1061  return _assuan_connect (ctx, sockfd, addr, addrlen);
1062  }

1067 /* Connect to HOST specified as host name on PORT. The current
1068  implementation requires that either the flags ASSUAN_SOCK_SOCKS or
1069  ASSUAN_SOCK_TOR are given in FLAGS. On success a new socket is
1070  returned; on error ASSUAN_INVALID_FD is returned and ERRNO set. If
1071  CREDENTIALS is not NULL, it is a string used for password based
1072  authentication. Username and password are separated by a colon.
1073  RESERVED must be 0. By passing HOST and PORT as 0 the function can
1074  be used to check for proxy availability: If the proxy is available
1075  a socket will be returned which the caller should then close. */
1076 assuan_fd_t
1077 _assuan_sock_connect_byname (assuan_context_t ctx, const char *host,
1078  unsigned short port, int reserved,
1079  const char *credentials, unsigned int flags)
1080 {
1081  assuan_fd_t fd;
1082  unsigned short socksport;
1083 
1084  if ((flags & ASSUAN_SOCK_TOR))
1085  socksport = TOR_PORT;
1086  else if ((flags & ASSUAN_SOCK_SOCKS))
1087  socksport = SOCKS_PORT;
1088  else
1089  {
1090  gpg_err_set_errno (ENOTSUP);
1091  return ASSUAN_INVALID_FD;
1092  }
1093 
1094  if (host && !*host)
1095  {
1096  /* Error out early on an empty host name. See below. */
1097  gpg_err_set_errno (EINVAL);
1098  return ASSUAN_INVALID_FD;
1099  }
1100 
1101  fd = _assuan_sock_new (ctx, AF_INET, SOCK_STREAM, 0);
1102  if (fd == ASSUAN_INVALID_FD)
1103  return fd;
1104 
1105  /* For HOST being NULL we pass an empty string which indicates to
1106  socks5_connect to stop midway during the proxy negotiation. Note
1107  that we can't pass NULL directly as this indicates IP address
1108  mode to the called function. */
1109  if (socks5_connect (ctx, fd, socksport,
1110  credentials, host? host:"", port, NULL, 0))
1111  {
1112  int save_errno = errno;
1113  assuan_sock_close (fd);
1114  gpg_err_set_errno (save_errno);
1115  return ASSUAN_INVALID_FD;
1116  }
1117 
1118  return fd;
1119 }

1314 int
1315 _assuan_sock_check_nonce (assuan_context_t ctx, assuan_fd_t fd,
1316  assuan_sock_nonce_t *nonce)
1317 {
1318 #ifdef HAVE_W32_SYSTEM
1319  char buffer[16];
1320  int n;
1321 
1322  if (sizeof nonce->nonce != 16)
1323  {
1324  gpg_err_set_errno (EINVAL);
1325  return -1;
1326  }
1327 
1328  if (nonce->length == 42 && nonce->nonce[0] == 42)
1329  return 0; /* Not a Unix domain socket. */
1330 
1331  if (nonce->length != 16)
1332  {
1333  gpg_err_set_errno (EINVAL);
1334  return -1;
1335  }
1336 
1337  if (do_readn (ctx, fd, buffer, 16))
1338  return -1;
1339  if (memcmp (buffer, nonce->nonce, 16))
1340  {
1341  gpg_err_set_errno (EACCES);
1342  return -1;
1343  }
1344  if (is_cygwin_fd (fd))
1345  {
1346  /* Send the nonce back to the client. */
1347  if (do_writen (ctx, fd, buffer, 16))
1348  return -1;
1349  /* Read the credentials. Cygwin uses the
1350  struct ucred { pid_t pid; uid_t uid; gid_t gid; };
1351  with pid_t being an int (4 bytes) and uid_t and gid_t being
1352  shorts (2 bytes). Thus we need to read 8 bytes. However we
1353  we ignore the values because they are not kernel controlled. */
1354  if (do_readn (ctx, fd, buffer, 8))
1355  return -1;
1356  /* Send our credentials: We use the uid and gid we received but
1357  our own pid. */
1358  n = getpid ();
1359  memcpy (buffer, &n, 4);
1360  if (do_writen (ctx, fd, buffer, 8))
1361  return -1;
1362  }
1363 
1364 #else
1365  (void)fd;
1366  (void)nonce;
1367 #endif
1368  return 0;
1369 }
