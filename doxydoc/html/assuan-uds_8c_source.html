66 #endif

68 #define CMSG_FIRSTHDR(mhdr) \

71 #endif

78 /* Read from a unix domain socket using sendmsg. */
79 static ssize_t
80 uds_reader (assuan_context_t ctx, void *buf, size_t buflen)
81 {
82 #ifndef HAVE_W32_SYSTEM
83  int len = 0;
84  /* This loop should be OK. As FDs are followed by data, the
85  readable status of the socket does not change and no new
86  select/event-loop round is necessary. */
87  while (!len) /* No data is buffered. */
88  {
89  struct msghdr msg;
90  struct iovec iovec;
91 #ifdef USE_DESCRIPTOR_PASSING
92  union {
93  struct cmsghdr cm;
94  char control[CMSG_SPACE(sizeof (int))];
95  } control_u;
96  struct cmsghdr *cmptr;
97 #endif /*USE_DESCRIPTOR_PASSING*/
98 
99  memset (&msg, 0, sizeof (msg));
100 
101  msg.msg_name = NULL;
102  msg.msg_namelen = 0;
103  msg.msg_iov = &iovec;
104  msg.msg_iovlen = 1;
105  iovec.iov_base = buf;
106  iovec.iov_len = buflen;
107 #ifdef USE_DESCRIPTOR_PASSING
108  msg.msg_control = control_u.control;
109  msg.msg_controllen = sizeof (control_u.control);
110 #endif
111 
112  len = _assuan_recvmsg (ctx, ctx->inbound.fd, &msg, 0);
113  if (len < 0)
114  return -1;
115  if (len == 0)
116  return 0;
117 
118 #ifdef USE_DESCRIPTOR_PASSING
119  cmptr = CMSG_FIRSTHDR (&msg);
120  if (cmptr && cmptr->cmsg_len == CMSG_LEN (sizeof(int)))
121  {
122  if (cmptr->cmsg_level != SOL_SOCKET
123  || cmptr->cmsg_type != SCM_RIGHTS)
124  TRACE0 (ctx, ASSUAN_LOG_SYSIO, "uds_reader", ctx,
125  "unexpected ancillary data received");
126  else
127  {
128  int fd;
129 
130  memcpy (&fd, CMSG_DATA (cmptr), sizeof (fd));
131 
132  if (ctx->uds.pendingfdscount >= DIM (ctx->uds.pendingfds))
133  {
134  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "uds_reader", ctx,
135  "too many descriptors pending - "
136  "closing received descriptor %d", fd);
137  _assuan_close (ctx, fd);
138  }
139  else
140  ctx->uds.pendingfds[ctx->uds.pendingfdscount++] = fd;
141  }
142  }
143 #endif /*USE_DESCRIPTOR_PASSING*/
144  }
145 
146  return len;
147 #else /*HAVE_W32_SYSTEM*/
148  int res = recvfrom (HANDLE2SOCKET(ctx->inbound.fd), buf, buflen, 0, NULL, NULL);
149  if (res < 0)
150  gpg_err_set_errno (_assuan_sock_wsa2errno (WSAGetLastError ()));
151  return res;
152 #endif /*HAVE_W32_SYSTEM*/
153 }

157 static ssize_t
158 uds_writer (assuan_context_t ctx, const void *buf, size_t buflen)
159 {
160 #ifndef HAVE_W32_SYSTEM
161  struct msghdr msg;
162  struct iovec iovec;
163  ssize_t len;
164 
165  memset (&msg, 0, sizeof (msg));
166 
167  msg.msg_name = NULL;
168  msg.msg_namelen = 0;
169  msg.msg_iovlen = 1;
170  msg.msg_iov = &iovec;
171  iovec.iov_base = (void*)buf;
172  iovec.iov_len = buflen;
173 
174  len = _assuan_sendmsg (ctx, ctx->outbound.fd, &msg, 0);
175 
176  return len;
177 #else /*HAVE_W32_SYSTEM*/
178  int res = sendto (HANDLE2SOCKET(ctx->outbound.fd), buf, buflen, 0,
179  (struct sockaddr *)&ctx->serveraddr,
180  sizeof (struct sockaddr_in));
181  if (res < 0)
182  gpg_err_set_errno ( _assuan_sock_wsa2errno (WSAGetLastError ()));
183  return res;
184 #endif /*HAVE_W32_SYSTEM*/
185 }

188 static gpg_error_t
189 uds_sendfd (assuan_context_t ctx, assuan_fd_t fd)
190 {
191 #ifdef USE_DESCRIPTOR_PASSING
192  struct msghdr msg;
193  struct iovec iovec;
194  union {
195  struct cmsghdr cm;
196  char control[CMSG_SPACE(sizeof (int))];
197  } control_u;
198  struct cmsghdr *cmptr;
199  int len;
200  char buffer[80];
201 
202  /* We need to send some real data so that a read won't return 0
203  which will be taken as an EOF. It also helps with debugging. */
204  snprintf (buffer, sizeof(buffer)-1, "# descriptor %d is in flight\n", fd);
205  buffer[sizeof(buffer)-1] = 0;
206 
207  memset (&msg, 0, sizeof (msg));
208 
209  memset (&control_u, 0, sizeof (control_u));
210 
211  msg.msg_name = NULL;
212  msg.msg_namelen = 0;
213  msg.msg_iovlen = 1;
214  msg.msg_iov = &iovec;
215  iovec.iov_base = buffer;
216  iovec.iov_len = strlen (buffer);
217 
218  msg.msg_control = control_u.control;
219  msg.msg_controllen = sizeof (control_u.control);
220  cmptr = CMSG_FIRSTHDR (&msg);
221  cmptr->cmsg_len = CMSG_LEN(sizeof(int));
222  cmptr->cmsg_level = SOL_SOCKET;
223  cmptr->cmsg_type = SCM_RIGHTS;
224 
225  memcpy (CMSG_DATA (cmptr), &fd, sizeof (fd));
226 
227  len = _assuan_sendmsg (ctx, ctx->outbound.fd, &msg, 0);
228  if (len < 0)
229  {
230  int saved_errno = errno;
231  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "uds_sendfd", ctx,
232  "uds_sendfd: %s", strerror (errno));
233  errno = saved_errno;
234  return _assuan_error (ctx, gpg_err_code_from_syserror ());
235  }
236  else
237  return 0;
238 #else
239  return _assuan_error (ctx, GPG_ERR_NOT_IMPLEMENTED);
240 #endif
241 }

244 static gpg_error_t
245 uds_receivefd (assuan_context_t ctx, assuan_fd_t *fd)
246 {
247 #ifdef USE_DESCRIPTOR_PASSING
248  int i;
249 
250  if (!ctx->uds.pendingfdscount)
251  {
252  TRACE0 (ctx, ASSUAN_LOG_SYSIO, "uds_receivefd", ctx,
253  "no pending file descriptors");
254  return _assuan_error (ctx, GPG_ERR_ASS_GENERAL);
255  }
256  assert (ctx->uds.pendingfdscount <= DIM(ctx->uds.pendingfds));
257 
258  *fd = ctx->uds.pendingfds[0];
259  for (i=1; i < ctx->uds.pendingfdscount; i++)
260  ctx->uds.pendingfds[i-1] = ctx->uds.pendingfds[i];
261  ctx->uds.pendingfdscount--;
262 
263  return 0;
264 #else
265  return _assuan_error (ctx, GPG_ERR_NOT_IMPLEMENTED);
266 #endif
267 }
