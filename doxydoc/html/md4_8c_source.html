187 /* The routine final terminates the message-digest computation and
188  * ends with the desired message digest in mdContext->digest[0...15].
189  * The handle is prepared for a new MD4 cycle.
190  * Returns 16 bytes representing the digest.
191  */

193 static void
194 md4_final( void *context )
195 {
196  MD4_CONTEXT *hd = context;
197  u32 t, th, msb, lsb;
198  byte *p;
199  unsigned int burn;
200 
201  _gcry_md_block_write(hd, NULL, 0); /* flush */;
202 
203  t = hd->bctx.nblocks;
204  if (sizeof t == sizeof hd->bctx.nblocks)
205  th = hd->bctx.nblocks_high;
206  else
207  th = hd->bctx.nblocks >> 32;
208 
209  /* multiply by 64 to make a byte count */
210  lsb = t << 6;
211  msb = (th << 6) | (t >> 26);
212  /* add the count */
213  t = lsb;
214  if( (lsb += hd->bctx.count) < t )
215  msb++;
216  /* multiply by 8 to make a bit count */
217  t = lsb;
218  lsb <<= 3;
219  msb <<= 3;
220  msb |= t >> 29;
221 
222  if( hd->bctx.count < 56 ) /* enough room */
223  {
224  hd->bctx.buf[hd->bctx.count++] = 0x80; /* pad */
225  while( hd->bctx.count < 56 )
226  hd->bctx.buf[hd->bctx.count++] = 0; /* pad */
227  }
228  else /* need one extra block */
229  {
230  hd->bctx.buf[hd->bctx.count++] = 0x80; /* pad character */
231  while( hd->bctx.count < 64 )
232  hd->bctx.buf[hd->bctx.count++] = 0;
233  _gcry_md_block_write(hd, NULL, 0); /* flush */;
234  memset(hd->bctx.buf, 0, 56 ); /* fill next block with zeroes */
235  }
236  /* append the 64 bit count */
237  buf_put_le32(hd->bctx.buf + 56, lsb);
238  buf_put_le32(hd->bctx.buf + 60, msb);
239  burn = transform( hd, hd->bctx.buf );
240  _gcry_burn_stack (burn);
241 
242  p = hd->bctx.buf;
243 #define X(a) do { *(u32*)p = le_bswap32((*hd).a) ; p += 4; } while(0)
244  X(A);
245  X(B);
246  X(C);
247  X(D);
248 #undef X
249 
250 }
