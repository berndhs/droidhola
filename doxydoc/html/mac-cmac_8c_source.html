31 static int
32 map_mac_algo_to_cipher (int mac_algo)
33 {
34  switch (mac_algo)
35  {
36  default:
37  return GCRY_CIPHER_NONE;
38  case GCRY_MAC_CMAC_AES:
39  return GCRY_CIPHER_AES;
40  case GCRY_MAC_CMAC_3DES:
41  return GCRY_CIPHER_3DES;
42  case GCRY_MAC_CMAC_CAMELLIA:
43  return GCRY_CIPHER_CAMELLIA128;
44  case GCRY_MAC_CMAC_IDEA:
45  return GCRY_CIPHER_IDEA;
46  case GCRY_MAC_CMAC_CAST5:
47  return GCRY_CIPHER_CAST5;
48  case GCRY_MAC_CMAC_BLOWFISH:

64 static gcry_err_code_t
65 cmac_open (gcry_mac_hd_t h)
66 {
67  gcry_err_code_t err;
68  gcry_cipher_hd_t hd;
69  int secure = (h->magic == CTX_MAGIC_SECURE);
70  int cipher_algo;
71  unsigned int flags;
72 
73  cipher_algo = map_mac_algo_to_cipher (h->spec->algo);
74  flags = (secure ? GCRY_CIPHER_SECURE : 0);
75 
76  err = _gcry_cipher_open_internal (&hd, cipher_algo, GCRY_CIPHER_MODE_CMAC,
77  flags);
78  if (err)
79  return err;
80 
81  h->u.cmac.cipher_algo = cipher_algo;
82  h->u.cmac.ctx = hd;
83  h->u.cmac.blklen = _gcry_cipher_get_algo_blklen (cipher_algo);
84  return 0;
85 }
