94 run_test (void)
95 {

98 
99  if (!CreatePipe (&hd[0], &hd[1], NULL, 0))

102  return;

104  log_info ("pipe created read=%p write=%p\n", hd[0], hd[1]);
105 
106  threads[0] = CreateThread (NULL, 0, reader_thread, hd[0], 0, NULL);
107  if (!threads[0])
108  log_fatal ("error creating reader thread: rc=%d\n", (int)GetLastError ());
109  else
110  log_info ("reader thread created\n");
111  threads[1] = CreateThread (NULL, 0, writer_thread, hd[1], 0, NULL);
112  if (!threads[0])
113  log_fatal ("error creating writer thread: rc=%d\n", (int)GetLastError ());
114  else
115  log_info ("writer thread created\n");
116 
117  switch (WaitForMultipleObjects (2, threads, FALSE, INFINITE))
118  {
119  case WAIT_OBJECT_0:
120  log_info ("reader thread finished first\n");
121  break;
122  case WAIT_OBJECT_0 + 1:
123  log_info ("writer thread finished first\n");
124  break;
125  default:
126  log_error ("WFMO failed: rc=%d\n", (int)GetLastError ());
127  break;
128  }
129

131

133  CloseHandle (threads[1]);
134 }
