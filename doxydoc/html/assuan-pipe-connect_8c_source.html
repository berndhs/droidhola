68 /* This should be called to make sure that SIGPIPE gets ignored. */
69 static void
70 fix_signals (void)
71 {
72 #ifndef HAVE_DOSISH_SYSTEM /* No SIGPIPE for these systems. */
73  static int fixed_signals;
74 
75  if (!fixed_signals)
76  {
77  struct sigaction act;
78 
79  sigaction (SIGPIPE, NULL, &act);
80  if (act.sa_handler == SIG_DFL)
81  {
82  act.sa_handler = SIG_IGN;
83  sigemptyset (&act.sa_mask);
84  act.sa_flags = 0;
85  sigaction (SIGPIPE, &act, NULL);
86  }
87  fixed_signals = 1;
88  /* FIXME: This is not MT safe */
89  }
90 #endif /*HAVE_DOSISH_SYSTEM*/
91 }

97 {

152 static gpg_error_t
153 pipe_connect (assuan_context_t ctx,
154  const char *name, const char **argv,
155  assuan_fd_t *fd_child_list,
156  void (*atfork) (void *opaque, int reserved),
157  void *atforkvalue, unsigned int flags)
158 {
159  gpg_error_t rc;
160  assuan_fd_t rp[2];
161  assuan_fd_t wp[2];
162  pid_t pid;
163  int res;
164  struct at_pipe_fork atp;
165  unsigned int spawn_flags;
166 
167  atp.user_atfork = atfork;
168  atp.user_atforkvalue = atforkvalue;
169  atp.parent_pid = getpid ();
170 
171  if (!ctx || !name || !argv || !argv[0])
172  return _assuan_error (ctx, GPG_ERR_ASS_INV_VALUE);
173 
174  if (! ctx->flags.no_fixsignals)
175  fix_signals ();
176 
177  if (_assuan_pipe (ctx, rp, 1) < 0)
178  return _assuan_error (ctx, gpg_err_code_from_syserror ());
179 
180  if (_assuan_pipe (ctx, wp, 0) < 0)
181  {
182  _assuan_close (ctx, rp[0]);
183  _assuan_close_inheritable (ctx, rp[1]);
184  return _assuan_error (ctx, gpg_err_code_from_syserror ());
185  }
186 
187  spawn_flags = 0;
188  if (flags & ASSUAN_PIPE_CONNECT_DETACHED)
189  spawn_flags |= ASSUAN_SPAWN_DETACHED;
190 
191  /* FIXME: Use atfork handler that closes child fds on Unix. */
192  res = _assuan_spawn (ctx, &pid, name, argv, wp[0], rp[1],
193  fd_child_list, at_pipe_fork_cb, &atp, spawn_flags);
194  if (res < 0)
195  {
196  rc = gpg_err_code_from_syserror ();
197  _assuan_close (ctx, rp[0]);
198  _assuan_close_inheritable (ctx, rp[1]);
199  _assuan_close_inheritable (ctx, wp[0]);
200  _assuan_close (ctx, wp[1]);
201  return _assuan_error (ctx, rc);
202  }
203 
204  /* Close the stdin/stdout child fds in the parent. */
205  _assuan_close_inheritable (ctx, rp[1]);
206  _assuan_close_inheritable (ctx, wp[0]);
207 
208  ctx->engine.release = _assuan_client_release;
209  ctx->engine.readfnc = _assuan_simple_read;
210  ctx->engine.writefnc = _assuan_simple_write;
211  ctx->engine.sendfd = NULL;
212  ctx->engine.receivefd = NULL;
213  ctx->finish_handler = _assuan_client_finish;
214  ctx->max_accepts = 1;
215  ctx->accept_handler = NULL;
216  ctx->inbound.fd = rp[0]; /* Our inbound is read end of read pipe. */
217  ctx->outbound.fd = wp[1]; /* Our outbound is write end of write pipe. */
218  ctx->pid = pid;
219 
220  rc = initial_handshake (ctx);
221  if (rc)
222  _assuan_reset (ctx);
223  return rc;
224 }

271  sets the I/O up to use sendmsg/recvmsg. */
272 static gpg_error_t
273 socketpair_connect (assuan_context_t ctx,
274  const char *name, const char **argv,
275  assuan_fd_t *fd_child_list,
276  void (*atfork) (void *opaque, int reserved),
277  void *atforkvalue)
278 {
279  gpg_error_t err;
280  int idx;
281  int fds[2];
282  char mypidstr[50];
283  pid_t pid;
284  int *child_fds = NULL;
285  int child_fds_cnt = 0;
286  struct at_socketpair_fork atp;
287  int rc;
288 
289  TRACE_BEG3 (ctx, ASSUAN_LOG_CTX, "socketpair_connect", ctx,
290  "name=%s,atfork=%p,atforkvalue=%p", name ? name : "(null)",
291  atfork, atforkvalue);
292 
293  atp.user_atfork = atfork;
294  atp.user_atforkvalue = atforkvalue;
295  atp.parent_pid = getpid ();
296 
297  if (!ctx
298  || (name && (!argv || !argv[0]))
299  || (!name && !argv))
300  return _assuan_error (ctx, GPG_ERR_ASS_INV_VALUE);
301 
302  if (! ctx->flags.no_fixsignals)
303  fix_signals ();
304 
305  sprintf (mypidstr, "%lu", (unsigned long)getpid ());
306 
307  if (fd_child_list)
308  while (fd_child_list[child_fds_cnt] != ASSUAN_INVALID_FD)
309  child_fds_cnt++;
310  child_fds = _assuan_malloc (ctx, (child_fds_cnt + 2) * sizeof (int));
311  if (! child_fds)
312  return TRACE_ERR (gpg_err_code_from_syserror ());
313  child_fds[1] = ASSUAN_INVALID_FD;
314  if (fd_child_list)
315  memcpy (&child_fds[1], fd_child_list, (child_fds_cnt + 1) * sizeof (int));
316 
317  if (_assuan_socketpair (ctx, AF_LOCAL, SOCK_STREAM, 0, fds))
318  {
319  TRACE_LOG1 ("socketpair failed: %s", strerror (errno));
320  _assuan_free (ctx, child_fds);
321  return TRACE_ERR (GPG_ERR_ASS_GENERAL);
322  }
323  atp.peer_fd = fds[1];
324  child_fds[0] = fds[1];
325 
326  rc = _assuan_spawn (ctx, &pid, name, argv, ASSUAN_INVALID_FD,
327  ASSUAN_INVALID_FD, child_fds, at_socketpair_fork_cb,
328  &atp, 0);
329  if (rc < 0)
330  {
331  err = gpg_err_code_from_syserror ();
332  _assuan_close (ctx, fds[0]);
333  _assuan_close (ctx, fds[1]);
334  _assuan_free (ctx, child_fds);
335  return TRACE_ERR (err);
336  }

338  /* For W32, the user needs to know the server-local names of the
339  inherited handles. Return them here. Note that the translation
340  of the peer socketpair fd (fd_child_list[0]) must be done by the
341  wrapper program based on the environment variable
342  _assuan_connection_fd. */
343  if (fd_child_list)
344  {
345  for (idx = 0; fd_child_list[idx] != -1; idx++)
346  /* We add 1 to skip over the socketpair end. */
347  fd_child_list[idx] = child_fds[idx + 1];
348  }
349 
350  _assuan_free (ctx, child_fds);
351 
352  /* If this is the server child process, exit early. */
353  if (! name && (*argv)[0] == 's')
354  {
355  _assuan_close (ctx, fds[0]);
356  return 0;
357  }
358 
359  _assuan_close (ctx, fds[1]);
360 
361  ctx->engine.release = _assuan_client_release;
362  ctx->finish_handler = _assuan_client_finish;
363  ctx->max_accepts = 1;
364  ctx->inbound.fd = fds[0];
365  ctx->outbound.fd = fds[0];
366  _assuan_init_uds_io (ctx);
367 
368  err = initial_handshake (ctx);
369  if (err)
370  _assuan_reset (ctx);
371  return err;

376 /* Connect to a server over a full-duplex socket (i.e. created by
377  socketpair), creating the assuan context and returning it in CTX.
378  The server filename is NAME, the argument vector in ARGV.
379  FD_CHILD_LIST is a -1 terminated list of file descriptors not to
380  close in the child. ATFORK is called in the child right after the
381  fork; ATFORKVALUE is passed as the first argument and 0 is passed
382  as the second argument. The ATFORK function should only act if the
383  second value is 0.
384 
385  FLAGS is a bit vector and controls how the function acts:
386  Bit 0: If cleared a simple pipe based server is expected and the
387  function behaves similar to `assuan_pipe_connect'.
388 
389  If set a server based on full-duplex pipes is expected. Such
390  pipes are usually created using the `socketpair' function.
391  It also enables features only available with such servers.
392 
393  Bit 7: If set and there is a need to start the server it will be
394  started as a background process. This flag is useful under
395  W32 systems, so that no new console is created and pops up a
396  console window when starting the server
397 
398 
399  If NAME is NULL, no exec is done but the same process is continued.
400  However all file descriptors are closed and some special
401  environment variables are set. To let the caller detect whether the
402  child or the parent continues, the child returns "client" or
403  "server" in *ARGV (but it is sufficient to check only the first
404  character). This feature is only available on POSIX platforms. */
405 gpg_error_t
406 assuan_pipe_connect (assuan_context_t ctx,
407  const char *name, const char *argv[],
408  assuan_fd_t *fd_child_list,
409  void (*atfork) (void *opaque, int reserved),
410  void *atforkvalue, unsigned int flags)
411 {
412  TRACE2 (ctx, ASSUAN_LOG_CTX, "assuan_pipe_connect", ctx,
413  "name=%s, flags=0x%x", name ? name : "(null)", flags);
414 
415  if (flags & ASSUAN_PIPE_CONNECT_FDPASSING)
416  {
417 #ifdef HAVE_W32_SYSTEM
418  return _assuan_error (ctx, GPG_ERR_NOT_IMPLEMENTED);
419 #else
420  return socketpair_connect (ctx, name, argv, fd_child_list,
421  atfork, atforkvalue);
422 #endif
423  }
424  else
425  return pipe_connect (ctx, name, argv, fd_child_list, atfork, atforkvalue,
426  flags);
427 }
