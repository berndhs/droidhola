135 void spoton_logviewer::slotTimeout(void)
136 {
137  QFileInfo fileInfo(spoton_misc::homePath() + QDir::separator() +
138  "error_log.dat");
139 
140  if(fileInfo.exists())
141  {
142  if(fileInfo.lastModified() >= m_lastModificationTime)
143  {
144  if(fileInfo.lastModified() == m_lastModificationTime)
145  m_lastModificationTime = fileInfo.lastModified().addMSecs(1);
146  else
147  m_lastModificationTime = fileInfo.lastModified();
148  }
149  else
150  return;
151  }
152  else
153  m_lastModificationTime = QDateTime();
154 
155  QFile file(fileInfo.absoluteFilePath());
156 
157  if(file.open(QIODevice::ReadOnly))
158  {
159  int vValue = ui.log->verticalScrollBar()->value();
160 
161  if(file.seek(qMax(static_cast<qint64> (0),
162  file.size() - 256 * 1024)))
163  {
164  ui.log->setPlainText(file.read(256 * 1024).trimmed());
165  ui.log->verticalScrollBar()->setValue(vValue);
166  }
167 
168  file.close();
169  }
170 }
