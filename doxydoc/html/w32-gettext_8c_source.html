1072 /* The so called `hashpjw' function by P.J. Weinberger
1073  [see Aho/Sethi/Ullman, COMPILERS: Principles, Techniques and Tools,
1074  1986, 1987 Bell Telephone Laboratories, Inc.] */
1075 static GPG_ERR_INLINE unsigned long
1076 hash_string (const char *str_param)
1077 {
1078  unsigned long int hval, g;
1079  const char *str = str_param;
1080 
1081  hval = 0;
1082  while (*str != '\0')
1083  {
1084  hval <<= 4;
1085  hval += (unsigned long int) *str++;
1086  g = hval & ((unsigned long int) 0xf << (HASHWORDBITS - 4));
1087  if (g != 0)
1088  {
1089  hval ^= g >> (HASHWORDBITS - 8);
1090  hval ^= g;
1091  }
1092  }
1093  return hval;
1094 }

1595 static const char*
1596 get_string (struct loaded_domain *domain, uint32_t idx,
1597  int use_plural, unsigned long nplural)
1598 {
1599  struct tls_space_s *tls = get_tls ();
1600  struct overflow_space_s *os;
1601  const char *trans; /* Pointer to the translated entry. */
1602  size_t translen; /* Length of that entry. */
1603 
1604  if (idx > 65534)
1605  return "ERROR in GETTEXT (too many strings)";
1606 
1607  if (tls->gt_use_utf8)
1608  {
1609  trans = (domain->data
1610  + SWAPIT(domain->must_swap, domain->trans_tab[idx].offset));
1611  translen = SWAPIT(domain->must_swap, domain->trans_tab[idx].length);
1612  }
1613  else if (!domain->mapped[idx])
1614  {
1615  /* Not yet mapped. Map from utf-8 to native encoding now. */
1616  const char *p_utf8;
1617  size_t plen_utf8, buflen;
1618  char *buf;
1619 
1620  p_utf8 = (domain->data
1621  + SWAPIT(domain->must_swap, domain->trans_tab[idx].offset));
1622  plen_utf8 = SWAPIT(domain->must_swap, domain->trans_tab[idx].length);
1623 
1624  /* We need to include the nul, so that the utf8->wchar->native
1625  conversion chain works correctly and the nul is stored after
1626  the conversion. */
1627  if (p_utf8[plen_utf8])
1628  {
1629  trans = "ERROR in MO file"; /* Terminating zero is missing. */
1630  translen = 0;
1631  goto leave;
1632  }
1633  plen_utf8++;
1634 
1635  buf = utf8_to_native (p_utf8, plen_utf8, &buflen);
1636  if (!buf)
1637  {
1638  trans = "ERROR in GETTEXT MALLOC";
1639  translen = 0;
1640  }
1641  else if (buflen <= plen_utf8 && buflen > 1)
1642  {
1643  /* Copy into the DATA_NATIVE area. */
1644  char *p_tmp;
1645 
1646  p_tmp = (domain->data_native
1647  + SWAPIT(domain->must_swap, domain->trans_tab[idx].offset));
1648  memcpy (p_tmp, buf, buflen);
1649  domain->mapped[idx] = buflen;
1650  trans = p_tmp;
1651  translen = buflen;
1652  }
1653  else
1654  {
1655  /* There is not enough space for the translation (or for
1656  whatever reason an empty string is used): Store it in the
1657  overflow_space and mark that in the mapped array.
1658  Because UTF-8 strings are in general longer than the
1659  Windows native encoding, we expect that this won't happen
1660  too often and thus we use a linked list to manage this
1661  space. */
1662  os = jnlib_malloc (sizeof *os + buflen);
1663  if (os)
1664  {
1665  os->idx = idx;
1666  memcpy (os->d, buf, buflen);
1667  os->length = buflen;
1668  os->next = domain->overflow_space;
1669  domain->overflow_space = os;
1670  domain->mapped[idx] = 1;
1671  trans = os->d;
1672  translen = os->length;
1673  }
1674  else
1675  {
1676  trans = "ERROR in GETTEXT MALLOC";
1677  translen = 0;
1678  }
1679  }
1680  if (translen)
1681  translen--; /* TRANSLEN shall be the size without the nul. */
1682  jnlib_free (buf);
1683  }
1684  else if (domain->mapped[idx] == 1)
1685  {
1686  /* The translated string is in the overflow_space. */
1687  for (os=domain->overflow_space; os; os = os->next)
1688  if (os->idx == idx)
1689  break;
1690  if (os)
1691  {
1692  trans = os->d;
1693  translen = os->length;
1694  }
1695  else
1696  {
1697  trans = "ERROR in GETTEXT (overflow space)\n";
1698  translen = 0;
1699  }
1700  }
1701  else
1702  {
1703  trans = (domain->data_native
1704  + SWAPIT(domain->must_swap, domain->trans_tab[idx].offset));
1705  translen = domain->mapped[idx];
1706  }
1707 
1708  leave:
1709  if (use_plural && translen)
1710  return get_plural (trans, translen, nplural);
1711  else
1712  return trans;
1713 }

1717 do_gettext (const char *domainname,
1718  const char *msgid, const char *msgid2, unsigned long nplural)
1719 {

1723  uint32_t top, bottom, nstr;
1724  char *filename;
1725 
1726  if (!domainname)
1727  domainname = current_domainname? current_domainname : "";
1728 
1729  /* FIXME: The whole locking stuff is a bit questionable because
1730  gettext does not claim to be thread-safe. We need to investigate
1731  this further. */
1732 
1733  load_failed = 0;
1734  domain = NULL;
1735  filename = NULL;
1736  EnterCriticalSection (&domainlist_access_cs);
1737  {
1738  for (dl = domainlist; dl; dl = dl->next)
1739  if (!strcmp (dl->name, domainname))
1740  {
1741  load_failed = dl->load_failed;
1742  domain = dl->domain;
1743  break;
1744  }
1745  if (dl && !domain && !load_failed && dl->fname)
1746  {
1747  filename = jnlib_malloc (strlen (dl->fname) + 1);
1748  if (filename)
1749  strcpy (filename, dl->fname);
1750  }
1751  }
1752  LeaveCriticalSection (&domainlist_access_cs);
1753  if (!dl)
1754  goto not_found; /* DOMAINNAME not bound. */
1755  if (filename)
1756  {
1757  /* No attempt so far to load the MO file. Try now. */
1758  int updated = 0;
1759 
1760  domain = load_domain (filename);
1761  jnlib_free (filename);
1762  filename = NULL;
1763  EnterCriticalSection (&domainlist_access_cs);
1764  {
1765  for (dl = domainlist; dl; dl = dl->next)
1766  if (!strcmp (dl->name, domainname))
1767  {
1768  if (domain)
1769  dl->domain = domain;
1770  else
1771  dl->load_failed = 1;
1772  updated = 1;
1773  break;
1774  }
1775  }
1776  LeaveCriticalSection (&domainlist_access_cs);
1777  if (!updated)
1778  {
1779  /* Ooops - lost the domain. */
1780  free_domain (domain);
1781  domain = NULL;
1782  }
1783  }
1784 
1785  if (!domain)
1786  goto not_found; /* No MO file. */
1787 
1788  /* First try to use the hash table. */
1789  if (domain->hash_size > 2 && domain->hash_tab)
1790  {
1791  /* Use the hashing table. */
1792  uint32_t len = strlen (msgid);
1793  uint32_t hash_val = hash_string (msgid);
1794  uint32_t idx = hash_val % domain->hash_size;
1795  uint32_t incr = 1 + (hash_val % (domain->hash_size - 2));
1796 
1797  while ( (nstr = SWAPIT (domain->must_swap, domain->hash_tab[idx])) )
1798  {
1799  nstr--;
1800  if (nstr < domain->nstrings
1801  && SWAPIT(domain->must_swap,
1802  domain->orig_tab[nstr].length) >= len
1803  && !strcmp (msgid, (domain->data
1804  + SWAPIT(domain->must_swap,
1805  domain->orig_tab[nstr].offset))))
1806  {
1807  return get_string (domain, nstr, !!msgid2, nplural);
1808  }
1809 
1810  if (idx >= domain->hash_size - incr)
1811  idx -= domain->hash_size - incr;
1812  else
1813  idx += incr;
1814  }
1815  }
1816 
1817  /* Now we try the default method: binary search in the sorted array
1818  of messages. */
1819  bottom = 0;
1820  top = domain->nstrings;
1821  while (bottom < top)
1822  {
1823  int cmp_val;
1824 
1825  nstr = (bottom + top) / 2;
1826  cmp_val = strcmp (msgid, (domain->data
1827  + SWAPIT(domain->must_swap,
1828  domain->orig_tab[nstr].offset)));
1829  if (cmp_val < 0)
1830  top = nstr;
1831  else if (cmp_val > 0)
1832  bottom = nstr + 1;
1833  else
1834  {
1835  return get_string (domain, nstr, !!msgid2, nplural);
1836  }
1837  }
1838 
1839  not_found:
1840  /* We use the standard Germanic rule if plural has been requested. */
1841  return msgid2? (nplural == 1? msgid : msgid2) : msgid;
1842 }
