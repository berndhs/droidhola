153 /* The structure containing information on random-data sources. Each
154  * record contains the source and a relative estimate of its usefulness
155  * (weighting) which is used to scale the number of kB of output from the
156  * source (total = data_bytes / usefulness). Usually the weighting is in the
157  * range 1-3 (or 0 for especially useless sources), resulting in a usefulness
158  * rating of 1...3 for each kB of source output (or 0 for the useless
159  * sources).
160  *
161  * If the source is constantly changing (certain types of network statistics
162  * have this characteristic) but the amount of output is small, the weighting
163  * is given as a negative value to indicate that the output should be treated
164  * as if a minimum of 1K of output had been obtained. If the source produces
165  * a lot of output then the scale factor is fractional, resulting in a
166  * usefulness rating of < 1 for each kB of source output.
167  *
168  * In order to provide enough randomness to satisfy the requirements for a
169  * slow poll, we need to accumulate at least 20 points of usefulness (a
170  * typical system should get about 30 points).
171  *
172  * Some potential options are missed out because of special considerations.
173  * pstat -i and pstat -f can produce amazing amounts of output (the record
174  * is 600K on an Oracle server) which floods the buffer and doesn't yield
175  * anything useful (apart from perhaps increasing the entropy of the vmstat
176  * output a bit), so we don't bother with this. pstat in general produces
177  * quite a bit of output, but it doesn't change much over time, so it gets
178  * very low weightings. netstat -s produces constantly-changing output but
179  * also produces quite a bit of it, so it only gets a weighting of 2 rather
180  * than 3. The same holds for netstat -in, which gets 1 rather than 2.
181  *
182  * Some binaries are stored in different locations on different systems so
183  * alternative paths are given for them. The code sorts out which one to
184  * run by itself, once it finds an exectable somewhere it moves on to the
185  * next source. The sources are arranged roughly in their order of
186  * usefulness, occasionally sources which provide a tiny amount of
187  * relatively useless data are placed ahead of ones which provide a large
188  * amount of possibly useful data because another 100 bytes can't hurt, and
189  * it means the buffer won't be swamped by one or two high-output sources.
190  * All the high-output sources are clustered towards the end of the list
191  * for this reason. Some binaries are checked for in a certain order, for
192  * example under Slowaris /usr/ucb/ps understands aux as an arg, but the
193  * others don't. Some systems have conditional defines enabling alternatives
194  * to commands which don't understand the usual options but will provide
195  * enough output (in the form of error messages) to look like they're the
196  * real thing, causing alternative options to be skipped (we can't check the
197  * return either because some commands return peculiar, non-zero status even
198  * when they're working correctly).
199  *
200  * In order to maximise use of the buffer, the code performs a form of run-
201  * length compression on its input where a repeated sequence of bytes is
202  * replaced by the occurrence count mod 256. Some commands output an awful
203  * lot of whitespace, this measure greatly increases the amount of data we
204  * can fit in the buffer.
205  *
206  * When we scale the weighting using the SC() macro, some preprocessors may
207  * give a division by zero warning for the most obvious expression
208  * 'weight ? 1024 / weight : 0' (and gcc 2.7.2.2 dies with a division by zero
209  * trap), so we define a value SC_0 which evaluates to zero when fed to
210  * '1024 / SC_0' */
211 
212 #define SC( weight ) ( 1024 / weight ) /* Scale factor */
213 #define SC_0 16384 /* SC( SC_0 ) evaluates to 0 */

385 /* Under SunOS popen() doesn't record the pid of the child process. When
386  * pclose() is called, instead of calling waitpid() for the correct child, it
387  * calls wait() repeatedly until the right child is reaped. The problem is
388  * that this reaps any other children that happen to have died at that
389  * moment, and when their pclose() comes along, the process hangs forever.
390  * The fix is to use a wrapper for popen()/pclose() which saves the pid in
391  * the dataSources structure (code adapted from GNU-libc's popen() call).
392  *
393  * Aut viam inveniam aut faciam */
394 
395 static FILE *
396 my_popen(struct RI *entry)
397 {
398  int pipedes[2];
399  FILE *stream;
400 
401  /* Create the pipe */
402  if (pipe(pipedes) < 0)
403  return (NULL);
404 
405  /* Fork off the child ("vfork() is like an OS orgasm. All OS's want to
406  * do it, but most just end up faking it" - Chris Wedgwood). If your OS
407  * supports it, you should try to use vfork() here because it's somewhat
408  * more efficient */
409 #if defined( sun ) || defined( __ultrix__ ) || defined( __osf__ ) || \
410  defined(__hpux)
411  entry->pid = vfork();
412 #else /* */
413  entry->pid = fork();
414 #endif /* Unixen which have vfork() */
415  if (entry->pid == (pid_t) - 1) {
416  /* The fork failed */
417  close(pipedes[0]);
418  close(pipedes[1]);
419  return (NULL);
420  }
421 
422  if (entry->pid == (pid_t) 0) {
423  struct passwd *passwd;
424 
425  /* We are the child. Make the read side of the pipe be stdout */
426  if (dup2(pipedes[STDOUT_FILENO], STDOUT_FILENO) < 0)
427  exit(127);
428 
429  /* Now that everything is set up, give up our permissions to make
430  * sure we don't read anything sensitive. If the getpwnam() fails,
431  * we default to -1, which is usually nobody */
432  if (gatherer_uid == (uid_t)-1 && \
433  (passwd = getpwnam("nobody")) != NULL)
434  gatherer_uid = passwd->pw_uid;
435 
436  setuid(gatherer_uid);
437 
438  /* Close the pipe descriptors */
439  close(pipedes[STDIN_FILENO]);
440  close(pipedes[STDOUT_FILENO]);
441 
442  /* Try and exec the program */
443  execl(entry->path, entry->path, entry->arg, NULL);
444 
445  /* Die if the exec failed */
446  exit(127);
447  }
448 
449  /* We are the parent. Close the irrelevant side of the pipe and open
450  * the relevant side as a new stream. Mark our side of the pipe to
451  * close on exec, so new children won't see it */
452  close(pipedes[STDOUT_FILENO]);
453 
454 #ifdef FD_CLOEXEC
455  fcntl(pipedes[STDIN_FILENO], F_SETFD, FD_CLOEXEC);
456 #endif
457 
458  stream = fdopen(pipedes[STDIN_FILENO], "r");
459 
460  if (stream == NULL) {
461  int savedErrno = errno;
462 
463  /* The stream couldn't be opened or the child structure couldn't be
464  * allocated. Kill the child and close the other side of the pipe */
465  kill(entry->pid, SIGKILL);
466  if (stream == NULL)
467  close(pipedes[STDOUT_FILENO]);
468  else
469  fclose(stream);
470 
471  waitpid(entry->pid, NULL, 0);
472 
473  entry->pid = 0;
474  errno = savedErrno;
475  return (NULL);
476  }
477 
478  return (stream);
479 }

481 static int
482 my_pclose(struct RI *entry)
483 {
484  int status = 0;
485 
486  if (fclose(entry->pipe))
487  return (-1);
488 
489  /* We ignore the return value from the process because some
490  programs return funny values which would result in the input
491  being discarded even if they executed successfully. This isn't
492  a problem because the result data size threshold will filter
493  out any programs which exit with a usage message without
494  producing useful output. */
495  if (waitpid(entry->pid, NULL, 0) != entry->pid)
496  status = -1;
497 
498  entry->pipe = NULL;
499  entry->pid = 0;
500  return (status);
501 }

505  *
506  * If a few of the randomness sources create a large amount of output then
507  * the slowPoll() stops once the buffer has been filled (but before all the
508  * randomness sources have been sucked dry) so that the 'usefulness' factor
509  * remains below the threshold. For this reason the gatherer buffer has to
510  * be fairly sizeable on moderately loaded systems. This is something of a
511  * bug since the usefulness should be influenced by the amount of output as
512  * well as the source type */
513

515 static int
516 slow_poll(FILE *dbgfp, int dbgall, size_t *nbytes )
517 {
518  int moreSources;
519  struct timeval tv;
520  fd_set fds;
521 #if defined( __hpux )
522  size_t maxFD = 0;
523 #else
524  int maxFD = 0;
525 #endif /* OS-specific brokenness */
526  int bufPos, i, usefulness = 0;
527  int last_so_far = 0;
528  int any_need_entropy = 0;
529  int delay;
530  int rc;
531 
532  /* Fire up each randomness source */
533  FD_ZERO(&fds);
534  for (i = 0; dataSources[i].path != NULL; i++) {
535  /* Since popen() is a fairly heavy function, we check to see whether
536  * the executable exists before we try to run it */
537  if (access(dataSources[i].path, X_OK)) {
538  if( dbgfp && dbgall )
539  fprintf(dbgfp, "%s not present%s\n", dataSources[i].path,
540  dataSources[i].hasAlternative ?
541  ", has alternatives" : "");
542  dataSources[i].pipe = NULL;
543  }
544  else
545  dataSources[i].pipe = my_popen(&dataSources[i]);
546 
547  if (dataSources[i].pipe != NULL) {
548  dataSources[i].pipeFD = fileno(dataSources[i].pipe);
549  if (dataSources[i].pipeFD > maxFD)
550  maxFD = dataSources[i].pipeFD;
551 
552 #ifdef O_NONBLOCK /* Ohhh what a hack (used for Atari) */
553  fcntl(dataSources[i].pipeFD, F_SETFL, O_NONBLOCK);
554 #else
555 #error O_NONBLOCK is missing
556 #endif
557  /* FIXME: We need to make sure that the fd is less than
558  FD_SETSIZE. */
559  FD_SET(dataSources[i].pipeFD, &fds);
560  dataSources[i].length = 0;
561 
562  /* If there are alternatives for this command, don't try and
563  * execute them */
564  while (dataSources[i].hasAlternative) {
565  if( dbgfp && dbgall )
566  fprintf(dbgfp, "Skipping %s\n", dataSources[i + 1].path);
567  i++;
568  }
569  }
570  }
571 
572 
573  /* Suck all the data we can get from each of the sources */
574  bufPos = 0;
575  moreSources = 1;
576  delay = 0; /* Return immediately (well, after 100ms) the first time. */
577  while (moreSources && bufPos <= gather_buffer_size) {
578  /* Wait for data to become available from any of the sources, with a
579  * timeout of 10 seconds. This adds even more randomness since data
580  * becomes available in a nondeterministic fashion. Kudos to HP's QA
581  * department for managing to ship a select() which breaks its own
582  * prototype */
583  tv.tv_sec = delay;
584  tv.tv_usec = delay? 0 : 100000;
585 
586 #if defined( __hpux ) && ( OS_VERSION == 9 )
587  rc = select(maxFD + 1, (int *)&fds, NULL, NULL, &tv);
588 #else /* */
589  rc = select(maxFD + 1, &fds, NULL, NULL, &tv);
590 #endif /* __hpux */
591  if (rc == -1)
592  break; /* Ooops; select failed. */
593 
594  if (!rc)
595  {
596  /* FIXME: Because we run several tools at once it is
597  unlikely that we will see a block in select at all. */
598  if (!any_need_entropy
599  || last_so_far != (gather_buffer_size - bufPos) )
600  {
601  last_so_far = gather_buffer_size - bufPos;
602  _gcry_random_progress ("need_entropy", 'X',
603  last_so_far,
604  gather_buffer_size);
605  any_need_entropy = 1;
606  }
607  delay = 10; /* Use 10 seconds henceforth. */
608  /* Note that the fd_set is setup again at the end of this loop. */
609  }
610 
611  /* One of the sources has data available, read it into the buffer */
612  for (i = 0; dataSources[i].path != NULL; i++) {
613  if( dataSources[i].pipe && FD_ISSET(dataSources[i].pipeFD, &fds)) {
614  size_t noBytes;
615 
616  if ((noBytes = fread(gather_buffer + bufPos, 1,
617  gather_buffer_size - bufPos,
618  dataSources[i].pipe)) == 0) {
619  if (my_pclose(&dataSources[i]) == 0) {
620  int total = 0;
621 
622  /* Try and estimate how much entropy we're getting
623  * from a data source */
624  if (dataSources[i].usefulness) {
625  if (dataSources[i].usefulness < 0)
626  total = (dataSources[i].length + 999)
627  / -dataSources[i].usefulness;
628  else
629  total = dataSources[i].length
630  / dataSources[i].usefulness;
631  }
632  if( dbgfp )
633  fprintf(dbgfp,
634  "%s %s contributed %d bytes, "
635  "usefulness = %d\n", dataSources[i].path,
636  (dataSources[i].arg != NULL) ?
637  dataSources[i].arg : "",
638  dataSources[i].length, total);
639  if( dataSources[i].length )
640  usefulness += total;
641  }
642  dataSources[i].pipe = NULL;
643  }
644  else {
645  int currPos = bufPos;
646  int endPos = bufPos + noBytes;
647 
648  /* Run-length compress the input byte sequence */
649  while (currPos < endPos) {
650  int ch = gather_buffer[currPos];
651 
652  /* If it's a single byte, just copy it over */
653  if (ch != gather_buffer[currPos + 1]) {
654  gather_buffer[bufPos++] = ch;
655  currPos++;
656  }
657  else {
658  int count = 0;
659 
660  /* It's a run of repeated bytes, replace them
661  * with the byte count mod 256 */
662  while ((ch == gather_buffer[currPos])
663  && currPos < endPos) {
664  count++;
665  currPos++;
666  }
667  gather_buffer[bufPos++] = count;
668  noBytes -= count - 1;
669  }
670  }
671 
672  /* Remember the number of (compressed) bytes of input we
673  * obtained */
674  dataSources[i].length += noBytes;
675  }
676  }
677  }
678 
679  /* Check if there is more input available on any of the sources */
680  moreSources = 0;
681  FD_ZERO(&fds);
682  for (i = 0; dataSources[i].path != NULL; i++) {
683  if (dataSources[i].pipe != NULL) {
684  FD_SET(dataSources[i].pipeFD, &fds);
685  moreSources = 1;
686  }
687  }
688  }
689 
690  if (any_need_entropy)
691  _gcry_random_progress ("need_entropy", 'X',
692  gather_buffer_size,
693  gather_buffer_size);
694 
695  if( dbgfp ) {
696  fprintf(dbgfp, "Got %d bytes, usefulness = %d\n", bufPos, usefulness);
697  fflush(dbgfp);
698  }
699  *nbytes = bufPos;
700  return usefulness;
701 }

703 /****************
704  * Start the gatherer process which writes messages of
705  * type GATHERER_MSG to pipedes
706  */
707 static void
708 start_gatherer( int pipefd )
709 {
710  FILE *dbgfp = NULL;
711  int dbgall;
712 
713  {
714  const char *s = getenv("GNUPG_RNDUNIX_DBG");
715  if( s ) {
716  dbgfp = (*s=='-' && !s[1])? stdout : fopen(s, "a");
717  if( !dbgfp )
718  log_info("can't open debug file `%s': %s\n",
719  s, strerror(errno) );
720  else
721  fprintf(dbgfp,"\nSTART RNDUNIX DEBUG pid=%d\n", (int)getpid());
722  }
723  dbgall = !!getenv("GNUPG_RNDUNIX_DBGALL");
724  }
725  /* close all files but the ones we need */
726  { int nmax, n1, n2, i;
727 #ifdef _SC_OPEN_MAX
728  if( (nmax=sysconf( _SC_OPEN_MAX )) < 0 ) {
729 #ifdef _POSIX_OPEN_MAX
730  nmax = _POSIX_OPEN_MAX;
731 #else
732  nmax = 20; /* assume a reasonable value */
733 #endif
734  }
735 #else
736  nmax = 20; /* assume a reasonable value */
737 #endif
738  n1 = fileno( stderr );
739  n2 = dbgfp? fileno( dbgfp ) : -1;
740  for(i=0; i < nmax; i++ ) {
741  if( i != n1 && i != n2 && i != pipefd )
742  close(i);
743  }
744  errno = 0;
745  }
746 
747 
748  /* Set up the buffer. Not ethat we use a plain standard malloc here. */
749  gather_buffer_size = GATHER_BUFSIZE;
750  gather_buffer = malloc( gather_buffer_size );
751  if( !gather_buffer ) {
752  log_error("out of core while allocating the gatherer buffer\n");
753  exit(2);
754  }
755 
756  /* Reset the SIGC(H)LD handler to the system default. This is necessary
757  * because if the program which cryptlib is a part of installs its own
758  * SIGC(H)LD handler, it will end up reaping the cryptlib children before
759  * cryptlib can. As a result, my_pclose() will call waitpid() on a
760  * process which has already been reaped by the installed handler and
761  * return an error, so the read data won't be added to the randomness
762  * pool. There are two types of SIGC(H)LD naming, the SysV SIGCLD and
763  * the BSD/Posix SIGCHLD, so we need to handle either possibility */
764 #ifdef SIGCLD
765  signal(SIGCLD, SIG_DFL);
766 #else
767  signal(SIGCHLD, SIG_DFL);
768 #endif
769 
770  fclose(stderr); /* Arrghh!! It's Stuart code!! */
771 
772  /* Mary goes to Berkeley: NetBSD emits warnings if the standard
773  descriptors are not open when running setuid program. Thus we
774  connect them to the bitbucket if they are not already open. */
775  {
776  struct stat statbuf;
777 
778  if (fstat (STDIN_FILENO, &statbuf) == -1 && errno == EBADF)
779  open ("/dev/null",O_RDONLY);
780  if (fstat (STDOUT_FILENO, &statbuf) == -1 && errno == EBADF)
781  open ("/dev/null",O_WRONLY);
782  if (fstat (STDERR_FILENO, &statbuf) == -1 && errno == EBADF)
783  open ("/dev/null",O_WRONLY);
784  }
785 
786  for(;;) {
787  GATHER_MSG msg;
788  size_t nbytes;
789  const char *p;
790 
791  msg.usefulness = slow_poll( dbgfp, dbgall, &nbytes );
792  p = (const char*)gather_buffer;
793  while( nbytes ) {
794  msg.ndata = nbytes > sizeof(msg.data)? sizeof(msg.data) : nbytes;
795  memcpy( msg.data, p, msg.ndata );
796  nbytes -= msg.ndata;
797  p += msg.ndata;
798 
799  while( write( pipefd, &msg, sizeof(msg) ) != sizeof(msg) ) {
800  if( errno == EINTR )
801  continue;
802  if( errno == EAGAIN ) {
803  struct timeval tv;
804  tv.tv_sec = 0;
805  tv.tv_usec = 50000;
806  select(0, NULL, NULL, NULL, &tv);
807  continue;
808  }
809  if( errno == EPIPE ) /* parent has exited, so give up */
810  exit(0);
811 
812  /* we can't do very much here because stderr is closed */
813  if( dbgfp )
814  fprintf(dbgfp, "gatherer can't write to pipe: %s\n",
815  strerror(errno) );
816  /* we start a new poll to give the system some time */
817  nbytes = 0;
818  break;
819  }
820  }
821  }
822  /* we are killed when the parent dies */
823 }

827 read_a_msg( int fd, GATHER_MSG *msg )
828 {
829  char *buffer = (char*)msg;
830  size_t length = sizeof( *msg );

832 
833  do {
834  do {

836  } while( n == -1 && errno == EINTR );

838  return -1;
839  buffer += n;

847  * Using a level of 0 should never block and better add nothing
848  * to the pool. So this is just a dummy for this gatherer.
849  */
850 int
851 _gcry_rndunix_gather_random (void (*add)(const void*, size_t,
852  enum random_origins),
853  enum random_origins origin,
854  size_t length, int level )
855 {
856  static pid_t gatherer_pid = 0;
857  static int pipedes[2];
858  GATHER_MSG msg;
859  size_t n;
860 
861  if( !level )
862  return 0;
863 
864  if( !gatherer_pid ) {
865  /* Make sure we are not setuid. */
866  if ( getuid() != geteuid() )
867  BUG();
868  /* time to start the gatherer process */
869  if( pipe( pipedes ) ) {
870  log_error("pipe() failed: %s\n", strerror(errno));
871  return -1;
872  }
873  gatherer_pid = fork();
874  if( gatherer_pid == -1 ) {
875  log_error("can't for gatherer process: %s\n", strerror(errno));
876  return -1;
877  }

879  start_gatherer( pipedes[1] );
880  /* oops, can't happen */
881  return -1;
882  }
883  }
884 
885  /* now read from the gatherer */
886  while( length ) {
887  int goodness;
888  ulong subtract;
889 
890  if( read_a_msg( pipedes[0], &msg ) ) {
891  log_error("reading from gatherer pipe failed: %s\n",
892  strerror(errno));
893  return -1;
894  }
895 
896 
897  if( level > 1 ) {
898  if( msg.usefulness > 30 )
899  goodness = 100;
900  else if ( msg.usefulness )
901  goodness = msg.usefulness * 100 / 30;
902  else
903  goodness = 0;
904  }
905  else if( level ) {
906  if( msg.usefulness > 15 )
907  goodness = 100;
908  else if ( msg.usefulness )
909  goodness = msg.usefulness * 100 / 15;
910  else
911  goodness = 0;
912  }
913  else
914  goodness = 100; /* goodness of level 0 is always 100 % */
915 
916  n = msg.ndata;
917  if( n > length )
918  n = length;
919  (*add)( msg.data, n, origin );
920 
921  /* this is the trick how we cope with the goodness */
922  subtract = (ulong)n * goodness / 100;
923  /* subtract at least 1 byte to avoid infinite loops */
924  length -= subtract ? subtract : 1;
925  }
926 
927  return 0;
928 }
