52 gpg_err_code_t
53 _gpgrt_lock_init (gpgrt_lock_t *lockhd)
54 {
55  _gpgrt_lock_t *lock = (_gpgrt_lock_t*)lockhd;
56 
57  /* If VERS is zero we assume that no static initialization has been
58  done, so we setup our ABI version right here. The caller might
59  have called us to test whether lock support is at all available. */
60  if (!lock->vers)
61  {
62  if (sizeof (gpgrt_lock_t) < sizeof (_gpgrt_lock_t))
63  abort ();
64  lock->vers = LOCK_ABI_VERSION;
65  }
66  else /* Run the usual check. */
67  {
68  lock = get_lock_object (lockhd);
69  if (sizeof (gpgrt_lock_t) < sizeof (_gpgrt_lock_t))
70  abort ();
71  }
72 
73  InitializeCriticalSection (&lock->csec);
74  lock->initdone = 1;
75  return 0;
76 }

79 gpg_err_code_t
80 _gpgrt_lock_lock (gpgrt_lock_t *lockhd)
81 {
82  _gpgrt_lock_t *lock = get_lock_object (lockhd);
83 
84  if (!lock->initdone)
85  {
86  if (!InterlockedIncrement (&lock->started))
87  {
88  /* The new value of started is 0. Because the initial value
89  if the variable was -1 we known that this thread is the
90  first who needs this lock. Thus we initialize now. All
91  other threads won't get 0 back from InterlockedIncrement
92  and thus fall into the wait loop below. We ignore that
93  STARTED may in theory overflow if this thread starves for
94  too long. */
95  gpgrt_lock_init (lockhd);
96  }
97  else
98  {
99  while (!lock->initdone)
100  Sleep (0);
101  }
102  }
103 
104  EnterCriticalSection (&lock->csec);
105  return 0;
106 }
