145 /* Check that forking won't return the same random. */
146 static void
147 check_forking (void)
148 {
149 #ifdef HAVE_W32_SYSTEM
150  if (verbose)
151  inf ("check_forking skipped: not applicable on Windows\n");
152 #else
153  pid_t pid;
154  int rp[2];
155  int i, status;
156  size_t nread;
157  char tmp1[16], tmp1c[16], tmp1p[16];
158 
159  if (verbose)
160  inf ("checking that a fork won't cause the same random output\n");
161 
162  /* We better make sure that the RNG has been initialzied. */
163  gcry_randomize (tmp1, sizeof tmp1, GCRY_STRONG_RANDOM);
164  if (verbose)
165  print_hex ("initial random: ", tmp1, sizeof tmp1);
166 
167  if (pipe (rp) == -1)
168  die ("pipe failed: %s\n", strerror (errno));
169 
170  pid = fork ();
171  if (pid == (pid_t)(-1))
172  die ("fork failed: %s\n", strerror (errno));
173  if (!pid)
174  {
175  gcry_randomize (tmp1c, sizeof tmp1c, GCRY_STRONG_RANDOM);
176  if (writen (rp[1], tmp1c, sizeof tmp1c))
177  die ("write failed: %s\n", strerror (errno));
178  if (verbose)
179  {
180  print_hex (" child random: ", tmp1c, sizeof tmp1c);
181  fflush (stdout);
182  }
183  _exit (0);
184  }
185  gcry_randomize (tmp1p, sizeof tmp1p, GCRY_STRONG_RANDOM);
186  if (verbose)
187  print_hex (" parent random: ", tmp1p, sizeof tmp1p);
188 
189  close (rp[1]);
190  if (readn (rp[0], tmp1c, sizeof tmp1c, &nread))
191  die ("read failed: %s\n", strerror (errno));
192  if (nread != sizeof tmp1c)
193  die ("read too short\n");
194 
195  while ( (i=waitpid (pid, &status, 0)) == -1 && errno == EINTR)
196  ;
197  if (i != (pid_t)(-1)
198  && WIFEXITED (status) && !WEXITSTATUS (status))
199  ;
200  else
201  die ("child failed\n");
202 
203  if (!memcmp (tmp1p, tmp1c, sizeof tmp1c))
204  die ("parent and child got the same random number\n");
205 #endif
206 }

433 /* Because we want to check initialization behaviour, we need to
434  fork/exec this program with several command line arguments. We use
435  system, so that these tests work also on Windows. */
436 static void
437 run_all_rng_tests (const char *program)
438 {
439  static const char *options[] = {
440  "--early-rng-check",
441  "--early-rng-check --prefer-standard-rng",
442  "--early-rng-check --prefer-fips-rng",
443  "--early-rng-check --prefer-system-rng",
444  "--prefer-standard-rng",
445  "--prefer-fips-rng",
446  "--prefer-system-rng",
447  NULL
448  };
449  int idx;
450  size_t len, maxlen;
451  char *cmdline;
452 
453  maxlen = 0;
454  for (idx=0; options[idx]; idx++)
455  {
456  len = strlen (options[idx]);
457  if (len > maxlen)
458  maxlen = len;
459  }
460  maxlen += strlen (program);
461  maxlen += strlen (" --in-recursion --verbose --debug --progress");
462  maxlen++;
463  cmdline = malloc (maxlen + 1);
464  if (!cmdline)
465  die ("out of core\n");
466 
467  for (idx=0; options[idx]; idx++)
468  {
469  if (verbose)
470  inf ("now running with options '%s'\n", options[idx]);
471  strcpy (cmdline, program);
472  strcat (cmdline, " --in-recursion");
473  if (verbose)
474  strcat (cmdline, " --verbose");
475  if (debug)
476  strcat (cmdline, " --debug");
477  if (with_progress)
478  strcat (cmdline, " --progress");
479  strcat (cmdline, " ");
480  strcat (cmdline, options[idx]);
481  if (system (cmdline))
482  die ("running '%s' failed\n", cmdline);
483  }
484 
485  free (cmdline);
486 }
