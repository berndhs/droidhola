124 static void
125 launch_thread (THREAD_RET_TYPE (*fnc)(void *), struct thread_arg *th)
126 {
127  int fd;
128 
129  th->stop_me = 0;
130  fd = es_fileno (th->stream);

132 
133  th->thread = CreateThread (NULL, 0, fnc, th, 0, NULL);
134  if (!th->thread)
135  die ("creating thread '%s' failed: rc=%d", th->name, (int)GetLastError ());
136  show ("thread '%s' launched (fd=%d)\n", th->name, fd);
137 
138 #elif USE_POSIX_THREADS
139 
140  if (pthread_create (&th->thread, NULL, fnc, th))
141  die ("creating thread '%s' failed: %s\n", th->name, strerror (errno));
142  show ("thread '%s' launched (fd=%d)\n", th->name, fd);
143 
144 # else /* no thread support */
145 
146  verbose++;

210 static void
211 test_poll (void)
212 {
213  int ret;
214  gpgrt_poll_t fds[3];
215  char buffer[16];
216  size_t used, nwritten;
217  int c;
218 
219  memset (fds, 0, sizeof fds);
220  fds[0].stream = test_stdin;
221  fds[0].want_read = 1;
222  fds[1].stream = test_stdout;
223  fds[1].want_write = 1;
224  /* FIXME: We don't use the next stream at all. */
225  fds[2].stream = test_stderr;
226  fds[2].want_write = 1;
227  fds[2].ignore = 1;
228 
229 
230  used = 0;
231  while (used || !fds[0].ignore)
232  {
233  ret = gpgrt_poll (fds, DIM(fds), -1);
234  if (ret == -1)
235  {
236  fail ("gpgrt_poll failed: %s\n", strerror (errno));
237  continue;
238  }
239  if (!ret)
240  {
241  fail ("gpgrt_poll unexpectedly timed out\n");
242  continue;
243  }
244  show ("gpgrt_poll detected %d events\n", ret);
245  if (fds[0].got_read)
246  {
247  /* Read from the producer. */
248  for (;;)
249  {
250  c = es_fgetc (fds[0].stream);
251  if (c == EOF)
252  {
253  if (es_feof (fds[0].stream))
254  {
255  show ("reading '%s': EOF\n", peer_stdin.name);
256  fds[0].ignore = 1; /* Not anymore needed. */
257  peer_stdin.stop_me = 1; /* Tell the thread to stop. */
258  }
259  else if (es_ferror (fds[0].stream))
260  {
261  fail ("error reading '%s': %s\n",
262  peer_stdin.name, strerror (errno));
263  fds[0].ignore = 1; /* Disable. */
264  peer_stdin.stop_me = 1; /* Tell the thread to stop. */
265  }
266  else
267  show ("reading '%s': EAGAIN\n", peer_stdin.name);
268  break;
269  }
270  else
271  {
272  if (used <= sizeof buffer -1)
273  buffer[used++] = c;
274  if (used == sizeof buffer)
275  {
276  show ("throttling reading from '%s'\n", peer_stdin.name);
277  fds[0].ignore = 1;
278  break;
279  }
280  }
281  }
282  show ("read from '%s': %zu bytes\n", peer_stdin.name, used);
283  if (used)
284  fds[1].ignore = 0; /* Data to send. */
285  }
286  if (fds[1].got_write)
287  {
288  if (used)
289  {
290  ret = es_write (fds[1].stream, buffer, used, &nwritten);
291  show ("result for writing to '%s': ret=%d, n=%zu, nwritten=%zu\n",
292  peer_stdout.name, ret, used, nwritten);
293  if (!ret)
294  {
295  assert (nwritten <= used);
296  /* Move the remaining data to the front of buffer. */
297  memmove (buffer, buffer + nwritten,
298  sizeof buffer - nwritten);
299  used -= nwritten;
300  }
301  ret = es_fflush (fds[1].stream);
302  if (ret)
303  fail ("Flushing for '%s' failed: %s\n",
304  peer_stdout.name, strerror (errno));
305  }
306  if (!used)
307  fds[1].ignore = 1; /* No need to send data. */
308  }
309 
310  if (used < sizeof buffer / 2 && !peer_stdin.stop_me && fds[0].ignore)
311  {
312  show ("accelerate reading from '%s'\n", peer_stdin.name);
313  fds[0].ignore = 0;
314  }
315  }
316 }
