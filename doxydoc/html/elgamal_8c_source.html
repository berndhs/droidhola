105 /****************
106  * Michael Wiener's table on subgroup sizes to match field sizes.
107  * (floating around somewhere, probably based on the paper from
108  * Eurocrypt 96, page 332)
109  */
110 static unsigned int
111 wiener_map( unsigned int n )
112 {
113  static struct { unsigned int p_n, q_n; } t[] =
114  { /* p q attack cost */
115  { 512, 119 }, /* 9 x 10^17 */
116  { 768, 145 }, /* 6 x 10^21 */
117  { 1024, 165 }, /* 7 x 10^24 */
118  { 1280, 183 }, /* 3 x 10^27 */
119  { 1536, 198 }, /* 7 x 10^29 */
120  { 1792, 212 }, /* 9 x 10^31 */
121  { 2048, 225 }, /* 8 x 10^33 */
122  { 2304, 237 }, /* 5 x 10^35 */
123  { 2560, 249 }, /* 3 x 10^37 */
124  { 2816, 259 }, /* 1 x 10^39 */
125  { 3072, 269 }, /* 3 x 10^40 */
126  { 3328, 279 }, /* 8 x 10^41 */
127  { 3584, 288 }, /* 2 x 10^43 */
128  { 3840, 296 }, /* 4 x 10^44 */
129  { 4096, 305 }, /* 7 x 10^45 */
130  { 4352, 313 }, /* 1 x 10^47 */
131  { 4608, 320 }, /* 2 x 10^48 */
132  { 4864, 328 }, /* 2 x 10^49 */
133  { 5120, 335 }, /* 3 x 10^50 */
134  { 0, 0 }
135  };
136  int i;
137 
138  for(i=0; t[i].p_n; i++ )
139  {
140  if( n <= t[i].p_n )
141  return t[i].q_n;
142  }
143  /* Not in table - use an arbitrary high number. */
144  return n / 8 + 200;
145 }

147 static int
148 test_keys ( ELG_secret_key *sk, unsigned int nbits, int nodie )
149 {
150  ELG_public_key pk;
151  gcry_mpi_t test = mpi_new ( 0 );
152  gcry_mpi_t out1_a = mpi_new ( nbits );
153  gcry_mpi_t out1_b = mpi_new ( nbits );
154  gcry_mpi_t out2 = mpi_new ( nbits );
155  int failed = 0;
156 
157  pk.p = sk->p;
158  pk.g = sk->g;
159  pk.y = sk->y;
160 
161  _gcry_mpi_randomize ( test, nbits, GCRY_WEAK_RANDOM );
162 
163  do_encrypt ( out1_a, out1_b, test, &pk );
164  decrypt ( out2, out1_a, out1_b, sk );
165  if ( mpi_cmp( test, out2 ) )
166  failed |= 1;
167 
168  sign ( out1_a, out1_b, test, sk );
169  if ( !verify( out1_a, out1_b, test, &pk ) )
170  failed |= 2;
171 
172  _gcry_mpi_release ( test );
173  _gcry_mpi_release ( out1_a );
174  _gcry_mpi_release ( out1_b );
175  _gcry_mpi_release ( out2 );
176 
177  if (failed && !nodie)
178  log_fatal ("Elgamal test key for %s %s failed\n",
179  (failed & 1)? "encrypt+decrypt":"",
180  (failed & 2)? "sign+verify":"");
181  if (failed && DBG_CIPHER)
182  log_debug ("Elgamal test key for %s %s failed\n",
183  (failed & 1)? "encrypt+decrypt":"",
184  (failed & 2)? "sign+verify":"");
185 
186  return failed;
187 }

190 /****************
191  * Generate a random secret exponent k from prime p, so that k is
192  * relatively prime to p-1. With SMALL_K set, k will be selected for
193  * better encryption performance - this must never be used signing!
194  */
195 static gcry_mpi_t
196 gen_k( gcry_mpi_t p, int small_k )
197 {
198  gcry_mpi_t k = mpi_alloc_secure( 0 );
199  gcry_mpi_t temp = mpi_alloc( mpi_get_nlimbs(p) );
200  gcry_mpi_t p_1 = mpi_copy(p);
201  unsigned int orig_nbits = mpi_get_nbits(p);
202  unsigned int nbits, nbytes;
203  char *rndbuf = NULL;
204 
205  if (small_k)
206  {
207  /* Using a k much lesser than p is sufficient for encryption and
208  * it greatly improves the encryption performance. We use
209  * Wiener's table and add a large safety margin. */
210  nbits = wiener_map( orig_nbits ) * 3 / 2;
211  if( nbits >= orig_nbits )
212  BUG();
213  }
214  else
215  nbits = orig_nbits;
216 
217 
218  nbytes = (nbits+7)/8;
219  if( DBG_CIPHER )
220  log_debug("choosing a random k\n");
221  mpi_sub_ui( p_1, p, 1);
222  for(;;)
223  {
224  if( !rndbuf || nbits < 32 )
225  {
226  xfree(rndbuf);
227  rndbuf = _gcry_random_bytes_secure( nbytes, GCRY_STRONG_RANDOM );
228  }
229  else
230  {
231  /* Change only some of the higher bits. We could improve
232  this by directly requesting more memory at the first call
233  to get_random_bytes() and use this the here maybe it is
234  easier to do this directly in random.c Anyway, it is
235  highly inlikely that we will ever reach this code. */
236  char *pp = _gcry_random_bytes_secure( 4, GCRY_STRONG_RANDOM );
237  memcpy( rndbuf, pp, 4 );
238  xfree(pp);
239  }
240  _gcry_mpi_set_buffer( k, rndbuf, nbytes, 0 );
241 
242  for(;;)
243  {
244  if( !(mpi_cmp( k, p_1 ) < 0) ) /* check: k < (p-1) */
245  {
246  if( DBG_CIPHER )
247  progress('+');
248  break; /* no */
249  }
250  if( !(mpi_cmp_ui( k, 0 ) > 0) ) /* check: k > 0 */
251  {
252  if( DBG_CIPHER )
253  progress('-');
254  break; /* no */
255  }
256  if (mpi_gcd( temp, k, p_1 ))
257  goto found; /* okay, k is relative prime to (p-1) */
258  mpi_add_ui( k, k, 1 );
259  if( DBG_CIPHER )
260  progress('.');
261  }
262  }
263  found:
264  xfree (rndbuf);
265  if( DBG_CIPHER )
266  progress('\n');
267  mpi_free(p_1);
268  mpi_free(temp);
269 
270  return k;
271 }

273 /****************
274  * Generate a key pair with a key of size NBITS
275  * Returns: 2 structures filled with all needed values
276  * and an array with n-1 factors of (p-1)
277  */
278 static gcry_err_code_t
279 generate ( ELG_secret_key *sk, unsigned int nbits, gcry_mpi_t **ret_factors )
280 {
281  gcry_err_code_t rc;
282  gcry_mpi_t p; /* the prime */
283  gcry_mpi_t p_min1;
284  gcry_mpi_t g;
285  gcry_mpi_t x; /* the secret exponent */
286  gcry_mpi_t y;
287  unsigned int qbits;
288  unsigned int xbits;
289  byte *rndbuf;
290 
291  p_min1 = mpi_new ( nbits );
292  qbits = wiener_map( nbits );
293  if( qbits & 1 ) /* better have a even one */
294  qbits++;
295  g = mpi_alloc(1);
296  rc = _gcry_generate_elg_prime (0, nbits, qbits, g, &p, ret_factors);
297  if (rc)
298  {
299  mpi_free (p_min1);
300  mpi_free (g);
301  return rc;
302  }
303  mpi_sub_ui(p_min1, p, 1);
304 
305 
306  /* Select a random number which has these properties:
307  * 0 < x < p-1
308  * This must be a very good random number because this is the
309  * secret part. The prime is public and may be shared anyway,
310  * so a random generator level of 1 is used for the prime.
311  *
312  * I don't see a reason to have a x of about the same size
313  * as the p. It should be sufficient to have one about the size
314  * of q or the later used k plus a large safety margin. Decryption
315  * will be much faster with such an x.
316  */
317  xbits = qbits * 3 / 2;
318  if( xbits >= nbits )
319  BUG();
320  x = mpi_snew ( xbits );
321  if( DBG_CIPHER )
322  log_debug("choosing a random x of size %u\n", xbits );
323  rndbuf = NULL;
324  do
325  {
326  if( DBG_CIPHER )
327  progress('.');
328  if( rndbuf )
329  { /* Change only some of the higher bits */
330  if( xbits < 16 ) /* should never happen ... */
331  {
332  xfree(rndbuf);
333  rndbuf = _gcry_random_bytes_secure ((xbits+7)/8,
334  GCRY_VERY_STRONG_RANDOM);
335  }
336  else
337  {
338  char *r = _gcry_random_bytes_secure (2, GCRY_VERY_STRONG_RANDOM);
339  memcpy(rndbuf, r, 2 );
340  xfree (r);
341  }
342  }
343  else
344  {
345  rndbuf = _gcry_random_bytes_secure ((xbits+7)/8,
346  GCRY_VERY_STRONG_RANDOM );
347  }
348  _gcry_mpi_set_buffer( x, rndbuf, (xbits+7)/8, 0 );
349  mpi_clear_highbit( x, xbits+1 );
350  }
351  while( !( mpi_cmp_ui( x, 0 )>0 && mpi_cmp( x, p_min1 )<0 ) );
352  xfree(rndbuf);
353 
354  y = mpi_new (nbits);
355  mpi_powm( y, g, x, p );
356 
357  if( DBG_CIPHER )
358  {
359  progress ('\n');
360  log_mpidump ("elg p", p );
361  log_mpidump ("elg g", g );
362  log_mpidump ("elg y", y );
363  log_mpidump ("elg x", x );
364  }
365 
366  /* Copy the stuff to the key structures */
367  sk->p = p;
368  sk->g = g;
369  sk->y = y;
370  sk->x = x;
371 
372  _gcry_mpi_release ( p_min1 );
373 
374  /* Now we can test our keys (this should never fail!) */
375  test_keys ( sk, nbits - 64, 0 );
376 
377  return 0;
378 }

381 /* Generate a key pair with a key of size NBITS not using a random
382  value for the secret key but the one given as X. This is useful to
383  implement a passphrase based decryption for a public key based
384  encryption. It has appliactions in backup systems.
385 
386  Returns: A structure filled with all needed values and an array
387  with n-1 factors of (p-1). */
388 static gcry_err_code_t
389 generate_using_x (ELG_secret_key *sk, unsigned int nbits, gcry_mpi_t x,
390  gcry_mpi_t **ret_factors )
391 {
392  gcry_err_code_t rc;
393  gcry_mpi_t p; /* The prime. */
394  gcry_mpi_t p_min1; /* The prime minus 1. */
395  gcry_mpi_t g; /* The generator. */
396  gcry_mpi_t y; /* g^x mod p. */
397  unsigned int qbits;
398  unsigned int xbits;
399 
400  sk->p = NULL;
401  sk->g = NULL;
402  sk->y = NULL;
403  sk->x = NULL;
404 
405  /* Do a quick check to see whether X is suitable. */
406  xbits = mpi_get_nbits (x);
407  if ( xbits < 64 || xbits >= nbits )
408  return GPG_ERR_INV_VALUE;
409 
410  p_min1 = mpi_new ( nbits );
411  qbits = wiener_map ( nbits );
412  if ( (qbits & 1) ) /* Better have an even one. */
413  qbits++;
414  g = mpi_alloc (1);
415  rc = _gcry_generate_elg_prime (0, nbits, qbits, g, &p, ret_factors );
416  if (rc)
417  {
418  mpi_free (p_min1);
419  mpi_free (g);
420  return rc;
421  }
422  mpi_sub_ui (p_min1, p, 1);
423 
424  if (DBG_CIPHER)
425  log_debug ("using a supplied x of size %u", xbits );
426  if ( !(mpi_cmp_ui ( x, 0 ) > 0 && mpi_cmp ( x, p_min1 ) <0 ) )
427  {
428  _gcry_mpi_release ( p_min1 );
429  _gcry_mpi_release ( p );
430  _gcry_mpi_release ( g );
431  return GPG_ERR_INV_VALUE;
432  }
433 
434  y = mpi_new (nbits);
435  mpi_powm ( y, g, x, p );
436 
437  if ( DBG_CIPHER )
438  {
439  progress ('\n');
440  log_mpidump ("elg p", p );
441  log_mpidump ("elg g", g );
442  log_mpidump ("elg y", y );
443  log_mpidump ("elg x", x );
444  }
445 
446  /* Copy the stuff to the key structures */
447  sk->p = p;
448  sk->g = g;
449  sk->y = y;
450  sk->x = mpi_copy (x);
451 
452  _gcry_mpi_release ( p_min1 );
453 
454  /* Now we can test our keys. */
455  if ( test_keys ( sk, nbits - 64, 1 ) )
456  {
457  _gcry_mpi_release ( sk->p ); sk->p = NULL;
458  _gcry_mpi_release ( sk->g ); sk->g = NULL;
459  _gcry_mpi_release ( sk->y ); sk->y = NULL;
460  _gcry_mpi_release ( sk->x ); sk->x = NULL;
461  return GPG_ERR_BAD_SECKEY;
462  }
463 
464  return 0;
465 }

522 static void
523 decrypt (gcry_mpi_t output, gcry_mpi_t a, gcry_mpi_t b, ELG_secret_key *skey )
524 {
525  gcry_mpi_t t1, t2, r;
526  unsigned int nbits = mpi_get_nbits (skey->p);
527 
528  mpi_normalize (a);
529  mpi_normalize (b);
530 
531  t1 = mpi_snew (nbits);
532 
533 #ifdef USE_BLINDING
534 
535  t2 = mpi_snew (nbits);
536  r = mpi_new (nbits);
537 
538  /* We need a random number of about the prime size. The random
539  number merely needs to be unpredictable; thus we use level 0. */
540  _gcry_mpi_randomize (r, nbits, GCRY_WEAK_RANDOM);
541 
542  /* t1 = r^x mod p */
543  mpi_powm (t1, r, skey->x, skey->p);
544  /* t2 = (a * r)^-x mod p */
545  mpi_mulm (t2, a, r, skey->p);
546  mpi_powm (t2, t2, skey->x, skey->p);
547  mpi_invm (t2, t2, skey->p);
548  /* t1 = (t1 * t2) mod p*/
549  mpi_mulm (t1, t1, t2, skey->p);
550 
551  mpi_free (r);
552  mpi_free (t2);
553 
554 #else
556  /* output = b/(a^x) mod p */
557  mpi_powm (t1, a, skey->x, skey->p);
558  mpi_invm (t1, t1, skey->p);
559 
560 #endif
562  mpi_mulm (output, b, t1, skey->p);
563 
564 #if 0
565  if( DBG_CIPHER )
566  {
567  log_mpidump ("elg decrypted x", skey->x);
568  log_mpidump ("elg decrypted p", skey->p);
569  log_mpidump ("elg decrypted a", a);
570  log_mpidump ("elg decrypted b", b);
571  log_mpidump ("elg decrypted M", output);
572  }
573 #endif
574  mpi_free (t1);
575 }

582 static void
583 sign(gcry_mpi_t a, gcry_mpi_t b, gcry_mpi_t input, ELG_secret_key *skey )
584 {
585  gcry_mpi_t k;
586  gcry_mpi_t t = mpi_alloc( mpi_get_nlimbs(a) );
587  gcry_mpi_t inv = mpi_alloc( mpi_get_nlimbs(a) );
588  gcry_mpi_t p_1 = mpi_copy(skey->p);
589 
590  /*
591  * b = (t * inv) mod (p-1)
592  * b = (t * inv(k,(p-1),(p-1)) mod (p-1)
593  * b = (((M-x*a) mod (p-1)) * inv(k,(p-1),(p-1))) mod (p-1)
594  *
595  */
596  mpi_sub_ui(p_1, p_1, 1);
597  k = gen_k( skey->p, 0 /* no small K ! */ );
598  mpi_powm( a, skey->g, k, skey->p );
599  mpi_mul(t, skey->x, a );
600  mpi_subm(t, input, t, p_1 );
601  mpi_invm(inv, k, p_1 );
602  mpi_mulm(b, t, inv, p_1 );
603 
604 #if 0
605  if( DBG_CIPHER )
606  {
607  log_mpidump ("elg sign p", skey->p);
608  log_mpidump ("elg sign g", skey->g);
609  log_mpidump ("elg sign y", skey->y);
610  log_mpidump ("elg sign x", skey->x);
611  log_mpidump ("elg sign k", k);
612  log_mpidump ("elg sign M", input);
613  log_mpidump ("elg sign a", a);
614  log_mpidump ("elg sign b", b);
615  }
616 #endif
617  mpi_free(k);
618  mpi_free(t);
619  mpi_free(inv);
620  mpi_free(p_1);
621 }

626  */
627 static int
628 verify(gcry_mpi_t a, gcry_mpi_t b, gcry_mpi_t input, ELG_public_key *pkey )
629 {

631  gcry_mpi_t t1;
632  gcry_mpi_t t2;
633  gcry_mpi_t base[4];
634  gcry_mpi_t ex[4];
635

638 
639  t1 = mpi_alloc( mpi_get_nlimbs(a) );
640  t2 = mpi_alloc( mpi_get_nlimbs(a) );
641

916  log_printmpi ("elg_decrypt x", sk.x);

962  _gcry_mpi_release (data_b);

964  _gcry_pk_util_free_encoding_ctx (&ctx);
965  if (DBG_CIPHER)
966  log_debug ("elg_decrypt => %s\n", gpg_strerror (rc));

968 }
