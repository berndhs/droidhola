58 /* Sleep for the given number of microseconds. Default
59  implementation. */
60 void
61 __assuan_usleep (assuan_context_t ctx, unsigned int usec)
62 {
63  if (! usec)
64  return;
65 
66 #ifdef HAVE_NANOSLEEP
67  {
68  struct timespec req;
69  struct timespec rem;
70 
71  req.tv_sec = 0;
72  req.tv_nsec = usec * 1000;
73 
74  while (nanosleep (&req, &rem) < 0 && errno == EINTR)
75  req = rem;
76  }
77 #else
78  {
79  struct timeval tv;
80 
81  tv.tv_sec = usec / 1000000;
82  tv.tv_usec = usec % 1000000;
83  select (0, NULL, NULL, NULL, &tv);
84  }
85 #endif
86 }

156 static int
157 writen (int fd, const char *buffer, size_t length)
158 {
159  while (length)
160  {
161  int nwritten = write (fd, buffer, length);
162 
163  if (nwritten < 0)
164  {
165  if (errno == EINTR)
166  continue;
167  return -1; /* write error */
168  }
169  length -= nwritten;
170  buffer += nwritten;
171  }
172  return 0; /* okay */
173 }

176 /* Return the maximum number of currently allowed open file
177  * descriptors. */
178 static int
179 get_max_fds (void)
180 {
181  int max_fds = -1;
182 
183 #ifdef HAVE_GETRLIMIT
184  struct rlimit rl;
185 
186  /* Under Linux we can figure out the highest used file descriptor by
187  * reading /proc/PID/fd. This is in the common cases much faster
188  * than for example doing 4096 close calls where almost all of them
189  * will fail. We use the same code in GnuPG and measured this: On a
190  * system with a limit of 4096 files and only 8 files open with the
191  * highest number being 10, we speedup close_all_fds from 125ms to
192  * 0.4ms including the readdir.
193  *
194  * Another option would be to close the file descriptors as returned
195  * from reading that directory - however then we need to snapshot
196  * that list before starting to close them. */
197 #ifdef __linux__
198  {
199  DIR *dir = NULL;
200  struct dirent *dir_entry;
201  const char *s;
202  int x;
203 
204  dir = opendir ("/proc/self/fd");
205  if (dir)
206  {
207  while ((dir_entry = readdir (dir)))
208  {
209  s = dir_entry->d_name;
210  if ( *s < '0' || *s > '9')
211  continue;
212  x = atoi (s);
213  if (x > max_fds)
214  max_fds = x;
215  }
216  closedir (dir);
217  }
218  if (max_fds != -1)
219  return max_fds + 1;
220  }
221 #endif /* __linux__ */
222 
223 # ifdef RLIMIT_NOFILE
224  if (!getrlimit (RLIMIT_NOFILE, &rl))
225  max_fds = rl.rlim_max;
226 # endif
227 
228 # ifdef RLIMIT_OFILE
229  if (max_fds == -1 && !getrlimit (RLIMIT_OFILE, &rl))
230  max_fds = rl.rlim_max;
231 
232 # endif
233 #endif /*HAVE_GETRLIMIT*/
234 
235 #ifdef _SC_OPEN_MAX
236  if (max_fds == -1)
237  {
238  long int scres = sysconf (_SC_OPEN_MAX);
239  if (scres >= 0)
240  max_fds = scres;
241  }
242 #endif
243 
244 #ifdef _POSIX_OPEN_MAX
245  if (max_fds == -1)
246  max_fds = _POSIX_OPEN_MAX;
247 #endif
248 
249 #ifdef OPEN_MAX
250  if (max_fds == -1)
251  max_fds = OPEN_MAX;
252 #endif
253 
254  if (max_fds == -1)
255  max_fds = 256; /* Arbitrary limit. */
256 
257  /* AIX returns INT32_MAX instead of a proper value. We assume that
258  this is always an error and use a more reasonable limit. */
259 #ifdef INT32_MAX
260  if (max_fds == INT32_MAX)
261  max_fds = 256;
262 #endif
263 
264  return max_fds;
265 }

268 int
269 __assuan_spawn (assuan_context_t ctx, pid_t *r_pid, const char *name,
270  const char **argv,
271  assuan_fd_t fd_in, assuan_fd_t fd_out,
272  assuan_fd_t *fd_child_list,
273  void (*atfork) (void *opaque, int reserved),
274  void *atforkvalue, unsigned int flags)
275 {
276  int pid;
277 
278  pid = fork ();
279  if (pid < 0)
280  return -1;
281 
282  if (pid == 0)
283  {
284  /* Child process (server side). */
285  int i;
286  int n;
287  char errbuf[512];
288  int *fdp;
289  int fdnul;
290 
291  if (atfork)
292  atfork (atforkvalue, 0);
293 
294  fdnul = open ("/dev/null", O_WRONLY);
295  if (fdnul == -1)
296  {
297  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_spawn", ctx,
298  "can't open `/dev/null': %s", strerror (errno));
299  _exit (4);
300  }
301 
302  /* Dup handles to stdin/stdout. */
303  if (fd_out != STDOUT_FILENO)
304  {
305  if (dup2 (fd_out == ASSUAN_INVALID_FD ? fdnul : fd_out,
306  STDOUT_FILENO) == -1)
307  {
308  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_spawn", ctx,
309  "dup2 failed in child: %s", strerror (errno));
310  _exit (4);
311  }
312  }
313 
314  if (fd_in != STDIN_FILENO)
315  {
316  if (dup2 (fd_in == ASSUAN_INVALID_FD ? fdnul : fd_in,
317  STDIN_FILENO) == -1)
318  {
319  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_spawn", ctx,
320  "dup2 failed in child: %s", strerror (errno));
321  _exit (4);
322  }
323  }
324 
325  /* Dup stderr to /dev/null unless it is in the list of FDs to be
326  passed to the child. */
327  fdp = fd_child_list;
328  if (fdp)
329  {
330  for (; *fdp != -1 && *fdp != STDERR_FILENO; fdp++)
331  ;
332  }
333  if (!fdp || *fdp == -1)
334  {
335  if (dup2 (fdnul, STDERR_FILENO) == -1)
336  {
337  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "pipe_connect_unix", ctx,
338  "dup2(dev/null, 2) failed: %s", strerror (errno));
339  _exit (4);
340  }
341  }
342  close (fdnul);
343 
344  /* Close all files which will not be duped and are not in the
345  fd_child_list. */
346  n = get_max_fds ();
347  for (i = 0; i < n; i++)
348  {
349  if (i == STDIN_FILENO || i == STDOUT_FILENO || i == STDERR_FILENO)
350  continue;
351  fdp = fd_child_list;
352  if (fdp)
353  {
354  while (*fdp != -1 && *fdp != i)
355  fdp++;
356  }
357 
358  if (!(fdp && *fdp != -1))
359  close (i);
360  }
361  gpg_err_set_errno (0);
362 
363  if (! name)
364  {
365  /* No name and no args given, thus we don't do an exec
366  but continue the forked process. */
367  *argv = "server";
368 
369  /* FIXME: Cleanup. */
370  return 0;
371  }
372 
373  execv (name, (char *const *) argv);
374 
375  /* oops - use the pipe to tell the parent about it */
376  snprintf (errbuf, sizeof(errbuf)-1,
377  "ERR %d can't exec `%s': %.50s\n",
378  _assuan_error (ctx, GPG_ERR_ASS_SERVER_START),
379  name, strerror (errno));
380  errbuf[sizeof(errbuf)-1] = 0;
381  writen (1, errbuf, strlen (errbuf));
382  _exit (4);
383  }
384 
385  if (! name)
386  *argv = "client";
387 
388  *r_pid = pid;
389 
390  return 0;
391 }
