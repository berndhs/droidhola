42  the loop becomes faster. */
43  j = -size;
44 
45  /* Offset the base pointers to compensate for the negative indices. */
46  s1_ptr -= j;
47  s2_ptr -= j;

56  cy = y < cy; /* get out carry from that addition */
57  y = x - y; /* main subtract */
