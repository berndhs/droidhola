35 assuan_set_hello_line (assuan_context_t ctx, const char *line)
36 {
37  if (!ctx)
38  return _assuan_error (ctx, GPG_ERR_ASS_INV_VALUE);
39  if (!line)
40  {
41  _assuan_free (ctx, ctx->hello_line);
42  ctx->hello_line = NULL;
43  }
44  else
45  {
46  char *buf = _assuan_malloc (ctx, 3 + strlen (line) + 1);
47  if (!buf)
48  return _assuan_error (ctx, gpg_err_code_from_syserror ());

55  }

58  }
59  return 0;
60 }

74 gpg_error_t
75 assuan_accept (assuan_context_t ctx)
76 {
77  gpg_error_t rc = 0;
78  const char *p, *pend;
79 
80  if (!ctx)
81  return _assuan_error (ctx, GPG_ERR_ASS_INV_VALUE);
82 
83  if (ctx->max_accepts != -1)
84  {
85  if (ctx->max_accepts-- == 0)
86  return -1; /* second invocation for pipemode -> terminate */
87  }
88  if (ctx->accept_handler)
89  {
90  /* FIXME: This should be superfluous, if everything else is
91  correct. */
92  ctx->finish_handler (ctx);
93  rc = ctx->accept_handler (ctx);
94  if (rc)
95  return rc;
96  }
97 
98  /* Send the hello. */
99  p = ctx->hello_line;
100  if (p && (pend = strchr (p, '\n')))
101  { /* This is a multi line hello. Send all but the last line as
102  comments. */
103  do
104  {
105  rc = _assuan_write_line (ctx, "# ", p, pend - p);
106  if (rc)
107  return rc;
108  p = pend + 1;
109  pend = strchr (p, '\n');
110  }
111  while (pend);
112  rc = _assuan_write_line (ctx, "OK ", p, strlen (p));
113  }
114  else if (p)
115  rc = assuan_write_line (ctx, p);
116  else
117  {
118  static char const okstr[] = "OK Pleased to meet you";
119  pid_t apid = assuan_get_pid (ctx);
120  if (apid != ASSUAN_INVALID_PID)
121  {
122  char tmpbuf[50];
123  snprintf (tmpbuf, sizeof tmpbuf, "%s, process %i", okstr, (int)apid);
124  rc = assuan_write_line (ctx, tmpbuf);
125  }
126  else
127  rc = assuan_write_line (ctx, okstr);
128  }
129  if (rc)
130  return rc;
131 
132  return 0;
133 }
