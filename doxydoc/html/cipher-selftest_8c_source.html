84  0x66,0x9A,0x00,0x7F,0xC7,0x6A,0x45,0x9F,

88  /* Allocate buffers, align first two elements to 16 bytes and latter to
89  block size. */
90  ctx_aligned_size = context_size + 15;
91  ctx_aligned_size -= ctx_aligned_size & 0xf;
92 
93  memsize = ctx_aligned_size + (blocksize * 2) + (blocksize * nblocks * 3) + 16;
94 
95  mem = xtrycalloc (1, memsize);
96  if (!mem)
97  return "failed to allocate memory";
98

100  ctx = (void*)(mem + offs);
101  iv = ctx + ctx_aligned_size;
102  iv2 = iv + blocksize;
103  plaintext = iv2 + blocksize;
104  plaintext2 = plaintext + nblocks * blocksize;
105  ciphertext = plaintext2 + nblocks * blocksize;
106 
107  /* Initialize ctx */
108  setkey_func (ctx, key, sizeof(key));
109 
110  /* Test single block code path */
111  memset (iv, 0x4e, blocksize);
112  memset (iv2, 0x4e, blocksize);
113  for (i = 0; i < blocksize; i++)
114  plaintext[i] = i;
115 
116  /* CBC manually. */
117  buf_xor (ciphertext, iv, plaintext, blocksize);
118  encrypt_one (ctx, ciphertext, ciphertext);
119  memcpy (iv, ciphertext, blocksize);
120 
121  /* CBC decrypt. */
122  bulk_cbc_dec (ctx, iv2, plaintext2, ciphertext, 1);
123  if (memcmp (plaintext2, plaintext, blocksize))
124  {
125  xfree (mem);
126 #ifdef HAVE_SYSLOG
127  syslog (LOG_USER|LOG_WARNING, "Libgcrypt warning: "
128  "%s-CBC-%d test failed (plaintext mismatch)", cipher,
129  blocksize * 8);
130 #else
131  (void)cipher; /* Not used. */
132 #endif
133  return "selftest for CBC failed - see syslog for details";
134  }
135 
136  if (memcmp (iv2, iv, blocksize))
137  {
138  xfree (mem);
139 #ifdef HAVE_SYSLOG
140  syslog (LOG_USER|LOG_WARNING, "Libgcrypt warning: "
141  "%s-CBC-%d test failed (IV mismatch)", cipher, blocksize * 8);
142 #endif
143  return "selftest for CBC failed - see syslog for details";
144  }
145 
146  /* Test parallelized code paths */
147  memset (iv, 0x5f, blocksize);
148  memset (iv2, 0x5f, blocksize);
149 
150  for (i = 0; i < nblocks * blocksize; i++)
151  plaintext[i] = i;
152 
153  /* Create CBC ciphertext manually. */
154  for (i = 0; i < nblocks * blocksize; i+=blocksize)
155  {
156  buf_xor (&ciphertext[i], iv, &plaintext[i], blocksize);
157  encrypt_one (ctx, &ciphertext[i], &ciphertext[i]);
158  memcpy (iv, &ciphertext[i], blocksize);
159  }
160 
161  /* Decrypt using bulk CBC and compare result. */
162  bulk_cbc_dec (ctx, iv2, plaintext2, ciphertext, nblocks);
163 
164  if (memcmp (plaintext2, plaintext, nblocks * blocksize))
165  {
166  xfree (mem);
167 #ifdef HAVE_SYSLOG
168  syslog (LOG_USER|LOG_WARNING, "Libgcrypt warning: "
169  "%s-CBC-%d test failed (plaintext mismatch, parallel path)",
170  cipher, blocksize * 8);
171 #endif
172  return "selftest for CBC failed - see syslog for details";
173  }
174  if (memcmp (iv2, iv, blocksize))
175  {
176  xfree (mem);
177 #ifdef HAVE_SYSLOG
178  syslog (LOG_USER|LOG_WARNING, "Libgcrypt warning: "
179  "%s-CBC-%d test failed (IV mismatch, parallel path)",
180  cipher, blocksize * 8);
181 #endif
182  return "selftest for CBC failed - see syslog for details";
183  }
184 
185  xfree (mem);
186  return NULL;
187 }
