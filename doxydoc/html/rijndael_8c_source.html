459 /* Perform the key setup. */
460 static gcry_err_code_t
461 do_setkey (RIJNDAEL_context *ctx, const byte *key, const unsigned keylen)
462 {
463  static int initialized = 0;
464  static const char *selftest_failed=0;
465  int rounds;
466  int i,j, r, t, rconpointer = 0;
467  int KC;
468 #if defined(USE_AESNI) || defined(USE_PADLOCK)
469  unsigned int hwfeatures;
470 #endif
471 
472  /* The on-the-fly self tests are only run in non-fips mode. In fips
473  mode explicit self-tests are required. Actually the on-the-fly
474  self-tests are not fully thread-safe and it might happen that a
475  failed self-test won't get noticed in another thread.
476 
477  FIXME: We might want to have a central registry of succeeded
478  self-tests. */
479  if (!fips_mode () && !initialized)
480  {
481  initialized = 1;
482  selftest_failed = selftest ();
483  if (selftest_failed)
484  log_error ("%s\n", selftest_failed );
485  }
486  if (selftest_failed)
487  return GPG_ERR_SELFTEST_FAILED;
488 
489 #if defined(USE_AESNI) || defined(USE_PADLOCK)
490  hwfeatures = _gcry_get_hw_features ();
491 #endif
492 
493  ctx->decryption_prepared = 0;
494 #ifdef USE_PADLOCK
495  ctx->use_padlock = 0;
496 #endif
497 #ifdef USE_AESNI
498  ctx->use_aesni = 0;
499 #endif
500 
501  if( keylen == 128/8 )
502  {
503  rounds = 10;
504  KC = 4;
505 
506  if (0)
507  {
508  ;
509  }
510 #ifdef USE_PADLOCK
511  else if (hwfeatures & HWF_PADLOCK_AES)
512  {
513  ctx->use_padlock = 1;
514  memcpy (ctx->padlockkey, key, keylen);
515  }
516 #endif
517 #ifdef USE_AESNI
518  else if (hwfeatures & HWF_INTEL_AESNI)
519  {
520  ctx->use_aesni = 1;
521  }
522 #endif
523  }
524  else if ( keylen == 192/8 )
525  {
526  rounds = 12;
527  KC = 6;
528 
529  if (0)
530  {
531  ;
532  }
533 #ifdef USE_AESNI
534  else if (hwfeatures & HWF_INTEL_AESNI)
535  {
536  ctx->use_aesni = 1;
537  }
538 #endif
539  }
540  else if ( keylen == 256/8 )
541  {
542  rounds = 14;
543  KC = 8;
544 
545  if (0)
546  {
547  ;
548  }
549 #ifdef USE_AESNI
550  else if (hwfeatures & HWF_INTEL_AESNI)
551  {
552  ctx->use_aesni = 1;
553  }
554 #endif
555  }
556  else
557  return GPG_ERR_INV_KEYLEN;
558 
559  ctx->rounds = rounds;
560 
561  /* NB: We don't yet support Padlock hardware key generation. */
562 
563  if (0)
564  {
565  ;
566  }
567 #ifdef USE_AESNI
568  else if (ctx->use_aesni)
569  aesni_do_setkey(ctx, key);
570 #endif
571  else
572  {
573  union
574  {
575  PROPERLY_ALIGNED_TYPE dummy;
576  byte data[MAXKC][4];
577  } k, tk;
578 #define k k.data
579 #define tk tk.data
580 #define W (ctx->keyschenc)
581  for (i = 0; i < keylen; i++)
582  {
583  k[i >> 2][i & 3] = key[i];
584  }
585 
586  for (j = KC-1; j >= 0; j--)
587  {
588  *((u32_a_t*)tk[j]) = *((u32_a_t*)k[j]);
589  }
590  r = 0;
591  t = 0;
592  /* Copy values into round key array. */
593  for (j = 0; (j < KC) && (r < rounds + 1); )
594  {
595  for (; (j < KC) && (t < 4); j++, t++)
596  {
597  *((u32_a_t*)W[r][t]) = *((u32_a_t*)tk[j]);
598  }
599  if (t == 4)
600  {
601  r++;
602  t = 0;
603  }
604  }
605 
606  while (r < rounds + 1)
607  {
608  /* While not enough round key material calculated calculate
609  new values. */
610  tk[0][0] ^= S[tk[KC-1][1]];
611  tk[0][1] ^= S[tk[KC-1][2]];
612  tk[0][2] ^= S[tk[KC-1][3]];
613  tk[0][3] ^= S[tk[KC-1][0]];
614  tk[0][0] ^= rcon[rconpointer++];
615 
616  if (KC != 8)
617  {
618  for (j = 1; j < KC; j++)
619  {
620  *((u32_a_t*)tk[j]) ^= *((u32_a_t*)tk[j-1]);
621  }
622  }
623  else
624  {
625  for (j = 1; j < KC/2; j++)
626  {
627  *((u32_a_t*)tk[j]) ^= *((u32_a_t*)tk[j-1]);
628  }
629  tk[KC/2][0] ^= S[tk[KC/2 - 1][0]];
630  tk[KC/2][1] ^= S[tk[KC/2 - 1][1]];
631  tk[KC/2][2] ^= S[tk[KC/2 - 1][2]];
632  tk[KC/2][3] ^= S[tk[KC/2 - 1][3]];
633  for (j = KC/2 + 1; j < KC; j++)
634  {
635  *((u32_a_t*)tk[j]) ^= *((u32_a_t*)tk[j-1]);
636  }
637  }
638 
639  /* Copy values into round key array. */
640  for (j = 0; (j < KC) && (r < rounds + 1); )
641  {
642  for (; (j < KC) && (t < 4); j++, t++)
643  {
644  *((u32_a_t*)W[r][t]) = *((u32_a_t*)tk[j]);
645  }
646  if (t == 4)
647  {
648  r++;
649  t = 0;
650  }
651  }
652  }
653 #undef W
654 #undef tk
655 #undef k
656  wipememory(&tk, sizeof(tk));
657  wipememory(&t, sizeof(t));
658  }
659 
660  return 0;

771 /* Encrypt one block. A and B need to be aligned on a 4 byte
772  boundary. A and B may be the same. */
773 static void
774 do_encrypt_aligned (const RIJNDAEL_context *ctx,
775  unsigned char *b, const unsigned char *a)
776 {
777 #ifdef USE_AMD64_ASM
778  _gcry_aes_amd64_encrypt_block(ctx->keyschenc, b, a, ctx->rounds);
779 #elif defined(USE_ARM_ASM)
780  _gcry_aes_arm_encrypt_block(ctx->keyschenc, b, a, ctx->rounds);
781 #else
782 #define rk (ctx->keyschenc)
783  int rounds = ctx->rounds;
784  int r;
785  union
786  {
787  u32 tempu32[4]; /* Force correct alignment. */
788  byte temp[4][4];
789  } u;
790 
791  *((u32_a_t*)u.temp[0]) = *((u32_a_t*)(a )) ^ *((u32_a_t*)rk[0][0]);
792  *((u32_a_t*)u.temp[1]) = *((u32_a_t*)(a+ 4)) ^ *((u32_a_t*)rk[0][1]);
793  *((u32_a_t*)u.temp[2]) = *((u32_a_t*)(a+ 8)) ^ *((u32_a_t*)rk[0][2]);
794  *((u32_a_t*)u.temp[3]) = *((u32_a_t*)(a+12)) ^ *((u32_a_t*)rk[0][3]);

797  ^ *((u32_a_t*)T3[u.temp[2][2]])
798  ^ *((u32_a_t*)T4[u.temp[3][3]]));
799  *((u32_a_t*)(b + 4)) = (*((u32_a_t*)T1[u.temp[1][0]])
800  ^ *((u32_a_t*)T2[u.temp[2][1]])
801  ^ *((u32_a_t*)T3[u.temp[3][2]])
802  ^ *((u32_a_t*)T4[u.temp[0][3]]));
803  *((u32_a_t*)(b + 8)) = (*((u32_a_t*)T1[u.temp[2][0]])

805  ^ *((u32_a_t*)T3[u.temp[0][2]])
806  ^ *((u32_a_t*)T4[u.temp[1][3]]));

808  ^ *((u32_a_t*)T2[u.temp[0][1]])
809  ^ *((u32_a_t*)T3[u.temp[1][2]])
810  ^ *((u32_a_t*)T4[u.temp[2][3]]));
811 
812  for (r = 1; r < rounds-1; r++)
813  {
814  *((u32_a_t*)u.temp[0]) = *((u32_a_t*)(b )) ^ *((u32_a_t*)rk[r][0]);
815  *((u32_a_t*)u.temp[1]) = *((u32_a_t*)(b+ 4)) ^ *((u32_a_t*)rk[r][1]);
816  *((u32_a_t*)u.temp[2]) = *((u32_a_t*)(b+ 8)) ^ *((u32_a_t*)rk[r][2]);
817  *((u32_a_t*)u.temp[3]) = *((u32_a_t*)(b+12)) ^ *((u32_a_t*)rk[r][3]);
818 
819  *((u32_a_t*)(b )) = (*((u32_a_t*)T1[u.temp[0][0]])
820  ^ *((u32_a_t*)T2[u.temp[1][1]])
821  ^ *((u32_a_t*)T3[u.temp[2][2]])
822  ^ *((u32_a_t*)T4[u.temp[3][3]]));
823  *((u32_a_t*)(b + 4)) = (*((u32_a_t*)T1[u.temp[1][0]])
824  ^ *((u32_a_t*)T2[u.temp[2][1]])
825  ^ *((u32_a_t*)T3[u.temp[3][2]])
826  ^ *((u32_a_t*)T4[u.temp[0][3]]));
827  *((u32_a_t*)(b + 8)) = (*((u32_a_t*)T1[u.temp[2][0]])
828  ^ *((u32_a_t*)T2[u.temp[3][1]])
829  ^ *((u32_a_t*)T3[u.temp[0][2]])
830  ^ *((u32_a_t*)T4[u.temp[1][3]]));
831  *((u32_a_t*)(b +12)) = (*((u32_a_t*)T1[u.temp[3][0]])
832  ^ *((u32_a_t*)T2[u.temp[0][1]])
833  ^ *((u32_a_t*)T3[u.temp[1][2]])
834  ^ *((u32_a_t*)T4[u.temp[2][3]]));
835  }
836 
837  /* Last round is special. */
838  *((u32_a_t*)u.temp[0]) = *((u32_a_t*)(b )) ^ *((u32_a_t*)rk[rounds-1][0]);
839  *((u32_a_t*)u.temp[1]) = *((u32_a_t*)(b+ 4)) ^ *((u32_a_t*)rk[rounds-1][1]);
840  *((u32_a_t*)u.temp[2]) = *((u32_a_t*)(b+ 8)) ^ *((u32_a_t*)rk[rounds-1][2]);
841  *((u32_a_t*)u.temp[3]) = *((u32_a_t*)(b+12)) ^ *((u32_a_t*)rk[rounds-1][3]);
842  b[ 0] = T1[u.temp[0][0]][1];
843  b[ 1] = T1[u.temp[1][1]][1];
844  b[ 2] = T1[u.temp[2][2]][1];
845  b[ 3] = T1[u.temp[3][3]][1];
846  b[ 4] = T1[u.temp[1][0]][1];
847  b[ 5] = T1[u.temp[2][1]][1];
848  b[ 6] = T1[u.temp[3][2]][1];
849  b[ 7] = T1[u.temp[0][3]][1];
850  b[ 8] = T1[u.temp[2][0]][1];
851  b[ 9] = T1[u.temp[3][1]][1];
852  b[10] = T1[u.temp[0][2]][1];
853  b[11] = T1[u.temp[1][3]][1];
854  b[12] = T1[u.temp[3][0]][1];
855  b[13] = T1[u.temp[0][1]][1];
856  b[14] = T1[u.temp[1][2]][1];
