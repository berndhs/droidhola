232 /* This function is used for 1 byte encodings to insert any required
233  quoting. It does not do the quoting for a space or hash mark at
234  the beginning of a string or a space as the last character of a
235  string. It will do steps of SKIP+1 characters, assuming that these
236  SKIP characters are null octets. */
237 static void
238 append_quoted (struct stringbuf *sb, const unsigned char *value, size_t length,
239  int skip)
240 {
241  unsigned char tmp[4];
242  const unsigned char *s = value;
243  size_t n = 0;
244 
245  for (;;)
246  {
247  for (value = s; n+skip < length; n++, s++)
248  {
249  s += skip;
250  n += skip;
251  if (*s < ' ' || *s > 126 || strchr (",+\"\\<>;", *s) )
252  break;
253  }
254 
255  if (s != value)
256  put_stringbuf_mem_skip (sb, value, s-value, skip);
257  if (n+skip >= length)
258  return; /* ready */
259  s += skip;
260  n += skip;
261  if ( *s < ' ' || *s > 126 )
262  {
263  snprintf (tmp, sizeof tmp, "\\%02X", *s);
264  put_stringbuf_mem (sb, tmp, 3);
265  }
266  else
267  {
268  tmp[0] = '\\';
269  tmp[1] = *s;
270  put_stringbuf_mem (sb, tmp, 2);
271  }
272  n++; s++;
273  }
274 }

277 /* Append VALUE of LENGTH and TYPE to SB. Do the required quoting. */
278 static void
279 append_utf8_value (const unsigned char *value, size_t length,
280  struct stringbuf *sb)
281 {
282  unsigned char tmp[6];
283  const unsigned char *s;
284  size_t n;
285  int i, nmore;
286 
287  if (length && (*value == ' ' || *value == '#'))
288  {
289  tmp[0] = '\\';
290  tmp[1] = *value;
291  put_stringbuf_mem (sb, tmp, 2);
292  value++;
293  length--;
294  }
295  if (length && value[length-1] == ' ')
296  {
297  tmp[0] = '\\';
298  tmp[1] = ' ';
299  put_stringbuf_mem (sb, tmp, 2);
300  length--;
301  }
302 
303  for (s=value, n=0;;)
304  {
305  for (value = s; n < length && !(*s & 0x80); n++, s++)
306  ;
307  if (s != value)
308  append_quoted (sb, value, s-value, 0);
309  if (n==length)
310  return; /* ready */
311  if (!(*s & 0x80))
312  nmore = 0; /* Not expected here: high bit not set. */
313  else if ( (*s & 0xe0) == 0xc0 ) /* 110x xxxx */
314  nmore = 1;
315  else if ( (*s & 0xf0) == 0xe0 ) /* 1110 xxxx */
316  nmore = 2;
317  else if ( (*s & 0xf8) == 0xf0 ) /* 1111 0xxx */
318  nmore = 3;
319  else if ( (*s & 0xfc) == 0xf8 ) /* 1111 10xx */
320  nmore = 4;
321  else if ( (*s & 0xfe) == 0xfc ) /* 1111 110x */
322  nmore = 5;
323  else /* Invalid encoding */
324  nmore = 0;
325 
326  if (!nmore)
327  {
328  /* Encoding error: We quote the bad byte. */
329  snprintf (tmp, sizeof tmp, "\\%02X", *s);
330  put_stringbuf_mem (sb, tmp, 3);
331  s++; n++;
332  }
333  else
334  {
335  tmp[0] = *s++; n++;
336  for (i=1; n < length && i <= nmore; i++)
337  {
338  if ( (*s & 0xc0) != 0x80)
339  break; /* Invalid encoding - let the next cycle detect this. */
340  tmp[i] = *s++;
341  n++;
342  }
343  put_stringbuf_mem (sb, tmp, i);
344  }
345  }
346 }

393 append_ucs4_value (const unsigned char *value, size_t length,
394  struct stringbuf *sb)

398  size_t n;
399  unsigned int c;
400  int i;
401

403  && (value[3] == ' ' || value[3] == '#'))
404  {
405  tmp[0] = '\\';
406  tmp[1] = *value;
407  put_stringbuf_mem (sb, tmp, 2);
408  value += 4;
409  length -= 4;
410  }
411  if (length>3 && !value[0] && !value[1] && !value[2] && value[3] == ' ')
412  {
413  tmp[0] = '\\';
414  tmp[1] = ' ';
415  put_stringbuf_mem (sb, tmp, 2);
416  length -= 4;
417  }
418 
419  for (s=value, n=0;;)
420  {
421  for (value = s; n+3 < length

423  ;

426  if (n>=length)

431  put_stringbuf_mem (sb, "\xff", 1);
432  return;

435  c |= *s++ << 16;
436  c |= *s++ << 8;
437  c |= *s++;
438  n += 4;
439  i=0;

441  {

446  {

456  tmp[i++] = 0x80 | ( c & 0x3f);
457  }
458  else if (c < (1<<26))
459  {

462  tmp[i++] = 0x80 | ((c >> 12) & 0x3f);

541 /* Append attribute and value. ROOT is the sequence */
542 static gpg_error_t
543 append_atv (const unsigned char *image, AsnNode root, struct stringbuf *sb)
544 {
545  AsnNode node = root->down;
546  const char *name;
547  int use_hex = 0;
548  int i;
549 
550  if (!node || node->type != TYPE_OBJECT_ID)
551  return gpg_error (GPG_ERR_UNEXPECTED_TAG);
552  if (node->off == -1)
553  return gpg_error (GPG_ERR_NO_VALUE); /* Hmmm, this might lead to misunderstandings */
554 
555  name = NULL;
556  for (i=0; oid_name_tbl[i].name; i++)
557  {
558  if (oid_name_tbl[i].source == 1
559  && node->len == oid_name_tbl[i].oidlen
560  && !memcmp (image+node->off+node->nhdr,
561  oid_name_tbl[i].oid, node->len))
562  {
563  name = oid_name_tbl[i].name;
564  break;
565  }
566  }
567  if (name)
568  put_stringbuf (sb, name);
569  else
570  { /* No name for the OID in the table; at least not DER encoded.
571  Now convert the OID to a string, try to find it in the table
572  again and use the string as last resort. */
573  char *p;
574 
575  p = ksba_oid_to_str (image+node->off+node->nhdr, node->len);
576  if (!p)
577  return gpg_error (GPG_ERR_ENOMEM);
578 
579  for (i=0; *p && oid_name_tbl[i].name; i++)
580  {
581  if (oid_name_tbl[i].source == 1
582  && !strcmp (p, oid_name_tbl[i].oidstr))
583  {
584  name = oid_name_tbl[i].name;
585  break;
586  }
587  }
588  if (name)
589  put_stringbuf (sb, name);
590  else
591  {
592  put_stringbuf (sb, p);
593  use_hex = 1;
594  }
595  xfree (p);
596  }
597  put_stringbuf (sb, "=");
598  node = node->right;
599  if (!node || node->off == -1)
600  return gpg_error (GPG_ERR_NO_VALUE);
601 
602  switch (use_hex? 0 : node->type)
603  {
604  case TYPE_UTF8_STRING:
605  append_utf8_value (image+node->off+node->nhdr, node->len, sb);
606  break;
607  case TYPE_PRINTABLE_STRING:
608  case TYPE_IA5_STRING:
609  /* we assume that wrong encodings are latin-1 */
610  case TYPE_TELETEX_STRING: /* Not correct, but mostly used as latin-1 */
611  append_latin1_value (image+node->off+node->nhdr, node->len, sb);
612  break;
613 
614  case TYPE_UNIVERSAL_STRING:
615  append_ucs4_value (image+node->off+node->nhdr, node->len, sb);
616  break;
617 
618  case TYPE_BMP_STRING:
619  append_ucs2_value (image+node->off+node->nhdr, node->len, sb);
620  break;
621 
622  case 0: /* forced usage of hex */
623  default:
624  put_stringbuf (sb, "#");
625  for (i=0; i < node->len; i++)
626  {
627  char tmp[3];
628  snprintf (tmp, sizeof tmp, "%02X", image[node->off+node->nhdr+i]);
629  put_stringbuf (sb, tmp);
630  }
631  break;
632  }
633 
634  return 0;
635 }

637 static gpg_error_t
638 dn_to_str (const unsigned char *image, AsnNode root, struct stringbuf *sb)
639 {
640  gpg_error_t err;
641  AsnNode nset;
642 
643  if (!root )
644  return 0; /* empty DN */
645  nset = root->down;
646  if (!nset)
647  return 0; /* consider this as empty */
648  if (nset->type != TYPE_SET_OF)
649  return gpg_error (GPG_ERR_UNEXPECTED_TAG);
650 
651  /* output in reverse order */
652  while (nset->right)
653  nset = nset->right;
654 
655  for (;;)
656  {
657  AsnNode nseq;
658 
659  if (nset->type != TYPE_SET_OF)
660  return gpg_error (GPG_ERR_UNEXPECTED_TAG);
661  for (nseq = nset->down; nseq; nseq = nseq->right)
662  {
663  if (nseq->type != TYPE_SEQUENCE)
