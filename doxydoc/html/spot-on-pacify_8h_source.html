39  spoton_pacify(const std::string &passphrase)
40  {
41  size_t n = passphrase.length();
42 
43  // Convert from UTF-8 to UTF-32.
44  // Note this is not an attack-safe implementation,
45  // this implementation allows characters that are
46  // encoded with an excessive number of bytes.
47 
48  // Count the number of characters.
49  // Byte Datum
50  // 0....... Single-byte character (Count these.)
51  // 11...... Initial byte in multibyte character (Count these.)
52  // 10...... Non-initial byte in multibyte character (Do not count these.)
53 
54  m_passphrase = 0;
55  m_passphrase_length = 0;
56 
57  for(size_t i = 0; i < n; i++)
58  if((static_cast<int> (passphrase.at(i)) & 0xc0) != 0x80)
59  m_passphrase_length += 1;
60 
61  if(m_passphrase_length > 0)
62  m_passphrase = new long int[m_passphrase_length];
63 
64  // Convert.
65  // 0.......
66  // 110..... 10......
67  // 1110.... 10...... 10......
68  // 11110... 10...... 10...... 10......
69  // 111110.. 10...... 10...... 10...... 10......
70  // 1111110. 10...... 10...... 10...... 10...... 10......
71 
72  for(size_t i = 0, j = 0; i < n; j++)
73  {
74  int c = static_cast<int> (passphrase.at(i++));
75 
76  if((c & 0x80) == 0)
77  {
78  // Single-byte character.
79 
80  if(m_passphrase)
81  m_passphrase[j] = static_cast<long int> (c);
82  }
83  else if((c & 0xc0) == 0xc0)
84  {
85  // Initial byte in multibyte-character.
86 
87  int b = 0;
88 
89  // Get width of character.
90  // 110..... 2-byte character 11-bit character
91  // 1110.... 3-byte character 16-bit character
92  // 11110... 4-byte character 21-bit character (New UTF-8 Limit)
93  // 111110.. 5-byte character 26-bit character
94  // 1111110. 6-byte character 31-bit character (Old UTF-8 Limit)
95 
96  while(c & 0x80)
97  b++, c <<= 1;
98 
99  if(b > 6)
100  {
101  if(m_passphrase)
102  m_passphrase[j] = 0; /* XXX invalid input. */
103 
104  break;
105  }
106 
107  // Get bits (represented by dots in the table above)
108  // stored in the start byte.
109 
110  c = (c & 0xff) >> b;
111 
112  if(m_passphrase)
113  m_passphrase[j] = static_cast<long int> (c);
114 
115  // We have read one of the bytes.
116 
117  b--;
118 
119  // Get bits stored in the continuation bytes.
120  // 10......
121 
122  while(b--)
123  {
124  if(i == n)
125  {
126  // Premature end of string.
127 
128  if(m_passphrase)
129  m_passphrase[j] = 0; /* XXX invalid input. */
130 
131  break;
132  }
133 
134  c = static_cast<int> (passphrase.at(i++));
135 
136  if((c & 0xc0) != 0x80)
137  {
138  // Premature end of multibyte-character byte sequence.
139 
140  if(m_passphrase)
141  m_passphrase[j] = 0; /* XXX invalid input. */
142 
143  i--;
144  break;
145  }
146  else
147  {
148  // Store the six lowest bits.
149 
150  if(m_passphrase)
151  {
152  m_passphrase[j] <<= 6;
153  m_passphrase[j] |= static_cast<long int> (c & 0x3f);
154  }
155  }
156  }
157  }
158  else
159  // Non-initial byte in multibyte-character at beginning of character.
160  // (Unexpected continuation byte.)
161 
162  j--; /* XXX invalid input. */
163  }
164  }

270  double char_class(const long int c) const
271  {
272  if(static_cast<long int> ('0') <= c && c <= static_cast<long int> ('9'))
273  return 1.0;
274  else if(static_cast<long int> ('a') <= c &&
275  c <= static_cast<long int> ('z'))
276  return 2.0;
277  else if(static_cast<long int> ('A') <= c &&
278  c <= static_cast<long int> ('Z'))
279  return 2.5;
280  else if(c < (1 << 7))
281  return 3.0;
282  else if(c < (1 << 8))
283  return 3.5;
284  else if(c < (1 << 10))
285  return 4.0;
286  else if(c < (1 << 14))
287  return 5.0;
288  else if(c < (1 << 16))
289  return 6.0;
290  else if(c < (1 << 18))
291  return 7.0;
292  else if(c < (1 << 22))
