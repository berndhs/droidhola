403 static int

416 /* Read COUNT bytes into buffer. BUFFER may be NULL to skip over
417  COUNT bytes. Return 0 on success or -1 on error. */
418 static int
419 read_buffer (ksba_reader_t reader, char *buffer, size_t count)
420 {
421  size_t nread;
422 
423  if (buffer)
424  {
425  while (count)
426  {
427  if (ksba_reader_read (reader, buffer, count, &nread))
428  return -1;
429  buffer += nread;
430  count -= nread;
431  }
432  }
433  else
434  {
435  char dummy[256];
436  size_t n;
437 
438  while (count)
439  {
440  n = count > DIM(dummy) ? DIM(dummy): count;
441  if (ksba_reader_read (reader, dummy, n, &nread))
442  return -1;
443  count -= nread;
444  }
445  }
446  return 0;
447 }

449 /* Return 0 for no match, 1 for a match and 2 for an ANY match of an
450  constructed type */
451 static int
452 cmp_tag (AsnNode node, const struct tag_info *ti)
453 {
454  if (node->flags.class != ti->class)
455  {
456  if (node->flags.class == CLASS_UNIVERSAL && node->type == TYPE_ANY)
457  return ti->is_constructed? 2:1;
458  return 0;
459  }
460  if (node->type == TYPE_TAG)
461  {
462  return_val_if_fail (node->valuetype == VALTYPE_ULONG, 0);
463  return node->value.v_ulong == ti->tag;
464  }
465  if (node->type == ti->tag)
466  return 1;
467  if (ti->class == CLASS_UNIVERSAL)
468  {
469  if (node->type == TYPE_SEQUENCE_OF && ti->tag == TYPE_SEQUENCE)
470  return 1;
471  if (node->type == TYPE_SET_OF && ti->tag == TYPE_SET)
472  return 1;
473  if (node->type == TYPE_ANY)
474  return _ksba_asn_is_primitive (ti->tag)? 1:2;
475  }
476 
477  return 0;
478 }

480 /* Find the node in the tree ROOT corresponding to TI and return that
481  node. Returns NULL if the node was not found */
482 static AsnNode
483 find_anchor_node (AsnNode root, const struct tag_info *ti)
484 {
485  AsnNode node = root;
486 
487  while (node)
488  {
489  if (cmp_tag (node, ti))
490  {
491  return node; /* found */
492  }
493 
494  if (node->down)
495  node = node->down;
496  else if (node == root)
497  return NULL; /* not found */
498  else if (node->right)
499  node = node->right;
500  else
501  { /* go up and right */
502  do
503  {
504  while (node->left && node->left->right == node)
505  node = node->left;
506  node = node->left;
507 
508  if (!node || node == root)
509  return NULL; /* back at the root -> not found */
510  }
511  while (!node->right);
512  node = node->right;
513  }
514  }
515 
516  return NULL;
517 }

519 static int
520 match_der (AsnNode root, const struct tag_info *ti,
521  DECODER_STATE ds, AsnNode *retnode, int debug)
522 {
523  int rc;
524  AsnNode node;
525 
526  *retnode = NULL;
527  node = ds->cur.node;
528  if (!node)
529  {
530  if (debug)
531  fputs (" looking for anchor\n", stderr);
532  node = find_anchor_node (root, ti);
533  if (!node)
534  fputs (" anchor node not found\n", stderr);
535  }
536  else if (ds->cur.again)
537  {
538  if (debug)
539  fputs (" doing last again\n", stderr);
540  ds->cur.again = 0;
541  }
542  else if (_ksba_asn_is_primitive (node->type) || node->type == TYPE_ANY
543  || node->type == TYPE_SIZE || node->type == TYPE_DEFAULT )
544  {
545  if (debug)
546  fputs (" primitive type - get next\n", stderr);
547  if (node->right)
548  node = node->right;
549  else if (!node->flags.in_choice)
550  {
551  if (node->flags.is_implicit)
552  {
553  if (debug)
554  fputs (" node was implicit - advancing\n", stderr);
555  while (node->left && node->left->right == node)
556  node = node->left;
557  node = node->left; /* this is the up pointer */
558  if (node)
559  node = node->right;
560  }
561  else
562  node = NULL;
563  }
564  else /* in choice */
565  {
566  if (debug)
567  fputs (" going up after choice - get next\n", stderr);
568  while (node->left && node->left->right == node)
569  node = node->left;
570  node = node->left; /* this is the up pointer */
571  if (node)
572  node = node->right;
573  }
574  }
575  else if (node->type == TYPE_SEQUENCE_OF || node->type == TYPE_SET_OF)
576  {
577  if (debug)
578  {
579  fprintf (stderr, " prepare for seq/set_of (%d %d) ",
580  ds->cur.length, ds->cur.nread);
581  fprintf (stderr, " cur: ("); _ksba_asn_node_dump (node, stderr);
582  fprintf (stderr, ")\n");
583  if (ds->cur.node->flags.in_array)
584  fputs (" This is in an array!\n", stderr);
585  if (ds->cur.went_up)
586  fputs (" And we are going up!\n", stderr);
587  }
588  if ((ds->cur.went_up && !ds->cur.node->flags.in_array) ||
589  (ds->idx && ds->cur.nread >= ds->stack[ds->idx-1].length))
590  {
591  if (debug)
592  fprintf (stderr, " advancing\n");
593  if (node->right)
594  node = node->right;
595  else
596  {
597  for (;;)
598  {
599  while (node->left && node->left->right == node)
600  node = node->left;
601  node = node->left; /* this is the up pointer */
602  if (!node)
603  break;
604  if (node->right)
605  {
606  node = node->right;
607  break;
608  }
609  }
610  }
611  }
612  else if (ds->cur.node->flags.in_array
613  && ds->cur.went_up)
614  {
615  if (debug)
616  fputs (" Reiterating\n", stderr);
617  node = _ksba_asn_insert_copy (node);
618  if (node)
619  prepare_copied_tree (node);
620  }
621  else
622  node = node->down;
623  }
624  else /* constructed */
625  {
626  if (debug)
627  {
628  fprintf (stderr, " prepare for constructed (%d %d) ",
629  ds->cur.length, ds->cur.nread);
630  fprintf (stderr, " cur: ("); _ksba_asn_node_dump (node, stderr);
631  fprintf (stderr, ")\n");
632  if (ds->cur.node->flags.in_array)
633  fputs (" This is in an array!\n", stderr);
634  if (ds->cur.went_up)
635  fputs (" And we are going up!\n", stderr);
636  }
637  ds->cur.in_seq_of = 0;
638 
639  if (ds->cur.node->flags.in_array
640  && ds->cur.went_up)
641  {
642  if (debug)
643  fputs (" Reiterating this\n", stderr);
644  node = _ksba_asn_insert_copy (node);
645  if (node)
646  prepare_copied_tree (node);
647  }
648  else if (ds->cur.went_up || ds->cur.next_tag || ds->cur.node->flags.skip_this)
649  {
650  if (node->right)
651  node = node->right;
652  else
653  {
654  for (;;)
655  {
656  while (node->left && node->left->right == node)
657  node = node->left;
658  node = node->left; /* this is the up pointer */
659  if (!node)
660  break;
661  if (node->right)
662  {
663  node = node->right;
664  break;
665  }
666  }
667  }
668  }
669  else
670  node = node->down;
671 
672  }
673  if (!node)
674  return -1;
675  ds->cur.node = node;
676  ds->cur.went_up = 0;
677  ds->cur.next_tag = 0;
678 
679  if (debug)
680  {
681  fprintf (stderr, " Expect ("); _ksba_asn_node_dump (node,stderr); fprintf (stderr, ")\n");
682  }
683 
684  if (node->flags.skip_this)
685  {
686  if (debug)
687  fprintf (stderr, " skipping this\n");
688  return 1;
689  }
690 
691  if (node->type == TYPE_SIZE)
692  {
693  if (debug)
694  fprintf (stderr, " skipping size tag\n");
695  return 1;
696  }
697  if (node->type == TYPE_DEFAULT)
698  {
699  if (debug)
700  fprintf (stderr, " skipping default tag\n");
701  return 1;
702  }
703 
704  if (node->flags.is_implicit)
705  {
706  if (debug)
707  fprintf (stderr, " dummy accept for implicit tag\n");
708  return 1; /* again */
709  }
710 
711  if ( (rc=cmp_tag (node, ti)))
712  {
713  *retnode = node;
714  return rc==2? 4:3;
715  }
716 
717  if (node->type == TYPE_CHOICE)
718  {
719  if (debug)
720  fprintf (stderr, " testing choice...\n");
721  for (node = node->down; node; node = node->right)
722  {
723  if (debug)
724  {
725  fprintf (stderr, " %s (", node->flags.skip_this? "skip":" cmp");
726  _ksba_asn_node_dump (node, stderr);
727  fprintf (stderr, ")\n");
728  }
729 
730  if (!node->flags.skip_this && cmp_tag (node, ti) == 1)
731  {
732  if (debug)
733  {
734  fprintf (stderr, " choice match <"); dump_tlv (ti, stderr);
735  fprintf (stderr, ">\n");
736  }
737  /* mark the remaining as done */
738  for (node=node->right; node; node = node->right)
739  node->flags.skip_this = 1;
740  return 1;
741  }
742  node->flags.skip_this = 1;
743 
744  }
745  node = ds->cur.node; /* reset */
746  }
747 
748  if (node->flags.in_choice)
749  {
750  if (debug)
751  fprintf (stderr, " skipping non matching choice\n");
752  return 1;
753  }
754 
755  if (node->flags.is_optional)
756  {
757  if (debug)
758  fprintf (stderr, " skipping optional element\n");
759  if (node->type == TYPE_TAG)
760  ds->cur.next_tag = 1;
761  return 1;
762  }
763 
764  if (node->flags.has_default)
765  {
766  if (debug)
767  fprintf (stderr, " use default value\n");
768  if (node->type == TYPE_TAG)
769  ds->cur.next_tag = 1;
770  *retnode = node;
771  return 2;
772  }
773 
774  return -1;
775 }

802 static gpg_error_t
803 decoder_next (BerDecoder d)
804 {
805  struct tag_info ti;
806  AsnNode node = NULL;
807  gpg_error_t err;
808  DECODER_STATE ds = d->ds;
809  int debug = d->debug;
810 
811  if (d->ignore_garbage && d->fast_stop)
812  {
813  /* I am not anymore sure why we have this ignore_garbage
814  machinery: The whole decoder needs and overhaul; it seems not
815  to honor the length specification and runs for longer than
816  expected.
817 
818  This here is another hack to not eat up an end tag - this is
819  required in in some cases and in theory should be used always
820  but we want to avoid any regression, thus this flag. */
821  return gpg_error (GPG_ERR_EOF);
822  }
823 
824  err = _ksba_ber_read_tl (d->reader, &ti);
825  if (err)
826  {
827  if (debug)
828  fprintf (stderr, "ber_read_tl error: %s (%s)\n",
829  gpg_strerror (err), ti.err_string? ti.err_string:"");
830  /* This is our actual hack to cope with some trailing garbage:
831  Only if we get an premature EOF and we know that we have read
832  the complete certificate we change the error to EOF. This
833  won't help with all kinds of garbage but it fixes the case
834  where just one byte is appended. This is for example the
835  case with current Siemens certificates. This approach seems
836  to be the least intrusive one. */
837  if (gpg_err_code (err) == GPG_ERR_BAD_BER
838  && d->ignore_garbage
839  && ti.err_string && !strcmp (ti.err_string, "premature EOF"))
840  err = gpg_error (GPG_ERR_EOF);
841  return err;
842  }
843 
844  if (debug)
845  {
846  fprintf (stderr, "ReadTLV <");
847  dump_tlv (&ti, stderr);
848  fprintf (stderr, ">\n");
849  }
850 
851  /* Check whether the trailing garbage hack is required. */
852  if (!d->first_tag_seen)
853  {
854  d->first_tag_seen = 1;
855  if (ti.tag == TYPE_SEQUENCE && ti.length && !ti.ndef)
856  d->outer_sequence_length = ti.length;
857  }
858 
859  /* Store stuff in the image buffer. */
860  if (d->use_image)
861  {
862  if (!d->image.buf)
863  {
864  /* We need some extra bytes to store the stuff we read ahead
865  * at the end of the module which is later pushed back. We
866  * also clear the buffer because there is no guarantee that
867  * we will copy data to all bytes of the buffer: A broken
868  * ASN.1 encoding may thus lead to access of uninitialized
869  * data even if we make sure that that access is not our of
870  * bounds. */
871  d->image.used = 0;
872  d->image.length = ti.length + 100;
873  if (d->image.length < ti.length)
874  return gpg_error (GPG_ERR_BAD_BER);
875  if (d->image.length > MAX_IMAGE_LENGTH)
876  return gpg_error (GPG_ERR_TOO_LARGE);
877  d->image.buf = xtrycalloc (1, d->image.length);
878  if (!d->image.buf)
879  return gpg_error (GPG_ERR_ENOMEM);
880  }
881 
882  if (sum_a1_a2_ge_b (ti.nhdr, d->image.used, d->image.length))
883  return set_error (d, NULL, "image buffer too short to store the tag");
884 
885  memcpy (d->image.buf + d->image.used, ti.buf, ti.nhdr);
886  d->image.used += ti.nhdr;
887  }
888 
889 
890  if (!d->bypass)
891  {
892  int again, endtag;
893 
894  do
895  {
896  again = endtag = 0;
897  switch ( ds->cur.in_any? 4
898  : (ti.class == CLASS_UNIVERSAL && !ti.tag)? (endtag=1,5)
899  : match_der (d->root, &ti, ds, &node, debug))
900  {
901  case -1:
902  if (debug)
903  {
904  fprintf (stderr, " FAIL <");
905  dump_tlv (&ti, stderr);
906  fprintf (stderr, ">\n");
907  }
908  if (d->honor_module_end)
909  {
910  /* We must push back the stuff we already read */
911  ksba_reader_unread (d->reader, ti.buf, ti.nhdr);
912  return gpg_error (GPG_ERR_EOF);
913  }
914  else
915  d->bypass = 1;
916  break;
917  case 0:
918  if (debug)
919  fputs (" End of description\n", stderr);
920  d->bypass = 1;
921  break;
922  case 1: /* again */
923  if (debug)
924  fprintf (stderr, " Again\n");
925  again = 1;
926  break;
927  case 2: /* Use default value + again */
928  if (debug)
929  fprintf (stderr, " Using default\n");
930  again = 1;
931  break;
932  case 4: /* Match of ANY on a constructed type */
933  if (debug)
934  fprintf (stderr, " ANY");
935  ds->cur.in_any = 1;
936  case 3: /* match */
937  case 5: /* end tag */
938  if (debug)
939  {
940  fprintf (stderr, " Match <");
941  dump_tlv (&ti, stderr);
942  fprintf (stderr, ">\n");
943  }
944  /* Increment by the header length */
945  ds->cur.nread += ti.nhdr;
946 
947  if (!ti.is_constructed)
948  ds->cur.nread += ti.length;
949 
950  ds->cur.went_up = 0;
951  do
952  {
953  if (debug)
954  fprintf (stderr, " (length %d nread %d) %s\n",
955  ds->idx? ds->stack[ds->idx-1].length:-1,
956  ds->cur.nread,
957  ti.is_constructed? "con":"pri");
958  if (d->outer_sequence_length
959  && ds->idx == 1
960  && ds->cur.nread == d->outer_sequence_length)
961  {
962  if (debug)
963  fprintf (stderr, " Need to stop now\n");
964  d->ignore_garbage = 1;
965  }
966 
967  if ( ds->idx
968  && !ds->stack[ds->idx-1].ndef_length
969  && (ds->cur.nread
970  > ds->stack[ds->idx-1].length))
971  {
972  fprintf (stderr, "ksba: ERROR: object length field "
973  "%d octects too large\n",
974  ds->cur.nread - ds->cur.length);
975  ds->cur.nread = ds->cur.length;
976  }
977  if ( ds->idx
978  && (endtag
979  || (!ds->stack[ds->idx-1].ndef_length
980  && (ds->cur.nread
981  >= ds->stack[ds->idx-1].length))))
982  {
983  int n = ds->cur.nread;
984  err = pop_decoder_state (ds);
985  if (err)
986  return err;
987  ds->cur.nread += n;
988  ds->cur.went_up++;
989  }
990  endtag = 0;
991  }
992  while ( ds->idx
993  && !ds->stack[ds->idx-1].ndef_length
994  && (ds->cur.nread
995  >= ds->stack[ds->idx-1].length));
996 
997  if (ti.is_constructed && (ti.length || ti.ndef))
998  {
999  /* prepare for the next level */
1000  ds->cur.length = ti.length;
1001  ds->cur.ndef_length = ti.ndef;
1002  err = push_decoder_state (ds);
1003  if (err)
1004  return err;
1005  ds->cur.length = 0;
1006  ds->cur.ndef_length = 0;
1007  ds->cur.nread = 0;
1008  }
1009  if (debug)
1010  fprintf (stderr, " (length %d nread %d) end\n",
1011  ds->idx? ds->stack[ds->idx-1].length:-1,
1012  ds->cur.nread);
1013  break;
1014  default:
1015  never_reached ();
1016  abort ();
1017  break;
1018  }
1019  }
1020  while (again);
1021  }
1022 
1023  d->val.primitive = !ti.is_constructed;
1024  d->val.length = ti.length;
1025  d->val.nhdr = ti.nhdr;
1026  d->val.tag = ti.tag; /* kludge to fix TYPE_ANY probs */
1027  d->val.is_endtag = (ti.class == CLASS_UNIVERSAL && !ti.tag);
1028  d->val.node = d->bypass? NULL : node;
1029  if (debug)
1030  dump_decoder_state (ds);
1031 
1032  return 0;
1033 }
