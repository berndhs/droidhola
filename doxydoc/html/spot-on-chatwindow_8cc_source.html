209 void spoton_chatwindow::sendMessage(bool *ok)
210 {
211  QByteArray message;
212  QByteArray name;
213  QDateTime now(QDateTime::currentDateTime());
214  QSettings settings;
215  QString error("");
216  QString msg("");
217 
218  if(m_kernelSocket->state() != QAbstractSocket::ConnectedState)
219  {
220  error = tr("The interface is not connected to the kernel.");
221  goto done_label;
222  }
223  else if(!m_kernelSocket->isEncrypted())
224  {
225  error = tr("The connection to the kernel is not encrypted.");
226  goto done_label;
227  }
228  else if(ui.message->toPlainText().isEmpty())
229  {
230  error = tr("Please provide a real message.");
231  goto done_label;
232  }
233 
234  if(m_keyType == "chat")
235  name = spoton::instance() ? spoton::instance()->m_settings.
236  value("gui/nodeName", "unknown").toByteArray() : "unknown";
237  else
238  name = spoton::instance() ? spoton::instance()->m_settings.
239  value("gui/poptasticName", "unknown@unknown.org").toByteArray() :
240  "unknown@unknown.org";
241 
242  msg.append
243  (QString("[%1/%2/%3 %4:%5<font color=gray>:%6</font>] ").
244  arg(now.toString("MM")).
245  arg(now.toString("dd")).
246  arg(now.toString("yyyy")).
247  arg(now.toString("hh")).
248  arg(now.toString("mm")).
249  arg(now.toString("ss")));
250  msg.append(tr("<b>me:</b> "));
251 
252  if(settings.value("gui/enableChatEmoticons", false).toBool())
253  msg.append
254  (spoton::mapIconToEmoticon(ui.message->toPlainText()));
255  else
256  msg.append(ui.message->toPlainText());
257 
258  ui.messages->append(msg);
259  ui.messages->verticalScrollBar()->setValue
260  (ui.messages->verticalScrollBar()->maximum());
261 
262  if(spoton::instance())
263  {
264  spoton::instance()->ui().messages->append(msg);
265  spoton::instance()->ui().messages->verticalScrollBar()->setValue
266  (spoton::instance()->ui().messages->verticalScrollBar()->maximum());
267  }
268 
269  if(name.isEmpty())
270  {
271  if(m_keyType == "chat")
272  name = "unknown";
273  else
274  name = "unknown@unknown.org";
275  }
276 
277  if(spoton::instance())
278  {
279  if(!spoton::instance()->m_chatSequenceNumbers.contains(m_id))
280  spoton::instance()->m_chatSequenceNumbers[m_id] = 0;
281 
282  spoton::instance()->m_chatSequenceNumbers[m_id] += 1;
283  }
284 
285  if(m_keyType == "chat")
286  message.append("message_");
287  else
288  message.append("poptasticmessage_");
289 
290  message.append(QString("%1_").arg(m_id));
291  message.append(name.toBase64());
292  message.append("_");
293  message.append(ui.message->toPlainText().toUtf8().toBase64());
294  message.append("_");
295  message.append
296  (QByteArray::number(spoton::instance() ?
297  spoton::instance()->m_chatSequenceNumbers[m_id] :
298  1).toBase64());
299  message.append("_");
300  message.append(QDateTime::currentDateTime().toUTC().
301  toString("MMddyyyyhhmmss").toLatin1().toBase64());
302  message.append("\n");
303 
304  if(spoton::instance())
305  spoton::instance()->addMessageToReplayQueue(msg, message, m_publicKeyHash);
306 
307  if(m_kernelSocket->write(message.constData(), message.length()) !=
308  message.length())
309  {
310  error = tr("An error occurred while writing to the "
311  "kernel socket.");
312  spoton_misc::logError
313  (QString("spoton_chatwindow::slotSendMessage(): write() failure for "
314  "%1:%2.").
315  arg(m_kernelSocket->peerAddress().toString()).
316  arg(m_kernelSocket->peerPort()));
317  goto done_label;
318  }
319  else
320  emit messageSent();
321 
322  ui.message->clear();
323 
324  done_label:
325 
326 #if QT_VERSION >= 0x050000
327  if(error.isEmpty())
328  {
329  QMediaPlayer *player = findChild<QMediaPlayer *> ();
330 
331  if(player)
332  player->deleteLater();
333 
334  if(spoton::instance() ? spoton::instance()->m_settings.
335  value("gui/play_sounds", false).toBool() : false)
336  {
337  QFileInfo fileInfo;
338  QString str
339  (QDir::cleanPath(QCoreApplication::applicationDirPath() +
340  QDir::separator() + "Sounds" + QDir::separator() +
341  "send.wav"));
342 
343  fileInfo.setFile(str);
344 
345  if(!(!fileInfo.isReadable() || fileInfo.size() < 8192))
346  {
347  player = new (std::nothrow) QMediaPlayer
348  (this, QMediaPlayer::LowLatency);
349 
350  if(player)
351  {
352  player->setMedia(QUrl::fromLocalFile(str));
353  player->setVolume(100);
354  player->play();
355  }
356  else
357  QApplication::beep();
358  }
359  }
360  }
361 #else
362  if(error.isEmpty())
363  QApplication::beep();
364 #endif
365 
366  if(!error.isEmpty())
367  {
368  if(ok)
369  *ok = false;
370  else
371  QMessageBox::critical(this, tr("%1: Error").
372  arg(SPOTON_APPLICATION_NAME), error);
373  }
374 }

516 void spoton_chatwindow::slotShareStarBeam(void)
517 {
518  QString error("");
519  spoton_crypt *crypt = spoton::instance() ?
520  spoton::instance()->crypts().value("chat", 0) : 0;
521 
522  if(!crypt)
523  {
524  error = tr("Invalid spoton_crypt object. This is a fatal flaw.");
525  showError(error);
526  return;
527  }
528 
529  /*
530  ** Some of this logic is redundant. Please see sendMessage().
531  */
532 
533  if(m_kernelSocket->state() != QAbstractSocket::ConnectedState)
534  {
535  error = tr("The interface is not connected to the kernel.");
536  showError(error);
537  return;
538  }
539  else if(!m_kernelSocket->isEncrypted())
540  {
541  error = tr("The connection to the kernel is not encrypted.");
542  showError(error);
543  return;
544  }
545 
546  /*
547  ** Select a file.
548  */
549 
550  QFileDialog dialog(this);
551 
552  dialog.setWindowTitle(tr("%1: Select StarBeam Transmit File").
553  arg(SPOTON_APPLICATION_NAME));
554  dialog.setFileMode(QFileDialog::ExistingFile);
555  dialog.setDirectory(QDir::homePath());
556  dialog.setLabelText(QFileDialog::Accept, tr("Select"));
557  dialog.setAcceptMode(QFileDialog::AcceptOpen);
558 #ifdef Q_OS_MAC
559 #if QT_VERSION < 0x050000
560  dialog.setAttribute(Qt::WA_MacMetalStyle, false);
561 #endif
562 #endif
563 
564  if(dialog.exec() != QDialog::Accepted)
565  return;
566 
567  QFileInfo fileInfo(dialog.selectedFiles().value(0));
568 
569  if(!fileInfo.exists() || !fileInfo.isReadable())
570  {
571  error = tr("The selected file is not readable.");
572  showError(error);
573  return;
574  }
575 
576  spoton::prepareDatabasesFromUI();
577 
578  /*
579  ** Create a StarBeam magnet.
580  */
581 
582  QByteArray eKey(spoton_crypt::strongRandomBytes(spoton_crypt::
583  cipherKeyLength("aes256")).
584  toBase64());
585  QByteArray mKey(spoton_crypt::
586  strongRandomBytes(spoton_crypt::
587  XYZ_DIGEST_OUTPUT_SIZE_IN_BYTES).
588  toBase64());
589  QByteArray magnet;
590  bool ok = true;
591 
592  magnet.append("magnet:?");
593  magnet.append("ct=aes256&");
594  magnet.append("ek=");
595  magnet.append(eKey);
596  magnet.append("&");
597  magnet.append("ht=sha512&");
598  magnet.append("mk=");
599  magnet.append(mKey);
600  magnet.append("&");
601  magnet.append("xt=urn:starbeam");
602  ui.message->setText(magnet);
603  sendMessage(&ok);
604 
605  if(!ok)
606  return;
607 
608  QString connectionName("");
609 
610  /*
611  ** Create a StarBeam database entry.
612  */
613 
614  {
615  QSqlDatabase db = spoton_misc::database(connectionName);
616 
617  db.setDatabaseName(spoton_misc::homePath() + QDir::separator() +
618  "starbeam.db");
619 
620  if(db.open())
621  {
622  QByteArray encryptedMosaic;
623  QByteArray mosaic
624  (spoton_crypt::strongRandomBytes(spoton_common::MOSAIC_SIZE).
625  toBase64());
626  QSqlQuery query(db);
627 
628  query.prepare("INSERT INTO transmitted "
629  "(file, hash, missing_links, mosaic, nova, "
630  "position, pulse_size, read_interval, "
631  "status_control, total_size) "
632  "VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
633  query.bindValue
634  (0, crypt->
635  encryptedThenHashed(fileInfo.absoluteFilePath().toUtf8(),
636  &ok).toBase64());
637 
638  if(ok)
639  query.bindValue
640  (1, crypt->
641  encryptedThenHashed
642  (spoton_crypt::
643  sha1FileHash(fileInfo.absoluteFilePath()).toHex(),
644  &ok).toBase64());
645 
646  if(ok)
647  query.bindValue
648  (2, crypt->
649  encryptedThenHashed(QByteArray(), &ok).toBase64());
650 
651  if(ok)
652  {
653  encryptedMosaic = crypt->encryptedThenHashed(mosaic, &ok);
654 
655  if(ok)
656  query.bindValue(3, encryptedMosaic.toBase64());
657  }
658 
659  if(ok)
660  query.bindValue
661  (4, crypt->encryptedThenHashed(QByteArray(), &ok).toBase64());
662 
663  if(ok)
664  query.bindValue
665  (5, crypt->encryptedThenHashed("0", &ok).toBase64());
666 
667  if(ok)
668  query.bindValue
669  (6, crypt->
670  encryptedThenHashed(QByteArray::number(30000),
671  &ok).toBase64());
672 
673  query.bindValue(7, 2.500);
674  query.bindValue(8, "transmitting");
675 
676  if(ok)
677  query.bindValue
678  (9, crypt->
679  encryptedThenHashed(QByteArray::number(fileInfo.size()),
680  &ok).toBase64());
681 
682  if(ok)
683  query.exec();
684 
685  query.prepare("INSERT INTO transmitted_magnets "
686  "(magnet, magnet_hash, transmitted_oid) "
687  "VALUES (?, ?, (SELECT OID FROM transmitted WHERE "
688  "mosaic = ?))");
689 
690  if(ok)
691  query.bindValue
692  (0, crypt->
693  encryptedThenHashed(magnet, &ok).toBase64());
694 
695  if(ok)
696  query.bindValue
697  (1, crypt->keyedHash(magnet, &ok).toBase64());
698 
699  if(ok)
700  query.bindValue(2, encryptedMosaic.toBase64());
701 
702  if(ok)
703  ok = query.exec();
704  }
705  else
706  ok = false;
707 
708  db.close();
709  }
710 
711  QSqlDatabase::removeDatabase(connectionName);
712 
713  if(!ok)
714  {
715  error = tr("An error occurred while attempting to "
716  "save the StarBeam data. Please enable "
717  "logging via the Log Viewer and try again.");
718  showError(error);
719  }

744 void spoton_chatwindow::slotLinkClicked(const QUrl &url)
745 {

747 
748  if(!(scheme == "ftp" || scheme == "http" || scheme == "https"))
749  return;
750

754  return;

758 
759  if(str.length() > 64)
760  str = str.mid(0, 24) + "..." + str.right(24);
761
