199 void spoton_pageviewer::setPage(const QByteArray &data, const QUrl &url,
200  const int compressedSize)
201 {
202  disconnect(m_ui.revision,
203  SIGNAL(activated(int)),
204  this,
205  SLOT(slotRevisionChanged(int)));
206  m_ui.revision->clear();
207 
208  if(m_database.isOpen())
209  {
210  QApplication::setOverrideCursor(QCursor(Qt::WaitCursor));
211 
212  QSqlQuery query(m_database);
213 
214  query.setForwardOnly(true);
215  query.prepare
216  (QString("SELECT content_hash, date_time_inserted "
217  "FROM spot_on_urls_revisions_%1 WHERE url_hash = ? "
218  "ORDER BY 2 DESC").
219  arg(m_urlHash.mid(0, 2)));
220  query.bindValue(0, m_urlHash);
221 
222  if(query.exec())
223  while(query.next())
224  m_ui.revision->addItem(query.value(1).toString(),
225  query.value(0).toByteArray());
226 
227  QApplication::restoreOverrideCursor();
228  }
229 
230  if(m_ui.revision->count() > 0)
231  {
232  m_ui.revision->insertItem(0, tr("Current"));
233  m_ui.revision->insertSeparator(1);
234  m_ui.revision->setCurrentIndex(0);
235  connect(m_ui.revision,
236  SIGNAL(activated(int)),
237  this,
238  SLOT(slotRevisionChanged(int)));
239  m_ui.revision->setEnabled(true);
240  }
241  else
242  {
243  m_ui.revision->addItem(tr("Current"));
244  m_ui.revision->setEnabled(false);
245  }
246 
247  /*
248  ** Fuzzy Wuzzy was a bear,
249  ** Fuzzy Wuzzy had no hair,
250  ** Fuzzy Wuzzy wasn't fuzzy,
251  ** was he?
252  */
253 
254  QLocale locale;
255 
256  m_ui.size->setText
257  (tr("%1 KiB, Compressed %2 KiB").
258  arg(locale.toString(data.length() / 1024)).
259  arg(locale.toString(compressedSize / 1024)));
260 
261  if(data.trimmed().isEmpty())
262  {
263  m_content = "Malformed content. Enjoy!";
264  m_webView->setContent(m_content);
265  }
266  else
267  {
268  m_content = data;
269 
270  /*
271  ** setContent() will not display
272  ** some characters. setHtml() may
273  ** produce network activity.
274  */
275 
276  m_webView->setHtml(m_content);
277  }
278 
279  m_webView->setFocus();
280  m_ui.url->setText(url.toString());
281 }

328 void spoton_pageviewer::slotRevisionChanged(int index)
329 {
330  spoton_crypt *crypt = spoton::instance() ?
331  spoton::instance()->urlCommonCrypt() : 0;
332 
333  if(!crypt || !m_database.isOpen())
334  {
335  disconnect(m_ui.revision,
336  SIGNAL(activated(int)),
337  this,
338  SLOT(slotRevisionChanged(int)));
339  m_ui.revision->setCurrentIndex(0);
340  m_ui.revision->setEnabled(false);
341  return;
342  }
343  else if(index == 1) // A separator.
344  return;
345 
346  QApplication::setOverrideCursor(QCursor(Qt::WaitCursor));
347 
348  QSqlQuery query(m_database);
349  QString dateTime(m_ui.revision->itemText(index));
350 
351  query.setForwardOnly(true);
352 
353  if(index == 0)
354  {
355  query.prepare(QString("SELECT content "
356  "FROM spot_on_urls_%1 WHERE url_hash = ?").
357  arg(m_urlHash.mid(0, 2)));
358  query.bindValue(0, m_urlHash);
359  }
360  else
361  {
362  query.prepare(QString("SELECT content "
363  "FROM spot_on_urls_revisions_%1 WHERE "
364  "date_time_inserted = ? AND url_hash = ?").
365  arg(m_urlHash.mid(0, 2)));
366  query.bindValue(0, dateTime);
367  query.bindValue(1, m_urlHash);
368  }
369 
370  if(query.exec())
371  if(query.next())
372  {
373  QByteArray content;
374  bool ok = true;
375 
376  content = crypt->decryptedAfterAuthenticated
377  (QByteArray::fromBase64(query.value(0).toByteArray()), &ok);
378 
379  if(ok)
380  {
381  content = qUncompress(content);
382 
383  if(content.trimmed().isEmpty())
384  {
385  m_content = "Malformed content. Enjoy!";
386  m_webView->setContent(m_content);
387  }
388  else
389  {
390  m_content = content;
391 
392  /*
393  ** setContent() will not
394  ** display some characters.
395  ** setHtml() may produce
396  ** network activity.
397  */
398 
399  m_webView->setHtml(m_content);
400  }
401 
402  QLocale locale;
403 
404  m_ui.size->setText
405  (tr("%1 KiB, Compressed %2 KiB").
406  arg(locale.toString(m_content.length() / 1024)).
407  arg(locale.toString(query.value(0).
408  toByteArray().length() / 1024)));
409  }
410  }
411 
412  QApplication::restoreOverrideCursor();
413 }
