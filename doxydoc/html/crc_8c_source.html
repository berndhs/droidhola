130 static u32
131 update_crc32 (u32 crc, const void *buf_arg, size_t len)
132 {
133  const char *buf = buf_arg;
134  size_t n;
135 
136  for (n = 0; n < len; n++)
137  crc = crc32_table[(crc ^ buf[n]) & 0xff] ^ (crc >> 8);
138 
139  return crc;
140 }

179 {

255 static void
256 crc24rfc2440_write (void *context, const void *inbuf_arg, size_t inlen)
257 {
258  const unsigned char *inbuf = inbuf_arg;
259  int i;
260  CRC_CONTEXT *ctx = (CRC_CONTEXT *) context;
261 
262  if (!inbuf)
263  return;
264 
265  while (inlen--) {
266  ctx->CRC ^= (*inbuf++) << 16;
267  for (i = 0; i < 8; i++) {
268  ctx->CRC <<= 1;
269  if (ctx->CRC & 0x1000000)
270  ctx->CRC ^= CRC24_POLY;
271  }
272  }
273 }
