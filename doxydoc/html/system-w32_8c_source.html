94 int
95 __assuan_pipe (assuan_context_t ctx, assuan_fd_t fd[2], int inherit_idx)
96 {
97  HANDLE rh;
98  HANDLE wh;
99  HANDLE th;
100  SECURITY_ATTRIBUTES sec_attr;
101 
102  memset (&sec_attr, 0, sizeof (sec_attr));
103  sec_attr.nLength = sizeof (sec_attr);
104  sec_attr.bInheritHandle = FALSE;
105 
106  if (!CreatePipe (&rh, &wh, &sec_attr, 0))
107  {
108  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_pipe", ctx,
109  "CreatePipe failed: %s", _assuan_w32_strerror (ctx, -1));
110  gpg_err_set_errno (EIO);
111  return -1;
112  }
113 
114  if (! DuplicateHandle (GetCurrentProcess(), (inherit_idx == 0) ? rh : wh,
115  GetCurrentProcess(), &th, 0,
116  TRUE, DUPLICATE_SAME_ACCESS ))
117  {
118  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_pipe", ctx,
119  "DuplicateHandle failed: %s", _assuan_w32_strerror (ctx, -1));
120  CloseHandle (rh);
121  CloseHandle (wh);
122  gpg_err_set_errno (EIO);
123  return -1;
124  }
125  if (inherit_idx == 0)
126  {
127  CloseHandle (rh);
128  rh = th;
129  }
130  else
131  {
132  CloseHandle (wh);
133  wh = th;
134  }
135 
136  fd[0] = rh;
137  fd[1] = wh;
138 
139  return 0;
140 }

164 /* Return true if HD refers to a socket. */
165 static int
166 is_socket (HANDLE hd)
167 {
168  /* We need to figure out whether we are working on a socket or on a
169  handle. A trivial way would be to check for the return code of
170  recv and see if it is WSAENOTSOCK. However the recv may block
171  after the server process died and thus the destroy_reader will
172  hang. Another option is to use getsockopt to test whether it is
173  a socket. The bug here is that once a socket with a certain
174  values has been opened, closed and later a CreatePipe returned
175  the same value (i.e. handle), getsockopt still believes it is a
176  socket. What we do now is to use a combination of GetFileType
177  and GetNamedPipeInfo. The specs say that the latter may be used
178  on anonymous pipes as well. Note that there are claims that
179  since winsocket version 2 ReadFile may be used on a socket but
180  only if it is supported by the service provider. Tests on a
181  stock XP using a local TCP socket show that it does not work. */
182  DWORD dummyflags, dummyoutsize, dummyinsize, dummyinst;
183  if (GetFileType (hd) == FILE_TYPE_PIPE
184  && !GetNamedPipeInfo (hd, &dummyflags, &dummyoutsize,
185  &dummyinsize, &dummyinst))
186  return 1; /* Function failed; thus we assume it is a socket. */
187  else
188  return 0; /* Success; this is not a socket. */
189 }

192 ssize_t
193 __assuan_read (assuan_context_t ctx, assuan_fd_t fd, void *buffer, size_t size)
194 {
195  int res;
196  int ec = 0;
197 
198  if (is_socket (fd))
199  {
200  int tries = 3;
201 
202  again:
203  ec = 0;
204  res = recv (HANDLE2SOCKET (fd), buffer, size, 0);
205  if (res == -1)
206  ec = WSAGetLastError ();
207  if (ec == WSAEWOULDBLOCK && tries--)
208  {
209  /* EAGAIN: Use select to wait for resources and try again.
210  We do this 3 times and then give up. The higher level
211  layer then needs to take care of EAGAIN. No need to
212  specify a timeout - the socket is not expected to be in
213  blocking mode. */
214  fd_set fds;
215 
216  FD_ZERO (&fds);
217  FD_SET (HANDLE2SOCKET (fd), &fds);
218  select (0, &fds, NULL, NULL, NULL);
219  goto again;
220  }
221  }
222  else
223  {
224  DWORD nread = 0;
225  if (!ReadFile (fd, buffer, size, &nread, NULL))
226  {
227  res = -1;
228  ec = GetLastError ();
229  }
230  else
231  res = nread;
232  }
233  if (res == -1)
234  {
235  switch (ec)
236  {
237  case WSAENOTSOCK:
238  gpg_err_set_errno (EBADF);
239  break;
240 
241  case WSAEWOULDBLOCK:
242  gpg_err_set_errno (EAGAIN);
243  break;
244 
245  case WSAECONNRESET: /* Due to the use of recv. */
246  case ERROR_BROKEN_PIPE:
247  gpg_err_set_errno (EPIPE);
248  break;
249 
250  default:
251  gpg_err_set_errno (EIO);
252  break;
253  }
254  }
255  return res;
256 }

353 /* Build a command line for use with W32's CreateProcess. On success
354  CMDLINE gets the address of a newly allocated string. */
355 static int
356 build_w32_commandline (assuan_context_t ctx, const char * const *argv,
357  char **cmdline)
358 {
359  int i, n;
360  const char *s;
361  char *buf, *p;
362 
363  *cmdline = NULL;
364  n = 0;
365  for (i=0; (s = argv[i]); i++)
366  {
367  n += strlen (s) + 1 + 2; /* (1 space, 2 quoting */
368  for (; *s; s++)
369  if (*s == '\"')
370  n++; /* Need to double inner quotes. */
371  }
372  n++;
373 
374  buf = p = _assuan_malloc (ctx, n);
375  if (! buf)
376  return -1;
377 
378  for (i = 0; argv[i]; i++)
379  {
380  if (i)
381  p = stpcpy (p, " ");
382  if (! *argv[i]) /* Empty string. */
383  p = stpcpy (p, "\"\"");
384  else if (strpbrk (argv[i], " \t\n\v\f\""))
385  {
386  p = stpcpy (p, "\"");
387  for (s = argv[i]; *s; s++)
388  {
389  *p++ = *s;
390  if (*s == '\"')
391  *p++ = *s;
392  }
393  *p++ = '\"';
394  *p = 0;
395  }
396  else
397  p = stpcpy (p, argv[i]);
398  }
399 
400  *cmdline= buf;
401  return 0;
402 }

405 int
406 __assuan_spawn (assuan_context_t ctx, pid_t *r_pid, const char *name,
407  const char **argv,
408  assuan_fd_t fd_in, assuan_fd_t fd_out,
409  assuan_fd_t *fd_child_list,
410  void (*atfork) (void *opaque, int reserved),
411  void *atforkvalue, unsigned int flags)
412 {
413  SECURITY_ATTRIBUTES sec_attr;
414  PROCESS_INFORMATION pi =
415  {
416  NULL, /* Returns process handle. */
417  0, /* Returns primary thread handle. */
418  0, /* Returns pid. */
419  0 /* Returns tid. */
420  };
421  STARTUPINFO si;
422  assuan_fd_t fd;
423  assuan_fd_t *fdp;
424  char *cmdline;
425  HANDLE nullfd = INVALID_HANDLE_VALUE;
426 
427  /* fixme: Actually we should set the "_assuan_pipe_connect_pid" env
428  variable. However this requires us to write a full environment
429  handler, because the strings are expected in sorted order. The
430  suggestion given in the MS Reference Library, to save the old
431  value, change it, create process and restore it, is not thread
432  safe. */
433 
434  /* Build the command line. */
435  if (build_w32_commandline (ctx, argv, &cmdline))
436  return -1;
437 
438  /* Start the process. */
439  memset (&sec_attr, 0, sizeof sec_attr);
440  sec_attr.nLength = sizeof sec_attr;
441  sec_attr.bInheritHandle = FALSE;
442 
443  memset (&si, 0, sizeof si);
444  si.cb = sizeof (si);
445  si.dwFlags = STARTF_USESTDHANDLES;
446  /* FIXME: Dup to nul if ASSUAN_INVALID_FD. */
447  si.hStdInput = fd_in;
448  si.hStdOutput = fd_out;
449 
450  /* Dup stderr to /dev/null unless it is in the list of FDs to be
451  passed to the child. */
452  fd = assuan_fd_from_posix_fd (fileno (stderr));
453  fdp = fd_child_list;
454  if (fdp)
455  {
456  for (; *fdp != ASSUAN_INVALID_FD && *fdp != fd; fdp++)
457  ;
458  }
459  if (!fdp || *fdp == ASSUAN_INVALID_FD)
460  {
461  nullfd = CreateFileW (L"nul", GENERIC_WRITE,
462  FILE_SHARE_READ | FILE_SHARE_WRITE,
463  NULL, OPEN_EXISTING, 0, NULL);
464  if (nullfd == INVALID_HANDLE_VALUE)
465  {
466  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "__assuan_spawn", ctx,
467  "can't open `nul': %s", _assuan_w32_strerror (ctx, -1));
468  _assuan_free (ctx, cmdline);
469  gpg_err_set_errno (EIO);
470  return -1;
471  }
472  si.hStdError = nullfd;
473  }
474  else
475  si.hStdError = fd;
476 
477  /* Note: We inherit all handles flagged as inheritable. This seems
478  to be a security flaw but there seems to be no way of selecting
479  handles to inherit. A fix for this would be to use a helper
480  process like we have in gpgme. */
481  /* _assuan_log_printf ("CreateProcess, path=`%s' cmdline=`%s'\n", */
482  /* name, cmdline); */
483  if (!CreateProcess (name, /* Program to start. */
484  cmdline, /* Command line arguments. */
485  &sec_attr, /* Process security attributes. */
486  &sec_attr, /* Thread security attributes. */
487  TRUE, /* Inherit handles. */
488  (CREATE_DEFAULT_ERROR_MODE
489  | ((flags & 128)? DETACHED_PROCESS : 0)
490  | GetPriorityClass (GetCurrentProcess ())
491  | CREATE_SUSPENDED), /* Creation flags. */
492  NULL, /* Environment. */
493  NULL, /* Use current drive/directory. */
494  &si, /* Startup information. */
495  &pi /* Returns process information. */
496  ))
497  {
498  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "pipe_connect_w32", ctx,
499  "CreateProcess failed: %s", _assuan_w32_strerror (ctx, -1));
500  _assuan_free (ctx, cmdline);
501  if (nullfd != INVALID_HANDLE_VALUE)
502  CloseHandle (nullfd);
503 
504  gpg_err_set_errno (EIO);
505  return -1;
506  }
507 
508  _assuan_free (ctx, cmdline);
509  if (nullfd != INVALID_HANDLE_VALUE)
510  CloseHandle (nullfd);
511 
512  ResumeThread (pi.hThread);
513  CloseHandle (pi.hThread);
514 
515  /* _assuan_log_printf ("CreateProcess ready: hProcess=%p hThread=%p" */
516  /* " dwProcessID=%d dwThreadId=%d\n", */
517  /* pi.hProcess, pi.hThread, */
518  /* (int) pi.dwProcessId, (int) pi.dwThreadId); */
519 
520  *r_pid = (pid_t) pi.hProcess;
521 
522  /* No need to modify peer process, as we don't change the handle
523  names. However this also means we are not safe, as we inherit
524  too many handles. Should use approach similar to gpgme and glib
525  using a helper process. */
526 
527  return 0;
528 }
