200 /* Our use of the ELF visibility feature works by passing
201  -fvisibiliy=hidden on the command line and by explicitly marking
202  all exported functions as visible.
203 
204  Note: When adding new functions, you need to add them to
205  libksba.vers and libksba.def as well. */
206 #ifdef KSBA_USE_VISIBILITY
207 # define _KSBA_VISIBILITY_DEFAULT __attribute__ ((visibility("default")))
208 #else
209 # define _KSBA_VISIBILITY_DEFAULT
210 #endif
211 
212 #ifdef _KSBA_INCLUDED_BY_VISIBILITY_C
213 # ifdef KSBA_USE_VISIBILITY
214 # define MARK_VISIBLE(name) \
215  extern __typeof__ (_##name) name _KSBA_VISIBILITY_DEFAULT;
216 # define MARK_VISIBLEX(name) \
217  extern __typeof__ (name) name _KSBA_VISIBILITY_DEFAULT;
218 # else
219 # define MARK_VISIBLE(name) /* */
220 # define MARK_VISIBLEX(name) /* */
221 # endif
