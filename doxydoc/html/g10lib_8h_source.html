311  unsigned char _vset=(_set); \

325 /* Following architectures can handle unaligned accesses fast. */
326 #if defined(__i386__) || defined(__x86_64__) || \
327  defined(__powerpc__) || defined(__powerpc64__) || \
328  (defined(__arm__) && defined(__ARM_FEATURE_UNALIGNED)) || \
329  defined(__aarch64__)
330 #define fast_wipememory2_unaligned_head(_ptr,_set,_len) /*do nothing*/
331 #else
332 #define fast_wipememory2_unaligned_head(_vptr,_vset,_vlen) do { \
333  while((size_t)(_vptr)&(sizeof(FASTWIPE_T)-1) && _vlen) \
334  { *_vptr=(_vset); _vptr++; _vlen--; } \
335  } while(0)
336 #endif
337 
338 /* fast_wipememory2 may leave tail bytes unhandled, in which case tail bytes
339  are handled by wipememory2. */
340 #define fast_wipememory2(_vptr,_vset,_vlen) do { \
341  FASTWIPE_T _vset_long = _vset; \
342  fast_wipememory2_unaligned_head(_vptr,_vset,_vlen); \
343  if (_vlen < sizeof(FASTWIPE_T)) \
344  break; \
345  _vset_long *= FASTWIPE_MULT; \
346  do { \
347  volatile FASTWIPE_T *_vptr_long = (volatile void *)_vptr; \
348  *_vptr_long = _vset_long; \
349  _vlen -= sizeof(FASTWIPE_T); \
350  _vptr += sizeof(FASTWIPE_T); \
351  } while (_vlen >= sizeof(FASTWIPE_T)); \
352  } while (0)
