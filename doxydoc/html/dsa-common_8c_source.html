32 /*
33  * Generate a random secret exponent K less than Q.
34  * Note that ECDSA uses this code also to generate D.
35  */
36 gcry_mpi_t
37 _gcry_dsa_gen_k (gcry_mpi_t q, int security_level)
38 {
39  gcry_mpi_t k = mpi_alloc_secure (mpi_get_nlimbs (q));
40  unsigned int nbits = mpi_get_nbits (q);
41  unsigned int nbytes = (nbits+7)/8;
42  char *rndbuf = NULL;
43 
44  /* To learn why we don't use mpi_mod to get the requested bit size,
45  read the paper: "The Insecurity of the Digital Signature
46  Algorithm with Partially Known Nonces" by Nguyen and Shparlinski.
47  Journal of Cryptology, New York. Vol 15, nr 3 (2003) */
48 
49  if (DBG_CIPHER)
50  log_debug ("choosing a random k of %u bits at seclevel %d\n",
51  nbits, security_level);
52  for (;;)
53  {
54  if ( !rndbuf || nbits < 32 )
55  {
56  xfree (rndbuf);
57  rndbuf = _gcry_random_bytes_secure (nbytes, security_level);
58  }
59  else
60  { /* Change only some of the higher bits. We could improve
61  this by directly requesting more memory at the first call
62  to get_random_bytes() and use these extra bytes here.
63  However the required management code is more complex and
64  thus we better use this simple method. */
65  char *pp = _gcry_random_bytes_secure (4, security_level);
66  memcpy (rndbuf, pp, 4);
67  xfree (pp);
68  }
69  _gcry_mpi_set_buffer (k, rndbuf, nbytes, 0);
70 
71  /* Make sure we have the requested number of bits. This code
72  looks a bit funny but it is easy to understand if you
73  consider that mpi_set_highbit clears all higher bits. We
74  don't have a clear_highbit, thus we first set the high bit
75  and then clear it again. */
76  if (mpi_test_bit (k, nbits-1))
77  mpi_set_highbit (k, nbits-1);
78  else
79  {
80  mpi_set_highbit (k, nbits-1);
81  mpi_clear_bit (k, nbits-1);
82  }
83 
84  if (!(mpi_cmp (k, q) < 0)) /* check: k < q */
85  {
86  if (DBG_CIPHER)
87  log_debug ("\tk too large - again\n");
88  continue; /* no */
89  }
90  if (!(mpi_cmp_ui (k, 0) > 0)) /* check: k > 0 */
91  {
92  if (DBG_CIPHER)
93  log_debug ("\tk is zero - again\n");
94  continue; /* no */
95  }
96  break; /* okay */
97  }
98  xfree (rndbuf);
99 
100  return k;
101 }

104 /* Turn VALUE into an octet string and store it in an allocated buffer
105  at R_FRAME. If the resulting octet string is shorter than NBYTES
106  the result will be left padded with zeroes. If VALUE does not fit
107  into NBYTES an error code is returned. */
108 static gpg_err_code_t
109 int2octets (unsigned char **r_frame, gcry_mpi_t value, size_t nbytes)
110 {
111  gpg_err_code_t rc;
112  size_t nframe, noff, n;
113  unsigned char *frame;
114 
115  rc = _gcry_mpi_print (GCRYMPI_FMT_USG, NULL, 0, &nframe, value);
116  if (rc)
117  return rc;
118  if (nframe > nbytes)
119  return GPG_ERR_TOO_LARGE; /* Value too long to fit into NBYTES. */
120 
121  noff = (nframe < nbytes)? nbytes - nframe : 0;
122  n = nframe + noff;
123  frame = mpi_is_secure (value)? xtrymalloc_secure (n) : xtrymalloc (n);
124  if (!frame)
125  return gpg_err_code_from_syserror ();
126  if (noff)
127  memset (frame, 0, noff);
128  nframe += noff;
129  rc = _gcry_mpi_print (GCRYMPI_FMT_USG, frame+noff, nframe-noff, NULL, value);
130  if (rc)
131  {
132  xfree (frame);
133  return rc;
134  }
135 
136  *r_frame = frame;
137  return 0;
138 }

141 /* Connert the bit string BITS of length NBITS into an octet string
142  with a length of (QBITS+7)/8 bytes. On success store the result at
143  R_FRAME. */
144 static gpg_err_code_t
145 bits2octets (unsigned char **r_frame,
146  const void *bits, unsigned int nbits,
147  gcry_mpi_t q, unsigned int qbits)
148 {
149  gpg_err_code_t rc;
150  gcry_mpi_t z1;
151 
152  /* z1 = bits2int (b) */
153  rc = _gcry_mpi_scan (&z1, GCRYMPI_FMT_USG, bits, (nbits+7)/8, NULL);
154  if (rc)
155  return rc;
156  if (nbits > qbits)
157  mpi_rshift (z1, z1, nbits - qbits);
158 
159  /* z2 - z1 mod q */
160  if (mpi_cmp (z1, q) >= 0)
161  mpi_sub (z1, z1, q);
162 
163  /* Convert to an octet string. */
164  rc = int2octets (r_frame, z1, (qbits+7)/8);
165 
166  mpi_free (z1);
167  return rc;
168 }

171 /*
172  * Generate a deterministic secret exponent K less than DSA_Q. H1 is
173  * the to be signed digest with a length of HLEN bytes. HALGO is the
174  * algorithm used to create the hash. On success the value for K is
175  * stored at R_K.
176  */
177 gpg_err_code_t
178 _gcry_dsa_gen_rfc6979_k (gcry_mpi_t *r_k,
179  gcry_mpi_t dsa_q, gcry_mpi_t dsa_x,
180  const unsigned char *h1, unsigned int hlen,
181  int halgo, unsigned int extraloops)
182 {
183  gpg_err_code_t rc;
184  unsigned char *V = NULL;
185  unsigned char *K = NULL;
186  unsigned char *x_buf = NULL;
187  unsigned char *h1_buf = NULL;
188  gcry_md_hd_t hd = NULL;
189  unsigned char *t = NULL;
190  gcry_mpi_t k = NULL;
191  unsigned int tbits, qbits;
192  int i;
193 
194  qbits = mpi_get_nbits (dsa_q);
195 
196  if (!qbits || !h1 || !hlen)
197  return GPG_ERR_EINVAL;
198 
199  if (_gcry_md_get_algo_dlen (halgo) != hlen)
200  return GPG_ERR_DIGEST_ALGO;
201 
202  /* Step b: V = 0x01 0x01 0x01 ... 0x01 */
203  V = xtrymalloc (hlen);
204  if (!V)
205  {
206  rc = gpg_err_code_from_syserror ();
207  goto leave;
208  }
209  for (i=0; i < hlen; i++)
210  V[i] = 1;
211 
212  /* Step c: K = 0x00 0x00 0x00 ... 0x00 */
213  K = xtrycalloc (1, hlen);
214  if (!K)
215  {
216  rc = gpg_err_code_from_syserror ();
217  goto leave;
218  }
219 
220  rc = int2octets (&x_buf, dsa_x, (qbits+7)/8);
221  if (rc)
222  goto leave;
223 
224  rc = bits2octets (&h1_buf, h1, hlen*8, dsa_q, qbits);
225  if (rc)
226  goto leave;
227 
228  /* Create a handle to compute the HMACs. */
229  rc = _gcry_md_open (&hd, halgo, (GCRY_MD_FLAG_SECURE | GCRY_MD_FLAG_HMAC));
230  if (rc)
231  goto leave;
232 
233  /* Step d: K = HMAC_K(V || 0x00 || int2octets(x) || bits2octets(h1) */
234  rc = _gcry_md_setkey (hd, K, hlen);
235  if (rc)
236  goto leave;
237  _gcry_md_write (hd, V, hlen);
238  _gcry_md_write (hd, "", 1);
239  _gcry_md_write (hd, x_buf, (qbits+7)/8);
240  _gcry_md_write (hd, h1_buf, (qbits+7)/8);
241  memcpy (K, _gcry_md_read (hd, 0), hlen);
242 
243  /* Step e: V = HMAC_K(V) */
244  rc = _gcry_md_setkey (hd, K, hlen);
245  if (rc)
246  goto leave;
247  _gcry_md_write (hd, V, hlen);
248  memcpy (V, _gcry_md_read (hd, 0), hlen);
249 
250  /* Step f: K = HMAC_K(V || 0x01 || int2octets(x) || bits2octets(h1) */
251  rc = _gcry_md_setkey (hd, K, hlen);
252  if (rc)
253  goto leave;
254  _gcry_md_write (hd, V, hlen);
255  _gcry_md_write (hd, "\x01", 1);
256  _gcry_md_write (hd, x_buf, (qbits+7)/8);
257  _gcry_md_write (hd, h1_buf, (qbits+7)/8);
258  memcpy (K, _gcry_md_read (hd, 0), hlen);
259 
260  /* Step g: V = HMAC_K(V) */
261  rc = _gcry_md_setkey (hd, K, hlen);
262  if (rc)
263  goto leave;
264  _gcry_md_write (hd, V, hlen);
265  memcpy (V, _gcry_md_read (hd, 0), hlen);
266 
267  /* Step h. */
268  t = xtrymalloc ((qbits+7)/8+hlen);
269  if (!t)
270  {
271  rc = gpg_err_code_from_syserror ();
272  goto leave;
273  }
274 
275  again:
276  for (tbits = 0; tbits < qbits;)
277  {
278  /* V = HMAC_K(V) */
279  rc = _gcry_md_setkey (hd, K, hlen);
280  if (rc)
281  goto leave;
282  _gcry_md_write (hd, V, hlen);
283  memcpy (V, _gcry_md_read (hd, 0), hlen);
284 
285  /* T = T || V */
286  memcpy (t+(tbits+7)/8, V, hlen);
287  tbits += 8*hlen;
288  }
289 
290  /* k = bits2int (T) */
291  mpi_free (k);
292  k = NULL;
293  rc = _gcry_mpi_scan (&k, GCRYMPI_FMT_USG, t, (tbits+7)/8, NULL);
294  if (rc)
295  goto leave;
296  if (tbits > qbits)
297  mpi_rshift (k, k, tbits - qbits);
298 
299  /* Check: k < q and k > 1 */
300  if (!(mpi_cmp (k, dsa_q) < 0 && mpi_cmp_ui (k, 0) > 0))
301  {
302  /* K = HMAC_K(V || 0x00) */
303  rc = _gcry_md_setkey (hd, K, hlen);
304  if (rc)
305  goto leave;
306  _gcry_md_write (hd, V, hlen);
307  _gcry_md_write (hd, "", 1);
308  memcpy (K, _gcry_md_read (hd, 0), hlen);
309 
310  /* V = HMAC_K(V) */
311  rc = _gcry_md_setkey (hd, K, hlen);
312  if (rc)
313  goto leave;
314  _gcry_md_write (hd, V, hlen);
315  memcpy (V, _gcry_md_read (hd, 0), hlen);
316 
317  goto again;
318  }
319 
320  /* The caller may have requested that we introduce some extra loops.
321  This is for example useful if the caller wants another value for
322  K because the last returned one yielded an R of 0. Becuase this
323  is very unlikely we implement it in a straightforward way. */
324  if (extraloops)
325  {
326  extraloops--;
327 
328  /* K = HMAC_K(V || 0x00) */
329  rc = _gcry_md_setkey (hd, K, hlen);
330  if (rc)
331  goto leave;
332  _gcry_md_write (hd, V, hlen);
333  _gcry_md_write (hd, "", 1);
334  memcpy (K, _gcry_md_read (hd, 0), hlen);
335 
336  /* V = HMAC_K(V) */
337  rc = _gcry_md_setkey (hd, K, hlen);
338  if (rc)
339  goto leave;
340  _gcry_md_write (hd, V, hlen);
341  memcpy (V, _gcry_md_read (hd, 0), hlen);
342 
343  goto again;
344  }
345 
346  /* log_mpidump (" k", k); */
347 
348  leave:
349  xfree (t);
350  _gcry_md_close (hd);
351  xfree (h1_buf);
352  xfree (x_buf);
353  xfree (K);
354  xfree (V);
355 
356  if (rc)
357  mpi_free (k);
358  else
359  *r_k = k;
360  return rc;
361 }

363 /*
364  * Truncate opaque hash value to qbits for DSA.
365  * Non-opaque input is not truncated, in hope that user
366  * knows what is passed. It is not possible to correctly
367  * trucate non-opaque inputs.
368  */
369 gpg_err_code_t
370 _gcry_dsa_normalize_hash (gcry_mpi_t input,
371  gcry_mpi_t *out,
372  unsigned int qbits)
373 {
374  gpg_err_code_t rc = 0;
375  const void *abuf;
376  unsigned int abits;
377  gcry_mpi_t hash;
378 
379  if (mpi_is_opaque (input))
380  {
381  abuf = mpi_get_opaque (input, &abits);
382  rc = _gcry_mpi_scan (&hash, GCRYMPI_FMT_USG, abuf, (abits+7)/8, NULL);
383  if (rc)
384  return rc;
385  if (abits > qbits)
386  mpi_rshift (hash, hash, abits - qbits);
387  }
388  else
389  hash = input;
390 
391  *out = hash;
392 
393  return rc;
394 }
