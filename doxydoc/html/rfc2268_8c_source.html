87 static void
88 do_encrypt (void *context, unsigned char *outbuf, const unsigned char *inbuf)
89 {
90  RFC2268_context *ctx = context;
91  register int i, j;
92  u16 word0 = 0, word1 = 0, word2 = 0, word3 = 0;
93 
94  word0 = (word0 << 8) | inbuf[1];
95  word0 = (word0 << 8) | inbuf[0];
96  word1 = (word1 << 8) | inbuf[3];
97  word1 = (word1 << 8) | inbuf[2];
98  word2 = (word2 << 8) | inbuf[5];
99  word2 = (word2 << 8) | inbuf[4];
100  word3 = (word3 << 8) | inbuf[7];
101  word3 = (word3 << 8) | inbuf[6];
102 
103  for (i = 0; i < 16; i++)
104  {
105  j = i * 4;
106  /* For some reason I cannot combine those steps. */
107  word0 += (word1 & ~word3) + (word2 & word3) + ctx->S[j];
108  word0 = rotl16(word0, 1);
109 
110  word1 += (word2 & ~word0) + (word3 & word0) + ctx->S[j + 1];
111  word1 = rotl16(word1, 2);
112 
113  word2 += (word3 & ~word1) + (word0 & word1) + ctx->S[j + 2];
114  word2 = rotl16(word2, 3);
115 
116  word3 += (word0 & ~word2) + (word1 & word2) + ctx->S[j + 3];
117  word3 = rotl16(word3, 5);
118 
119  if (i == 4 || i == 10)
120  {
121  word0 += ctx->S[word3 & 63];
122  word1 += ctx->S[word0 & 63];
123  word2 += ctx->S[word1 & 63];
124  word3 += ctx->S[word2 & 63];
125  }
126 
127  }
128 
129  outbuf[0] = word0 & 255;
130  outbuf[1] = word0 >> 8;
131  outbuf[2] = word1 & 255;
132  outbuf[3] = word1 >> 8;
133  outbuf[4] = word2 & 255;
134  outbuf[5] = word2 >> 8;
135  outbuf[6] = word3 & 255;
136  outbuf[7] = word3 >> 8;
137 }

146 static void
147 do_decrypt (void *context, unsigned char *outbuf, const unsigned char *inbuf)
148 {
149  RFC2268_context *ctx = context;
150  register int i, j;
151  u16 word0 = 0, word1 = 0, word2 = 0, word3 = 0;
152 
153  word0 = (word0 << 8) | inbuf[1];
154  word0 = (word0 << 8) | inbuf[0];
155  word1 = (word1 << 8) | inbuf[3];
156  word1 = (word1 << 8) | inbuf[2];
157  word2 = (word2 << 8) | inbuf[5];
158  word2 = (word2 << 8) | inbuf[4];
159  word3 = (word3 << 8) | inbuf[7];
160  word3 = (word3 << 8) | inbuf[6];
161 
162  for (i = 15; i >= 0; i--)
163  {
164  j = i * 4;
165 
166  word3 = rotr16(word3, 5);
167  word3 -= (word0 & ~word2) + (word1 & word2) + ctx->S[j + 3];
168 
169  word2 = rotr16(word2, 3);
170  word2 -= (word3 & ~word1) + (word0 & word1) + ctx->S[j + 2];
171 
172  word1 = rotr16(word1, 2);
173  word1 -= (word2 & ~word0) + (word3 & word0) + ctx->S[j + 1];
174 
175  word0 = rotr16(word0, 1);
176  word0 -= (word1 & ~word3) + (word2 & word3) + ctx->S[j];
177 
178  if (i == 5 || i == 11)
179  {
180  word3 = word3 - ctx->S[word2 & 63];
181  word2 = word2 - ctx->S[word1 & 63];
182  word1 = word1 - ctx->S[word0 & 63];
183  word0 = word0 - ctx->S[word3 & 63];
184  }
185 
186  }
187 
188  outbuf[0] = word0 & 255;
189  outbuf[1] = word0 >> 8;
190  outbuf[2] = word1 & 255;
191  outbuf[3] = word1 >> 8;
192  outbuf[4] = word2 & 255;
193  outbuf[5] = word2 >> 8;
194  outbuf[6] = word3 & 255;
195  outbuf[7] = word3 >> 8;
196 }

206 static gpg_err_code_t
207 setkey_core (void *context, const unsigned char *key, unsigned int keylen, int with_phase2)
208 {
209  static int initialized;
210  static const char *selftest_failed;
211  RFC2268_context *ctx = context;
212  unsigned int i;
213  unsigned char *S, x;
214  int len;
215  int bits = keylen * 8;
216 
217  if (!initialized)
218  {
219  initialized = 1;
220  selftest_failed = selftest ();
221  if (selftest_failed)
222  log_error ("RFC2268 selftest failed (%s).\n", selftest_failed);
223  }
224  if (selftest_failed)
225  return GPG_ERR_SELFTEST_FAILED;
226 
227  if (keylen < 40 / 8) /* We want at least 40 bits. */
228  return GPG_ERR_INV_KEYLEN;
229 
230  S = (unsigned char *) ctx->S;
231 
232  for (i = 0; i < keylen; i++)
233  S[i] = key[i];
234 
235  for (i = keylen; i < 128; i++)
236  S[i] = rfc2268_sbox[(S[i - keylen] + S[i - 1]) & 255];
237 
238  S[0] = rfc2268_sbox[S[0]];
239 
240  /* Phase 2 - reduce effective key size to "bits". This was not
241  * discussed in Gutmann's paper. I've copied that from the public
242  * domain code posted in sci.crypt. */
243  if (with_phase2)
244  {
245  len = (bits + 7) >> 3;
246  i = 128 - len;
247  x = rfc2268_sbox[S[i] & (255 >> (7 & -bits))];
248  S[i] = x;
249 
250  while (i--)
251  {
252  x = rfc2268_sbox[x ^ S[i + len]];
253  S[i] = x;
254  }
255  }
256 
257  /* Make the expanded key, endian independent. */
258  for (i = 0; i < 64; i++)
259  ctx->S[i] = ( (u16) S[i * 2] | (((u16) S[i * 2 + 1]) << 8));
260 
261  return 0;
262 }
