68 /* Convert the the buffer BUFFER into a block BLOCK, using I as
69  counter. */
70 #define buffer_to_block(buffer, block, i) \
71  for (i = 0; i < 8; i++) \
72  (block)[i] = buf_get_be64((buffer) + i * 8);

74 /* Convert the block BLOCK into a buffer BUFFER, using I as
75  counter. */
76 #define block_to_buffer(buffer, block, i) \
77  for (i = 0; i < 8; i++) \
78  buf_put_be64((buffer) + i * 8, (block)[i]);

80 /* Copy the block BLOCK_SRC to BLOCK_DST, using I as counter. */
81 #define block_copy(block_dst, block_src, i) \
82  for (i = 0; i < 8; i++) \
83  block_dst[i] = block_src[i];

85 /* XOR the block BLOCK_SRC into BLOCK_DST, using I as counter. */
86 #define block_xor(block_dst, block_src, i) \
87  for (i = 0; i < 8; i++) \
88  block_dst[i] ^= block_src[i];
