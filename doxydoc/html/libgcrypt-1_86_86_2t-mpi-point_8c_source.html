260 /* Convert STRING consisting of hex characters into its binary
261  representation and return it as an allocated buffer. The valid
262  length of the buffer is returned at R_LENGTH. The string is
263  delimited by end of string. The function returns NULL on
264  error. */
265 static void *
266 hex2buffer (const char *string, size_t *r_length)
267 {
268  const char *s;
269  unsigned char *buffer;
270  size_t length;
271 
272  buffer = xmalloc (strlen(string)/2+1);
273  length = 0;
274  for (s=string; *s; s +=2 )
275  {
276  if (!hexdigitp (s) || !hexdigitp (s+1))
277  return NULL; /* Invalid hex digits. */
278  ((unsigned char*)buffer)[length++] = xtoi_2 (s);
279  }
280  *r_length = length;
281  return buffer;
282 }

808 static void
809 twistededwards_math (void)
810 {

812  gcry_ctx_t ctx;
813  gcry_mpi_point_t G, Q;
814  gcry_mpi_t k;
815  gcry_mpi_t w, a, x, y, z, p, n, b, I;
816 
817  wherestr = "twistededwards_math";
818  show ("checking basic Twisted Edwards math\n");
819 
820  err = gcry_mpi_ec_new (&ctx, NULL, "Ed25519");
821  if (err)
822  die ("gcry_mpi_ec_new failed: %s\n", gpg_strerror (err));
823 
824  k = hex2mpi
825  ("2D3501E723239632802454EE5DDC406EFB0BDF18486A5BDE9C0390A9C2984004"
826  "F47252B628C953625B8DEB5DBCB8DA97AA43A1892D11FA83596F42E0D89CB1B6");
827  G = gcry_mpi_ec_get_point ("g", ctx, 1);
828  if (!G)
829  die ("gcry_mpi_ec_get_point(G) failed\n");
830  Q = gcry_mpi_point_new (0);
831 
832 
833  w = gcry_mpi_new (0);
834  a = gcry_mpi_new (0);
835  x = gcry_mpi_new (0);
836  y = gcry_mpi_new (0);
837  z = gcry_mpi_new (0);
838  I = gcry_mpi_new (0);
839  p = gcry_mpi_ec_get_mpi ("p", ctx, 1);
840  n = gcry_mpi_ec_get_mpi ("n", ctx, 1);
841  b = gcry_mpi_ec_get_mpi ("b", ctx, 1);
842 
843  /* Check: 2^{p-1} mod p == 1 */
844  gcry_mpi_sub_ui (a, p, 1);
845  gcry_mpi_powm (w, GCRYMPI_CONST_TWO, a, p);
846  if (gcry_mpi_cmp_ui (w, 1))
847  fail ("failed assertion: 2^{p-1} mod p == 1\n");
848 
849  /* Check: p % 4 == 1 */
850  gcry_mpi_mod (w, p, GCRYMPI_CONST_FOUR);
851  if (gcry_mpi_cmp_ui (w, 1))
852  fail ("failed assertion: p % 4 == 1\n");
853 
854  /* Check: 2^{n-1} mod n == 1 */
855  gcry_mpi_sub_ui (a, n, 1);
856  gcry_mpi_powm (w, GCRYMPI_CONST_TWO, a, n);
857  if (gcry_mpi_cmp_ui (w, 1))
858  fail ("failed assertion: 2^{n-1} mod n == 1\n");
859 
860  /* Check: b^{(p-1)/2} mod p == p-1 */
861  gcry_mpi_sub_ui (a, p, 1);
862  gcry_mpi_div (x, NULL, a, GCRYMPI_CONST_TWO, -1);
863  gcry_mpi_powm (w, b, x, p);
864  gcry_mpi_abs (w);
865  if (gcry_mpi_cmp (w, a))
866  fail ("failed assertion: b^{(p-1)/2} mod p == p-1\n");
867 
868  /* I := 2^{(p-1)/4} mod p */
869  gcry_mpi_sub_ui (a, p, 1);
870  gcry_mpi_div (x, NULL, a, GCRYMPI_CONST_FOUR, -1);
871  gcry_mpi_powm (I, GCRYMPI_CONST_TWO, x, p);
872 
873  /* Check: I^2 mod p == p-1 */
874  gcry_mpi_powm (w, I, GCRYMPI_CONST_TWO, p);
875  if (gcry_mpi_cmp (w, a))
876  fail ("failed assertion: I^2 mod p == p-1\n");
877 
878  /* Check: G is on the curve */
879  if (!gcry_mpi_ec_curve_point (G, ctx))
880  fail ("failed assertion: G is on the curve\n");
881 
882  /* Check: nG == (0,1) */
883  gcry_mpi_ec_mul (Q, n, G, ctx);
884  if (gcry_mpi_ec_get_affine (x, y, Q, ctx))
885  fail ("failed to get affine coordinates\n");

887  fail ("failed assertion: nG == (0,1)\n");

889  /* Now two arbitrary point operations taken from the ed25519.py
890  sample data. */
891  gcry_mpi_release (a);
892  a = hex2mpi
893  ("4f71d012df3c371af3ea4dc38385ca5bb7272f90cb1b008b3ed601c76de1d496"
894  "e30cbf625f0a756a678d8f256d5325595cccc83466f36db18f0178eb9925edd3");
895  gcry_mpi_ec_mul (Q, a, G, ctx);
896  if (gcry_mpi_ec_get_affine (x, y, Q, ctx))
897  fail ("failed to get affine coordinates\n");
898  if (cmp_mpihex (x, ("157f7361c577aad36f67ed33e38dc7be"
899  "00014fecc2165ca5cee9eee19fe4d2c1"))
900  || cmp_mpihex (y, ("5a69dbeb232276b38f3f5016547bb2a2"
901  "4025645f0b820e72b8cad4f0a909a092")))
902  {

904  print_mpi ("r", a);
