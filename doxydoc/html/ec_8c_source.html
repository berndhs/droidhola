563 /* Compute the affine coordinates from the projective coordinates in
564  POINT. Set them into X and Y. If one coordinate is not required,
565  X or Y may be passed as NULL. CTX is the usual context. Returns: 0
566  on success or !0 if POINT is at infinity. */
567 int
568 _gcry_mpi_ec_get_affine (gcry_mpi_t x, gcry_mpi_t y, mpi_point_t point,
569  mpi_ec_t ctx)
570 {
571  if (!mpi_cmp_ui (point->z, 0))
572  return -1;
573 
574  switch (ctx->model)
575  {
576  case MPI_EC_WEIERSTRASS: /* Using Jacobian coordinates. */
577  {
578  gcry_mpi_t z1, z2, z3;
579 
580  z1 = mpi_new (0);
581  z2 = mpi_new (0);
582  ec_invm (z1, point->z, ctx); /* z1 = z^(-1) mod p */
583  ec_mulm (z2, z1, z1, ctx); /* z2 = z^(-2) mod p */
584 
585  if (x)
586  ec_mulm (x, point->x, z2, ctx);
587 
588  if (y)
589  {
590  z3 = mpi_new (0);
591  ec_mulm (z3, z2, z1, ctx); /* z3 = z^(-3) mod p */
592  ec_mulm (y, point->y, z3, ctx);
593  mpi_free (z3);
594  }
595 
596  mpi_free (z2);
597  mpi_free (z1);
598  }
599  return 0;
600 
601  case MPI_EC_MONTGOMERY:
602  {
603  log_fatal ("%s: %s not yet supported\n",
604  "_gcry_mpi_ec_get_affine", "Montgomery");
605  }
606  return -1;
607 
608  case MPI_EC_TWISTEDEDWARDS:
609  {
610  gcry_mpi_t z;
611 
612  z = mpi_new (0);
613  ec_invm (z, point->z, ctx);
614 
615  if (x)
616  ec_mulm (x, point->x, z, ctx);
617  if (y)
618  ec_mulm (y, point->y, z, ctx);
619 
620  _gcry_mpi_release (z);
621  }
622  return 0;
623 
624  default:
625  return -1;
626  }
627 }

631 /* RESULT = 2 * POINT (Weierstrass version). */
632 static void
633 dup_point_weierstrass (mpi_point_t result, mpi_point_t point, mpi_ec_t ctx)
634 {
635 #define x3 (result->x)
636 #define y3 (result->y)
637 #define z3 (result->z)
638 #define t1 (ctx->t.scratch[0])
639 #define t2 (ctx->t.scratch[1])
640 #define t3 (ctx->t.scratch[2])
641 #define l1 (ctx->t.scratch[3])
642 #define l2 (ctx->t.scratch[4])
643 #define l3 (ctx->t.scratch[5])
644 
645  if (!mpi_cmp_ui (point->y, 0) || !mpi_cmp_ui (point->z, 0))
646  {
647  /* P_y == 0 || P_z == 0 => [1:1:0] */
648  mpi_set_ui (x3, 1);
649  mpi_set_ui (y3, 1);
650  mpi_set_ui (z3, 0);
651  }
652  else
653  {
654  if (ec_get_a_is_pminus3 (ctx)) /* Use the faster case. */
655  {
656  /* L1 = 3(X - Z^2)(X + Z^2) */
657  /* T1: used for Z^2. */
658  /* T2: used for the right term. */
659  ec_pow2 (t1, point->z, ctx);
660  ec_subm (l1, point->x, t1, ctx);
661  ec_mulm (l1, l1, mpi_const (MPI_C_THREE), ctx);
662  ec_addm (t2, point->x, t1, ctx);
663  ec_mulm (l1, l1, t2, ctx);
664  }
665  else /* Standard case. */
666  {
667  /* L1 = 3X^2 + aZ^4 */
668  /* T1: used for aZ^4. */
669  ec_pow2 (l1, point->x, ctx);
670  ec_mulm (l1, l1, mpi_const (MPI_C_THREE), ctx);
671  ec_powm (t1, point->z, mpi_const (MPI_C_FOUR), ctx);
672  ec_mulm (t1, t1, ctx->a, ctx);

824  mpi_point_t p1, mpi_point_t p2,
825  mpi_ec_t ctx)

828 #define y1 (p1->y )
829 #define z1 (p1->z )
830 #define x2 (p2->x )
831 #define y2 (p2->y )
832 #define z2 (p2->z )
833 #define x3 (result->x)
834 #define y3 (result->y)
835 #define z3 (result->z)
836 #define l1 (ctx->t.scratch[0])
837 #define l2 (ctx->t.scratch[1])
838 #define l3 (ctx->t.scratch[2])
839 #define l4 (ctx->t.scratch[3])
840 #define l5 (ctx->t.scratch[4])
841 #define l6 (ctx->t.scratch[5])
842 #define l7 (ctx->t.scratch[6])
843 #define l8 (ctx->t.scratch[7])
844 #define l9 (ctx->t.scratch[8])
845 #define t1 (ctx->t.scratch[9])
846 #define t2 (ctx->t.scratch[10])
847 
848  if ( (!mpi_cmp (x1, x2)) && (!mpi_cmp (y1, y2)) && (!mpi_cmp (z1, z2)) )
849  {
850  /* Same point; need to call the duplicate function. */
851  _gcry_mpi_ec_dup_point (result, p1, ctx);
852  }
853  else if (!mpi_cmp_ui (z1, 0))
854  {
855  /* P1 is at infinity. */
856  mpi_set (x3, p2->x);
857  mpi_set (y3, p2->y);
858  mpi_set (z3, p2->z);
859  }
860  else if (!mpi_cmp_ui (z2, 0))
861  {
862  /* P2 is at infinity. */
863  mpi_set (x3, p1->x);
864  mpi_set (y3, p1->y);
865  mpi_set (z3, p1->z);
866  }
867  else
868  {
869  int z1_is_one = !mpi_cmp_ui (z1, 1);
870  int z2_is_one = !mpi_cmp_ui (z2, 1);
871 
872  /* l1 = x1 z2^2 */
873  /* l2 = x2 z1^2 */
874  if (z2_is_one)
875  mpi_set (l1, x1);
876  else
877  {
878  ec_pow2 (l1, z2, ctx);
879  ec_mulm (l1, l1, x1, ctx);
880  }
881  if (z1_is_one)
882  mpi_set (l2, x2);
883  else
884  {
885  ec_pow2 (l2, z1, ctx);
886  ec_mulm (l2, l2, x2, ctx);
887  }
888  /* l3 = l1 - l2 */
889  ec_subm (l3, l1, l2, ctx);
890  /* l4 = y1 z2^3 */
891  ec_powm (l4, z2, mpi_const (MPI_C_THREE), ctx);
892  ec_mulm (l4, l4, y1, ctx);
893  /* l5 = y2 z1^3 */
894  ec_powm (l5, z1, mpi_const (MPI_C_THREE), ctx);
895  ec_mulm (l5, l5, y2, ctx);
896  /* l6 = l4 - l5 */
897  ec_subm (l6, l4, l5, ctx);
898 
899  if (!mpi_cmp_ui (l3, 0))
900  {
901  if (!mpi_cmp_ui (l6, 0))
902  {
903  /* P1 and P2 are the same - use duplicate function. */
904  _gcry_mpi_ec_dup_point (result, p1, ctx);
905  }
906  else
907  {
908  /* P1 is the inverse of P2. */
909  mpi_set_ui (x3, 1);
910  mpi_set_ui (y3, 1);
911  mpi_set_ui (z3, 0);
912  }
913  }
914  else
915  {
916  /* l7 = l1 + l2 */
917  ec_addm (l7, l1, l2, ctx);
918  /* l8 = l4 + l5 */
919  ec_addm (l8, l4, l5, ctx);
920  /* z3 = z1 z2 l3 */
921  ec_mulm (z3, z1, z2, ctx);
922  ec_mulm (z3, z3, l3, ctx);
923  /* x3 = l6^2 - l7 l3^2 */
924  ec_pow2 (t1, l6, ctx);
925  ec_pow2 (t2, l3, ctx);
926  ec_mulm (t2, t2, l7, ctx);
927  ec_subm (x3, t1, t2, ctx);
928  /* l9 = l7 l3^2 - 2 x3 */
929  ec_mul2 (t1, x3, ctx);
930  ec_subm (l9, t2, t1, ctx);
931  /* y3 = (l9 l6 - l8 l3^3)/2 */
932  ec_mulm (l9, l9, l6, ctx);
933  ec_powm (t1, l3, mpi_const (MPI_C_THREE), ctx); /* fixme: Use saved value*/
934  ec_mulm (t1, t1, l8, ctx);
935  ec_subm (y3, l9, t1, ctx);
936  ec_mulm (y3, y3, ec_get_two_inv_p (ctx), ctx);
937  }
938  }

979 static void
980 add_points_twistededwards (mpi_point_t result,
981  mpi_point_t p1, mpi_point_t p2,
982  mpi_ec_t ctx)

985 #define Y1 (p1->y)

987 #define X2 (p2->x)
988 #define Y2 (p2->y)

994 #define B (ctx->t.scratch[1])

998 #define F (ctx->t.scratch[5])
999 #define G (ctx->t.scratch[6])

1005  ec_mulm (A, Z1, Z2, ctx);

1008  ec_pow2 (B, A, ctx);
1009 
1010  /* C = X1 · X2 */

1012 
1013  /* D = Y1 · Y2 */
1014  ec_mulm (D, Y1, Y2, ctx);
1015 
1016  /* E = d · C · D */
1017  ec_mulm (E, ctx->b, C, ctx);
1018  ec_mulm (E, E, D, ctx);
1019 
1020  /* F = B - E */
1021  ec_subm (F, B, E, ctx);
1022 
1023  /* G = B + E */
1024  ec_addm (G, B, E, ctx);

1028  ec_addm (X3, X2, Y2, ctx);

1030  ec_subm (X3, X3, C, ctx);
1031  ec_subm (X3, X3, D, ctx);
1032  ec_mulm (X3, X3, F, ctx);
1033  ec_mulm (X3, X3, A, ctx);

1041  _gcry_mpi_neg (Y3, Y3);

1045  {
1046  ec_mulm (Y3, ctx->a, C, ctx);
1047  ec_subm (Y3, D, Y3, ctx);
1048  }
1049  ec_mulm (Y3, Y3, G, ctx);
1050  ec_mulm (Y3, Y3, A, ctx);
1051 
1052  /* Z_3 = F · G */
1053  ec_mulm (Z3, F, G, ctx);
1054

1097 /* Scalar point multiplication - the main function for ECC. If takes
1098  an integer SCALAR and a POINT as well as the usual context CTX.
1099  RESULT will be set to the resulting point. */
1100 void
1101 _gcry_mpi_ec_mul_point (mpi_point_t result,
1102  gcry_mpi_t scalar, mpi_point_t point,
1103  mpi_ec_t ctx)
1104 {
1105  gcry_mpi_t x1, y1, z1, k, h, yy;
1106  unsigned int i, loops;
1107  mpi_point_struct p1, p2, p1inv;
1108 
1109  if (ctx->model == MPI_EC_TWISTEDEDWARDS
1110  || (ctx->model == MPI_EC_WEIERSTRASS
1111  && mpi_is_secure (scalar)))
1112  {
1113  /* Simple left to right binary method. GECC Algorithm 3.27 */
1114  unsigned int nbits;
1115  int j;
1116 
1117  nbits = mpi_get_nbits (scalar);
1118  if (ctx->model == MPI_EC_WEIERSTRASS)
1119  {
1120  mpi_set_ui (result->x, 1);
1121  mpi_set_ui (result->y, 1);
1122  mpi_set_ui (result->z, 0);
1123  }
1124  else
1125  {
1126  mpi_set_ui (result->x, 0);
1127  mpi_set_ui (result->y, 1);
1128  mpi_set_ui (result->z, 1);
1129  }
1130 
1131  if (mpi_is_secure (scalar))
1132  {
1133  /* If SCALAR is in secure memory we assume that it is the
1134  secret key we use constant time operation. */
1135  mpi_point_struct tmppnt;
1136 
1137  point_init (&tmppnt);
1138  for (j=nbits-1; j >= 0; j--)
1139  {
1140  _gcry_mpi_ec_dup_point (result, result, ctx);
1141  _gcry_mpi_ec_add_points (&tmppnt, result, point, ctx);
1142  if (mpi_test_bit (scalar, j))
1143  point_set (result, &tmppnt);
1144  }
1145  point_free (&tmppnt);
1146  }
1147  else
1148  {
1149  for (j=nbits-1; j >= 0; j--)
1150  {
1151  _gcry_mpi_ec_dup_point (result, result, ctx);
1152  if (mpi_test_bit (scalar, j))
1153  _gcry_mpi_ec_add_points (result, result, point, ctx);
1154  }
1155  }
1156  return;
1157  }
1158 
1159  x1 = mpi_alloc_like (ctx->p);
1160  y1 = mpi_alloc_like (ctx->p);
1161  h = mpi_alloc_like (ctx->p);
1162  k = mpi_copy (scalar);
1163  yy = mpi_copy (point->y);
1164 
1165  if ( mpi_has_sign (k) )
1166  {
1167  k->sign = 0;
1168  ec_invm (yy, yy, ctx);
1169  }
1170 
1171  if (!mpi_cmp_ui (point->z, 1))
1172  {
1173  mpi_set (x1, point->x);
1174  mpi_set (y1, yy);
1175  }
1176  else
1177  {
1178  gcry_mpi_t z2, z3;
1179 
1180  z2 = mpi_alloc_like (ctx->p);
1181  z3 = mpi_alloc_like (ctx->p);
1182  ec_mulm (z2, point->z, point->z, ctx);
1183  ec_mulm (z3, point->z, z2, ctx);
1184  ec_invm (z2, z2, ctx);
1185  ec_mulm (x1, point->x, z2, ctx);
1186  ec_invm (z3, z3, ctx);
1187  ec_mulm (y1, yy, z3, ctx);
1188  mpi_free (z2);
1189  mpi_free (z3);
1190  }
1191  z1 = mpi_copy (mpi_const (MPI_C_ONE));
1192 
1193  mpi_mul (h, k, mpi_const (MPI_C_THREE)); /* h = 3k */
1194  loops = mpi_get_nbits (h);
1195  if (loops < 2)
1196  {
1197  /* If SCALAR is zero, the above mpi_mul sets H to zero and thus
1198  LOOPs will be zero. To avoid an underflow of I in the main
1199  loop we set LOOP to 2 and the result to (0,0,0). */
1200  loops = 2;
1201  mpi_clear (result->x);
1202  mpi_clear (result->y);
1203  mpi_clear (result->z);
1204  }
1205  else
1206  {
1207  mpi_set (result->x, point->x);
1208  mpi_set (result->y, yy);
1209  mpi_set (result->z, point->z);
1210  }
1211  mpi_free (yy); yy = NULL;
1212 
1213  p1.x = x1; x1 = NULL;
1214  p1.y = y1; y1 = NULL;
1215  p1.z = z1; z1 = NULL;
1216  point_init (&p2);
1217  point_init (&p1inv);
1218 
1219  /* Invert point: y = p - y mod p */
1220  point_set (&p1inv, &p1);
1221  ec_subm (p1inv.y, ctx->p, p1inv.y, ctx);
1222 
1223  for (i=loops-2; i > 0; i--)
1224  {
1225  _gcry_mpi_ec_dup_point (result, result, ctx);
1226  if (mpi_test_bit (h, i) == 1 && mpi_test_bit (k, i) == 0)
1227  {
1228  point_set (&p2, result);
1229  _gcry_mpi_ec_add_points (result, &p2, &p1, ctx);
1230  }
1231  if (mpi_test_bit (h, i) == 0 && mpi_test_bit (k, i) == 1)
1232  {
1233  point_set (&p2, result);
1234  _gcry_mpi_ec_add_points (result, &p2, &p1inv, ctx);
1235  }
1236  }
1237 
1238  point_free (&p1);
1239  point_free (&p2);
1240  point_free (&p1inv);
1241  mpi_free (h);
1242  mpi_free (k);
1243 }

1246 /* Return true if POINT is on the curve described by CTX. */
1247 int
1248 _gcry_mpi_ec_curve_point (gcry_mpi_point_t point, mpi_ec_t ctx)
1249 {
1250  int res = 0;
1251  gcry_mpi_t x, y, w;
1252 
1253  x = mpi_new (0);
1254  y = mpi_new (0);
1255  w = mpi_new (0);
1256 
1257  if (_gcry_mpi_ec_get_affine (x, y, point, ctx))
1258  goto leave;
1259 
1260  switch (ctx->model)
1261  {
1262  case MPI_EC_WEIERSTRASS:
1263  {
1264  gcry_mpi_t xxx = mpi_new (0);
1265 
1266  /* y^2 == x^3 + a·x + b */
1267  ec_pow2 (y, y, ctx);
1268 
1269  ec_pow3 (xxx, x, ctx);
1270  ec_mulm (w, ctx->a, x, ctx);
1271  ec_addm (w, w, ctx->b, ctx);
1272  ec_addm (w, w, xxx, ctx);
1273 
1274  if (!mpi_cmp (y, w))
1275  res = 1;
1276 
1277  _gcry_mpi_release (xxx);
1278  }
1279  break;
1280  case MPI_EC_MONTGOMERY:
1281  log_fatal ("%s: %s not yet supported\n",
1282  "_gcry_mpi_ec_curve_point", "Montgomery");
1283  break;
1284  case MPI_EC_TWISTEDEDWARDS:
