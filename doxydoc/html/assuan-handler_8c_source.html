75 static const char std_help_option[] =
76  "OPTION <NAME> [ [=] <VALUE> ]\n"
77  "\n"
78  "Set option <NAME> to configure server operation. Leading and\n"
79  "trailing spaces around <NAME> and <VALUE> are allowed but should be\n"
80  "ignored. For compatibility reasons, <NAME> may be prefixed with two\n"
81  "dashes. The use of the equal sign is optional but suggested if\n"
82  "<VALUE> is given.";
83 static gpg_error_t
84 std_handler_option (assuan_context_t ctx, char *line)
85 {
86  char *key, *value, *p;
87 
88  for (key=line; spacep (key); key++)
89  ;
90  if (!*key)
91  return
92  PROCESS_DONE (ctx, set_error (ctx, GPG_ERR_ASS_SYNTAX, "argument required"));
93  if (*key == '=')
94  return
95  PROCESS_DONE (ctx, set_error (ctx, GPG_ERR_ASS_SYNTAX,
96  "no option name given"));
97  for (value=key; *value && !spacep (value) && *value != '='; value++)
98  ;
99  if (*value)
100  {
101  if (spacep (value))
102  *value++ = 0; /* terminate key */
103  for (; spacep (value); value++)
104  ;
105  if (*value == '=')
106  {
107  *value++ = 0; /* terminate key */
108  for (; spacep (value); value++)
109  ;
110  if (!*value)
111  return
112  PROCESS_DONE (ctx, set_error (ctx, GPG_ERR_ASS_SYNTAX,
113  "option argument expected"));
114  }
115  if (*value)
116  {
117  for (p = value + strlen(value) - 1; p > value && spacep (p); p--)
118  ;
119  if (p > value)
120  *++p = 0; /* strip trailing spaces */
121  }
122  }
123 
124  if (*key == '-' && key[1] == '-' && key[2])
125  key += 2; /* the double dashes are optional */
126  if (*key == '-')
127  return PROCESS_DONE (ctx,
128  set_error (ctx, GPG_ERR_ASS_SYNTAX,
129  "option should not begin with one dash"));
130 
131  if (ctx->option_handler_fnc)
132  return PROCESS_DONE (ctx, ctx->option_handler_fnc (ctx, key, value));
133  return PROCESS_DONE (ctx, 0);
134 }

191 std_handler_help (assuan_context_t ctx, char *line)
192 {
193  unsigned int i;
194  char buf[ASSUAN_LINELENGTH];
195  const char *helpstr;
196  size_t n;
197 
198  n = strcspn (line, " \t\n");

203  help string. */

212  snprintf (buf, sizeof (buf), "# %.*s", (int)n, helpstr);
213  else
214  snprintf (buf, sizeof (buf), "# %s", ctx->cmdtbl[i].name);
215  buf[ASSUAN_LINELENGTH - 1] = '\0';
216  assuan_write_line (ctx, buf);
217  }
218  }
219  else
220  {
221  /* Print the help for the given command. */
222  int c = line[n];
223  line[n] = 0;
224  for (i=0; ctx->cmdtbl[i].name; i++)
225  if (!my_strcasecmp (line, ctx->cmdtbl[i].name))
226  break;
227  line[n] = c;
228  if (!ctx->cmdtbl[i].name)
229  return PROCESS_DONE (ctx, set_error (ctx,GPG_ERR_UNKNOWN_COMMAND,NULL));
230  helpstr = ctx->cmdtbl[i].helpstr;
231  if (!helpstr)
232  return PROCESS_DONE (ctx, set_error (ctx, GPG_ERR_NOT_FOUND, NULL));
233  do
234  {
235  n = strcspn (helpstr, "\n");
236  snprintf (buf, sizeof (buf), "# %.*s", (int)n, helpstr);
237  helpstr += n;
238  if (*helpstr == '\n')
239  helpstr++;
240  buf[ASSUAN_LINELENGTH - 1] = '\0';
241  assuan_write_line (ctx, buf);
242  }
243  while (*helpstr);
244  }
245 
246  return PROCESS_DONE (ctx, 0);

260 gpg_error_t
261 assuan_command_parse_fd (assuan_context_t ctx, char *line, assuan_fd_t *rfd)
262 {
263  char *endp;
264 
265  if ((strncmp (line, "FD", 2) && strncmp (line, "fd", 2))
266  || (line[2] != '=' && line[2] != '\0' && !spacep(&line[2])))
267  return set_error (ctx, GPG_ERR_ASS_SYNTAX, "FD[=<n>] expected");
268  line += 2;
269  if (*line == '=')
270  {
271  line ++;
272  if (!digitp (*line))
273  return set_error (ctx, GPG_ERR_ASS_SYNTAX, "number required");
274 #if HAVE_W64_SYSTEM
275  *rfd = (void*)strtoull (line, &endp, 10);
276 #elif HAVE_W32_SYSTEM
277  *rfd = (void*)strtoul (line, &endp, 10);
278 #else
279  *rfd = strtoul (line, &endp, 10);
280 #endif
281  /* Remove that argument so that a notify handler won't see it. */
282  memset (line, ' ', endp? (endp-line):strlen(line));
283 
284  if (*rfd == ctx->inbound.fd)
285  return set_error (ctx, GPG_ERR_ASS_PARAMETER, "fd same as inbound fd");
286  if (*rfd == ctx->outbound.fd)
287  return set_error (ctx, GPG_ERR_ASS_PARAMETER, "fd same as outbound fd");
288  return 0;
289  }
290  else
291  /* Our peer has sent the file descriptor. */
292  return assuan_receivefd (ctx, rfd);
293 }

415 gpg_error_t
416 assuan_register_command (assuan_context_t ctx, const char *cmd_name,
417  assuan_handler_t handler, const char *help_string)
418 {
419  int i, cmd_index = -1;
420  const char *s;
421 
422  if (cmd_name && !*cmd_name)
423  cmd_name = NULL;
424 
425  if (!cmd_name)
426  return _assuan_error (ctx, GPG_ERR_ASS_INV_VALUE);
427 
428  if (!handler)
429  { /* find a default handler. */
430  for (i=0; (s=std_cmd_table[i].name) && strcmp (cmd_name, s); i++)
431  ;
432  if (!s)
433  { /* Try again but case insensitive. */
434  for (i=0; (s=std_cmd_table[i].name)
435  && my_strcasecmp (cmd_name, s); i++)
436  ;
437  }
438  if (s)
439  handler = std_cmd_table[i].handler;
440  if (!handler)
441  handler = dummy_handler; /* Last resort is the dummy handler. */
442  }
443 
444  if (!ctx->cmdtbl)
445  {
446  ctx->cmdtbl_size = 50;
447  ctx->cmdtbl = _assuan_calloc (ctx, ctx->cmdtbl_size, sizeof *ctx->cmdtbl);
448  if (!ctx->cmdtbl)
449  return _assuan_error (ctx, gpg_err_code_from_syserror ());
450  ctx->cmdtbl_used = 0;
451  }
452  else if (ctx->cmdtbl_used >= ctx->cmdtbl_size)
453  {
454  struct cmdtbl_s *x;
455 
456  x = _assuan_realloc (ctx, ctx->cmdtbl, (ctx->cmdtbl_size+10) * sizeof *x);
457  if (!x)
458  return _assuan_error (ctx, gpg_err_code_from_syserror ());
459  ctx->cmdtbl = x;
460  ctx->cmdtbl_size += 50;
461  }
462 
463  for (i=0; i<ctx->cmdtbl_used; i++)
464  {
465  if (!my_strcasecmp (cmd_name, ctx->cmdtbl[i].name))
466  {
467  cmd_index = i;
468  break;
469  }
470  }
471 
472  if (cmd_index == -1)
473  cmd_index = ctx->cmdtbl_used++;
474 
475  ctx->cmdtbl[cmd_index].name = cmd_name;
476  ctx->cmdtbl[cmd_index].handler = handler;
477  ctx->cmdtbl[cmd_index].helpstr = help_string;
478  return 0;
479 }

598 /* like ascii_strcasecmp but assume that B is already uppercase */
599 static int
600 my_strcasecmp (const char *a, const char *b)
601 {
602  if (a == b)
603  return 0;
604 
605  for (; *a && *b; a++, b++)
606  {
607  if (((*a >= 'a' && *a <= 'z')? (*a&~0x20):*a) != *b)
608  break;
609  }
610  return *a == *b? 0 : (((*a >= 'a' && *a <= 'z')? (*a&~0x20):*a) - *b);
611 }

614 /* Parse the line, break out the command, find it in the command
615  table, remove leading and white spaces from the arguments, call the
616  handler with the argument line and return the error. */
617 static gpg_error_t
618 dispatch_command (assuan_context_t ctx, char *line, int linelen)
619 {
620  gpg_error_t err;
621  char *p;
622  const char *s;
623  int shift, i;
624 
625  /* Note that as this function is invoked by assuan_process_next as
626  well, we need to hide non-critical errors with PROCESS_DONE. */
627 
628  if (*line == 'D' && line[1] == ' ') /* divert to special handler */
629  /* FIXME: Depending on the final implementation of
630  handle_data_line, this may be wrong here. For example, if a
631  user callback is invoked, and that callback is responsible for
632  calling assuan_process_done, then this is wrong. */
633  return PROCESS_DONE (ctx, handle_data_line (ctx, line+2, linelen-2));
634 
635  for (p=line; *p && *p != ' ' && *p != '\t'; p++)
636  ;
637  if (p==line)
638  return PROCESS_DONE
639  (ctx, set_error (ctx, GPG_ERR_ASS_SYNTAX, "leading white-space"));
640  if (*p)
641  { /* Skip over leading WS after the keyword */
642  *p++ = 0;
643  while ( *p == ' ' || *p == '\t')
644  p++;
645  }
646  shift = p - line;
647 
648  for (i=0; (s=ctx->cmdtbl[i].name); i++)
649  {
650  if (!strcmp (line, s))
651  break;
652  }
653  if (!s)
654  { /* and try case insensitive */
655  for (i=0; (s=ctx->cmdtbl[i].name); i++)
656  {
657  if (!my_strcasecmp (line, s))
658  break;
659  }
660  }
661  if (!s)

682 gpg_error_t
683 assuan_process_done (assuan_context_t ctx, gpg_error_t rc)
684 {
685  if (!ctx->in_command)
686  return _assuan_error (ctx, GPG_ERR_ASS_GENERAL);
687 
688  if (ctx->flags.force_close)
689  ctx->process_complete = 1;
690 
691  ctx->in_command = 0;
692 
693  /* Check for data write errors. */
694  if (ctx->outbound.data.fp)
695  {
696  /* Flush the data lines. */
697  fclose (ctx->outbound.data.fp);
698  ctx->outbound.data.fp = NULL;
699  if (!rc && ctx->outbound.data.error)
700  rc = ctx->outbound.data.error;
701  }
702  else
703  {
704  /* Flush any data send without using the data FP. */
705  assuan_send_data (ctx, NULL, 0);
706  if (!rc && ctx->outbound.data.error)
707  rc = ctx->outbound.data.error;
708  }
709 
710  /* Error handling. */
711  if (!rc)
712  {
713  if (ctx->process_complete)
714  {
715  /* No error checking because the peer may have already
716  disconnect. */
717  assuan_write_line (ctx, "OK closing connection");
718  ctx->finish_handler (ctx);
719  }
720  else
721  rc = assuan_write_line (ctx, ctx->okay_line ? ctx->okay_line : "OK");
722  }
723  else
724  {
725  char errline[300];
726  const char *text = ctx->err_no == rc ? ctx->err_str : NULL;
727  char ebuf[50];
728 
729  if (ctx->flags.force_close)
730  text = "[closing connection]";
731 
732  gpg_strerror_r (rc, ebuf, sizeof (ebuf));
733  snprintf (errline, sizeof errline, "ERR %d %.50s <%.30s>%s%.100s",
734  rc, ebuf, gpg_strsource (rc),
735  text? " - ":"", text?text:"");
736 
737  rc = assuan_write_line (ctx, errline);
738 
739  if (ctx->flags.force_close)
740  ctx->finish_handler (ctx);
741  }
742 
743  if (ctx->post_cmd_notify_fnc)
744  ctx->post_cmd_notify_fnc (ctx, rc);
745 
746  ctx->flags.confidential = 0;
747  if (ctx->okay_line)
748  {
749  _assuan_free (ctx, ctx->okay_line);
750  ctx->okay_line = NULL;
751  }
752 
753  return rc;

757 static gpg_error_t
758 process_next (assuan_context_t ctx)
759 {
760  gpg_error_t rc;
761 
762  /* What the next thing to do is depends on the current state.
763  However, we will always first read the next line. The client is
764  required to write full lines without blocking long after starting
765  a partial line. */
766  rc = _assuan_read_line (ctx);
767  if (_assuan_error_is_eagain (ctx, rc))
768  return 0;
769  if (gpg_err_code (rc) == GPG_ERR_EOF)
770  {
771  ctx->process_complete = 1;
772  return 0;
773  }
774  if (rc)
775  return rc;
776  if (*ctx->inbound.line == '#' || !ctx->inbound.linelen)
777  /* Comment lines are ignored. */
778  return 0;
779 
780  /* Now we have a line that really means something. It could be one
781  of the following things: First, if we are not in a command
782  already, it is the next command to dispatch. Second, if we are
783  in a command, it can only be the response to an INQUIRE
784  reply. */
785 
786  if (!ctx->in_command)
787  {
788  ctx->in_command = 1;
789 
790  ctx->outbound.data.error = 0;
791  ctx->outbound.data.linelen = 0;
792  /* Dispatch command and return reply. */
793  ctx->in_process_next = 1;
794  rc = dispatch_command (ctx, ctx->inbound.line, ctx->inbound.linelen);
795  ctx->in_process_next = 0;
796  }
797  else if (ctx->in_inquire)
798  {
799  /* FIXME: Pick up the continuation. */
800  rc = _assuan_inquire_ext_cb (ctx);
801  }
802  else
803  {
804  /* Should not happen. The client is sending data while we are
805  in a command and not waiting for an inquire. We log an error
806  and discard it. */
807  TRACE0 (ctx, ASSUAN_LOG_DATA, "process_next", ctx,
808  "unexpected client data");
809  rc = 0;
810  }
811 
812  return rc;
813 }

816 /* This function should be invoked when the assuan connected FD is
817  ready for reading. If the equivalent to EWOULDBLOCK is returned
818  (this should be done by the command handler), assuan_process_next
819  should be invoked the next time the connected FD is readable.
820  Eventually, the caller will finish by invoking assuan_process_done.
821  DONE is set to 1 if the connection has ended. */
822 gpg_error_t
823 assuan_process_next (assuan_context_t ctx, int *done)
824 {
825  gpg_error_t rc;
826 
827  if (done)
828  *done = 0;
829  ctx->process_complete = 0;
830  do
831  {
832  rc = process_next (ctx);
833  }
834  while (!rc && !ctx->process_complete && assuan_pending_line (ctx));
835 
836  if (done)
837  *done = !!ctx->process_complete;
838 
839  return rc;
840 }

844 static gpg_error_t
845 process_request (assuan_context_t ctx)
846 {
847  gpg_error_t rc;
848 
849  if (ctx->in_inquire)
850  return _assuan_error (ctx, GPG_ERR_ASS_NESTED_COMMANDS);
851 
852  do
853  {
854  rc = _assuan_read_line (ctx);
855  }
856  while (_assuan_error_is_eagain (ctx, rc));
857  if (gpg_err_code (rc) == GPG_ERR_EOF)
858  {
859  ctx->process_complete = 1;
860  return 0;
861  }
862  if (rc)
863  return rc;
864  if (*ctx->inbound.line == '#' || !ctx->inbound.linelen)
865  return 0; /* comment line - ignore */
866 
867  ctx->in_command = 1;
868  ctx->outbound.data.error = 0;
869  ctx->outbound.data.linelen = 0;
870  /* dispatch command and return reply */
871  rc = dispatch_command (ctx, ctx->inbound.line, ctx->inbound.linelen);
872 
873  return assuan_process_done (ctx, rc);
874 }
