103 /* Clear the error and eof indicators for READER, so that it can be
104  continued to use. Also dicards any unread bytes. This is usually
105  required if the upper layer want to send to send an EOF to indicate
106  the logical end of one part of a file. If BUFFER and BUFLEN are
107  not NULL, possible unread data is copied to a newly allocated
108  buffer and this buffer is assigned to BUFFER, BUFLEN will be set to
109  the length of the unread bytes. */
110 gpg_error_t
111 ksba_reader_clear (ksba_reader_t r, unsigned char **buffer, size_t *buflen)
112 {
113  size_t n;
114 
115  if (!r)
116  return gpg_error (GPG_ERR_INV_VALUE);
117 
118  r->eof = 0;
119  r->error = 0;
120  r->nread = 0;
121  n = r->unread.length;
122  r->unread.length = 0;
123 
124  if (buffer && buflen)
125  {
126  *buffer = NULL;
127  *buflen = 0;
128  if (n)
129  {
130  *buffer = xtrymalloc (n);
131  if (!*buffer)
132  return gpg_error_from_errno (errno);
133  memcpy (*buffer, r->unread.buf, n);
134  *buflen = n;
135  }
136  }
137 
138  return 0;
139 }

307 gpg_error_t
308 ksba_reader_read (ksba_reader_t r, char *buffer, size_t length, size_t *nread)
309 {
310  size_t nbytes;
311 
312  if (!r || !nread)
313  return gpg_error (GPG_ERR_INV_VALUE);
314 
315 
316  if (!buffer)
317  {
318  if (r->type != READER_TYPE_MEM)
319  return gpg_error (GPG_ERR_NOT_IMPLEMENTED);
320  *nread = r->u.mem.size - r->u.mem.readpos;
321  if (r->unread.buf)
322  *nread += r->unread.length - r->unread.readpos;
323  return *nread? 0 : gpg_error (GPG_ERR_EOF);
324  }
325 
326  *nread = 0;
327 
328  if (r->unread.buf && r->unread.length)
329  {
330  nbytes = r->unread.length - r->unread.readpos;
331  if (!nbytes)
332  return gpg_error (GPG_ERR_BUG);
333 
334  if (nbytes > length)
335  nbytes = length;
336  memcpy (buffer, r->unread.buf + r->unread.readpos, nbytes);
337  r->unread.readpos += nbytes;
338  if (r->unread.readpos == r->unread.length)
339  r->unread.readpos = r->unread.length = 0;
340  *nread = nbytes;
341  r->nread += nbytes;
342  return 0;
343  }
344 
345 
346  if (!r->type)
347  {
348  r->eof = 1;
349  return gpg_error (GPG_ERR_EOF);
350  }
351  else if (r->type == READER_TYPE_MEM)
352  {
353  nbytes = r->u.mem.size - r->u.mem.readpos;
354  if (!nbytes)
355  {
356  r->eof = 1;
357  return gpg_error (GPG_ERR_EOF);
358  }
359 
360  if (nbytes > length)
361  nbytes = length;
362  memcpy (buffer, r->u.mem.buffer + r->u.mem.readpos, nbytes);
363  *nread = nbytes;
364  r->nread += nbytes;
365  r->u.mem.readpos += nbytes;
366  }
367  else if (r->type == READER_TYPE_FILE)
368  {
369  size_t n;
370 
371  if (r->eof)
372  return gpg_error (GPG_ERR_EOF);
373 
374  if (!length)
375  {
376  *nread = 0;
377  return 0;
378  }
379 
380  n = fread (buffer, 1, length, r->u.file);
381  if (n)
382  {
383  r->nread += n;
384  *nread = n;
385  }
386  else
387  *nread = 0;
388  if (n < length)
389  {
390  if (ferror(r->u.file))
391  r->error = errno;
392  r->eof = 1;
393  if (!n)
394  return gpg_error (GPG_ERR_EOF);
395  }
396  }
397  else if (r->type == READER_TYPE_CB)
398  {
399  if (r->eof)
400  return gpg_error (GPG_ERR_EOF);
401 
402  if (r->u.cb.fnc (r->u.cb.value, buffer, length, nread))
403  {
404  *nread = 0;
405  r->eof = 1;
406  return gpg_error (GPG_ERR_EOF);
407  }
408  r->nread += *nread;
409  }
410  else
411  return gpg_error (GPG_ERR_BUG);
412 
413  return 0;
414 }

416 gpg_error_t
417 ksba_reader_unread (ksba_reader_t r, const void *buffer, size_t count)
418 {
419  if (!r || !buffer)
420  return gpg_error (GPG_ERR_INV_VALUE);
421  if (!count)
422  return 0;
423 
424  /* Make sure that we do not push more bytes back than we have read.
425  Otherwise r->nread won't have a clear semantic. */
426  if (r->nread < count)
427  return gpg_error (GPG_ERR_CONFLICT);
428 
429  if (!r->unread.buf)
430  {
431  r->unread.size = count + 100;
432  r->unread.buf = xtrymalloc (r->unread.size);
433  if (!r->unread.buf)
434  return gpg_error (GPG_ERR_ENOMEM);
435  r->unread.length = count;
436  r->unread.readpos = 0;
437  memcpy (r->unread.buf, buffer, count);
438  r->nread -= count;
439  }
440  else if (r->unread.length + count < r->unread.size)
441  {
442  memcpy (r->unread.buf+r->unread.length, buffer, count);
443  r->unread.length += count;
444  r->nread -= count;
445  }
446  else
447  return gpg_error (GPG_ERR_NOT_IMPLEMENTED); /* fixme: easy to do */
448 
449  return 0;
450 }
