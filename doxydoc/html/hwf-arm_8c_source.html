43 static int
44 get_hwcap(unsigned int *hwcap)
45 {
46  struct { unsigned int a_type; unsigned int a_val; } auxv;
47  FILE *f;
48  int err = -1;
49  static int hwcap_initialized = 0;
50  static unsigned int stored_hwcap;
51 
52  if (hwcap_initialized)
53  {
54  *hwcap = stored_hwcap;
55  return 0;
56  }
57 
58  f = fopen("/proc/self/auxv", "r");
59  if (!f)
60  {
61  *hwcap = stored_hwcap;
62  return -1;
63  }
64 
65  while (fread(&auxv, sizeof(auxv), 1, f) > 0)
66  {
67  if (auxv.a_type != AT_HWCAP)
68  continue;
69 
70  stored_hwcap = auxv.a_val;
71  hwcap_initialized = 1;
72  err = 0;
73  break;
74  }
75 
76  fclose(f);
77  *hwcap = stored_hwcap;
78  return err;
79 }

100 static unsigned int
101 detect_arm_proc_cpuinfo(void)
102 {
103  char buf[1024]; /* large enough */
104  char *str_features, *str_neon;
105  FILE *f;
106  int readlen, i;
107  static int cpuinfo_initialized = 0;
108  static unsigned int stored_cpuinfo_features;
109 
110  if (cpuinfo_initialized)
111  return stored_cpuinfo_features;
112 
113  f = fopen("/proc/cpuinfo", "r");
114  if (!f)
115  return 0;
116 
117  memset (buf, 0, sizeof(buf));
118  readlen = fread (buf, 1, sizeof(buf), f);
119  fclose (f);
120  if (readlen <= 0 || readlen > sizeof(buf))
121  return 0;
122 
123  buf[sizeof(buf) - 1] = '\0';
124 
125  cpuinfo_initialized = 1;
126  stored_cpuinfo_features = 0;
127 
128  /* Find features line. */
129  str_features = strstr(buf, "Features");
130  if (!str_features)
131  return stored_cpuinfo_features;
132 
133  /* Lines to strings. */
134  for (i = 0; i < sizeof(buf); i++)
135  if (buf[i] == '\n')
136  buf[i] = '\0';
137 
138  /* Check for NEON. */
139  str_neon = strstr(str_features, " neon");
140  if (str_neon && (str_neon[5] == ' ' || str_neon[5] == '\0'))
141  stored_cpuinfo_features |= HWF_ARM_NEON;
142 
143  return stored_cpuinfo_features;
144 }
