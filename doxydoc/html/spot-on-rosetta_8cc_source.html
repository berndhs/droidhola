329 void spoton_rosetta::slotAddContact(void)
330 {
331  spoton_crypt *eCrypt = spoton::instance() ? spoton::instance()->crypts().
332  value("rosetta", 0) : 0;
333  spoton_crypt *sCrypt = spoton::instance() ? spoton::instance()->crypts().
334  value("rosetta-signature", 0) : 0;
335 
336  if(!eCrypt || !sCrypt)
337  {
338  QMessageBox::critical(this, tr("%1: Error").
339  arg(SPOTON_APPLICATION_NAME),
340  tr("Invalid spoton_crypt object(s). This is "
341  "a fatal flaw."));
342  return;
343  }
344 
345  QByteArray key
346  (ui.newContact->toPlainText().toLatin1());
347 
348  if(key.isEmpty())
349  {
350  QMessageBox::critical(this, tr("%1: Error").
351  arg(SPOTON_APPLICATION_NAME),
352  tr("Empty key(s). Really?"));
353  return;
354  }
355 
356  if(!(key.startsWith("K") || key.startsWith("k")))
357  {
358  QMessageBox::critical
359  (this, tr("%1: Error").
360  arg(SPOTON_APPLICATION_NAME),
361  tr("Invalid key(s). The provided text "
362  "must start with either the letter K or the letter k."));
363  return;
364  }
365 
366  QList<QByteArray> list(key.mid(1).split('@'));
367 
368  if(list.size() != 6)
369  {
370  QMessageBox::critical
371  (this, tr("%1: Error").
372  arg(SPOTON_APPLICATION_NAME),
373  tr("Irregular data. Expecting 6 entries, received %1.").
374  arg(list.size()));
375  return;
376  }
377 
378  QByteArray keyType(list.value(0));
379 
380  keyType = QByteArray::fromBase64(keyType);
381 
382  if(keyType != "rosetta")
383  {
384  QMessageBox::critical
385  (this, tr("%1: Error").
386  arg(SPOTON_APPLICATION_NAME),
387  tr("Invalid key type. Expecting 'rosetta'."));
388  return;
389  }
390 
391  QByteArray mPublicKey(list.value(2));
392  QByteArray mSignature(list.value(3));
393  QByteArray myPublicKey;
394  QByteArray mySPublicKey;
395  bool ok = true;
396 
397  mPublicKey = QByteArray::fromBase64(mPublicKey);
398  myPublicKey = eCrypt->publicKey(&ok);
399 
400  if(!ok)
401  {
402  QMessageBox mb(this);
403 
404 #ifdef Q_OS_MAC
405 #if QT_VERSION < 0x050000
406  mb.setAttribute(Qt::WA_MacMetalStyle, true);
407 #endif
408 #endif
409  mb.setIcon(QMessageBox::Question);
410  mb.setWindowTitle(tr("%1: Confirmation").
411  arg(SPOTON_APPLICATION_NAME));
412  mb.setWindowModality(Qt::WindowModal);
413  mb.setStandardButtons(QMessageBox::No | QMessageBox::Yes);
414  mb.setText(tr("Unable to retrieve your %1 "
415  "public key for comparison. Continue?").
416  arg(keyType.constData()));
417 
418  if(mb.exec() != QMessageBox::Yes)
419  return;
420  }
421 
422  mySPublicKey = sCrypt->publicKey(&ok);
423 
424  if(!ok)
425  {
426  QMessageBox mb(this);
427 
428 #ifdef Q_OS_MAC
429 #if QT_VERSION < 0x050000
430  mb.setAttribute(Qt::WA_MacMetalStyle, true);
431 #endif
432 #endif
433  mb.setIcon(QMessageBox::Question);
434  mb.setWindowTitle(tr("%1: Confirmation").
435  arg(SPOTON_APPLICATION_NAME));
436  mb.setWindowModality(Qt::WindowModal);
437  mb.setStandardButtons(QMessageBox::No | QMessageBox::Yes);
438  mb.setText(tr("Unable to retrieve your %1 signature "
439  "public key for comparison. Continue?").
440  arg(keyType.constData()));
441 
442  if(mb.exec() != QMessageBox::Yes)
443  return;
444  }
445 
446  if((mPublicKey == myPublicKey && !myPublicKey.isEmpty()) ||
447  (mSignature == mySPublicKey && !mySPublicKey.isEmpty()))
448  {
449  QMessageBox::critical
450  (this, tr("%1: Error").
451  arg(SPOTON_APPLICATION_NAME),
452  tr("You're attempting to add your own '%1' keys. "
453  "Please do not do this!").arg(keyType.constData()));
454  return;
455  }
456 
457  mSignature = QByteArray::fromBase64(mSignature);
458 
459  if(!spoton_crypt::isValidSignature(mPublicKey, mPublicKey,
460  mSignature))
461  {
462  QMessageBox::critical
463  (this, tr("%1: Error").
464  arg(SPOTON_APPLICATION_NAME),
465  tr("Invalid 'rosetta' public key signature."));
466  return;
467  }
468 
469  QByteArray sPublicKey(list.value(4));
470  QByteArray sSignature(list.value(5));
471 
472  sPublicKey = QByteArray::fromBase64(sPublicKey);
473  sSignature = QByteArray::fromBase64(sSignature);
474 
475  if(!spoton_crypt::isValidSignature(sPublicKey, sPublicKey,
476  sSignature))
477  {
478  QMessageBox::critical
479  (this, tr("%1: Error").
480  arg(SPOTON_APPLICATION_NAME),
481  tr("Invalid signature public key signature."));
482  return;
483  }
484 
485  QString connectionName("");
486 
487  {
488  QSqlDatabase db = spoton_misc::database(connectionName);
489 
490  db.setDatabaseName(spoton_misc::homePath() + QDir::separator() +
491  "friends_public_keys.db");
492 
493  if(db.open())
494  {
495  QByteArray name(list.value(1));
496 
497  name = QByteArray::fromBase64(name);
498 
499  if((ok = spoton_misc::saveFriendshipBundle(keyType,
500  name,
501  mPublicKey,
502  sPublicKey,
503  -1,
504  db,
505  eCrypt)))
506  if((ok = spoton_misc::saveFriendshipBundle(keyType + "-signature",
507  name,
508  sPublicKey,
509  QByteArray(),
510  -1,
511  db,
512  eCrypt)))
513  ui.newContact->selectAll();
514  }
515  else
516  ok = false;
517 
518  db.close();
519  }
520 
521  QSqlDatabase::removeDatabase(connectionName);
522 
523  if(!ok)
524  QMessageBox::critical
525  (this, tr("%1: Error").
526  arg(SPOTON_APPLICATION_NAME),

530  populateContacts();
531 }

547 void spoton_rosetta::populateContacts(void)
548 {
549  QString connectionName("");
550 
551  {
552  QSqlDatabase db = spoton_misc::database(connectionName);
553 
554  db.setDatabaseName(spoton_misc::homePath() + QDir::separator() +
555  "friends_public_keys.db");
556 
557  if(db.open())
558  {
559  QMultiMap<QString, QByteArray> names;
560  QSqlQuery query(db);
561  bool ok = true;
562  spoton_crypt *eCrypt = spoton::instance() ?
563  spoton::instance()->crypts().value("rosetta", 0) : 0;
564 
565  ui.contacts->clear();
566  query.setForwardOnly(true);
567  query.prepare("SELECT name, public_key FROM friends_public_keys "
568  "WHERE key_type_hash = ?");
569 
570  if(eCrypt)
571  query.bindValue(0, eCrypt->keyedHash(QByteArray("rosetta"), &ok).
572  toBase64());
573 
574  if(ok && query.exec())
575  while(query.next())
576  {
577  QByteArray name;
578  QByteArray publicKey;
579  bool ok = true;
580 
581  if(eCrypt)
582  name = eCrypt->decryptedAfterAuthenticated
583  (QByteArray::fromBase64(query.value(0).
584  toByteArray()),
585  &ok);
586  else
587  ok = false;
588 
589  if(ok)
590  {
591  if(eCrypt)
592  publicKey = eCrypt->decryptedAfterAuthenticated
593  (QByteArray::fromBase64(query.value(1).
594  toByteArray()),
595  &ok);
596  else
597  ok = false;
598  }
599 
600  if(ok)
601  names.insert(name, publicKey);
602  }
603 
604  QMapIterator<QString, QByteArray> it(names);
605 
606  while(it.hasNext())
607  {
608  it.next();
609 
610  QString str(it.key().trimmed());
611 
612  if(str.isEmpty())
613  ui.contacts->addItem("unknown", it.value());
614  else
615  ui.contacts->addItem(str, it.value());
616  }
617  }
618 
619  db.close();
620  }
621 
622  QSqlDatabase::removeDatabase(connectionName);
623 
624  if(ui.contacts->count() == 0)
625  ui.contacts->addItem("Empty");
626 }

628 void spoton_rosetta::slotConvert(void)
629 {
630  spoton_crypt *eCrypt = spoton::instance() ? spoton::instance()->crypts().
631  value("rosetta", 0) : 0;
632  spoton_crypt *sCrypt = spoton::instance() ? spoton::instance()->crypts().
633  value("rosetta-signature", 0) : 0;
634 
635  if(!eCrypt || !sCrypt)
636  {
637  QMessageBox::critical(this, tr("%1: Error").
638  arg(SPOTON_APPLICATION_NAME),
639  tr("Invalid spoton_crypt object(s). This is "
640  "a fatal flaw."));
641  return;
642  }
643 
644  if(ui.encrypt->isChecked())
645  {
646  QByteArray data;
647  QByteArray encryptionKey;
648  QByteArray hashKey;
649  QByteArray keyInformation;
650  QByteArray messageCode;
651  QByteArray myPublicKey;
652  QByteArray myPublicKeyHash;
653  QByteArray publicKey;
654  QByteArray signature;
655  QDataStream stream(&keyInformation, QIODevice::WriteOnly);
656  QScopedPointer<spoton_crypt> crypt;
657  QString error("");
658  bool ok = true;
659  size_t encryptionKeyLength = 0;
660 
661  if(ui.contacts->itemData(ui.contacts->currentIndex()).isNull())
662  {
663  error = tr("Invalid item data. This is a serious flaw.");
664  goto done_label1;
665  }
666 
667  if(ui.input->toPlainText().isEmpty())
668  {
669  error = tr("Please provide an actual message!");
670  goto done_label1;
671  }
672 
673  encryptionKeyLength = spoton_crypt::cipherKeyLength
674  (ui.cipher->currentText().toLatin1());
675 
676  if(encryptionKeyLength == 0)
677  {
678  error = tr("The method spoton_crypt::cipherKeyLength() "
679  "failed.");
680  goto done_label1;
681  }
682 
683  QApplication::setOverrideCursor(QCursor(Qt::WaitCursor));
684  encryptionKey.resize(static_cast<int> (encryptionKeyLength));
685  encryptionKey = spoton_crypt::veryStrongRandomBytes
686  (static_cast<size_t> (encryptionKey.length()));
687  hashKey.resize(spoton_crypt::XYZ_DIGEST_OUTPUT_SIZE_IN_BYTES);
688  hashKey = spoton_crypt::veryStrongRandomBytes
689  (static_cast<size_t> (hashKey.length()));
690  publicKey = ui.contacts->itemData(ui.contacts->currentIndex()).
691  toByteArray();
692  stream << encryptionKey
693  << hashKey
694  << ui.cipher->currentText().toLatin1()
695  << ui.hash->currentText().toLatin1();
696 
697  if(stream.status() != QDataStream::Ok)
698  ok = false;
699 
700  if(ok)
701  keyInformation = spoton_crypt::publicKeyEncrypt
702  (keyInformation, publicKey, &ok);
703 
704  if(!ok)
705  {
706  QApplication::restoreOverrideCursor();
707  error = tr("The method spoton_crypt::publicKeyEncrypt() failed or "
708  "an error occurred with the QDataStream object.");
709  goto done_label1;
710  }
711 
712  crypt.reset(new spoton_crypt(ui.cipher->currentText(),
713  ui.hash->currentText(),
714  QByteArray(),
715  encryptionKey,
716  hashKey,
717  0,
718  0,
719  ""));
720 
721  if(ui.sign->isChecked())
722  {
723  if(ok)
724  myPublicKey = eCrypt->publicKey(&ok);
725 
726  if(ok)
727  myPublicKeyHash = spoton_crypt::sha512Hash(myPublicKey, &ok);
728 
729  if(ok)
730  signature = sCrypt->digitalSignature
731  (myPublicKeyHash + ui.input->toPlainText().toUtf8(),
732  &ok);
733  }
734 
735  if(ok)
736  {
737  QDataStream stream(&data, QIODevice::WriteOnly);
738 
739  stream << myPublicKeyHash
740  << ui.input->toPlainText().toUtf8()
741  << signature;
742 
743  if(stream.status() != QDataStream::Ok)
744  ok = false;
745 
746  if(ok)
747  data = crypt->encrypted(data, &ok);
748  }
749 
750  if(ok)
751  messageCode = crypt->keyedHash(data, &ok);
752
