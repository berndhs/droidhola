68 void spoton_documentation::slotAnchorClicked(const QUrl &url)
69 {
70  QString scheme(url.scheme().toLower().trimmed());
71 
72  if(scheme != "qrc")
73  {
74  QMessageBox mb(this);
75 
76 #ifdef Q_OS_MAC
77 #if QT_VERSION < 0x050000
78  mb.setAttribute(Qt::WA_MacMetalStyle, true);
79 #endif
80 #endif
81  mb.setIcon(QMessageBox::Question);
82  mb.setStandardButtons(QMessageBox::No | QMessageBox::Yes);
83  mb.setText(tr("Are you sure that you wish to open %1?").
84  arg(url.toString()));
85  mb.setWindowModality(Qt::WindowModal);
86  mb.setWindowTitle(tr("%1: Confirmation").arg(SPOTON_APPLICATION_NAME));
87 
88  if(mb.exec() == QMessageBox::Yes)
89  QDesktopServices::openUrl(url);
90 
91  if(m_ui.textBrowser->isBackwardAvailable())
92  QTimer::singleShot(250, m_ui.textBrowser, SLOT(backward(void)));
93  else
94  QTimer::singleShot(250, this, SLOT(slotReload(void)));
95  }
96 }

113 {

116 
117 #ifdef Q_OS_MAC

121  connect(&printDialog,

125  QApplication::setOverrideCursor(QCursor(Qt::WaitCursor));
126  printDialog.show();
