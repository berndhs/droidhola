339 static void
340 check_snprintf (void)
341 {
342  char buffer[20];
343  int rc, rc2;
344  size_t tmplen, blen, blen2;
345 
346  rc = gpgrt_snprintf (buffer, 0, "%*s", 18, "");
347  if (rc != 18)
348  printf ("rc=%d\n", rc );
349  rc = gpgrt_snprintf (buffer, sizeof buffer, "%*s", 18, "");
350  if (rc != 18)
351  printf ("rc=%d, strlen(buffer)=%d\n", rc, (int)strlen (buffer));
352  rc = gpgrt_snprintf (buffer, sizeof buffer, "%*s", 19, "");
353  if (rc != 19)
354  printf ("rc=%d, strlen(buffer)=%d\n", rc, (int)strlen (buffer));
355  rc = gpgrt_snprintf (buffer, sizeof buffer, "%*s", 20, "");
356  if (rc != 20)
357  printf ("rc=%d, strlen(buffer)=%d\n", rc, (int)strlen (buffer));
358  rc = gpgrt_snprintf (buffer, sizeof buffer, "%*s", 21, "");
359  if (rc != 21)
360  printf ("rc=%d, strlen(buffer)=%d\n", rc, (int)strlen (buffer));
361 
362  for (tmplen = 0; tmplen <= sizeof buffer; tmplen++)
363  {
364  rc = gpgrt_snprintf (buffer, tmplen, "%04d%02d%02dT%02d%02d%02d",
365  1998, 9, 7, 16, 56, 05);
366  blen = strlen (buffer);
367  rc2 = snprintf (buffer, tmplen, "%04d%02d%02dT%02d%02d%02d",
368  1998, 9, 7, 16, 56, 05);
369  blen2 = strlen (buffer);
370  if (rc != rc2 || blen != blen2)
371  printf ("snprintf test with len %u gives %d instead of %d (%u,%u)\n",
372  (unsigned int)tmplen, rc, rc2,
373  (unsigned int)blen, (unsigned int)blen2);
374  }
