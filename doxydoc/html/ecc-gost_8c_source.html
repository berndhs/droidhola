36 /* Compute an GOST R 34.10-01/-12 signature.
37  * Return the signature struct (r,s) from the message hash. The caller
38  * must have allocated R and S.
39  */
40 gpg_err_code_t
41 _gcry_ecc_gost_sign (gcry_mpi_t input, ECC_secret_key *skey,
42  gcry_mpi_t r, gcry_mpi_t s)
43 {
44  gpg_err_code_t rc = 0;
45  gcry_mpi_t k, dr, sum, ke, x, e;
46  mpi_point_struct I;
47  gcry_mpi_t hash;
48  const void *abuf;
49  unsigned int abits, qbits;
50  mpi_ec_t ctx;
51 
52  if (DBG_CIPHER)
53  log_mpidump ("gost sign hash ", input );
54 
55  qbits = mpi_get_nbits (skey->E.n);
56 
57  /* Convert the INPUT into an MPI if needed. */
58  if (mpi_is_opaque (input))
59  {
60  abuf = mpi_get_opaque (input, &abits);
61  rc = _gcry_mpi_scan (&hash, GCRYMPI_FMT_USG, abuf, (abits+7)/8, NULL);
62  if (rc)
63  return rc;
64  if (abits > qbits)
65  mpi_rshift (hash, hash, abits - qbits);
66  }
67  else
68  hash = input;
69 
70 
71  k = NULL;
72  dr = mpi_alloc (0);
73  sum = mpi_alloc (0);
74  ke = mpi_alloc (0);
75  e = mpi_alloc (0);
76  x = mpi_alloc (0);
77  point_init (&I);
78 
79  ctx = _gcry_mpi_ec_p_internal_new (skey->E.model, skey->E.dialect, 0,
80  skey->E.p, skey->E.a, skey->E.b);
81 
82  mpi_mod (e, input, skey->E.n); /* e = hash mod n */
83 
84  if (!mpi_cmp_ui (e, 0))
85  mpi_set_ui (e, 1);
86 
87  /* Two loops to avoid R or S are zero. This is more of a joke than
88  a real demand because the probability of them being zero is less
89  than any hardware failure. Some specs however require it. */
90  do
91  {
92  do
93  {
94  mpi_free (k);
95  k = _gcry_dsa_gen_k (skey->E.n, GCRY_STRONG_RANDOM);
96 
97  _gcry_mpi_ec_mul_point (&I, k, &skey->E.G, ctx);
98  if (_gcry_mpi_ec_get_affine (x, NULL, &I, ctx))
99  {
100  if (DBG_CIPHER)
101  log_debug ("ecc sign: Failed to get affine coordinates\n");
102  rc = GPG_ERR_BAD_SIGNATURE;
103  goto leave;
104  }
105  mpi_mod (r, x, skey->E.n); /* r = x mod n */
106  }
107  while (!mpi_cmp_ui (r, 0));
108  mpi_mulm (dr, skey->d, r, skey->E.n); /* dr = d*r mod n */
109  mpi_mulm (ke, k, e, skey->E.n); /* ke = k*e mod n */
110  mpi_addm (s, ke, dr, skey->E.n); /* sum = (k*e+ d*r) mod n */
111  }
112  while (!mpi_cmp_ui (s, 0));
113 
114  if (DBG_CIPHER)
115  {
116  log_mpidump ("gost sign result r ", r);
117  log_mpidump ("gost sign result s ", s);
118  }
119 
120  leave:
121  _gcry_mpi_ec_free (ctx);
122  point_free (&I);
123  mpi_free (x);
124  mpi_free (e);
125  mpi_free (ke);
126  mpi_free (sum);
127  mpi_free (dr);
128  mpi_free (k);
129 
130  if (hash != input)
131  mpi_free (hash);
132 
133  return rc;
134 }

205  if (mpi_cmp (x, r)) /* x != r */
