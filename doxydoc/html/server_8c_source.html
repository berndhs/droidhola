31 void
32 _assuan_server_finish (assuan_context_t ctx)
33 {
34  if (ctx->inbound.fd != ASSUAN_INVALID_FD)
35  {
36  _assuan_close (ctx, ctx->inbound.fd);
37  if (ctx->inbound.fd == ctx->outbound.fd)
38  ctx->outbound.fd = ASSUAN_INVALID_FD;
39  ctx->inbound.fd = ASSUAN_INVALID_FD;
40  }
41  if (ctx->outbound.fd != ASSUAN_INVALID_FD)
42  {
43  _assuan_close (ctx, ctx->outbound.fd);
44  ctx->outbound.fd = ASSUAN_INVALID_FD;
45  }
46  if (ctx->pid != ASSUAN_INVALID_PID && ctx->pid)
47  {
48  _assuan_waitpid (ctx, ctx->pid, ctx->flags.no_waitpid, NULL, 0);
49  ctx->pid = ASSUAN_INVALID_PID;
50  }
51 
52  _assuan_uds_deinit (ctx);
53 
54  _assuan_inquire_release (ctx);
55 }
