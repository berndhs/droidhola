294 static gpg_error_t
295 do_writer_write (ksba_writer_t w, const void *buffer, size_t length)
296 {
297  if (!w->type)
298  {
299  w->error = EINVAL;
300  return gpg_error_from_errno (w->error);
301  }
302  else if (w->type == WRITER_TYPE_MEM)
303  {
304  if (w->error == ENOMEM)
305  return gpg_error (GPG_ERR_ENOMEM); /* it does not make sense to proceed then */
306 
307  if (w->nwritten + length > w->u.mem.size)
308  {
309  size_t newsize = w->nwritten + length;
310  char *p;
311 
312  newsize = ((newsize + 4095)/4096)*4096;
313  if (newsize < 16384)
314  newsize += 4096;
315  else
316  newsize += 16384;
317 
318  p = xtryrealloc (w->u.mem.buffer, newsize);
319  if (!p)
320  {
321  /* Keep an error flag so that the user does not need to
322  check the return code of a write but instead use
323  ksba_writer_error() to check for it or even figure
324  this state out when using ksba_writer_get_mem() */
325  w->error = ENOMEM;
326  return gpg_error (GPG_ERR_ENOMEM);
327  }
328  w->u.mem.buffer = p;
329  w->u.mem.size = newsize;
330  /* Better check again in case of an overwrap. */
331  if (w->nwritten + length > w->u.mem.size)
332  return gpg_error (GPG_ERR_ENOMEM);
333  }
334  memcpy (w->u.mem.buffer + w->nwritten, buffer, length);
335  w->nwritten += length;
336  }
337  else if (w->type == WRITER_TYPE_FILE)
338  {
339  if (!length)
340  return 0;
341 
342  if ( fwrite (buffer, length, 1, w->u.file) == 1)
343  {
344  w->nwritten += length;
345  }
346  else
347  {
348  w->error = errno;
349  return gpg_error_from_errno (errno);
350  }
351  }
352  else if (w->type == WRITER_TYPE_CB)
353  {
354  int err = w->u.cb.fnc (w->u.cb.value, buffer, length);
355  if (err)
356  return err;
357  w->nwritten += length;
358  }
359  else
360  return gpg_error (GPG_ERR_BUG);
361 
362  return 0;
363 }

375 gpg_error_t
376 ksba_writer_write (ksba_writer_t w, const void *buffer, size_t length)
377 {
378  gpg_error_t err=0;
379 
380  if (!w)
381  return gpg_error (GPG_ERR_INV_VALUE);
382 
383  if (!buffer)
384  return gpg_error (GPG_ERR_NOT_IMPLEMENTED);
385 
386  if (w->filter)
387  {
388  char outbuf[4096];
389  size_t nin, nout;
390  const char *p = buffer;
391 
392  while (length)
393  {
394  err = w->filter (w->filter_arg, p, length, &nin,
395  outbuf, sizeof (outbuf), &nout);
396  if (err)
397  break;
398  if (nin > length || nout > sizeof (outbuf))
399  return gpg_error (GPG_ERR_BUG); /* tsss, someone else made an error */
400  err = do_writer_write (w, outbuf, nout);
401  if (err)
402  break;
403  length -= nin;
404  p += nin;
405  }
406  }
407  else
408  {
409  err = do_writer_write (w, buffer, length);
410  }
411 
412  return err;
413 }

415 /* Write LENGTH bytes of BUFFER to W while encoding it as an BER
416  encoded octet string. With FLUSH set to 1 the octet stream will be
417  terminated. If the entire octet string is available in BUFFER it
418  is a good idea to set FLUSH to 1 so that the function does not need
419  to encode the string partially. */
420 gpg_error_t
421 ksba_writer_write_octet_string (ksba_writer_t w,
422  const void *buffer, size_t length, int flush)
423 {
424  gpg_error_t err = 0;
425 
426  if (!w)
427  return gpg_error (GPG_ERR_INV_VALUE);
428 
429  if (buffer && length)
430  {
431  if (!w->ndef_is_open && !flush)
432  {
433  err = _ksba_ber_write_tl (w, TYPE_OCTET_STRING,
434  CLASS_UNIVERSAL, 1, 0);
435  if (err)
436  return err;
437  w->ndef_is_open = 1;
438  }
439 
440  err = _ksba_ber_write_tl (w, TYPE_OCTET_STRING,
441  CLASS_UNIVERSAL, 0, length);
442  if (!err)
443  err = ksba_writer_write (w, buffer, length);
444  }
445 
446  if (!err && flush && w->ndef_is_open) /* write an end tag */
447  err = _ksba_ber_write_tl (w, 0, 0, 0, 0);
448 
449  if (flush) /* Reset it even in case of an error. */
450  w->ndef_is_open = 1;
451 
452  return err;
453 }
