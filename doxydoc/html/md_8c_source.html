130 /* Return the spec structure for the hash algorithm ALGO. For an
131  unknown algorithm NULL is returned. */
132 static gcry_md_spec_t *
133 spec_from_algo (int algo)
134 {
135  int idx;
136  gcry_md_spec_t *spec;
137 
138  algo = map_algo (algo);
139 
140  for (idx = 0; (spec = digest_list[idx]); idx++)
141  if (algo == spec->algo)
142  return spec;
143  return NULL;
144 }

165 static gcry_md_spec_t *
166 spec_from_oid (const char *oid)
167 {
168  gcry_md_spec_t *spec;
169  gcry_md_oid_spec_t *oid_specs;
170  int idx, j;
171 
172  for (idx=0; (spec = digest_list[idx]); idx++)
173  {
174  oid_specs = spec->oids;
175  if (oid_specs)
176  {
177  for (j = 0; oid_specs[j].oidstring; j++)
178  if (!stricmp (oid, oid_specs[j].oidstring))
179  return spec;
180  }
181  }

183  return NULL;

187 static gcry_md_spec_t *
188 search_oid (const char *oid, gcry_md_oid_spec_t *oid_spec)
189 {
190  gcry_md_spec_t *spec;
191  int i;
192 
193  if (oid && ((! strncmp (oid, "oid.", 4))
194  || (! strncmp (oid, "OID.", 4))))
195  oid += 4;
196 
197  spec = spec_from_oid (oid);
198  if (spec && spec->oids)
199  {
200  for (i = 0; spec->oids[i].oidstring; i++)
201  if (!stricmp (oid, spec->oids[i].oidstring))
202  {
203  if (oid_spec)
204  *oid_spec = spec->oids[i];
205  return spec;
206  }
207  }
208 
209  return NULL;

213 /****************
214  * Map a string to the digest algo
215  */
216 int
217 _gcry_md_map_name (const char *string)
218 {
219  gcry_md_spec_t *spec;
220 
221  if (!string)
222  return 0;
223 
224  /* If the string starts with a digit (optionally prefixed with
225  either "OID." or "oid."), we first look into our table of ASN.1
226  object identifiers to figure out the algorithm */
227  spec = search_oid (string, NULL);

270 /****************
271  * Open a message digest handle for use with algorithm ALGO.
272  * More algorithms may be added by md_enable(). The initial algorithm
273  * may be 0.
274  */
275 static gcry_err_code_t
276 md_open (gcry_md_hd_t *h, int algo, unsigned int flags)
277 {
278  gcry_err_code_t err = 0;
279  int secure = !!(flags & GCRY_MD_FLAG_SECURE);
280  int hmac = !!(flags & GCRY_MD_FLAG_HMAC);
281  int bufsize = secure ? 512 : 1024;
282  struct gcry_md_context *ctx;
283  gcry_md_hd_t hd;
284  size_t n;
285 
286  /* Allocate a memory area to hold the caller visible buffer with it's
287  * control information and the data required by this module. Set the
288  * context pointer at the beginning to this area.
289  * We have to use this strange scheme because we want to hide the
290  * internal data but have a variable sized buffer.
291  *
292  * +---+------+---........------+-------------+
293  * !ctx! bctl ! buffer ! private !
294  * +---+------+---........------+-------------+
295  * ! ^
296  * !---------------------------!
297  *
298  * We have to make sure that private is well aligned.
299  */
300  n = sizeof (struct gcry_md_handle) + bufsize;
301  n = ((n + sizeof (PROPERLY_ALIGNED_TYPE) - 1)
302  / sizeof (PROPERLY_ALIGNED_TYPE)) * sizeof (PROPERLY_ALIGNED_TYPE);
303 
304  /* Allocate and set the Context pointer to the private data */
305  if (secure)
306  hd = xtrymalloc_secure (n + sizeof (struct gcry_md_context));
307  else
308  hd = xtrymalloc (n + sizeof (struct gcry_md_context));
309 
310  if (! hd)
311  err = gpg_err_code_from_errno (errno);
312 
313  if (! err)
314  {
315  hd->ctx = ctx = (struct gcry_md_context *) ((char *) hd + n);
316  /* Setup the globally visible data (bctl in the diagram).*/
317  hd->bufsize = n - sizeof (struct gcry_md_handle) + 1;
318  hd->bufpos = 0;
319 
320  /* Initialize the private data. */
321  memset (hd->ctx, 0, sizeof *hd->ctx);
322  ctx->magic = secure ? CTX_MAGIC_SECURE : CTX_MAGIC_NORMAL;
323  ctx->actual_handle_size = n + sizeof (struct gcry_md_context);
324  ctx->flags.secure = secure;
325  ctx->flags.bugemu1 = !!(flags & GCRY_MD_FLAG_BUGEMU1);
326 
327  if (hmac)
328  {
329  switch (algo)
330  {
331  case GCRY_MD_SHA384:
332  case GCRY_MD_SHA512:
333  ctx->macpads_Bsize = 128;
334  break;
335  case GCRY_MD_GOSTR3411_94:
336  ctx->macpads_Bsize = 32;
337  break;
338  default:
339  ctx->macpads_Bsize = 64;
340  break;
341  }
342  ctx->macpads = xtrymalloc_secure (2*(ctx->macpads_Bsize));
343  if (!ctx->macpads)
344  {
345  err = gpg_err_code_from_errno (errno);
346  md_close (hd);
347  }
348  }
349  }
350 
351  if (! err)
352  {
353  /* Hmmm, should we really do that? - yes [-wk] */
354  _gcry_fast_random_poll ();
355 
356  if (algo)
357  {
358  err = md_enable (hd, algo);
359  if (err)
360  md_close (hd);
361  }
362  }
363 
364  if (! err)
365  *h = hd;
366 
367  return err;
368 }

370 /* Create a message digest object for algorithm ALGO. FLAGS may be
371  given as an bitwise OR of the gcry_md_flags values. ALGO may be
372  given as 0 if the algorithms to be used are later set using
373  gcry_md_enable. H is guaranteed to be a valid handle or NULL on
374  error. */
375 gcry_err_code_t
376 _gcry_md_open (gcry_md_hd_t *h, int algo, unsigned int flags)
377 {
378  gcry_err_code_t rc;
379  gcry_md_hd_t hd;
380 
381  if ((flags & ~(GCRY_MD_FLAG_SECURE
382  | GCRY_MD_FLAG_HMAC
383  | GCRY_MD_FLAG_BUGEMU1)))
384  rc = GPG_ERR_INV_ARG;
385  else
386  rc = md_open (&hd, algo, flags);
387 
388  *h = rc? NULL : hd;
389  return rc;

394 static gcry_err_code_t
395 md_enable (gcry_md_hd_t hd, int algorithm)
396 {
397  struct gcry_md_context *h = hd->ctx;
398  gcry_md_spec_t *spec;
399  GcryDigestEntry *entry;
400  gcry_err_code_t err = 0;
401 
402  for (entry = h->list; entry; entry = entry->next)
403  if (entry->spec->algo == algorithm)
404  return 0; /* Already enabled */
405 
406  spec = spec_from_algo (algorithm);
407  if (!spec)
408  {
409  log_debug ("md_enable: algorithm %d not available\n", algorithm);
410  err = GPG_ERR_DIGEST_ALGO;
411  }
412 
413 
414  if (!err && algorithm == GCRY_MD_MD5 && fips_mode ())
415  {
416  _gcry_inactivate_fips_mode ("MD5 used");
417  if (_gcry_enforced_fips_mode () )
418  {
419  /* We should never get to here because we do not register
420  MD5 in enforced fips mode. But better throw an error. */
421  err = GPG_ERR_DIGEST_ALGO;
422  }
423  }
424 
425  if (!err)
426  {
427  size_t size = (sizeof (*entry)
428  + spec->contextsize
429  - sizeof (entry->context));
430 
431  /* And allocate a new list entry. */
432  if (h->flags.secure)
433  entry = xtrymalloc_secure (size);
434  else
435  entry = xtrymalloc (size);
436 
437  if (! entry)
438  err = gpg_err_code_from_errno (errno);
439  else
440  {
441  entry->spec = spec;
442  entry->next = h->list;
443  entry->actual_struct_size = size;
444  h->list = entry;
445 
446  /* And init this instance. */
447  entry->spec->init (&entry->context.c,
448  h->flags.bugemu1? GCRY_MD_FLAG_BUGEMU1:0);
449  }
450  }
451 
452  return err;
453 }

463 static gcry_err_code_t
464 md_copy (gcry_md_hd_t ahd, gcry_md_hd_t *b_hd)
465 {
466  gcry_err_code_t err = 0;
467  struct gcry_md_context *a = ahd->ctx;
468  struct gcry_md_context *b;
469  GcryDigestEntry *ar, *br;
470  gcry_md_hd_t bhd;
471  size_t n;
472 
473  if (ahd->bufpos)
474  md_write (ahd, NULL, 0);
475 
476  n = (char *) ahd->ctx - (char *) ahd;
477  if (a->flags.secure)
478  bhd = xtrymalloc_secure (n + sizeof (struct gcry_md_context));
479  else
480  bhd = xtrymalloc (n + sizeof (struct gcry_md_context));
481 
482  if (! bhd)
483  err = gpg_err_code_from_errno (errno);
484 
485  if (! err)
486  {
487  bhd->ctx = b = (struct gcry_md_context *) ((char *) bhd + n);
488  /* No need to copy the buffer due to the write above. */
489  gcry_assert (ahd->bufsize == (n - sizeof (struct gcry_md_handle) + 1));
490  bhd->bufsize = ahd->bufsize;
491  bhd->bufpos = 0;
492  gcry_assert (! ahd->bufpos);
493  memcpy (b, a, sizeof *a);
494  b->list = NULL;
495  b->debug = NULL;
496  if (a->macpads)
497  {
498  b->macpads = xtrymalloc_secure (2*(a->macpads_Bsize));
499  if (! b->macpads)
500  {
501  err = gpg_err_code_from_errno (errno);
502  md_close (bhd);
503  }
504  else
505  memcpy (b->macpads, a->macpads, (2*(a->macpads_Bsize)));
506  }
507  }
508 
509  /* Copy the complete list of algorithms. The copied list is
510  reversed, but that doesn't matter. */
511  if (!err)
512  {
513  for (ar = a->list; ar; ar = ar->next)
514  {
515  if (a->flags.secure)
516  br = xtrymalloc_secure (sizeof *br
517  + ar->spec->contextsize
518  - sizeof(ar->context));
519  else
520  br = xtrymalloc (sizeof *br
521  + ar->spec->contextsize
522  - sizeof (ar->context));
523  if (!br)
524  {
525  err = gpg_err_code_from_errno (errno);
526  md_close (bhd);
527  break;
528  }
529 
530  memcpy (br, ar, (sizeof (*br) + ar->spec->contextsize
531  - sizeof (ar->context)));
532  br->next = b->list;
533  b->list = br;
534  }
535  }
536 
537  if (a->debug && !err)
538  md_start_debug (bhd, "unknown");
539 
540  if (!err)
541  *b_hd = bhd;
542 
543  return err;
544 }

648 static void
649 md_final (gcry_md_hd_t a)
650 {
651  GcryDigestEntry *r;
652 
653  if (a->ctx->flags.finalized)
654  return;
655 
656  if (a->bufpos)
657  md_write (a, NULL, 0);
658 
659  for (r = a->ctx->list; r; r = r->next)
660  (*r->spec->final) (&r->context.c);
661 
662  a->ctx->flags.finalized = 1;
663 
664  if (a->ctx->macpads)
665  {
666  /* Finish the hmac. */
667  int algo = md_get_algo (a);
668  byte *p = md_read (a, algo);
669  size_t dlen = md_digest_length (algo);
670  gcry_md_hd_t om;
671  gcry_err_code_t err;
672 
673  err = md_open (&om, algo,
674  ((a->ctx->flags.secure? GCRY_MD_FLAG_SECURE:0)
675  | (a->ctx->flags.bugemu1? GCRY_MD_FLAG_BUGEMU1:0)));
676  if (err)
677  _gcry_fatal_error (err, NULL);
678  md_write (om,
679  (a->ctx->macpads)+(a->ctx->macpads_Bsize),
680  a->ctx->macpads_Bsize);
681  md_write (om, p, dlen);
682  md_final (om);
683  /* Replace our digest with the mac (they have the same size). */
684  memcpy (p, md_read (om, algo), dlen);
685  md_close (om);
686  }
687 }

690 static gcry_err_code_t
691 prepare_macpads (gcry_md_hd_t hd, const unsigned char *key, size_t keylen)
692 {
693  int i;
694  int algo = md_get_algo (hd);
695  unsigned char *helpkey = NULL;
696  unsigned char *ipad, *opad;
697 
698  if (!algo)
699  return GPG_ERR_DIGEST_ALGO; /* Might happen if no algo is enabled. */
700 
701  if ( keylen > hd->ctx->macpads_Bsize )
702  {
703  helpkey = xtrymalloc_secure (md_digest_length (algo));
704  if (!helpkey)
705  return gpg_err_code_from_errno (errno);
706  _gcry_md_hash_buffer (algo, helpkey, key, keylen);
707  key = helpkey;
708  keylen = md_digest_length (algo);
709  gcry_assert ( keylen <= hd->ctx->macpads_Bsize );
710  }
711 
712  memset ( hd->ctx->macpads, 0, 2*(hd->ctx->macpads_Bsize) );
713  ipad = hd->ctx->macpads;
714  opad = (hd->ctx->macpads)+(hd->ctx->macpads_Bsize);
715  memcpy ( ipad, key, keylen );
716  memcpy ( opad, key, keylen );
717  for (i=0; i < hd->ctx->macpads_Bsize; i++ )

846 /*
847  * Shortcut function to hash a buffer with a given algo. The only
848  * guaranteed supported algorithms are RIPE-MD160 and SHA-1. The
849  * supplied digest buffer must be large enough to store the resulting
850  * hash. No error is returned, the function will abort on an invalid
851  * algo. DISABLED_ALGOS are ignored here. */
852 void
853 _gcry_md_hash_buffer (int algo, void *digest,
854  const void *buffer, size_t length)
855 {
856  if (algo == GCRY_MD_SHA1)
857  _gcry_sha1_hash_buffer (digest, buffer, length);
858  else if (algo == GCRY_MD_RMD160 && !fips_mode () )
859  _gcry_rmd160_hash_buffer (digest, buffer, length);
860  else
861  {
862  /* For the others we do not have a fast function, so we use the
863  normal functions. */
864  gcry_md_hd_t h;
865  gpg_err_code_t err;
866 
867  if (algo == GCRY_MD_MD5 && fips_mode ())
868  {
869  _gcry_inactivate_fips_mode ("MD5 used");
870  if (_gcry_enforced_fips_mode () )
871  {
872  /* We should never get to here because we do not register
873  MD5 in enforced fips mode. */
874  _gcry_fips_noreturn ();
875  }
876  }
877 
878  err = md_open (&h, algo, 0);
879  if (err)
880  log_bug ("gcry_md_open failed for algo %d: %s",
881  algo, gpg_strerror (gcry_error(err)));
882  md_write (h, (byte *) buffer, length);
883  md_final (h);
884  memcpy (digest, md_read (h, algo), md_digest_length (algo));
885  md_close (h);
886  }
887 }

890 /* Shortcut function to hash multiple buffers with a given algo. In
891  contrast to gcry_md_hash_buffer, this function returns an error on
892  invalid arguments or on other problems; disabled algorithms are
893  _not_ ignored but flagged as an error.
894 
895  The data to sign is taken from the array IOV which has IOVCNT items.
896 
897  The only supported flag in FLAGS is GCRY_MD_FLAG_HMAC which turns
898  this function into a HMAC function; the first item in IOV is then
899  used as the key.
900 
901  On success 0 is returned and resulting hash or HMAC is stored at
902  DIGEST which must have been provided by the caller with an
903  appropriate length. */
904 gpg_err_code_t
905 _gcry_md_hash_buffers (int algo, unsigned int flags, void *digest,
906  const gcry_buffer_t *iov, int iovcnt)
907 {
908  int hmac;
909 
910  if (!iov || iovcnt < 0)
911  return GPG_ERR_INV_ARG;
912  if (flags & ~(GCRY_MD_FLAG_HMAC))
913  return GPG_ERR_INV_ARG;
914 
915  hmac = !!(flags & GCRY_MD_FLAG_HMAC);
916  if (hmac && iovcnt < 1)
917  return GPG_ERR_INV_ARG;
918 
919  if (algo == GCRY_MD_SHA1 && !hmac)
920  _gcry_sha1_hash_buffers (digest, iov, iovcnt);
921  else
922  {
923  /* For the others we do not have a fast function, so we use the
924  normal functions. */
925  gcry_md_hd_t h;
926  gpg_err_code_t rc;
927 
928  if (algo == GCRY_MD_MD5 && fips_mode ())
929  {
930  _gcry_inactivate_fips_mode ("MD5 used");
931  if (_gcry_enforced_fips_mode () )
932  {
933  /* We should never get to here because we do not register
934  MD5 in enforced fips mode. */
935  _gcry_fips_noreturn ();
936  }
937  }
938 
939  rc = md_open (&h, algo, (hmac? GCRY_MD_FLAG_HMAC:0));
940  if (rc)
941  return rc;
942 
943  if (hmac)
944  {
945  rc = _gcry_md_setkey (h,
946  (const char*)iov[0].data + iov[0].off,
947  iov[0].len);
948  if (rc)
949  {
950  md_close (h);
951  return rc;
952  }
953  iov++; iovcnt--;
954  }
955  for (;iovcnt; iov++, iovcnt--)
956  md_write (h, (const char*)iov[0].data + iov[0].off, iov[0].len);
957  md_final (h);
958  memcpy (digest, md_read (h, algo), md_digest_length (algo));
959  md_close (h);
960  }

1035 /****************
1036  * Return information about the given cipher algorithm
1037  * WHAT select the kind of information returned:
1038  * GCRYCTL_TEST_ALGO:
1039  * Returns 0 when the specified algorithm is available for use.
1040  * buffer and nbytes must be zero.
1041  * GCRYCTL_GET_ASNOID:
1042  * Return the ASNOID of the algorithm in buffer. if buffer is NULL, only
1043  * the required length is returned.
1044  * GCRYCTL_SELFTEST
1045  * Helper for the regression tests - shall not be used by applications.
1046  *
1047  * Note: Because this function is in most cases used to return an
1048  * integer value, we can make it easier for the caller to just look at
1049  * the return value. The caller will in all cases consult the value
1050  * and thereby detecting whether a error occurred or not (i.e. while checking
1051  * the block size)
1052  */
1053 gcry_err_code_t
1054 _gcry_md_algo_info (int algo, int what, void *buffer, size_t *nbytes)
1055 {
1056  gcry_err_code_t rc;
1057 
1058  switch (what)
1059  {
1060  case GCRYCTL_TEST_ALGO:
1061  if (buffer || nbytes)
1062  rc = GPG_ERR_INV_ARG;
1063  else
1064  rc = check_digest_algo (algo);
1065  break;
1066 
1067  case GCRYCTL_GET_ASNOID:
1068  /* We need to check that the algo is available because
1069  md_asn_oid would otherwise raise an assertion. */
1070  rc = check_digest_algo (algo);
1071  if (!rc)
1072  {
1073  const char unsigned *asn;
1074  size_t asnlen;
1075 
1076  asn = md_asn_oid (algo, &asnlen, NULL);
1077  if (buffer && (*nbytes >= asnlen))
1078  {
1079  memcpy (buffer, asn, asnlen);
1080  *nbytes = asnlen;
1081  }
1082  else if (!buffer && nbytes)
1083  *nbytes = asnlen;
1084  else
1085  {
1086  if (buffer)
1087  rc = GPG_ERR_TOO_SHORT;
1088  else
1089  rc = GPG_ERR_INV_ARG;
1090  }
1091  }
1092  break;

1095  /* Helper function for the regression tests. */
1096  rc = gpg_err_code (_gcry_md_selftest (algo, nbytes? (int)*nbytes : 0,
1097  NULL));
1098  break;
1099

1101  rc = GPG_ERR_INV_OP;
1102  break;
1103  }
