58 void *
59 _assuan_calloc (assuan_context_t ctx, size_t cnt, size_t elsize)
60 {
61  void *ptr;
62  size_t nbytes;
63 
64  nbytes = cnt * elsize;
65 
66  /* Check for overflow. */
67  if (elsize && nbytes / elsize != cnt)
68  {
69  gpg_err_set_errno (ENOMEM);
70  return NULL;
71  }
72 
73  ptr = ctx->malloc_hooks.malloc (nbytes);
74  if (ptr)
75  memset (ptr, 0, nbytes);
76  return ptr;
77 }

97 /* Copy the system hooks struct, paying attention to version
98  differences. SRC is usually from the user, DST MUST be from the
99  library. */
100 void
101 _assuan_system_hooks_copy (assuan_system_hooks_t dst,
102  assuan_system_hooks_t src)
103 
104 {
105  /* Reset the defaults. */
106  if (dst != &_assuan_system_hooks)
107  memcpy (dst, &_assuan_system_hooks, sizeof (*dst));
108 
109  dst->version = ASSUAN_SYSTEM_HOOKS_VERSION;
110  if (src->version >= 1)
111  {
112  dst->usleep = src->usleep;
113  dst->pipe = src->pipe;
114  dst->close = src->close;
115  dst->read = src->read;
116  dst->write = src->write;
117  dst->sendmsg = src->sendmsg;
118  dst->recvmsg = src->recvmsg;
119  dst->spawn = src->spawn;
120  dst->waitpid = src->waitpid;
121  dst->socketpair = src->socketpair;
122  }
123  if (src->version >= 2)
124  {
125  dst->socket = src->socket;
126  dst->connect = src->connect;
127  }
128  if (src->version > 2)
129  /* FIXME. Application uses newer version of the library. What to
130  do? */
131  ;
132 }

297 /* Create a new process from NAME and ARGV. Provide FD_IN and FD_OUT
298  as stdin and stdout. Inherit the ASSUAN_INVALID_FD-terminated
299  FD_CHILD_LIST as given (no remapping), which must be inheritable.
300  On Unix, call ATFORK with ATFORKVALUE after fork and before exec. */
301 int
302 _assuan_spawn (assuan_context_t ctx, pid_t *r_pid, const char *name,
303  const char **argv,
304  assuan_fd_t fd_in, assuan_fd_t fd_out,
305  assuan_fd_t *fd_child_list,
306  void (*atfork) (void *opaque, int reserved),
307  void *atforkvalue, unsigned int flags)
308 {
309  int res;
310  int i;
311  TRACE_BEG6 (ctx, ASSUAN_LOG_CTX, "_assuan_spawn", ctx,
312  "name=%s,fd_in=0x%x,fd_out=0x%x,"
313  "atfork=%p,atforkvalue=%p,flags=%i",
314  name ? name : "(null)", fd_in, fd_out,
315  atfork, atforkvalue, flags);
316 
317  if (name)
318  {
319  i = 0;
320  while (argv[i])
321  {
322  TRACE_LOG2 ("argv[%2i] = %s", i, argv[i]);
323  i++;
324  }
325  }
326  i = 0;
327  if (fd_child_list)
328  {
329  while (fd_child_list[i] != ASSUAN_INVALID_FD)
330  {
331  TRACE_LOG2 ("fd_child_list[%2i] = 0x%x", i, fd_child_list[i]);
332  i++;
333  }
334  }
335 
336  res = (ctx->system.spawn) (ctx, r_pid, name, argv, fd_in, fd_out,
337  fd_child_list, atfork, atforkvalue, flags);
338 
339  if (name)
340  {
341  TRACE_LOG1 ("pid = 0x%x", *r_pid);
342  }
343  else
344  {
345  TRACE_LOG2 ("pid = 0x%x (%s)", *r_pid, *argv);
346  }
347 
348  return TRACE_SYSERR (res);
349 }
