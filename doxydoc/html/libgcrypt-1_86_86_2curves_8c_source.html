96 static void
97 check_matching (void)
98 {
99  gpg_error_t err;
100  gcry_sexp_t key;
101  const char *name;
102  unsigned int nbits;
103 
104  err = gcry_sexp_new (&key, sample_key_1, 0, 1);
105  if (err)
106  die ("parsing s-expression string failed: %s\n", gpg_strerror (err));
107  name = gcry_pk_get_curve (key, 0, &nbits);
108  if (!name)
109  fail ("curve name not found for sample_key_1\n");
110  else if (strcmp (name, sample_key_1_curve))
111  fail ("expected curve name %s but got %s for sample_key_1\n",
112  sample_key_1_curve, name);
113  else if (nbits != sample_key_1_nbits)
114  fail ("expected curve size %u but got %u for sample_key_1\n",
115  sample_key_1_nbits, nbits);
116 
117  gcry_sexp_release (key);
118 
119 }
