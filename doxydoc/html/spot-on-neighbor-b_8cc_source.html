433 void spoton_neighbor::parsePrivateApplicationData
434 (const QByteArray &data,
435  const QByteArray &privateApplicationCredentials,
436  const qint64 maximumContentLength)
437 {
438  /*
439  ** The container data contains Spot-On data, that is, data does
440  ** not contain raw application data.
441  */
442 
443  if(privateApplicationCredentials.isEmpty())
444  return;
445 
446  int a = data.indexOf("Content-Length: ");
447 
448  if(a >= 0)
449  {
450  QByteArray contentLength;
451  int b = data.indexOf("\r\n", a);
452  int length = 0;
453 
454  if(b > 0)
455  {
456  a += static_cast<int> (qstrlen("Content-Length: "));
457 
458  if(a < b)
459  contentLength = data.mid(a, b - a);
460  }
461 
462  /*
463  ** toInt() returns zero on failure.
464  */
465 
466  length = contentLength.toInt();
467 
468  if(length > 0 && length <= maximumContentLength)
469  if((a = data.indexOf("content=", a)) > 0)
470  {
471  QByteArray bytes(data.mid(a));
472 
473  if(bytes.length() == length)
474  {
475  bytes = bytes.mid
476  (static_cast<int> (qstrlen("content="))).trimmed();
477 
478  if((a = bytes.lastIndexOf('\n')) > 0)
479  {
480  spoton_crypt *crypt = spoton_misc::
481  parsePrivateApplicationMagnet
482  (privateApplicationCredentials);
483 
484  if(crypt)
485  {
486  bool ok = true;
487 
488  bytes = crypt->decryptedAfterAuthenticated
489  (QByteArray::fromBase64(bytes.mid(0, a)), &ok);
490  delete crypt;
491 
492  if(ok)
493  {
494  /*
495  ** In-order:
496  ** 1
497  ** 2
498  ** 3
499  ** 4
500  ** 5
501  ** Out-of-order:
502  ** 2
503  ** 5
504  ** 3
505  ** 1
506  ** 4
507  */
508 
509  QByteArray sequencer(bytes.mid(0, 20));
510  quint64 sequence = sequencer.toULongLong();
511 
512  if(m_isUserDefined)
513  {
514  QMutexLocker locker(&m_privateApplicationMutex);
515 
516  if(m_privateApplicationSequences.first ==
517  sequence)
518  {
519  emit writeParsedApplicationData
520  (bytes.mid(20));
521  m_privateApplicationSequences.first += 1;
522  }
523  else if(m_privateApplicationSequences.first <
524  sequence)
525  /*
526  ** The received sequence number
527  ** should always increase, yes? Our
528  ** initial value is one. It is increased
529  ** if the received sequence equals the
530  ** initial value.
531  */
532 
533  m_privateApplicationMap[sequence] =
534  bytes.mid(20);
535 
536  /*
537  ** Determine if we can distribute more data.
538  */
539 
540  while(true)
541  if(m_privateApplicationMap.
542  contains(m_privateApplicationSequences.
543  first))
544  {
545  emit writeParsedApplicationData
546  (m_privateApplicationMap.
547  take(m_privateApplicationSequences.
548  first));
549  m_privateApplicationSequences.first += 1;
550  }
551  else
552  break;
553  }
554  else
555  {
556  QMutexLocker locker(&m_privateApplicationMutex);
557 
558  if(m_privateApplicationSequences.second ==
559  sequence)
560  {
561  emit writeParsedApplicationData
562  (bytes.mid(20));
563  m_privateApplicationSequences.second += 1;
564  }
565  else if(m_privateApplicationSequences.second <
566  sequence)
567  /*
568  ** The received sequence number
569  ** should always increase, yes? Our
570  ** initial value is one. It is increased
571  ** if the received sequence equals the
572  ** initial value.
573  */
574 
575  m_privateApplicationMap[sequence] =
576  bytes.mid(20);
577 
578  /*
579  ** Determine if we can distribute more data.
580  */
581 
582  while(true)
583  if(m_privateApplicationMap.
584  contains(m_privateApplicationSequences.
585  second))
586  {
587  emit writeParsedApplicationData
588  (m_privateApplicationMap.
589  take(m_privateApplicationSequences.
590  second));
591  m_privateApplicationSequences.second += 1;
592  }
593  else
594  break;
595  }
596  }
597  }
598  }
599  }
600  }
601  }
