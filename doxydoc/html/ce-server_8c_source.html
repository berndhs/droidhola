103 /* A wrapper around read to make it work under Windows with HANDLES
104  and socket descriptors. Takes care of EINTR on POSIX. */
105 static int
106 my_read (assuan_fd_t fd, void *buffer, size_t size)
107 {
108  int res;
109 
110 #ifdef HAVE_W32_SYSTEM
111  res = recv (HANDLE2SOCKET (fd), buffer, size, 0);
112  if (res == -1)
113  {
114  switch (WSAGetLastError ())
115  {
116  case WSAENOTSOCK:
117  {
118  DWORD nread = 0;
119 
120  res = ReadFile (fd, buffer, size, &nread, NULL);
121  if (!res)
122  {
123  switch (GetLastError ())
124  {
125  case ERROR_BROKEN_PIPE:
126  gpg_err_set_errno (EPIPE);
127  break;
128 
129  default:
130  gpg_err_set_errno (EIO);
131  }
132  res = -1;
133  }
134  else
135  res = (int) nread;
136  }
137  break;
138 
139  case WSAEWOULDBLOCK:
140  gpg_err_set_errno (EAGAIN);
141  break;
142 
143  case ERROR_BROKEN_PIPE:
144  gpg_err_set_errno (EPIPE);
145  break;
146 
147  default:
148  gpg_err_set_errno (EIO);
149  break;
150  }
151  }
152  return res;
153 #else
154  do
155  res = read (fd, buffer, size);
156  while (res == -1 && errno == EINTR);
157  return res;
158 #endif
159 }

162 /* Extended version of write(2) to guarantee that all bytes are
163  written. Returns 0 on success or -1 and ERRNO on failure. NOTE:
164  This function does not return the number of bytes written, so any
165  error must be treated as fatal for this connection as the state of
166  the receiver is unknown. This works best if blocking is allowed
167  (so EAGAIN cannot occur). Under Windows this function handles
168  socket descriptors and system handles. */
169 static int
170 my_writen (assuan_fd_t fd, const char *buffer, size_t length)
171 {
172  while (length)
173  {
174  int nwritten;
175 #ifdef HAVE_W32_SYSTEM
176  nwritten = send (HANDLE2SOCKET (fd), buffer, length, 0);
177  if (nwritten == -1 && WSAGetLastError () == WSAENOTSOCK)
178  {
179  DWORD nwrite;
180 
181  nwritten = WriteFile (fd, buffer, length, &nwrite, NULL);
182  if (!nwritten)
183  {
184  switch (GetLastError ())
185  {
186  case ERROR_BROKEN_PIPE:
187  case ERROR_NO_DATA:
188  gpg_err_set_errno (EPIPE);
189  break;
190 
191  default:
192  gpg_err_set_errno (EIO);
193  break;
194  }
195  nwritten= -1;
196  }
197  else
198  nwritten = (int)nwrite;
199  }
200 #else
201  nwritten = write (fd, buffer, length);
202 #endif
203  if (nwritten < 0)
204  {
205  if (errno == EINTR)
206  continue;
207  return -1; /* write error */
208  }
209  length -= nwritten;
210  buffer += nwritten;
211  }
212  return 0; /* okay */
213 }

457 static const char hlp_cat[] =
458  "CAT [<filename>]\n"
459  "\n"
460  "Copy the content of FILENAME to the descriptor set by the OUTPUT\n"
461  "command. If no OUTPUT command has been given, send the content\n"
462  "using data lines. Without FILENAME take the content from the\n"
463  "descriptor set by the INPUT command; if a DATAPORT has been set\n"
464  "this descriptor is used for I/O and the INOPUT/OUTPUT descriptors\n"
465  "are not touched.";
466 static gpg_error_t
467 cmd_cat (assuan_context_t ctx, char *line)
468 {
469  state_t state = assuan_get_pointer (ctx);
470  gpg_error_t err = 0;
471  assuan_fd_t fd_in = ASSUAN_INVALID_FD;
472  assuan_fd_t fd_out = ASSUAN_INVALID_FD;
473  FILE *fp_in = NULL;
474  char buf[256];
475  size_t nread;
476  int use_dataport = 0;
477 
478  if (*line)
479  {
480  fp_in = fopen (line, "rb");
481  if (!fp_in)
482  err = gpg_error_from_syserror ();
483  else
484  fd_out = assuan_get_output_fd (ctx);
485  }
486  else if (state->dataport_fd != ASSUAN_INVALID_FD)
487  {
488  use_dataport = 1;
489  fd_in = state->dataport_fd;
490  fd_out = state->dataport_fd;
491  }
492  else if ((fd_in = assuan_get_input_fd (ctx)) != ASSUAN_INVALID_FD)
493  {
494  /* This FD is actually a socket descriptor. We can't fdopen it
495  because under Windows we ust use recv(2) instead of read(2).
496  Note that on POSIX systems there is no difference between
497  libc file descriptors and socket descriptors. */
498 
499  fd_out = assuan_get_output_fd (ctx);
500  }
501  else
502  err = gpg_error (GPG_ERR_ASS_NO_INPUT);
503  if (err)
504  goto leave;
505 
506  do
507  {
508  if (fp_in)
509  {
510  nread = fread (buf, 1, sizeof buf, fp_in);
511  if (nread < sizeof buf)
512  {
513  if (ferror (fp_in))
514  err = gpg_error_from_syserror ();
515  else if (feof (fp_in))
516  err = gpg_error (GPG_ERR_EOF);
517  }
518  }
519  else
520  {
521  int n;
522 
523  nread = 0;
524  n = my_read (fd_in, buf, sizeof buf);
525  if (n < 0)
526  err = gpg_error_from_syserror ();
527  else if (!n)
528  err = gpg_error (GPG_ERR_EOF);
529  else
530  nread = n;
531  }
532 
533 
534  if (fd_out != ASSUAN_INVALID_FD)
535  {
536  if (nread && my_writen (fd_out, buf, nread))
537  err = gpg_error_from_syserror ();
538  }
539  else if (nread)
540  err = assuan_send_data (ctx, buf, nread);
541  }
542  while (!err);
543  if (gpg_err_code (err) == GPG_ERR_EOF)
544  err = 0;
545 
546 leave:
547  if (fp_in)
548  fclose (fp_in);
549  if (use_dataport)
550  {
551  if (state->dataport_fd != ASSUAN_INVALID_FD)
552  {
553  assuan_sock_close (state->dataport_fd);
554  state->dataport_fd = ASSUAN_INVALID_FD;
555  }
556  }
557  else
558  {
559  assuan_close_input_fd (ctx);
560  assuan_close_output_fd (ctx);
561  }
562  return leave_cmd (ctx, err);
563 }

786  }

789  {

791  {
792  err = gpg_error_from_syserror ();
793  log_error ("SetStdioPathW(%d) failed: %d\n", idx, GetLastError ());
794  goto leave;
795  }

797  {
798  err = gpg_error_from_syserror ();

800  goto leave;
801  }
802  }

805  w32ret = CreateProcess (pgmname, /* Program to start. */
806  cmdline, /* Command line arguments. */
807  NULL, /* Process security. Not used. */
808  NULL, /* Thread security. Not used. */
809  FALSE, /* Inherit handles. Not used. */
810  CREATE_SUSPENDED, /* Creation flags. */
811  NULL, /* Environment. Not used. */
812  NULL, /* Use current dir. Not used. */
813  NULL, /* Startup information. Not used. */
814  &pi /* Returns process information. */
815  );

817  {

819  {

822  idx, GetLastError ());

825  }

830  log_error ("CreateProcess failed: %d\n", GetLastError ());

836  " dwProcessID=%d dwThreadId=%d\n",

841  CloseHandle (pi.hThread);

849  (int)pi.dwProcessId, GetLastError ());

864  else
865  {
866  err = 0;
867  }
868  break;

870  default:
871  err = gpg_error_from_syserror ();;
872  log_error ("WaitForSingleObject returned unexpected "
873  "code %d for pid %d\n", code, (int)pi.dwProcessId);
874  break;
875  }
876  CloseHandle (pi.hProcess);

880  {

882  {

885  idx, GetLastError ());

893  CloseHandle (pipes[idx].hd[0]);
894  if (pipes[idx].hd[1] != INVALID_HANDLE_VALUE)
895  CloseHandle (pipes[idx].hd[1]);
896  }

898  xfree (pgmname);
899  return leave_cmd (ctx, err);

1211 /* Startup the server. */
1212 static void
1213 server (void)
1214 {
1215  gpg_error_t err;
1216  assuan_fd_t server_fd;
1217  assuan_sock_nonce_t server_nonce;
1218  int one = 1;
1219  struct sockaddr_in name;
1220  assuan_context_t ctx;
1221  state_t state = NULL;
1222 
1223  err = assuan_new (&ctx);
1224  if (err)
1225  log_fatal ("assuan_new failed: %s\n", gpg_strerror (err));
1226 
1227  server_fd = assuan_sock_new (PF_INET, SOCK_STREAM, 0);
1228  if (server_fd == ASSUAN_INVALID_FD)
1229  log_fatal ("socket() failed: %s\n", strerror (errno));
1230 
1231  if (setsockopt (HANDLE2SOCKET (server_fd),
1232  SOL_SOCKET, SO_REUSEADDR, (void*)&one, sizeof one))
1233  log_error ("setsockopt(SO_REUSEADDR) failed: %s\n", strerror (errno));
1234 
1235  name.sin_family = AF_INET;
1236  name.sin_port = htons (server_port);
1237  name.sin_addr.s_addr = htonl (INADDR_ANY);
1238  if (assuan_sock_bind (server_fd, (struct sockaddr *) &name, sizeof name))
1239  log_fatal ("bind() failed: %s\n", strerror (errno));
1240  if (assuan_sock_get_nonce ((struct sockaddr*)&name, sizeof name,
1241  &server_nonce))
1242  log_fatal ("assuan_sock_get_nonce failed: %s\n", strerror (errno));
1243 
1244  /* Register the nonce with the context so that assuan_accept knows
1245  about it. We can't do that directly in assuan_sock_bind because
1246  we want these socket wrappers to be context neutral and drop in
1247  replacement for the standard socket functions. */
1248  assuan_set_sock_nonce (ctx, &server_nonce);
1249 
1250  if (listen (HANDLE2SOCKET (server_fd), 5))
1251  log_fatal ("listen() failed: %s\n", strerror (errno));
1252 
1253  log_info ("server listening on port %hd\n", server_port);
1254 
1255  err = assuan_init_socket_server (ctx, server_fd, 0);
1256  if (err)
1257  log_fatal ("assuan_init_socket_server failed: %s\n", gpg_strerror (err));
1258 
1259  err = register_commands (ctx);
1260  if (err)
1261  log_fatal ("register_commands failed: %s\n", gpg_strerror(err));
1262 
1263  if (debug)
1264  assuan_set_log_stream (ctx, stderr);
1265 
1266  assuan_register_reset_notify (ctx, reset_notify);
1267  assuan_register_input_notify (ctx, input_notify);
1268  assuan_register_output_notify (ctx, output_notify);
1269 
1270 
1271  state = new_state ();
1272 
1273  assuan_set_pointer (ctx, state);
1274 
1275  while (!shutdown_pending)
1276  {
1277  int done;
1278 
1279  err = assuan_accept (ctx);
1280  if (err)
1281  {
1282  if (gpg_err_code (err) == GPG_ERR_EOF || err == -1)
1283  log_error ("assuan_accept failed: %s\n", gpg_strerror (err));
1284  break;
1285  }
1286 
1287  log_info ("client connected. Client's pid is %ld\n",
1288  (long)assuan_get_pid (ctx));
1289  do
1290  {
1291  /* We need to use select here so that we can accept
1292  supplemental connections from the client as requested by
1293  the DATAPORT command. */
1294  fd_set rfds;
1295  int connfd, datafd, max_fd;
1296 
1297  connfd = HANDLE2SOCKET (get_connection_fd (ctx));
1298  FD_ZERO (&rfds);
1299  FD_SET (connfd, &rfds);
1300  max_fd = connfd;
1301 
1302  if (state->dataport_listen_fd != ASSUAN_INVALID_FD)
1303  {
1304  datafd = HANDLE2SOCKET (state->dataport_listen_fd);
1305  FD_SET (datafd, &rfds);
1306  if (datafd > max_fd)
1307  max_fd = datafd;
1308  }
1309  else
1310  datafd = -1;
1311 
1312  if (select (max_fd + 1, &rfds, NULL, NULL, NULL) > 0)
1313  {
1314  if (datafd != -1 && FD_ISSET (datafd, &rfds))
1315  {
1316  struct sockaddr_in clnt_addr;
1317  socklen_t len = sizeof clnt_addr;
1318  int fd;
1319 
1320  fd = accept (datafd, (struct sockaddr*)&clnt_addr, &len);
1321  if (fd == -1)
1322  {
1323  err = gpg_err_code_from_syserror ();
1324  assuan_sock_close (state->dataport_listen_fd);
1325  state->dataport_listen_fd = ASSUAN_INVALID_FD;
1326  log_error ("accepting on dataport failed: %s\n",
1327  gpg_strerror (err));
1328  state->dataport_accept_err = err;
1329  err = 0;
1330  }
1331  else
1332  {
1333  /* No more need for the listening socket. */
1334  assuan_sock_close (state->dataport_listen_fd);
1335  state->dataport_listen_fd = ASSUAN_INVALID_FD;
1336  /* Record the accepted fd. */
1337  state->dataport_accept_err = 0;
1338  state->dataport_accepted_fd = SOCKET2HANDLE (fd);
1339  }
1340  }
1341 
1342  if (FD_ISSET (connfd, &rfds))
1343  {
1344  err = assuan_process_next (ctx, &done);
1345  }
1346  }
1347  }
1348  while (!err && !done && !shutdown_pending);
1349  if (err)
1350  log_error ("assuan_process failed: %s\n", gpg_strerror (err));
1351  }
1352 
1353  assuan_sock_close (server_fd);
1354  assuan_release (ctx);
1355  release_state (state);
1356 }
