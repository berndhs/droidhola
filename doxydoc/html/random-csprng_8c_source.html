309 static void
310 initialize(void)
311 {
312  /* Although the basic initialization should have happened already,
313  we call it here to make sure that all prerequisites are met. */
314  initialize_basics ();
315 
316  /* Now we can look the pool and complete the initialization if
317  necessary. */
318  lock_pool ();
319  if (!rndpool)

326  : xcalloc (1, POOLSIZE + BLOCKLEN));

465 /* Add BUFLEN bytes from BUF to the internal random pool. QUALITY
466  should be in the range of 0..100 to indicate the goodness of the
467  entropy added, or -1 for goodness not known. */
468 gcry_error_t
469 _gcry_rngcsprng_add_bytes (const void *buf, size_t buflen, int quality)
470 {
471  size_t nbytes;
472  const char *bufptr;
473 
474  if (quality == -1)
475  quality = 35;
476  else if (quality > 100)
477  quality = 100;
478  else if (quality < 0)
479  quality = 0;
480 
481  if (!buf)
482  return gpg_error (GPG_ERR_INV_ARG);
483 
484  if (!buflen || quality < 10)
485  return 0; /* Take a shortcut. */
486 
487  /* Because we don't increment the entropy estimation with FASTPOLL,
488  we don't need to take lock that estimation while adding from an
489  external source. This limited entropy estimation also means that
490  we can't take QUALITY into account. */
491  initialize_basics ();
492  bufptr = buf;
493  while (buflen)
494  {
495  nbytes = buflen > POOLSIZE? POOLSIZE : buflen;
496  lock_pool ();
497  if (rndpool)
498  add_randomness (bufptr, nbytes, RANDOM_ORIGIN_EXTERNAL);
499  unlock_pool ();
500  bufptr += nbytes;
501  buflen -= nbytes;
502  }
503  return 0;
504 }

507 /* Public function to fill the buffer with LENGTH bytes of
508  cryptographically strong random bytes. Level GCRY_WEAK_RANDOM is
509  not very strong, GCRY_STRONG_RANDOM is strong enough for most
510  usage, GCRY_VERY_STRONG_RANDOM is good for key generation stuff but

512 void
513 _gcry_rngcsprng_randomize (void *buffer, size_t length,
514  enum gcry_random_level level)
515 {
516  unsigned char *p;
517 
518  /* Make sure we are initialized. */
519  initialize ();
520 
521  /* Handle our hack used for regression tests of Libgcrypt. */
522  if ( quick_test && level > GCRY_STRONG_RANDOM )
523  level = GCRY_STRONG_RANDOM;
524 
525  /* Make sure the level is okay. */
526  level &= 3;
527 
528 #ifdef USE_RANDOM_DAEMON
529  if (allow_daemon
530  && !_gcry_daemon_randomize (daemon_socket_name, buffer, length, level))
531  return; /* The daemon succeeded. */
532  allow_daemon = 0; /* Daemon failed - switch off. */
533 #endif /*USE_RANDOM_DAEMON*/
534 
535  /* Acquire the pool lock. */
536  lock_pool ();
537 
538  /* Update the statistics. */
539  if (level >= GCRY_VERY_STRONG_RANDOM)
540  {
541  rndstats.getbytes2 += length;
542  rndstats.ngetbytes2++;
543  }
544  else
545  {
546  rndstats.getbytes1 += length;
547  rndstats.ngetbytes1++;
548  }
549

551  for (p = buffer; length > 0;)
552  {
553  size_t n;

555  n = length > POOLSIZE? POOLSIZE : length;
556  read_pool (p, n, level);
557  length -= n;

559  }

562  unlock_pool ();
563 }

568 /*
569  * Mix the 600 byte pool. Note that the 64 byte scratch area directly
570  * follows the pool. The numbers in the diagram give the number of
571  * bytes.
572  * <................600...............> <.64.>
573  * pool |------------------------------------| |------|
574  * <20><.24.> <20>
575  * | | +-----+
576  * +-----|-------------------------------|-+
577  * +-------------------------------|-|-+
578  * v v v
579  * |------|
580  * <hash>
581  * +---------------------------------------+
582  * v
583  * <20>
584  * pool' |------------------------------------|
585  * <20><20><.24.>
586  * +---|-----|---------------------------+
587  * +-----|---------------------------|-+
588  * +---------------------------|-|-+
589  * v v v
590  * |------|
591  * <hash>
592  * |
593  * +-----------------------------------+
594  * v
595  * <20>
596  * pool'' |------------------------------------|
597  * <20><20><20><.24.>
598  * +---|-----|-----------------------+
599  * +-----|-----------------------|-+
600  * +-----------------------|-|-+
601  * v v v
602  *
603  * and so on until we did this for all 30 blocks.
604  *
605  * To better protect against implementation errors in this code, we
606  * xor a digest of the entire pool into the pool before mixing.
607  *
608  * Note: this function must only be called with a locked pool.
609  */
610 static void
611 mix_pool(unsigned char *pool)
612 {
613  static unsigned char failsafe_digest[DIGESTLEN];
614  static int failsafe_digest_valid;
615 
616  unsigned char *hashbuf = pool + POOLSIZE;
617  unsigned char *p, *pend;
618  int i, n;
619  RMD160_CONTEXT md;
620 
621 #if DIGESTLEN != 20
622 #error must have a digest length of 20 for ripe-md-160
623 #endif
624 
625  gcry_assert (pool_is_locked);
626  _gcry_rmd160_init( &md );
627 
628  /* pool_0 -> pool'. */
629  pend = pool + POOLSIZE;
630  memcpy (hashbuf, pend - DIGESTLEN, DIGESTLEN);
631  memcpy (hashbuf+DIGESTLEN, pool, BLOCKLEN-DIGESTLEN);
632  _gcry_rmd160_mixblock (&md, hashbuf);
633  memcpy (pool, hashbuf, DIGESTLEN);
634 
635  if (failsafe_digest_valid && pool == rndpool)
636  {
637  for (i=0; i < DIGESTLEN; i++)
638  pool[i] ^= failsafe_digest[i];
639  }
640 
641  /* Loop for the remaining iterations. */
642  p = pool;
643  for (n=1; n < POOLBLOCKS; n++)
644  {
645  if (p + BLOCKLEN < pend)
646  memcpy (hashbuf, p, BLOCKLEN);
647  else
648  {
649  unsigned char *pp = p;
650 
651  for (i=0; i < BLOCKLEN; i++ )
652  {
653  if ( pp >= pend )
654  pp = pool;
655  hashbuf[i] = *pp++;
656  }
657  }
658 
659  _gcry_rmd160_mixblock (&md, hashbuf);
660  p += DIGESTLEN;
661  memcpy (p, hashbuf, DIGESTLEN);
662  }
663 
664  /* Our hash implementation does only leave small parts (64 bytes)
665  of the pool on the stack, so it is okay not to require secure
666  memory here. Before we use this pool, it will be copied to the
667  help buffer anyway. */
668  if ( pool == rndpool)
669  {
670  _gcry_rmd160_hash_buffer (failsafe_digest, pool, POOLSIZE);
671  failsafe_digest_valid = 1;
672  }
673 
674  _gcry_burn_stack (384); /* for the rmd160_mixblock(), rmd160_hash_buffer */
675 }

687 /* Lock an open file identified by file descriptor FD and wait a
688  reasonable time to succeed. With FOR_WRITE set to true a write
689  lock will be taken. FNAME is used only for diagnostics. Returns 0
690  on success or -1 on error. */
691 static int
692 lock_seed_file (int fd, const char *fname, int for_write)
693 {
694 #ifdef __GCC__
695 #warning Check whether we can lock on Windows.
696 #endif
697 #if LOCK_SEED_FILE
698  struct flock lck;
699  struct timeval tv;
700  int backoff=0;
701 
702  /* We take a lock on the entire file. */
703  memset (&lck, 0, sizeof lck);
704  lck.l_type = for_write? F_WRLCK : F_RDLCK;
705  lck.l_whence = SEEK_SET;
706 
707  while (fcntl (fd, F_SETLK, &lck) == -1)
708  {

729 /* Read in a seed from the random_seed file and return true if this
730  was successful.
731 
732  Note: Multiple instances of applications sharing the same random
733  seed file can be started in parallel, in which case they will read
734  out the same pool and then race for updating it (the last update
735  overwrites earlier updates). They will differentiate only by the
736  weak entropy that is added in read_seed_file based on the PID and
737  clock, and up to 16 bytes of weak random non-blockingly. The
738  consequence is that the output of these different instances is
739  correlated to some extent. In the perfect scenario, the attacker
740  can control (or at least guess) the PID and clock of the
741  application, and drain the system's entropy pool to reduce the "up
742  to 16 bytes" above to 0. Then the dependencies of the initial
743  states of the pools are completely known. */
