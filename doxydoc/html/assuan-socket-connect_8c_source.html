79 /* Returns true if STR represents a valid port number in decimal
80  notation and no garbage is following. */
81 static int
82 parse_portno (const char *str, uint16_t *r_port)
83 {
84  unsigned int value;
85 
86  for (value=0; *str && (*str >= '0' && *str <= '9'); str++)
87  {
88  value = value * 10 + (*str - '0');
89  if (value > 65535)
90  return 0;
91  }
92  if (*str || !value)
93  return 0;
94 
95  *r_port = value;
96  return 1;
97 }

100 static gpg_error_t
101 _assuan_connect_finalize (assuan_context_t ctx, assuan_fd_t fd,
102  unsigned int flags)
103 {
104  gpg_error_t err;
105 
106  ctx->engine.release = _assuan_client_release;
107  ctx->engine.readfnc = _assuan_simple_read;
108  ctx->engine.writefnc = _assuan_simple_write;
109  ctx->engine.sendfd = NULL;
110  ctx->engine.receivefd = NULL;
111  ctx->finish_handler = _assuan_client_finish;
112  ctx->inbound.fd = fd;
113  ctx->outbound.fd = fd;
114  ctx->max_accepts = -1;
115 
116  if (flags & ASSUAN_SOCKET_CONNECT_FDPASSING)
117  _assuan_init_uds_io (ctx);
118 
119  /* initial handshake */
120  {
121  assuan_response_t response;
122  int off;
123 
124  err = _assuan_read_from_server (ctx, &response, &off, 0);
125  if (err)
126  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "assuan_socket_connect", ctx,
127  "can't connect to server: %s\n", gpg_strerror (err));
128  else if (response != ASSUAN_RESPONSE_OK)
129  {
130  char *sname = _assuan_encode_c_string (ctx, ctx->inbound.line);
131  if (sname)
132  {
133  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "assuan_socket_connect", ctx,
134  "can't connect to server: %s", sname);
135  _assuan_free (ctx, sname);
136  }
137  err = _assuan_error (ctx, GPG_ERR_ASS_CONNECT_FAILED);
138  }
139  }
140 
141  return err;
142 }

169 /* Make a connection to the Unix domain socket NAME and return a new
170  Assuan context in CTX. SERVER_PID is currently not used but may
171  become handy in the future. Defined flag bits are:
172 
173  ASSUAN_SOCKET_CONNECT_FDPASSING
174  sendmsg and recvmsg are used.
175 
176  NAME must either start with a slash and optional with a drive
177  prefix ("c:") or use one of these URL schemata:
178 
179  file://<fname>
180 
181  This is the same as the default just with an explicit schemata.
182 
183  assuan://<ipaddr>:<port>
184  assuan://[<ip6addr>]:<port>
185 
186  Connect using TCP to PORT of the server with the numerical
187  IPADDR. Note that '[' and ']' are literal characters.
188 
189  */
190 gpg_error_t
191 assuan_socket_connect (assuan_context_t ctx, const char *name,
192  pid_t server_pid, unsigned int flags)
193 {
194  gpg_error_t err = 0;
195  assuan_fd_t fd;
196 #ifdef WITH_IPV6
197  struct sockaddr_in6 srvr_addr_in6;
198 #endif
199  struct sockaddr_un srvr_addr_un;
200  struct sockaddr_in srvr_addr_in;
201  struct sockaddr *srvr_addr = NULL;
202  uint16_t port = 0;
203  size_t len = 0;
204  const char *s;
205  int af = AF_LOCAL;
206  int pf = PF_LOCAL;
207 
208  TRACE2 (ctx, ASSUAN_LOG_CTX, "assuan_socket_connect", ctx,
209  "name=%s, flags=0x%x", name ? name : "(null)", flags);
210 
211  if (!ctx || !name)
212  return _assuan_error (ctx, GPG_ERR_ASS_INV_VALUE);
213 
214  if (!strncmp (name, "file://", 7) && name[7])
215  name += 7;
216  else if (!strncmp (name, "assuan://", 9) && name[9])
217  {
218  name += 9;
219  af = AF_INET;
220  pf = PF_INET;
221  }
222  else /* Default. */
223  {
224  /* We require that the name starts with a slash if no URL
225  schemata is used. To make things easier we allow an optional
226  drive prefix. */
227  s = name;
228  if (*s && s[1] == ':')
229  s += 2;
230  if (*s != DIRSEP_C && *s != '/')
231  return _assuan_error (ctx, GPG_ERR_ASS_INV_VALUE);
232  }
233 
234  if (af == AF_LOCAL)
235  {
236  int redirected;
237 
238  if (_assuan_sock_set_sockaddr_un (name, (struct sockaddr *)&srvr_addr_un,
239  &redirected))
240  return _assuan_error (ctx, gpg_err_code_from_syserror ());
241 
242  len = SUN_LEN (&srvr_addr_un);
243  srvr_addr = (struct sockaddr *)&srvr_addr_un;
244  }
245  else
246  {
247  char *addrstr, *p;
248 #ifdef HAVE_INET_PTON
249  void *addrbuf = NULL;
250 #endif
251 
252  addrstr = _assuan_malloc (ctx, strlen (name) + 1);
253  if (!addrstr)
254  return _assuan_error (ctx, gpg_err_code_from_syserror ());
255 
256  if (*name == '[')
257  {
258  strcpy (addrstr, name+1);
259  p = strchr (addrstr, ']');
260  if (!p || p[1] != ':' || !parse_portno (p+2, &port))
261  err = _assuan_error (ctx, GPG_ERR_BAD_URI);
262  else
263  {
264  *p = 0;
265 #ifdef WITH_IPV6
266  af = AF_INET6;
267  pf = PF_INET6;
268  memset (&srvr_addr_in6, 0, sizeof srvr_addr_in6);
269  srvr_addr_in6.sin6_family = af;
270  srvr_addr_in6.sin6_port = htons (port);
271 #ifdef HAVE_INET_PTON
272  addrbuf = &srvr_addr_in6.sin6_addr;
273 #endif
274  srvr_addr = (struct sockaddr *)&srvr_addr_in6;
275  len = sizeof srvr_addr_in6;
276 #else
277  err = _assuan_error (ctx, GPG_ERR_EAFNOSUPPORT);
278 #endif
279  }
280  }
281  else
282  {
283  strcpy (addrstr, name);
284  p = strchr (addrstr, ':');
285  if (!p || !parse_portno (p+1, &port))
286  err = _assuan_error (ctx, GPG_ERR_BAD_URI);
287  else
288  {
289  *p = 0;
290  memset (&srvr_addr_in, 0, sizeof srvr_addr_in);
291  srvr_addr_in.sin_family = af;
292  srvr_addr_in.sin_port = htons (port);
293 #ifdef HAVE_INET_PTON
294  addrbuf = &srvr_addr_in.sin_addr;
295 #endif
296  srvr_addr = (struct sockaddr *)&srvr_addr_in;
297  len = sizeof srvr_addr_in;
298  }
299  }
300 
301  if (!err)
302  {
303 #ifdef HAVE_INET_PTON
304  switch (inet_pton (af, addrstr, addrbuf))
305  {
306  case 1: break;
307  case 0: err = _assuan_error (ctx, GPG_ERR_BAD_URI); break;
308  default: err = _assuan_error (ctx, gpg_err_code_from_syserror ());
309  }
310 #else
311  /* We need to use the old function. If we are here v6
312  support isn't enabled anyway and thus we can do fine
313  without. Note that Windows as a compatible inet_pton
314  function named inetPton, but only since Vista. */
315  srvr_addr_in.sin_addr.s_addr = inet_addr (addrstr);
316  if (srvr_addr_in.sin_addr.s_addr == INADDR_NONE)
317  err = _assuan_error (ctx, GPG_ERR_BAD_URI);
318 #endif
319  }
320 
321  _assuan_free (ctx, addrstr);
322  if (err)
323  return err;
324  }
325 
326  fd = _assuan_sock_new (ctx, pf, SOCK_STREAM, 0);
327  if (fd == ASSUAN_INVALID_FD)
328  {
329  err = _assuan_error (ctx, gpg_err_code_from_syserror ());
330  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "assuan_socket_connect", ctx,
331  "can't create socket: %s", strerror (errno));
332  return err;
333  }
334 
335  if (_assuan_sock_connect (ctx, fd, srvr_addr, len) == -1)
336  {
337  TRACE2 (ctx, ASSUAN_LOG_SYSIO, "assuan_socket_connect", ctx,
338  "can't connect to `%s': %s\n", name, strerror (errno));
339  _assuan_close (ctx, fd);
340  return _assuan_error (ctx, GPG_ERR_ASS_CONNECT_FAILED);
341  }
342 
343  err = _assuan_connect_finalize (ctx, fd, flags);
344 
345  if (err)
346  _assuan_reset (ctx);
347 
348  return err;
349 }
