459 static void
460 do_encrypt_block( CAST5_context *c, byte *outbuf, const byte *inbuf )
461 {
462  u32 l, r, t;
463  u32 I; /* used by the Fx macros */
464  u32 *Km;
465  byte *Kr;
466 
467  Km = c->Km;
468  Kr = c->Kr;
469 
470  /* (L0,R0) <-- (m1...m64). (Split the plaintext into left and
471  * right 32-bit halves L0 = m1...m32 and R0 = m33...m64.)
472  */
473  l = buf_get_be32(inbuf + 0);
474  r = buf_get_be32(inbuf + 4);
475 
476  /* (16 rounds) for i from 1 to 16, compute Li and Ri as follows:
477  * Li = Ri-1;
478  * Ri = Li-1 ^ f(Ri-1,Kmi,Kri), where f is defined in Section 2.2
479  * Rounds 1, 4, 7, 10, 13, and 16 use f function Type 1.
480  * Rounds 2, 5, 8, 11, and 14 use f function Type 2.
481  * Rounds 3, 6, 9, 12, and 15 use f function Type 3.
482  */
483 
484  t = l; l = r; r = t ^ F1(r, Km[ 0], Kr[ 0]);
485  t = l; l = r; r = t ^ F2(r, Km[ 1], Kr[ 1]);
486  t = l; l = r; r = t ^ F3(r, Km[ 2], Kr[ 2]);
487  t = l; l = r; r = t ^ F1(r, Km[ 3], Kr[ 3]);
488  t = l; l = r; r = t ^ F2(r, Km[ 4], Kr[ 4]);
489  t = l; l = r; r = t ^ F3(r, Km[ 5], Kr[ 5]);
490  t = l; l = r; r = t ^ F1(r, Km[ 6], Kr[ 6]);
491  t = l; l = r; r = t ^ F2(r, Km[ 7], Kr[ 7]);
492  t = l; l = r; r = t ^ F3(r, Km[ 8], Kr[ 8]);
493  t = l; l = r; r = t ^ F1(r, Km[ 9], Kr[ 9]);
494  t = l; l = r; r = t ^ F2(r, Km[10], Kr[10]);
495  t = l; l = r; r = t ^ F3(r, Km[11], Kr[11]);
496  t = l; l = r; r = t ^ F1(r, Km[12], Kr[12]);
497  t = l; l = r; r = t ^ F2(r, Km[13], Kr[13]);
498  t = l; l = r; r = t ^ F3(r, Km[14], Kr[14]);
499  t = l; l = r; r = t ^ F1(r, Km[15], Kr[15]);
500 
501  /* c1...c64 <-- (R16,L16). (Exchange final blocks L16, R16 and
502  * concatenate to form the ciphertext.) */
503  buf_put_be32(outbuf + 0, r);
504  buf_put_be32(outbuf + 4, l);
505 }
