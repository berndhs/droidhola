40 static gpg_error_t
41 cmd_echo (assuan_context_t ctx, char *line)
42 {
43  int fd;
44  int c;
45  FILE *fp;
46  int nbytes;
47 
48  log_info ("got ECHO command (%s)\n", line);
49 
50  fd = assuan_get_input_fd (ctx);
51  if (fd == -1)
52  return gpg_error (GPG_ERR_ASS_NO_INPUT);
53  fp = fdopen (fd, "r");
54  if (!fp)
55  {
56  log_error ("fdopen failed on input fd: %s\n", strerror (errno));
57  return gpg_error (GPG_ERR_ASS_GENERAL);
58  }
59  nbytes = 0;
60  while ( (c=getc (fp)) != -1)
61  {
62  putc (c, stdout);
63  nbytes++;
64  }
65  fflush (stdout);
66  log_info ("done printing %d bytes to stdout\n", nbytes);
67 
68  fclose (fp);
69  return 0;
70 }

152 /* Client main. If true is returned, a disconnect has not been done. */
153 static int
154 client (assuan_context_t ctx, const char *fname)
155 {
156  int rc;
157  FILE *fp;
158  int i;
159 
160  log_info ("client started. Servers's pid is %ld\n",
161  (long)assuan_get_pid (ctx));
162 
163  for (i=0; i < 6; i++)
164  {
165  fp = fopen (fname, "r");
166  if (!fp)
167  {
168  log_error ("failed to open `%s': %s\n", fname,
169  strerror (errno));
170  return -1;
171  }
172 
173  rc = assuan_sendfd (ctx, fileno (fp));
174  if (rc)
175  {
176  log_error ("assuan_sendfd failed: %s\n", gpg_strerror (rc));
177  return -1;
178  }
179  fclose (fp);
180 
181  rc = assuan_transact (ctx, "INPUT FD", NULL, NULL, NULL, NULL,
182  NULL, NULL);
183  if (rc)
184  {
185  log_error ("sending INPUT FD failed: %s\n", gpg_strerror (rc));
186  return -1;
187  }
188 
189  rc = assuan_transact (ctx, "ECHO", NULL, NULL, NULL, NULL, NULL, NULL);
190  if (rc)
191  {
192  log_error ("sending ECHO failed: %s\n", gpg_strerror (rc));
193  return -1;
194  }
195  }
196 
197  /* Give us some time to check with lsof that all descriptors are closed. */
198 /* sleep (10); */
199 
200  assuan_release (ctx);
201  return 0;
202 }
