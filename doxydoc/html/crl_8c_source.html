55 static inline void
56 do_hash (ksba_crl_t crl, const void *buffer, size_t length)
57 {
58  while (length)
59  {
60  size_t n = length;
61 
62  if (crl->hashbuf.used + n > sizeof crl->hashbuf.buffer)
63  n = sizeof crl->hashbuf.buffer - crl->hashbuf.used;
64  memcpy (crl->hashbuf.buffer+crl->hashbuf.used, buffer, n);
65  crl->hashbuf.used += n;
66  if (crl->hashbuf.used == sizeof crl->hashbuf.buffer)
67  {
68  if (crl->hash_fnc)
69  crl->hash_fnc (crl->hash_fnc_arg,
70  crl->hashbuf.buffer, crl->hashbuf.used);
71  crl->hashbuf.used = 0;
72  }
73  buffer = (const char *)buffer + n;
74  length -= n;
75  }
76 }
