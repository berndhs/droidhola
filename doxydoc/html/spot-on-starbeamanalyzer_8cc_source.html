258 void spoton_starbeamanalyzer::analyze(const QString &fileName,
259  const QString &pulseSize,
260  const QString &totalSize,
261  QAtomicInt *interrupt)
262 {
263  int ps = pulseSize.trimmed().toInt();
264 
265  if(ps <= 0)
266  {
267  emit updatePercent(fileName, 0);
268  return;
269  }
270 
271  QFile file(fileName);
272 
273  if(file.open(QIODevice::ReadOnly))
274  {
275  QByteArray bytes(ps, 0);
276  bool excessive = false;
277  bool first = true;
278  bool interrupted = false;
279  int percent = 0;
280  int problems = 0;
281  qint64 nPulses = 0;
282  qint64 pos = 0;
283  qint64 rc = 0;
284  qint64 ts = qMax(static_cast<long long> (1),
285  qMax(file.size(), totalSize.trimmed().toLongLong()));
286 
287  nPulses = ts / ps;
288 
289  while((rc = file.read(bytes.data(), bytes.length())) > 0)
290  {
291  if(bytes.count('\0') == bytes.length())
292  {
293  problems += 1;
294 
295  double p = 100 * (static_cast<double> (problems) /
296  static_cast<double> (nPulses));
297 
298  if(p >= 75.00)
299  {
300  excessive = true;
301  break;
302  }
303 
304  /*
305  ** Potential problem.
306  */
307 
308  if(first)
309  {
310  if(pos - ps >= 0)
311  emit potentialProblem(fileName, pos - ps);
312 
313  first = false;
314  }
315 
316  emit potentialProblem(fileName, pos);
317  }
318 
319  pos += ps;
320  percent = static_cast<int>
321  (qMin(static_cast<double> (100),
322  100 * static_cast<double> (file.pos()) /
323  static_cast<double> (ts)));
324 
325  if(percent > 0 && percent % 5 == 0)
326  emit updatePercent(fileName, percent);
327 
328  if(interrupt && interrupt->fetchAndAddRelaxed(0))
329  {
330  interrupted = true;
331  break;
332  }
333  }
334 
335  file.close();
336  first = true;
337 
338  /*
339  ** Now that we've reviewed the file, let's review shadow portions.
340  */
341 
342  if(!excessive && !interrupted)
343  while(percent < 100)
344  {
345  problems += 1;
346 
347  double p = 100 * (static_cast<double> (problems) /
348  static_cast<double> (nPulses));
349 
350  if(p >= 75.00)
351  {
352  excessive = true;
353  break;
354  }
355 
356  if(first)
357  {
358  if(pos - ps >= 0)
359  emit potentialProblem(fileName, pos - ps);
360 
361  first = false;
362  }
363 
364  emit potentialProblem(fileName, pos);
365  pos += ps;
366  percent = static_cast<int>
367  (qMin(static_cast<double> (100),
368  100 * static_cast<double> (pos) /
369  static_cast<double> (ts)));
370 
371  if(percent > 0 && percent % 5 == 0)
372  emit updatePercent(fileName, percent);
373 
374  if(interrupt && interrupt->fetchAndAddRelaxed(0))
375  {
376  interrupted = true;
377  break;
378  }
379  }
380 
381  if(excessive)
382  emit excessiveProblems(fileName);
383  else if(!interrupted)
384  {
385  if(rc == -1)
386  emit updatePercent(fileName, 0);
387  else
388  emit updatePercent(fileName, 100);
389  }
390  }
391  else
392  emit updatePercent(fileName, 0);
393 }
