123 /* Default log handler. */
124 int
125 _assuan_log_handler (assuan_context_t ctx, void *hook, unsigned int cat,
126  const char *msg)
127 {
128  FILE *fp;
129  const char *prf;
130  int saved_errno = errno;
131 
132  /* For now. */
133  if (msg == NULL)
134  return TEST_LOG_CAT (cat);
135 
136  if (! TEST_LOG_CAT (cat))
137  return 0;
138 
139  fp = ctx->log_fp ? ctx->log_fp : _assuan_log;
140  if (!fp)
141  return 0;
142 
143  prf = assuan_get_assuan_log_prefix ();
144  if (*prf)
145  fprintf (fp, "%s[%u]: ", prf, (unsigned int)getpid ());
146 
147  fprintf (fp, "%s", msg);
148  /* If the log stream is a file, the output would be buffered. This
149  is bad for debugging, thus we flush the stream if FORMAT ends
150  with a LF. */
151  if (msg && *msg && msg[strlen (msg) - 1] == '\n')
152  fflush (fp);
153  gpg_err_set_errno (saved_errno);
154 
155  return 0;
156 }

160 /* Log a control channel message. This is either a STRING with a
161  diagnostic or actual data in (BUFFER1,LENGTH1) and
162  (BUFFER2,LENGTH2). If OUTBOUND is true the data is intended for
163  the peer. */
164 void
165 _assuan_log_control_channel (assuan_context_t ctx, int outbound,
166  const char *string,
167  const void *buffer1, size_t length1,
168  const void *buffer2, size_t length2)
169 {
170  int res;
171  char *outbuf;
172  int saved_errno;
173 
174  /* Check whether logging is enabled and do a quick check to see
175  whether the callback supports our category. */
176  if (!ctx
177  || !ctx->log_cb
178  || ctx->flags.no_logging
179  || !(*ctx->log_cb) (ctx, ctx->log_cb_data, ASSUAN_LOG_CONTROL, NULL))
180  return;
181 
182  saved_errno = errno;
183 
184  /* Note that we use the inbound channel fd as the printed channel
185  number for both directions. */
186 #ifdef HAVE_W32_SYSTEM
187 # define CHANNEL_FMT "%p"
188 #else
189 # define CHANNEL_FMT "%d"
190 #endif
191 #define TOHEX(val) (((val) < 10) ? ((val) + '0') : ((val) - 10 + 'a'))
192 
193  if (!buffer1 && buffer2)
194  {
195  buffer1 = buffer2;
196  length1 = length2;
197  buffer2 = NULL;
198  length2 = 0;
199  }
200 
201  if (ctx->flags.confidential && !string && buffer1)
202  string = "[Confidential data not shown]";
203 
204  if (string)
205  {
206  /* Print the diagnostic. */
207  res = gpgrt_asprintf (&outbuf, "chan_" CHANNEL_FMT " %s [%s]\n",
208  ctx->inbound.fd, outbound? "->":"<-", string);
209  }
210  else if (buffer1)
211  {
212  /* Print the control channel data. */
213  const unsigned char *s;
214  unsigned int n, x;
215 
216  for (n = length1, s = buffer1; n; n--, s++)
217  if ((!isascii (*s) || iscntrl (*s) || !isprint (*s) || !*s)
218  && !(*s >= 0x80))
219  break;
220  if (!n && buffer2)
221  {
222  for (n = length2, s = buffer2; n; n--, s++)
223  if ((!isascii (*s) || iscntrl (*s) || !isprint (*s) || !*s)
224  && !(*s >= 0x80))
225  break;
226  }
227  if (!buffer2)
228  length2 = 0;
229 
230  if (!n && (length1 && *(const char*)buffer1 != '['))
231  {
232  /* No control characters and not starting with our error
233  message indicator. Log it verbatim. */
234  res = gpgrt_asprintf (&outbuf, "chan_" CHANNEL_FMT " %s %.*s%.*s\n",
235  ctx->inbound.fd, outbound? "->":"<-",
236  (int)length1, (const char*)buffer1,
237  (int)length2, buffer2? (const char*)buffer2:"");
238  }
239  else
240  {
241  /* The buffer contains control characters - do a hex dump.
242  Even in full logging mode we limit the line length -
243  however this is no real limit because the provided
244  buffers will never be larger than the maximum assuan line
245  length. */
246  char *hp;
247  unsigned int nbytes;
248  unsigned int maxbytes = full_logging? (2*LINELENGTH) : 16;
249 
250  nbytes = length1 + length2;
251  if (nbytes > maxbytes)
252  nbytes = maxbytes;
253 
254  if (!(outbuf = malloc (50 + 3*nbytes + 60 + 3 + 1)))
255  res = -1;
256  else
257  {
258  res = 0;
259  hp = outbuf;
260  snprintf (hp, 50, "chan_" CHANNEL_FMT " %s [",
261  ctx->inbound.fd, outbound? "->":"<-");
262  hp += strlen (hp);
263  n = 0;
264  for (s = buffer1, x = 0; x < length1 && n < nbytes; x++, n++)
265  {
266  *hp++ = ' ';
267  *hp++ = TOHEX (*s >> 4);
268  *hp++ = TOHEX (*s & 0x0f);
269  s++;
270  }
271  for (s = buffer2, x = 0; x < length2 && n < nbytes; x++, n++)
272  {
273  *hp++ = ' ';
274  *hp++ = TOHEX (*s >> 4);
275  *hp++ = TOHEX (*s & 0x0f);
276  s++;
277  }
278  if (nbytes < length1 + length2)
279  {
280  snprintf (hp, 60, " ...(%u byte(s) skipped)",
281  (unsigned int)((length1+length2) - nbytes));
282  hp += strlen (hp);
283  }
284  strcpy (hp, " ]\n");
285  }
286  }
287  }
288  else
289  {
290  res = 0;
291  outbuf = NULL;
292  }
293  if (res < 0)
294  ctx->log_cb (ctx, ctx->log_cb_data, ASSUAN_LOG_CONTROL,
295  "[libassuan failed to format the log message]");
296  else if (outbuf)
297  {
298  ctx->log_cb (ctx, ctx->log_cb_data, ASSUAN_LOG_CONTROL, outbuf);
299  free (outbuf);
300  }
301 #undef TOHEX
302 #undef CHANNEL_FMT
303  gpg_err_set_errno (saved_errno);
304 }
