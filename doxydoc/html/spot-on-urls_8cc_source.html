54  "urls_key_information.db");
55 
56  if(db.open())
57  {
58  QSqlQuery query(db);
59  QStringList queries;
60  bool ok = true;
61  int counts[2] = {0, 0};
62 
63  query.setForwardOnly(true);
64  queries << "SELECT * FROM import_key_information"
65  << "SELECT * FROM remote_key_information";
66

68  if(query.exec(queries.at(i)) && query.next())
69  for(int j = 0; j < query.record().count(); j++)
70  {
71  crypt->decryptedAfterAuthenticated
72  (QByteArray::fromBase64(query.value(j).
73  toByteArray()),
74  &ok);
75 
76  if(ok)
77  counts[i] = 1;
78  else
79  {
80  counts[i] = 0;
81  break;
82  }
83  }
84 
85  importCount = counts[0];
86  remoteCount = counts[1];
87  }
88 
89  db.close();
90  }
91 
92  QSqlDatabase::removeDatabase(connectionName);
93  }
94 
95  if(importCount > 0)
96  m_ui.importCredentialsLabel->setText
97  (tr("Import credentials have been prepared."));
