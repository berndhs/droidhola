122 QHostAddress spoton_sctp_socket::localAddressAndPort(quint16 *port) const
123 {
124 #ifdef SPOTON_SCTP_ENABLED
125  if(port)
126  *port = 0;
127 
128  if(m_socketDescriptor < 0)
129  return QHostAddress();
130 
131  QHostAddress address;
132  socklen_t length = 0;
133  struct sockaddr_storage peeraddr;
134 
135  length = sizeof(peeraddr);
136 
137  if(getsockname(m_socketDescriptor, (struct sockaddr *) &peeraddr,
138  &length) == 0)
139  {
140  if(peeraddr.ss_family == AF_INET)
141  {
142  spoton_type_punning_sockaddr_t *sockaddr =
143  (spoton_type_punning_sockaddr_t *) &peeraddr;
144 
145  if(sockaddr)
146  {
147  address.setAddress
148  (ntohl(sockaddr->sockaddr_in.sin_addr.s_addr));
149 
150  if(port)
151  *port = ntohs(sockaddr->sockaddr_in.sin_port);
152  }
153  }
154  else
155  {
156  spoton_type_punning_sockaddr_t *sockaddr =
157  (spoton_type_punning_sockaddr_t *) &peeraddr;
158 
159  if(sockaddr)
160  {
161  Q_IPV6ADDR temp;
162 
163  memcpy(&temp.c, &sockaddr->sockaddr_in6.sin6_addr.s6_addr,
164  qMin(sizeof(sockaddr->sockaddr_in6.sin6_addr.s6_addr),
165  sizeof(temp.c)));
166  address.setAddress(temp);
167  address.setScopeId
168  (QString::number(sockaddr->sockaddr_in6.sin6_scope_id));
169 
170  if(port)
171  *port = ntohs(sockaddr->sockaddr_in6.sin6_port);
172  }
173  }
174  }
175 
176  return address;
177 #else
178  if(port)
179  *port = 0;
180 
181  return QHostAddress();
182 #endif
183 }

306 int spoton_sctp_socket::setSocketBlockingOrNon(void)
307 {
308 #ifdef SPOTON_SCTP_ENABLED
309  if(m_socketDescriptor < 0)
310  {
311  spoton_misc::logError("spoton_sctp_socket::setSocketBlockingOrNon(): "
312  "m_socketDescriptor is less than zero.");
313  return -1;
314  }
315 
316  int rc = 0;
317 
318 #ifdef Q_OS_WIN32
319  unsigned long int enabled = 1;
320 
321  rc = ioctlsocket(m_socketDescriptor, FIONBIO, &enabled);
322 
323  if(rc != 0)
324  {
325  QString errorstr("setSocketBlockingOrNon()::ioctlsocket()");
326 
327  emit error(errorstr, UnknownSocketError);
328  return -1;
329  }
330 #else
331  rc = fcntl(m_socketDescriptor, F_GETFL, 0);
332 
333  if(rc == -1)
334  {
335  QString errorstr(QString("setSocketBlockingOrNon()::fcntl()::"
336  "errno=%1").
337  arg(errno));
338 
339  emit error(errorstr, UnknownSocketError);
340  return -1;
341  }
342 
343  rc = fcntl(m_socketDescriptor, F_SETFL, O_NONBLOCK | rc);
344 
345  if(rc == -1)
346  {
347  QString errorstr(QString("setSocketBlockingOrNon()::fcntl()::"
348  "errno=%1").
349  arg(errno));
350 
351  emit error(errorstr, UnknownSocketError);
352  return -1;
353  }
354 #endif
355  return 0;
356 #else
357  return -1;
358 #endif
359 }

366 qint64 spoton_sctp_socket::read(char *data, const qint64 size)
367 {
368 #ifdef SPOTON_SCTP_ENABLED
369  if(m_socketDescriptor < 0 || m_state != ConnectedState)
370  return -1;
371  else if(!data || size < 0)
372  return -1;
373  else if(size == 0)
374  return 0;
375 
376  fd_set efds, rfds, wfds;
377  ssize_t rc = 0;
378  struct timeval tv;
379 
380  FD_ZERO(&efds);
381  FD_ZERO(&rfds);
382  FD_ZERO(&wfds);
383  FD_SET(m_socketDescriptor, &efds);
384  FD_SET(m_socketDescriptor, &rfds);
385  FD_SET(m_socketDescriptor, &wfds);
386  tv.tv_sec = 0;
387  tv.tv_usec = 25000; // 25 milliseconds.
388 
389  if(select(m_socketDescriptor + 1, &rfds, &wfds, &efds, &tv) > 0)
390  {
391  if(FD_ISSET(m_socketDescriptor, &rfds))
392  rc = recv
393  (m_socketDescriptor, data, static_cast<size_t> (size), 0);
394  else
395 #ifdef Q_OS_WIN32
396  WSASetLastError(WSAEWOULDBLOCK);
397 #else
398  errno = EWOULDBLOCK;
399 #endif
400  }
401  else
402 #ifdef Q_OS_WIN32
403  WSASetLastError(WSAEWOULDBLOCK);
404 #else
405  errno = EWOULDBLOCK;
406 #endif
407 
408  if(rc == -1)
409  {
410 #ifdef Q_OS_WIN32
411  QString errorstr(QString("read()::recv()::error=%1").
412  arg(WSAGetLastError()));
413 
414  if(WSAGetLastError() == WSAEWOULDBLOCK)
415  /*
416  ** We'll ignore this condition.
417  */
418 
419  rc = 0;
420  else
421  emit error(errorstr, UnknownSocketError);
422 #else
423  QString errorstr(QString("read()::recv()::errno=%1").
424  arg(errno));
425 
426  if(errno == EAGAIN || errno == EINPROGRESS || errno == EWOULDBLOCK)
427  /*
428  ** We'll ignore this condition.
429  */
430 
431  rc = 0;
432  else if(errno == ECONNREFUSED)
433  emit error(errorstr, ConnectionRefusedError);
434  else if(errno == ECONNRESET)
435  emit error(errorstr, RemoteHostClosedError);
436  else if(errno == ENOBUFS ||
437  errno == ENOMEM)
438  emit error(errorstr, SocketResourceError);
439  else if(errno == ENOTCONN)
440  emit error(errorstr, NetworkError);
441  else if(errno == EOPNOTSUPP)
442  emit error(errorstr, UnsupportedSocketOperationError);
443  else
444  emit error(errorstr, UnknownSocketError);
445 #endif
446  }
447 
448  return static_cast<qint64> (rc);
449 #else
450  Q_UNUSED(data);
451  Q_UNUSED(size);
452  return 0;
453 #endif
454 }

456 qint64 spoton_sctp_socket::write(const char *data, const qint64 size)
457 {
458 #ifdef SPOTON_SCTP_ENABLED
459  if(m_socketDescriptor < 0 || m_state != ConnectedState)
460  return -1;
461  else if(!data || size < 0)
462  return -1;
463  else if(size == 0)
464  return 0;
465 
466  qint64 writeSize = 4096;
467  ssize_t sent = 0;
468 
469  /*
470  ** We'll send a fraction of the desired buffer size. Otherwise,
471  ** our process may become exhausted.
472  */
473 
474 #ifdef Q_OS_WIN32
475  sent = send
476  (m_socketDescriptor, data,
477  static_cast<size_t> (qMin(size, writeSize)), 0);
478 #else
479  sent = send
480  (m_socketDescriptor, data,
481  static_cast<size_t> (qMin(size, writeSize)), MSG_DONTWAIT);
482 #endif
483 
484  if(sent == -1)
485 #ifdef Q_OS_WIN32
486  if(WSAGetLastError() == WSAEWOULDBLOCK)
487  sent = 0;
488  #else
489  if(errno == EAGAIN || errno == EWOULDBLOCK)
490  sent = 0;
491 #endif
492 
493  if(sent == -1)
494  {
495 #ifdef Q_OS_WIN32
496  QString errorstr(QString("write()::send()::error=%1").
497  arg(WSAGetLastError()));
498 
499  emit error(errorstr, UnknownSocketError);
500 #else
501  QString errorstr(QString("write()::send()::errno=%1").
502  arg(errno));
503 
504  if(errno == EACCES)
505  emit error(errorstr, SocketAccessError);
506  else if(errno == ECONNRESET)
507  emit error(errorstr, RemoteHostClosedError);
508  else if(errno == EMSGSIZE || errno == ENOBUFS || errno == ENOMEM)
509  emit error(errorstr, SocketResourceError);
510  else if(errno == EHOSTUNREACH || errno == ENETDOWN ||
511  errno == ENETUNREACH || errno == ENOTCONN)
512  emit error(errorstr, NetworkError);
513  else if(errno == EOPNOTSUPP)
514  emit error(errorstr, UnsupportedSocketOperationError);
515  else
516  emit error(errorstr, UnknownSocketError);
517 #endif
518  }
519 
520  return static_cast<qint64> (sent);
521 #else
522  Q_UNUSED(data);
523  Q_UNUSED(size);
524  return 0;
525 #endif
526 }

616 void spoton_sctp_socket::connectToHostImplementation(void)
617 {
618 #ifdef SPOTON_SCTP_ENABLED
619  NetworkLayerProtocol protocol = IPv4Protocol;
620  int optval = 0;
621  int rc = 0;
622  socklen_t optlen = sizeof(optval);
623 
624  if(QHostAddress(m_ipAddress).protocol() ==
625  QAbstractSocket::NetworkLayerProtocol(IPv6Protocol))
626  protocol = IPv6Protocol;
627 
628  if(protocol == IPv4Protocol)
629  m_socketDescriptor = rc = socket(AF_INET, SOCK_STREAM, IPPROTO_SCTP);
630  else
631  m_socketDescriptor = rc = socket(AF_INET6, SOCK_STREAM, IPPROTO_SCTP);
632 
633  if(rc == -1)
634  {
635 #ifdef Q_OS_WIN32
636  QString errorstr
637  (QString("connectToHostImplementation()::socket()::error=%1").
638  arg(WSAGetLastError()));
639 
640  emit error(errorstr, UnknownSocketError);
641 #else
642  QString errorstr
643  (QString("connectToHostImplementation()::socket()::errno=%1").
644  arg(errno));
645 
646  if(errno == EACCES)
647  emit error(errorstr, SocketAccessError);
648  else if(errno == EAFNOSUPPORT || errno == EPROTONOSUPPORT)
649  emit error(errorstr, UnsupportedSocketOperationError);
650  else if(errno == EISCONN || errno == EMFILE ||
651  errno == ENFILE || errno == ENOBUFS ||
652  errno == ENOMEM)
653  emit error(errorstr, SocketResourceError);
654  else
655  emit error(errorstr, UnknownSocketError);
656 #endif
657  goto done_label;
658  }
659 
660  if((rc = setSocketBlockingOrNon()) == -1)
661  goto done_label;
662 
663  /*
664  ** Set the read and write buffer sizes.
665  */
666 
667  optval = m_bufferSize;
668 #ifdef Q_OS_WIN32
669  rc = setsockopt(m_socketDescriptor, SOL_SOCKET,
670  SO_RCVBUF, (const char *) &optval, (int) optlen);
671 #else
672  rc = setsockopt(m_socketDescriptor, SOL_SOCKET, SO_RCVBUF, &optval, optlen);
673 #endif
674 
675  if(rc != 0)
676  spoton_misc::logError
677  ("spoton_sctp_socket::connectToHostImplementation(): "
678  "setsockopt() failure, SO_RCVBUF.");
679 
680  optval = m_bufferSize;
681 #ifdef Q_OS_WIN32
682  rc = setsockopt(m_socketDescriptor, SOL_SOCKET,
683  SO_SNDBUF, (const char *) &optval, (int) optlen);
684 #else
685  rc = setsockopt(m_socketDescriptor, SOL_SOCKET, SO_SNDBUF, &optval, optlen);
686 #endif
687 
688  if(rc != 0)
689  spoton_misc::logError
690  ("spoton_sctp_socket::connectToHostImplementation(): "
691  "setsockopt() failure, SO_SNDBUF.");
692 
693  if(protocol == IPv4Protocol)
694  {
695  socklen_t length = 0;
696  struct sockaddr_in serveraddr;
697 
698  length = sizeof(serveraddr);
699  memset(&serveraddr, 0, sizeof(serveraddr));
700  serveraddr.sin_addr.s_addr = htonl(INADDR_ANY);
701  serveraddr.sin_family = AF_INET;
702  serveraddr.sin_port = htons(m_connectToPeerPort);
703 
704 #ifdef Q_OS_WIN32
705  rc = WSAStringToAddressA((LPSTR) m_ipAddress.toLatin1().data(),
706  AF_INET, 0, (LPSOCKADDR) &serveraddr, &length);
707 
708  if(rc != 0)
709  {
710  emit error("connectToHostImplementation()::WSAStringToAddressA()",
711  UnknownSocketError);
712  goto done_label;
713  }
714 
715  /*
716  ** Reset sin_port.
717  */
718 
719  serveraddr.sin_port = htons(m_connectToPeerPort);
720 #else
721  rc = inet_pton(AF_INET, m_ipAddress.toLatin1().constData(),
722  &serveraddr.sin_addr);
723 
724  if(rc != 1)
725  {
726  if(rc == -1)
727  {
728  if(errno == EAFNOSUPPORT)
729  emit error("connectToHostImplementation()::inet_pton()",
730  UnsupportedSocketOperationError);
731  else
732  emit error("connectToHostImplementation()::inet_pton()",
733  UnknownSocketError);
734  }
735  else
736  emit error("connectToHostImplementation()::inet_pton()",
737  UnknownSocketError);
738 
739  goto done_label;
740  }
741 #endif
742  m_state = ConnectingState;
743  rc = ::connect
744  (m_socketDescriptor, (const struct sockaddr *) &serveraddr, length);
745 
746  if(rc == 0)
747  {
748  /*
749  ** The connection was established immediately.
750  */
751 
752  m_state = ConnectedState;
753  emit connected();
754  }
755  else
756 #ifdef Q_OS_WIN32
757  rc = inspectConnectResult(rc, WSAGetLastError());
758 #else
759  rc = inspectConnectResult(rc, errno);
760 #endif
761  }
762  else
763  {
764  socklen_t length = 0;
765  struct sockaddr_in6 serveraddr;
766 
767  length = sizeof(serveraddr);
