55 static gpg_error_t
56 cmd_cat (assuan_context_t ctx, char *line)
57 {
58  assuan_fd_t fd, fdout;
59  int c;
60  FILE *fp, *fpout;
61  int nbytes;
62 
63  log_info ("got CAT command (%s)\n", line);
64 
65  fd = assuan_get_input_fd (ctx);
66  if (fd == ASSUAN_INVALID_FD)
67  return gpg_error (GPG_ERR_ASS_NO_INPUT);
68  fdout = assuan_get_output_fd (ctx);
69  if (fdout == ASSUAN_INVALID_FD)
70  return gpg_error (GPG_ERR_ASS_NO_OUTPUT);
71  fp = fdopen (fd, "r");
72  if (!fp)
73  {
74  log_error ("fdopen failed on input fd: %s\n", strerror (errno));
75  return gpg_error (GPG_ERR_ASS_GENERAL);
76  }
77 
78  fpout = fdopen (fdout, "w");
79  if (!fpout)
80  {
81  log_error ("fdopen failed on output fd: %s\n", strerror (errno));
82  fclose (fp);
83  return gpg_error (GPG_ERR_ASS_GENERAL);
84  }
85 
86  nbytes = 0;
87  while ( (c=getc (fp)) != -1)
88  {
89  putc (c, fpout);
90  nbytes++;
91  }
92  log_info ("done printing %d bytes to output fd\n", nbytes);
93 
94  /* Fixme: This also closes the original fd. */
95  fclose (fp);
96  fclose (fpout);
97  return 0;
98 }

247 static void
248 parse_std_file_handles (int *argcp, char ***argvp)
249 {
250 #ifdef HAVE_W32CE_SYSTEM
251  int argc = *argcp;
252  char **argv = *argvp;
253  const char *s;
254  assuan_fd_t fd;
255  int i;
256  int fixup = 0;
257 
258  if (!argc)
259  return;
260 
261  for (argc--, argv++; argc; argc--, argv++)
262  {
263  s = *argv;
264  if (*s == '-' && s[1] == '&' && s[2] == 'S'
265  && (s[3] == '0' || s[3] == '1' || s[3] == '2')
266  && s[4] == '='
267  && (strchr ("-01234567890", s[5]) || !strcmp (s+5, "null")))
268  {
269  if (s[5] == 'n')
270  fd = ASSUAN_INVALID_FD;
271  else
272  fd = _assuan_w32ce_finish_pipe (atoi (s+5), s[3] != '0');
273  switch (s[3] - '0')
274  {
275  case 0: my_stdin = fd; break;
276  case 1: my_stdout = fd; break;
277  case 2: my_stderr = fd; break;
278  }
279 
280  fixup++;
281  }
282  else
283  break;
284  }
285 
286  if (fixup)
287  {
288  argc = *argcp;
289  argc -= fixup;
290  *argcp = argc;
291 
292  argv = *argvp;
293  for (i=1; i < argc; i++)
294  argv[i] = argv[i + fixup];
295  for (; i < argc + fixup; i++)
296  argv[i] = NULL;
297  }
298 #else
299  (void)argcp;
300  (void)argvp;
301  my_stdin = 0;
302  my_stdout = 1;
303  my_stderr = 2;
304 #endif
305 }

394
