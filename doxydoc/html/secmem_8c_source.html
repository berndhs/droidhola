104 /* Check whether P points into the pool. */
105 static int
106 ptr_into_pool_p (const void *p)
107 {
108  /* We need to convert pointers to addresses. This is required by
109  C-99 6.5.8 to avoid undefined behaviour. Using size_t is at
110  least only implementation defined. See also
111  http://lists.gnupg.org/pipermail/gcrypt-devel/2007-February/001102.html
112  */
113  size_t p_addr = (size_t)p;
114  size_t pool_addr = (size_t)pool;
115 
116  return p_addr >= pool_addr && p_addr < pool_addr+pool_size;
117 }

135 /* Return the block following MB or NULL, if MB is the last block. */
136 static memblock_t *
137 mb_get_next (memblock_t *mb)
138 {
139  memblock_t *mb_next;
140 
141  mb_next = (memblock_t *) ((char *) mb + BLOCK_HEAD_SIZE + mb->size);
142 
143  if (! ptr_into_pool_p (mb_next))
144  mb_next = NULL;

149 /* Return the block preceding MB or NULL, if MB is the first
150  block. */
151 static memblock_t *
152 mb_get_prev (memblock_t *mb)
153 {
154  memblock_t *mb_prev, *mb_next;
155 
156  if (mb == pool)
157  mb_prev = NULL;
158  else
159  {
160  mb_prev = (memblock_t *) pool;
161  while (1)
162  {
163  mb_next = mb_get_next (mb_prev);
164  if (mb_next == mb)
165  break;
166  else
167  mb_prev = mb_next;
168  }
169  }
170 
171  return mb_prev;
172 }

174 /* If the preceding block of MB and/or the following block of MB
175  exist and are not active, merge them to form a bigger block. */
176 static void
177 mb_merge (memblock_t *mb)
178 {
179  memblock_t *mb_prev, *mb_next;
180 
181  mb_prev = mb_get_prev (mb);
182  mb_next = mb_get_next (mb);
183 
184  if (mb_prev && (! (mb_prev->flags & MB_FLAG_ACTIVE)))
185  {
186  mb_prev->size += BLOCK_HEAD_SIZE + mb->size;
187  mb = mb_prev;
188  }
189  if (mb_next && (! (mb_next->flags & MB_FLAG_ACTIVE)))
190  mb->size += BLOCK_HEAD_SIZE + mb_next->size;
191 }

193 /* Return a new block, which can hold SIZE bytes. */
194 static memblock_t *
195 mb_get_new (memblock_t *block, size_t size)
196 {
197  memblock_t *mb, *mb_split;
198 
199  for (mb = block; ptr_into_pool_p (mb); mb = mb_get_next (mb))
200  if (! (mb->flags & MB_FLAG_ACTIVE) && mb->size >= size)
201  {
202  /* Found a free block. */
203  mb->flags |= MB_FLAG_ACTIVE;
204 
205  if (mb->size - size > BLOCK_HEAD_SIZE)
206  {
207  /* Split block. */
208 
209  mb_split = (memblock_t *) (((char *) mb) + BLOCK_HEAD_SIZE + size);
210  mb_split->size = mb->size - size - BLOCK_HEAD_SIZE;
211  mb_split->flags = 0;
212 
213  mb->size = size;
214 
215  mb_merge (mb_split);
216 
217  }
218 
219  break;
220  }
221 
222  if (! ptr_into_pool_p (mb))
223  {
224  gpg_err_set_errno (ENOMEM);
225  mb = NULL;
226  }
227 
228  return mb;
229 }

239 /* Lock the memory pages into core and drop privileges. */
240 static void
241 lock_pool (void *p, size_t n)
242 {
243 #if defined(USE_CAPABILITIES) && defined(HAVE_MLOCK)
244  int err;
245 
246  {
247  cap_t cap;
248 
249  if (!no_priv_drop)
250  {
251  cap = cap_from_text ("cap_ipc_lock+ep");
252  cap_set_proc (cap);
253  cap_free (cap);
254  }
255  err = no_mlock? 0 : mlock (p, n);
256  if (err && errno)
257  err = errno;
258  if (!no_priv_drop)
259  {
260  cap = cap_from_text ("cap_ipc_lock+p");
261  cap_set_proc (cap);
262  cap_free(cap);
263  }
264  }
265 
266  if (err)
267  {
268  if (errno != EPERM
269 #ifdef EAGAIN /* OpenBSD returns this */
270  && errno != EAGAIN
271 #endif
272 #ifdef ENOSYS /* Some SCOs return this (function not implemented) */
273  && errno != ENOSYS
274 #endif
275 #ifdef ENOMEM /* Linux might return this. */
276  && errno != ENOMEM
277 #endif
278  )
279  log_error ("can't lock memory: %s\n", strerror (err));
280  show_warning = 1;
281  not_locked = 1;
282  }
283 
284 #elif defined(HAVE_MLOCK)
285  uid_t uid;
286  int err;
287 
288  uid = getuid ();
289 
290 #ifdef HAVE_BROKEN_MLOCK
291  /* Under HP/UX mlock segfaults if called by non-root. Note, we have
292  noch checked whether mlock does really work under AIX where we
293  also detected a broken nlock. Note further, that using plock ()
294  is not a good idea under AIX. */
295  if (uid)
296  {
297  errno = EPERM;
298  err = errno;
299  }
300  else
301  {
302  err = no_mlock? 0 : mlock (p, n);
303  if (err && errno)
304  err = errno;
305  }
306 #else /* !HAVE_BROKEN_MLOCK */
307  err = no_mlock? 0 : mlock (p, n);
308  if (err && errno)
309  err = errno;
310 #endif /* !HAVE_BROKEN_MLOCK */
311 
312  /* Test whether we are running setuid(0). */
313  if (uid && ! geteuid ())
314  {
315  /* Yes, we are. */
316  if (!no_priv_drop)
317  {
318  /* Check that we really dropped the privs.
319  * Note: setuid(0) should always fail */
320  if (setuid (uid) || getuid () != geteuid () || !setuid (0))
321  log_fatal ("failed to reset uid: %s\n", strerror (errno));
322  }
323  }
324 
325  if (err)
326  {
327  if (errno != EPERM
328 #ifdef EAGAIN /* OpenBSD returns this. */
329  && errno != EAGAIN
330 #endif
331 #ifdef ENOSYS /* Some SCOs return this (function not implemented). */
332  && errno != ENOSYS
333 #endif
334 #ifdef ENOMEM /* Linux might return this. */
335  && errno != ENOMEM
336 #endif
337  )
338  log_error ("can't lock memory: %s\n", strerror (err));
339  show_warning = 1;
340  not_locked = 1;
341  }
342 
343 #elif defined ( __QNX__ )
344  /* QNX does not page at all, so the whole secure memory stuff does
345  * not make much sense. However it is still of use because it
346  * wipes out the memory on a free().
347  * Therefore it is sufficient to suppress the warning. */
348  (void)p;
349  (void)n;
350 #elif defined (HAVE_DOSISH_SYSTEM) || defined (__CYGWIN__)
351  /* It does not make sense to print such a warning, given the fact that
352  * this whole Windows !@#$% and their user base are inherently insecure. */
353  (void)p;
354  (void)n;
355 #elif defined (__riscos__)
356  /* No virtual memory on RISC OS, so no pages are swapped to disc,
357  * besides we don't have mmap, so we don't use it! ;-)
358  * But don't complain, as explained above. */
359  (void)p;
360  (void)n;
361 #else
362  (void)p;
363  (void)n;
364  if (!no_mlock)
365  log_info ("Please note that you don't have secure memory on this system\n");
366 #endif
367 }

369 /* Initialize POOL. */
370 static void
371 init_pool (size_t n)
372 {
373  memblock_t *mb;
374 
375  pool_size = n;
376 
377  if (disable_secmem)
378  log_bug ("secure memory is disabled");
379 
380 
381 #if HAVE_MMAP
382  {
383  size_t pgsize;
384  long int pgsize_val;
385 
386 # if defined(HAVE_SYSCONF) && defined(_SC_PAGESIZE)
387  pgsize_val = sysconf (_SC_PAGESIZE);
388 # elif defined(HAVE_GETPAGESIZE)
389  pgsize_val = getpagesize ();
390 # else
391  pgsize_val = -1;
392 # endif
393  pgsize = (pgsize_val != -1 && pgsize_val > 0)? pgsize_val:DEFAULT_PAGE_SIZE;
394 
395  pool_size = (pool_size + pgsize - 1) & ~(pgsize - 1);
396 # ifdef MAP_ANONYMOUS
397  pool = mmap (0, pool_size, PROT_READ | PROT_WRITE,
398  MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
399 # else /* map /dev/zero instead */
400  {
401  int fd;
402 
403  fd = open ("/dev/zero", O_RDWR);
404  if (fd == -1)
405  {
406  log_error ("can't open /dev/zero: %s\n", strerror (errno));
407  pool = (void *) -1;
408  }
409  else
410  {
411  pool = mmap (0, pool_size,
412  (PROT_READ | PROT_WRITE), MAP_PRIVATE, fd, 0);
413  close (fd);
414  }
415  }
416 # endif
417  if (pool == (void *) -1)
418  log_info ("can't mmap pool of %u bytes: %s - using malloc\n",
419  (unsigned) pool_size, strerror (errno));
420  else
421  {
422  pool_is_mmapped = 1;
423  pool_okay = 1;
424  }
425  }
426 #endif /*HAVE_MMAP*/
427 
428  if (!pool_okay)
429  {
430  pool = malloc (pool_size);
431  if (!pool)
432  log_fatal ("can't allocate memory pool of %u bytes\n",
433  (unsigned) pool_size);
434  else
435  pool_okay = 1;
436  }
437 
438  /* Initialize first memory block. */
439  mb = (memblock_t *) pool;
440  mb->size = pool_size;
441  mb->flags = 0;
442 }

496  {
497  cap_t cap;
498

503 
504 #elif !defined(HAVE_DOSISH_SYSTEM)
505  uid_t uid;
506 
507  disable_secmem = 1;

510  {
511  if (setuid (uid) || getuid () != geteuid () || !setuid (0))
512  log_fatal ("failed to drop setuid\n");
513  }
514 #endif
515  }
516  else
517  {

520  if (! pool_okay)

522  init_pool (n);
523  lock_pool (pool, n);
524  }
525  else
526  log_error ("Oops, secure memory pool already initialized\n");
527  }
528 }

561 _gcry_secmem_malloc_internal (size_t size)
562 {
563  memblock_t *mb;
564 
565  if (!pool_okay)
566  {
567  /* Try to initialize the pool if the user forgot about it. */
568  secmem_init (STANDARD_POOL_SIZE);
569  if (!pool_okay)
570  {
571  log_info (_("operation is not possible without "

575  }
576  }

578  {

580  gpg_err_set_errno (ENOMEM);

584  {
585  show_warning = 0;
586  print_warn ();
587  }
588

590  size = ((size + 31) / 32) * 32;

593  if (mb)

612 _gcry_secmem_free_internal (void *a)
613 {
614  memblock_t *mb;
615  int size;
616 
617  if (!a)
618  return;
619 
620  mb = ADDR_TO_BLOCK (a);

622 
623  /* This does not make much sense: probably this memory is held in the
624  * cache. We do it anyway: */
