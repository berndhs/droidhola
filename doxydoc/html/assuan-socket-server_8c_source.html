54 static gpg_error_t
55 accept_connection_bottom (assuan_context_t ctx)
56 {
57  assuan_fd_t fd = ctx->connected_fd;
58 
59  TRACE (ctx, ASSUAN_LOG_SYSIO, "accept_connection_bottom", ctx);
60 
61  ctx->peercred_valid = 0;
62 #ifdef HAVE_SO_PEERCRED
63  {
64  struct ucred cr;
65  socklen_t cl = sizeof cr;
66 
67  if ( !getsockopt (fd, SOL_SOCKET, SO_PEERCRED, &cr, &cl))
68  {
69  ctx->peercred.pid = cr.pid;
70  ctx->peercred.uid = cr.uid;
71  ctx->peercred.gid = cr.gid;
72  ctx->peercred_valid = 1;
73 
74  /* This overrides any already set PID if the function returns
75  a valid one. */
76  if (cr.pid != ASSUAN_INVALID_PID && cr.pid)
77  ctx->pid = cr.pid;
78  }
79  }
80 #elif defined (HAVE_GETPEERUCRED)
81  {
82  ucred_t *ucred = NULL;
83 
84  if (getpeerucred (fd, &ucred) != -1)
85  {
86  ctx->peercred.uid = ucred_geteuid (ucred);
87  ctx->peercred.gid = ucred_getegid (ucred);
88  ctx->peercred.pid = ucred_getpid (ucred);
89  ctx->peercred_valid = 1;
90  ucred_free (ucred);
91  }
92  }
93 #elif defined (HAVE_LOCAL_PEEREID)
94  {
95  struct unpcbid unp;
96  socklen_t unpl = sizeof unp;
97 
98  if (getsockopt (fd, 0, LOCAL_PEEREID, &unp, &unpl) != -1)
99  {
100  ctx->peercred.pid = unp.unp_pid;
101  ctx->peercred.uid = unp.unp_euid;
102  ctx->peercred.gid = unp.unp_egid;
103  ctx->peercred_valid = 1;
104  }
105  }
106 #elif defined(HAVE_GETPEEREID)
107  {
108  if (getpeereid (fd, &ctx->peercred.uid, &ctx->peercred.gid) != -1)
109  {
110  ctx->peercred.pid = ASSUAN_INVALID_PID;
111  ctx->peercred_valid = 1;
112  }
113  }
114 #endif
115 
116  ctx->inbound.fd = fd;
117  ctx->inbound.eof = 0;
118  ctx->inbound.linelen = 0;
119  ctx->inbound.attic.linelen = 0;
120  ctx->inbound.attic.pending = 0;
121 
122  ctx->outbound.fd = fd;
123  ctx->outbound.data.linelen = 0;
124  ctx->outbound.data.error = 0;
125 
126  ctx->flags.confidential = 0;
127 
128  return 0;
129 }

132 static gpg_error_t
133 accept_connection (assuan_context_t ctx)
134 {
135  assuan_fd_t fd;
136  struct sockaddr_un clnt_addr;
137  socklen_t len = sizeof clnt_addr;
138 
139  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "accept_connection", ctx,
140  "listen_fd=0x%x", ctx->listen_fd);
141 
142  fd = SOCKET2HANDLE(accept (HANDLE2SOCKET(ctx->listen_fd),
143  (struct sockaddr*)&clnt_addr, &len ));
144  if (fd == ASSUAN_INVALID_FD)
145  {
146  return _assuan_error (ctx, gpg_err_code_from_syserror ());
147  }
148  TRACE1 (ctx, ASSUAN_LOG_SYSIO, "accept_connection", ctx,
149  "fd->0x%x", fd);
150  if (_assuan_sock_check_nonce (ctx, fd, &ctx->listen_nonce))
151  {
152  _assuan_close (ctx, fd);
153  return _assuan_error (ctx, GPG_ERR_ASS_ACCEPT_FAILED);
154  }
155 
156  ctx->connected_fd = fd;
157  return accept_connection_bottom (ctx);
158 }

165 gpg_error_t
166 assuan_init_socket_server (assuan_context_t ctx, assuan_fd_t fd,
167  unsigned int flags)
168 {
169  gpg_error_t rc;
170  TRACE_BEG2 (ctx, ASSUAN_LOG_CTX, "assuan_init_socket_server", ctx,
171  "fd=0x%x, flags=0x%x", fd, flags);
172 
173  rc = _assuan_register_std_commands (ctx);
174  if (rc)
175  return TRACE_ERR (rc);
176 
177  ctx->engine.release = _assuan_server_release;
178  ctx->engine.readfnc = _assuan_simple_read;
179  ctx->engine.writefnc = _assuan_simple_write;
180  ctx->engine.sendfd = NULL;
181  ctx->engine.receivefd = NULL;
182  ctx->is_server = 1;
183  if (flags & ASSUAN_SOCKET_SERVER_ACCEPTED)
184  /* We want a second accept to indicate EOF. */
185  ctx->max_accepts = 1;
186  else
187  ctx->max_accepts = -1;
188  ctx->input_fd = ASSUAN_INVALID_FD;
189  ctx->output_fd = ASSUAN_INVALID_FD;
190 
191  ctx->inbound.fd = ASSUAN_INVALID_FD;
192  ctx->outbound.fd = ASSUAN_INVALID_FD;
193 
194  if (flags & ASSUAN_SOCKET_SERVER_ACCEPTED)
195  {
196  ctx->listen_fd = ASSUAN_INVALID_FD;
197  ctx->connected_fd = fd;
198  }
199  else
200  {
201  ctx->listen_fd = fd;
202  ctx->connected_fd = ASSUAN_INVALID_FD;
203  }
204  ctx->accept_handler = ((flags & ASSUAN_SOCKET_SERVER_ACCEPTED)
205  ? accept_connection_bottom
206  : accept_connection);
207  ctx->finish_handler = _assuan_server_finish;
208 
209  if (flags & ASSUAN_SOCKET_SERVER_FDPASSING)
210  _assuan_init_uds_io (ctx);
211 
212  rc = _assuan_register_std_commands (ctx);
213  if (rc)
214  _assuan_reset (ctx);
215  return TRACE_ERR (rc);
216 }
